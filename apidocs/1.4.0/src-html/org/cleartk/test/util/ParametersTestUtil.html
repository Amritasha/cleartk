049  public static void testParameterDefinitions(String outputDirectory, String... excludeFiles)
050      throws ClassNotFoundException {
051    IOFileFilter includeFilter = new SuffixFileFilter(".java");
052
053    if (excludeFiles != null) {
054      IOFileFilter excludeFilter = FileFilterUtils.notFileFilter(new SuffixFileFilter(excludeFiles));
055      includeFilter = FileFilterUtils.and(excludeFilter, includeFilter);
056    }
057
058    Iterator<File> files = org.apache.commons.io.FileUtils.iterateFiles(
059        new File(outputDirectory),
060        includeFilter,
061        TrueFileFilter.INSTANCE);
062    testParameterDefinitions(files);
063  }
064
065  public static void testParameterDefinitions(Iterator<File> files) throws ClassNotFoundException {
066    List<String> badParameters = new ArrayList<String>();
067    List<String> missingParameterNameFields = new ArrayList<String>();
068
069    while (files.hasNext()) {
070      File file = files.next();
071      String className = file.getPath();
072      className = className.substring(14);
073      className = className.substring(0, className.length() - 5);
074      className = className.replace(File.separatorChar, '.');
075      Class<?> cls = Class.forName(className);
076      Field[] fields = cls.getDeclaredFields();
077      for (Field field : fields) {
078        if (ConfigurationParameterFactory.isConfigurationParameterField(field)) {
079          org.uimafit.descriptor.ConfigurationParameter annotation = field.getAnnotation(org.uimafit.descriptor.ConfigurationParameter.class);
080          String parameterName = annotation.name();
081          String expectedName = className + "." + field.getName();
082          if (parameterName.equals(org.uimafit.descriptor.ConfigurationParameter.USE_FIELD_NAME))
083            expectedName = org.uimafit.descriptor.ConfigurationParameter.USE_FIELD_NAME;
084          if (!expectedName.equals(parameterName)) {
085            badParameters.add("'" + parameterName + "' should be '" + expectedName + "'");
086          }
087
088          expectedName = className + "." + field.getName(); // change it back to the real parameter
089                                                            // name
090          String fieldName = getParameterNameField(expectedName);
091          try {
092            Field fld = cls.getDeclaredField(fieldName);
093            if ((fld.getModifiers() & Modifier.PUBLIC) == 0
094                || (fld.getModifiers() & Modifier.FINAL) == 0
095                || (fld.getModifiers() & Modifier.PUBLIC) == 0) {
096              missingParameterNameFields.add(expectedName);
097            } else if (!fld.get(null).equals(expectedName)) {
098              missingParameterNameFields.add(expectedName);
099            }
100          } catch (Exception e) {
101            missingParameterNameFields.add(expectedName);
102          }
103        }
104      }
105    }
106
107    if (badParameters.size() > 0 || missingParameterNameFields.size() > 0) {
108      String message = String.format(
109          "%d descriptor parameters with bad names and %d descriptor parameters with no name field. ",
110          badParameters.size(),
111          missingParameterNameFields.size());
112      System.err.println(message);
113      System.err.println("descriptor parameters with bad names: ");
114      for (String badParameter : badParameters) {
115        System.err.println(badParameter);
116      }
117      System.err.println("each configuration parameter should have a public static final String that specifies its name.  The missing fields are: ");
118      for (String missingParameterNameField : missingParameterNameFields) {
119        System.err.println(missingParameterNameField + " should be named by "
120            + missingParameterNameField.substring(0, missingParameterNameField.lastIndexOf('.'))
121            + "." + getParameterNameField(missingParameterNameField));
122      }
123      Assert.fail(message);
124    }
125  }
126
127  private static String getParameterNameField(String parameterName) {
128    String parameterNameField = "PARAM";
129
130    String fieldName = parameterName.substring(parameterName.lastIndexOf('.') + 1);
131    String[] fieldNameParts = fieldName.split("(?=[A-Z]++)");
132    for (String fieldNamePart : fieldNameParts) {
133      parameterNameField += "_" + fieldNamePart.toUpperCase();
134    }
135    return parameterNameField;
136  }
