063public class MaltParser extends JCasAnnotator_ImplBase {
064
065  @Deprecated
066  public static TypeSystemDescription TYPE_SYSTEM_DESCRIPTION = TypeSystemDescriptionFactory.createTypeSystemDescription("org.cleartk.TypeSystem");
067
068  public static final String ENGMALT_RESOURCE_NAME = "/models/engmalt.linear-1.7.mco";
069
070  public static AnalysisEngineDescription getDescription() throws ResourceInitializationException {
071    // get the resource path and strip the ".mco" suffix
072    String fileName = MaltParser.class.getResource(ENGMALT_RESOURCE_NAME).getFile();
073    String fileBase = fileName.substring(0, fileName.length() - 4);
074    return getDescription(fileBase);
075  }
076
077  public static AnalysisEngineDescription getDescription(String modelFileName)
078      throws ResourceInitializationException {
079    return AnalysisEngineFactory.createPrimitiveDescription(
080        MaltParser.class,
081        TYPE_SYSTEM_DESCRIPTION,
082        PARAM_MODEL_FILE_NAME,
083        modelFileName);
084  }
085
086  @ConfigurationParameter(
087      description = "The path to the model file, without the .mco suffix.",
088      mandatory = true)
089  private String modelFileName;
090
091  public static final String PARAM_MODEL_FILE_NAME = ConfigurationParameterFactory.createConfigurationParameterName(
092      MaltParser.class,
093      "modelFileName");
094
095  private MaltParserService service;
096
097  @Override
098  public void initialize(UimaContext context) throws ResourceInitializationException {
099    super.initialize(context);
100    try {
101      OptionManager.instance().loadOptionDescriptionFile();
102      OptionManager.instance().getOptionDescriptions().generateMaps();
103      this.service = new MaltParserService();
104      File modelFile = new File(this.modelFileName);
105      String fileName = modelFile.getName();
106      String workingDirectory = modelFile.getParent();
107      String command = String.format("-c %s -m parse -w %s", fileName, workingDirectory);
108      this.service.initializeParserModel(command);
109    } catch (MaltChainedException e) {
110      throw new ResourceInitializationException(e);
111    }
112  }
113
114  @Override
115  public void collectionProcessComplete() throws AnalysisEngineProcessException {
116    super.collectionProcessComplete();
117    try {
118      this.service.terminateParserModel();
119    } catch (MaltChainedException e) {
120      throw new AnalysisEngineProcessException(e);
121    }
122  }
123
124  @Override
125  public void process(JCas jCas) throws AnalysisEngineProcessException {
126    for (Sentence sentence : JCasUtil.select(jCas, Sentence.class)) {
127      List<Token> tokens = JCasUtil.selectCovered(jCas, Token.class, sentence);
128
129      // convert tokens into MaltParser input array
130      List<String> inputStrings = new ArrayList<String>();
131      int lineNo = -1;
132      for (Token token : tokens) {
133        lineNo += 1;
134        String text = token.getCoveredText();
135        String pos = token.getPos();
136
137        // line format is <index>\t<word>\t_\t<pos>\t<pos>\t_
138        String lineFormat = "%1$d\t%2$s\t_\t%3$s\t%3$s\t_";
139        inputStrings.add(String.format(lineFormat, lineNo + 1, text, pos));
140      }
141
142      try {
143        // parse with MaltParser
144        String[] input = inputStrings.toArray(new String[inputStrings.size()]);
145        DependencyStructure graph = this.service.parse(input);
146
147        // convert MaltParser structure into annotations
148        Map<Integer, DependencyNode> nodes = new HashMap<Integer, DependencyNode>();
149        SortedSet<Integer> tokenIndices = graph.getTokenIndices();
150        for (int i : tokenIndices) {
151          org.maltparser.core.syntaxgraph.node.DependencyNode maltNode = graph.getTokenNode(i);
152          Token token = tokens.get(maltNode.getIndex() - 1);
153          DependencyNode node;
154          if (maltNode.getHead().getIndex() != 0) {
155            node = new DependencyNode(jCas, token.getBegin(), token.getEnd());
156          } else {
157            node = new TopDependencyNode(jCas, token.getBegin(), token.getEnd());
158          }
159          nodes.put(i, node);
160        }
161
162        // add relation annotations
163        Map<DependencyNode, List<DependencyRelation>> headRelations;
164        headRelations = new HashMap<DependencyNode, List<DependencyRelation>>();
165        Map<DependencyNode, List<DependencyRelation>> childRelations;
166        childRelations = new HashMap<DependencyNode, List<DependencyRelation>>();
167        SymbolTable table = graph.getSymbolTables().getSymbolTable("DEPREL");
168        for (int i : tokenIndices) {
169          org.maltparser.core.syntaxgraph.node.DependencyNode maltNode = graph.getTokenNode(i);
170          int headIndex = maltNode.getHead().getIndex();
171          if (headIndex != 0) {
172            String label = maltNode.getHeadEdge().getLabelSymbol(table);
173            DependencyNode node = nodes.get(i);
174            DependencyNode head = nodes.get(headIndex);
175            DependencyRelation rel = new DependencyRelation(jCas);
176            rel.setHead(head);
177            rel.setChild(node);
178            rel.setRelation(label);
179            rel.addToIndexes();
180            if (!headRelations.containsKey(node)) {
181              headRelations.put(node, new ArrayList<DependencyRelation>());
182            }
183            headRelations.get(node).add(rel);
184            if (!childRelations.containsKey(head)) {
185              childRelations.put(head, new ArrayList<DependencyRelation>());
186            }
187            childRelations.get(head).add(rel);
188          }
189        }
190
191        // finalize nodes: add links between nodes and relations
192        for (DependencyNode node : nodes.values()) {
193          List<DependencyRelation> heads = headRelations.get(node);
194          node.setHeadRelations(new FSArray(jCas, heads == null ? 0 : heads.size()));
195          if (heads != null) {
196            FSCollectionFactory.fillArrayFS(node.getHeadRelations(), heads);
197          }
198          List<DependencyRelation> children = childRelations.get(node);
199          node.setChildRelations(new FSArray(jCas, children == null ? 0 : children.size()));
200          if (children != null) {
201            FSCollectionFactory.fillArrayFS(node.getChildRelations(), children);
202          }
203          node.addToIndexes();
204        }
205
206      } catch (MaltChainedException e) {
207        throw new AnalysisEngineProcessException(e);
208      }
209    }
210  }
