019package org.cleartk.syntax.berkeley;
020
021import java.util.ArrayList;
022import java.util.List;
023
024import org.apache.uima.jcas.JCas;
025import org.cleartk.token.type.Sentence;
026import org.cleartk.token.type.Token;
027import org.uimafit.util.JCasUtil;
028
029/**
030 * <br>
031 * Copyright (c) 2012, Regents of the University of Colorado <br>
032 * All rights reserved.
033 * <p>
034 * 
035 * @author Philip Ogren
036 */
037
038public class DefaultInputTypesHelper extends InputTypesHelper<Token, Sentence> {
039
040  public List<Token> getTokens(JCas jCas, Sentence sentence) {
041    return JCasUtil.selectCovered(jCas, Token.class, sentence);
042  }
043
044  @Override
045  public String getPosTag(Token token) {
046    return token.getPos();
047  }
048
049  @Override
050  public void setPosTag(Token token, String tag) {
051    token.setPos(tag);
052  }
053
054  public List<Sentence> getSentences(JCas jCas) {
055    return new ArrayList<Sentence>(JCasUtil.select(jCas, Sentence.class));
056  }
057
058}
