024package org.cleartk.corpus.conll2005;
025
026import java.io.BufferedReader;
027import java.io.File;
028import java.io.FileInputStream;
029import java.io.IOException;
030import java.io.InputStream;
031import java.io.InputStreamReader;
032import java.net.URI;
033import java.net.URISyntaxException;
034import java.util.zip.GZIPInputStream;
035
036import org.apache.uima.UimaContext;
037import org.apache.uima.cas.CASException;
038import org.apache.uima.collection.CollectionException;
039import org.apache.uima.collection.CollectionReader;
040import org.apache.uima.jcas.JCas;
041import org.apache.uima.resource.ResourceInitializationException;
042import org.apache.uima.util.Progress;
043import org.apache.uima.util.ProgressImpl;
044import org.cleartk.util.ViewURIUtil;
045import org.uimafit.component.JCasCollectionReader_ImplBase;

047import org.uimafit.descriptor.SofaCapability;
048import org.uimafit.factory.CollectionReaderFactory;
049import org.uimafit.factory.ConfigurationParameterFactory;

056@SofaCapability(outputSofas = { Conll2005Constants.CONLL_2005_VIEW, ViewURIUtil.URI })
057public class Conll2005GoldReader extends JCasCollectionReader_ImplBase {
058
059  public static CollectionReader getCollectionReader(String conll2005DataFile)
060      throws ResourceInitializationException {
061    return CollectionReaderFactory.createCollectionReader(
062        Conll2005GoldReader.class,
063        PARAM_CONLL2005_DATA_FILE,
064        conll2005DataFile);
065  }
066
067  @ConfigurationParameter(mandatory = true, description = "the path of the CoNLL 2005 data file")
068  private File conll2005DataFile;
069
070  public static final String PARAM_CONLL2005_DATA_FILE = ConfigurationParameterFactory.createConfigurationParameterName(
071      Conll2005GoldReader.class,
072      "conll2005DataFile");
073
074  private BufferedReader reader;
075
076  private boolean finished = false;
077
078  private int documentNumber;
079
080  private int totalDocuments;
081
082  @Override
083  public void initialize(UimaContext context) throws ResourceInitializationException {
084    try {
085      this.reader = this.getBufferedReader();
086      String line;
087      this.totalDocuments = 0;
088      do {
089        line = this.reader.readLine();
090        while (line != null && line.trim().length() == 0) {
091          line = this.reader.readLine();
092        }
093        if (line == null) {
094          break;
095        }
096        this.totalDocuments += 1;
097        while (line != null && line.trim().length() > 0) {
098          line = this.reader.readLine();
099        }
100      } while (line != null);
101      this.reader.close();
102
103      this.reader = this.getBufferedReader();
104      documentNumber = 0;
105
106    } catch (IOException e) {
107      throw new ResourceInitializationException(e);
108    }
109  }
110
111  private BufferedReader getBufferedReader() throws IOException {
112    InputStream in;
113    if (this.conll2005DataFile.getName().endsWith(".gz")) {
114      in = new GZIPInputStream(new FileInputStream(this.conll2005DataFile));
115    } else {
116      in = new FileInputStream(this.conll2005DataFile);
117    }
118    return new BufferedReader(new InputStreamReader(in));
119  }
120
121  public void getNext(JCas jCas) throws IOException, CollectionException {
122    try {
123      JCas conllView = jCas.createView(Conll2005Constants.CONLL_2005_VIEW);
124
125      String lineBuffer;
126      StringBuffer docBuffer = new StringBuffer();
127
128      lineBuffer = reader.readLine();
129      while (lineBuffer != null && lineBuffer.trim().length() == 0) {
130        lineBuffer = reader.readLine();
131      }
132
133      if (lineBuffer == null) {
134        throw new CollectionException("unexpected end of input", null);
135      }
136
137      while (lineBuffer != null && lineBuffer.trim().length() != 0) {
138        docBuffer.append(lineBuffer.trim());
139        docBuffer.append("\n");
140        lineBuffer = reader.readLine();
141      }
142
143      documentNumber += 1;
144
145      if (documentNumber == totalDocuments) {
146        finished = true;
147      }
148
149      conllView.setSofaDataString(docBuffer.toString(), "text/plain");
150      URI fileURI = this.conll2005DataFile.toURI();
151      String fragment = String.valueOf(this.documentNumber);
152      URI uri;
153      try {
154        uri = new URI(fileURI.getScheme(), fileURI.getHost(), fileURI.getPath(), fragment);
155      } catch (URISyntaxException e) {
156        // should never reach this; fragment should always be valid since it's just a number
157        throw new RuntimeException(e);
158      }
159      ViewURIUtil.setURI(jCas, uri);
160    } catch (CASException e) {
161      throw new CollectionException(e);
162    }
163  }
164
165  public void close() throws IOException {
166    reader.close();
167  }
168
169  public Progress[] getProgress() {
170    return new Progress[] { new ProgressImpl(documentNumber, totalDocuments, Progress.ENTITIES) };
171  }
172
173  public boolean hasNext() throws IOException, CollectionException {
174    return !finished;
175  }
176}
