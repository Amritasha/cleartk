052/**
053 * <br>
054 * Copyright (c) 2007-2008, Regents of the University of Colorado <br>
055 * All rights reserved.
056 * 
057 * 
058 * <p>
059 * PropbankCollectionReader reads all <tt>.mrg</tt> files of the WSJ part of Treebank in lexical
060 * order, then reads the corresponding Propbank entries, and populates the "TreebankView" and
061 * "PropbankView" SOFAs.
062 * </p>
063 * 
064 * @author Philip Ogren, Philipp Wetzler
065 */
066
067@SofaCapability(outputSofas = {
068    PropbankConstants.PROPBANK_VIEW,
069    PennTreebankReader.TREEBANK_VIEW,
070    ViewURIUtil.URI })
071public class PropbankGoldReader extends JCasCollectionReader_ImplBase {
072
073  public static final String PARAM_PROPBANK_FILE_NAME = ConfigurationParameterFactory.createConfigurationParameterName(
074      PropbankGoldReader.class,
075      "propbankFileName");
076
077  @ConfigurationParameter(description = "points to propbank data file", mandatory = true)
078  private String propbankFileName;
079
080  public static final String PARAM_PENNTREEBANK_DIRECTORY_NAME = ConfigurationParameterFactory.createConfigurationParameterName(
081      PropbankGoldReader.class,
082      "penntreebankDirectoryName");
083
084  private static final String PENN_TREEBANK_DIRECTORY_DESCRIPTION = "points to the PennTreebank corpus. "
085      + "The directory should contain subdirectories corresponding to the sections (e.g. \"00\", \"01\", etc.)  "
086      + "That is, if a local copy of PennTreebank sits at C:/Data/PTB/wsj/mrg, then the subdirectory C:/Data/PTB/wsj/mrg/00 should exist. "
087      + "There are 24 sections in PTB corresponding to the directories 00, 01, 02, ... 24.";
088
089  @ConfigurationParameter(description = PENN_TREEBANK_DIRECTORY_DESCRIPTION, mandatory = true)
090  private String penntreebankDirectoryName;
091
092  public static final String PARAM_WSJ_SECTIONS = ConfigurationParameterFactory.createConfigurationParameterName(
093      PropbankGoldReader.class,
094      "wsjSections");
095
096  @ConfigurationParameter(
097      description = "Determines which sections of WSJ will be used.  The format allows for comma-separated section numbers and section ranges, for example \"02,07-12,16\".",
098      mandatory = true)
099  private String wsjSections;
100
101  /**
102   * holds all of the propbank data from props.txt. One entry per line in the file
103   */
104  protected LinkedList<String> propbankData;
105
106  protected File treebankDirectory;
107
108  protected LinkedList<File> treebankFiles;
109
110  protected int totalTreebankFiles = 0;
111
112  protected ListSpecification wsjSpecification;
113
114  @Override
115  public void initialize(UimaContext context) throws ResourceInitializationException {
116    try {
117      this.wsjSpecification = new ListSpecification(wsjSections);
118
119      File propbankFile = new File(propbankFileName);
120      if (!propbankFile.exists()) {
121        throw CleartkInitializationException.fileNotFound(propbankFile);
122      }
123      BufferedReader reader = new BufferedReader(new FileReader(propbankFile));
124      try {
125        propbankData = new LinkedList<String>();
126        String line;
127        while ((line = reader.readLine()) != null) {
128          propbankData.add(line);
129        }
130      } finally {
131        reader.close();
132      }
133      Collections.sort(propbankData);
134
135      this.treebankFiles = new LinkedList<File>();
136      treebankDirectory = new File(penntreebankDirectoryName);
137      // don't forget that the paths in props.txt have "wsj" in the name.
138      File wsjDirectory = new File(treebankDirectory, "wsj");
139      if (!wsjDirectory.exists()) {
140        throw CleartkInitializationException.fileNotFound(wsjDirectory);
141      }
142      PennTreebankReader.collectSections(wsjDirectory, this.treebankFiles, this.wsjSpecification);
143      Collections.sort(treebankFiles);
144      this.totalTreebankFiles = treebankFiles.size();
145
146    } catch (FileNotFoundException fnfe) {
147      throw new ResourceInitializationException(fnfe);
148    } catch (IOException ioe) {
149      throw new ResourceInitializationException(ioe);
150    }
151  }
152
153  /**
154   * Reads the next file and stores its text in <b>cas</b> as the "TreebankView" SOFA. Then stores
155   * the corresponding Propbank entries in the "PropbankView" SOFA.
156   */
157  public void getNext(JCas jCas) throws IOException, CollectionException {
158    JCas tbView, pbView;
159    try {
160      tbView = jCas.createView(PennTreebankReader.TREEBANK_VIEW);
161      pbView = jCas.createView(PropbankConstants.PROPBANK_VIEW);
162    } catch (CASException ce) {
163      throw new CollectionException(ce);
164    }
165
166    File treebankFile = treebankFiles.removeFirst();
167    ViewURIUtil.setURI(jCas, treebankFile.toURI());
168
169    StringBuffer propbankText = new StringBuffer();
170
171    /*
172     * The logic here is rather fragile and should be rewritten and/or unit tested. I changed the
173     * code so that the comparison is between the canonical paths. (PVO)
174     */
175    while (propbankData.size() > 0) {
176      File nextPbFile = new File(treebankDirectory.getPath() + File.separator
177          + Propbank.filenameFromString(propbankData.getFirst())).getCanonicalFile();
178
179      int c = treebankFile.getCanonicalFile().compareTo(nextPbFile);
180      if (c < 0) {
181        break;
182      } else if (c > 0) {
183        propbankData.removeFirst();
184        continue;
185      }
186
187      propbankText.append(propbankData.removeFirst() + "\n");
188    }
189
190    tbView.setSofaDataString(FileUtils.file2String(treebankFile), "text/plain");
191    pbView.setSofaDataString(propbankText.toString(), "text/plain");
192  }
193
194  public void close() throws IOException {
195  }
196
197  public Progress[] getProgress() {
198    return new Progress[] { new ProgressImpl(
199        totalTreebankFiles - treebankFiles.size(),
200        totalTreebankFiles,
201        Progress.ENTITIES) };
202  }
203
204  public boolean hasNext() throws IOException, CollectionException {
205    if (treebankFiles.size() > 0)
206      return true;
207    else
208      return false;
209  }
210
211  public void setPropbankFileName(String propbankFileName) {
212    this.propbankFileName = propbankFileName;
213  }
214
215  public void setPenntreebankDirectoryName(String treebankDirectoryName) {
216    this.penntreebankDirectoryName = treebankDirectoryName;
217  }
218
219  public void setWsjSections(String wsjSections) {
220    this.wsjSections = wsjSections;
221  }
