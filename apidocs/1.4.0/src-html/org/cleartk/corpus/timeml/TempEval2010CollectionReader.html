087  public static CollectionReader getCollectionReader(String... dataPaths)
088      throws ResourceInitializationException {
089    List<File> dirs = new ArrayList<File>();
090    for (String path : dataPaths) {
091      dirs.add(new File(path));
092    }
093    return getCollectionReader(dirs);
094  }

101  public static CollectionReader getCollectionReader(
102      List<File> dataDirectories,
103      Set<String> selectedFileNames) throws ResourceInitializationException {
104    // workaround UimaFIT limitation
105    List<String> dirsList = new ArrayList<String>();
106    for (File dir : dataDirectories) {
107      dirsList.add(dir.getPath());
108    }
109    String[] dirs = dirsList.toArray(new String[dirsList.size()]);
110    String[] names = selectedFileNames == null
111        ? null
112        : selectedFileNames.toArray(new String[selectedFileNames.size()]);
113    return CollectionReaderFactory.createCollectionReader(
114        TempEval2010CollectionReader.class,
115        null,
116        PARAM_DATA_DIRECTORIES,
117        dirs,
118        PARAM_SELECTED_FILE_NAMES,
119        names);
120  }

142  private Map<String, Map<String, String>> viewFileTexts;
143
144  @Override
145  public void initialize(UimaContext context) throws ResourceInitializationException {
146    super.initialize(context);
147
148    try {
149      // assemble URIs for each file in the data
150      this.uriIndex = 0;
151      this.uris = new ArrayList<URI>();
152      for (File dataDirectory : this.dataDirectories) {
153        URI dataURI = dataDirectory.toURI();
154        for (String fileName : getAnnotatedFileNames(dataDirectory)) {
155          if (this.selectedFileNames == null || this.selectedFileNames.contains(fileName)) {
156            URI uri = new URI(dataURI.getScheme(), dataURI.getHost(), dataURI.getPath(), fileName);
157            this.uris.add(uri);
158          }
159        }
160      }
161
162      // group lines by filename
163      this.viewFileTexts = new HashMap<String, Map<String, String>>();
164      for (String viewName : Arrays.asList(
165          BASE_SEGMENTATION_VIEW_NAME,
166          DCT_VIEW_NAME,
167          EVENT_EXTENTS_VIEW_NAME,
168          EVENT_ATTRIBUTES_VIEW_NAME,
169          TIMEX_EXTENTS_VIEW_NAME,
170          TIMEX_ATTRIBUTES_VIEW_NAME,
171          TLINK_DCT_EVENT_VIEW_NAME,
172          TLINK_MAIN_EVENTS_VIEW_NAME,
173          TLINK_SUBORDINATED_EVENTS_VIEW_NAME,
174          TLINK_TIMEX_EVENT_VIEW_NAME)) {
175        // assumes view names are the same as the .tab file names
176        this.viewFileTexts.put(viewName, this.textByFileName(viewName));
177      }
178    } catch (IOException e) {
179      throw new ResourceInitializationException(e);
180    } catch (URISyntaxException e) {
181      throw new ResourceInitializationException(e);
182    }
183  }
184
185  @Override
186  public boolean hasNext() throws IOException, CollectionException {
187    return this.uriIndex < this.uris.size();
188  }
189
190  @Override
191  public void getNext(JCas jCas) throws IOException, CollectionException {
192    URI uri = this.uris.get(this.uriIndex);
193    this.uriIndex += 1;
194    ViewURIUtil.setURI(jCas, uri);
195
196    String fileName = uri.getFragment();
197    for (String viewName : this.viewFileTexts.keySet()) {
198      JCas view;
199      try {
200        view = jCas.createView(viewName);
201      } catch (CASException e) {
202        throw new CollectionException(e);
203      }
204      String text = this.viewFileTexts.get(viewName).get(fileName);
205      view.setDocumentText(text == null ? "" : text);
206    }
207  }
208
209  @Override
210  public Progress[] getProgress() {
211    return new Progress[] { new ProgressImpl(this.uriIndex, this.uris.size(), Progress.ENTITIES) };
212  }

214  private Map<String, String> textByFileName(String tabFileName) throws IOException {
215    // get all variants of the file under all subdirectories
216    List<File> files = new ArrayList<File>();
217    for (File dir : this.dataDirectories) {
218      files.addAll(getTempEvalFiles(dir, tabFileName));
219    }
220
221    // map each file name to its lines
222    ListMultimap<String, String> fileLines = ArrayListMultimap.create();
223    for (File file : files) {
224      for (String line : Files.readLines(file, Charsets.US_ASCII)) {
225        String fileName = getAnnotatedFileName(line);
226        fileLines.put(fileName, line);
227      }
228    }
229
230    // convert lists of lines back into text
231    Map<String, String> fileTexts = new HashMap<String, String>();
232    for (String fileName : fileLines.keySet()) {
233      StringBuilder builder = new StringBuilder();
234      for (String line : fileLines.get(fileName)) {
235        builder.append(line).append('\n');
236      }
237      fileTexts.put(fileName, builder.toString());
238    }
239    return fileTexts;
240  }

242  private static List<File> getTempEvalFiles(File dataDirectory, String tabFileName)
243      throws FileNotFoundException {
244
245    // subdirectory is "data" for training, and both "entities" and "relations" for testing
246    List<File> files = new ArrayList<File>();
247    for (String subDir : Arrays.asList("data", "key")) {
248      files.add(new File(new File(dataDirectory, subDir), tabFileName));
249    }
250    // weird special case: dct.txt is dct-en.txt in testing base directory
251    files.add(new File(dataDirectory, tabFileName.replaceAll("\\.txt", "-en.txt")));
252
253    // filter existing files
254    List<File> existingFiles = new ArrayList<File>();
255    for (File file : files) {
256      if (file.exists()) {
257        existingFiles.add(file);
258      }
259    }
260
261    // error if we didn't find at least one
262    if (existingFiles.size() == 0) {
263      throw new FileNotFoundException("Could not find any of " + files);
264    }
265    return existingFiles;
266  }
267
268  protected static String getAnnotatedFileName(String line) {
269    // the filename is the first column
270    String[] parts = line.split("\t", 2);
271    if (parts.length != 2) {
272      throw new IllegalArgumentException("Expected <filename>\t..., found " + line);
273    }
274    return parts[0];
275  }
276
277  public static List<String> getAnnotatedFileNames(File dataDirectory) throws IOException {
278    // look for file names in all the base segmentation files
279    List<String> fileNames = new ArrayList<String>();
280    Set<String> seenFileNames = new HashSet<String>();
281    for (File tabFile : getTempEvalFiles(dataDirectory, "base-segmentation.tab")) {
282      for (String line : Files.readLines(tabFile, Charsets.US_ASCII)) {
283
284        // add the filename to the list if we haven't already seen it
285        String fileName = getAnnotatedFileName(line);
286        if (!seenFileNames.contains(fileName)) {
287          seenFileNames.add(fileName);
288          fileNames.add(fileName);
289        }
290      }
291    }
292    return fileNames;
293  }
