024package org.cleartk.timeml.event;
025
026import org.apache.uima.UimaContext;
027import org.apache.uima.analysis_engine.AnalysisEngineDescription;
028import org.apache.uima.resource.ResourceInitializationException;
029import org.cleartk.classifier.feature.extractor.CleartkExtractor;
030import org.cleartk.classifier.feature.extractor.CleartkExtractor.Bag;
031import org.cleartk.classifier.feature.extractor.CleartkExtractor.Covered;
032import org.cleartk.classifier.feature.extractor.CleartkExtractor.Preceding;
033import org.cleartk.classifier.feature.extractor.simple.TypePathExtractor;
034import org.cleartk.classifier.liblinear.LIBLINEARStringOutcomeDataWriter;
035import org.cleartk.timeml.type.Event;
036import org.cleartk.timeml.util.CleartkInternalModelFactory;
037import org.cleartk.feature.TextSliceExtractor;
038import org.cleartk.feature.token.TokenTextForSelectedPOSExtractor;
039import org.cleartk.token.type.Token;
040import org.uimafit.factory.AnalysisEngineFactory;
041
042/**
043 * <br>
044 * Copyright (c) 2010, Regents of the University of Colorado <br>
045 * All rights reserved.
046 * 
047 * Annotator for the "tense" attribute of TimeML EVENTs.
048 * 
049 * @author Steven Bethard
050 */
051public class EventTenseAnnotator extends EventAttributeAnnotator<String> {
052
053  public static final CleartkInternalModelFactory FACTORY = new CleartkInternalModelFactory() {
054    @Override
055    public Class<?> getAnnotatorClass() {
056      return EventTenseAnnotator.class;
057    }
058
059    @Override
060    public Class<?> getDataWriterClass() {
061      return LIBLINEARStringOutcomeDataWriter.class;
062    }
063
064    @Override
065    public AnalysisEngineDescription getBaseDescription() throws ResourceInitializationException {
066      return AnalysisEngineFactory.createPrimitiveDescription(EventTenseAnnotator.class);
067    }
068  };
069
070  @Override
071  public void initialize(UimaContext context) throws ResourceInitializationException {
072    super.initialize(context);
073    this.eventFeatureExtractors.add(new TextSliceExtractor(-2));
074    this.eventFeatureExtractors.add(new CleartkExtractor(Token.class, new TypePathExtractor(
075        Token.class,
076        "pos"), new Bag(new Covered())));
077
078    this.contextExtractors.add(new CleartkExtractor(
079        Token.class,
080        new TokenTextForSelectedPOSExtractor("MD", "TO", "IN", "VB"),
081        new Preceding(3)));
082  }
083
084  @Override
085  protected String getDefaultValue() {
086    return "NONE";
087  }
088
089  @Override
090  protected String getAttribute(Event event) {
091    return event.getTense();
092  }
093
094  @Override
095  protected void setAttribute(Event event, String value) {
096    event.setTense(value);
097  }
098}
