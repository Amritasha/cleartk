024package org.cleartk.timeml.event;
025
026import java.util.ArrayList;
027import java.util.List;
028
029import org.apache.uima.UimaContext;
030import org.apache.uima.analysis_engine.AnalysisEngineProcessException;
031import org.apache.uima.jcas.JCas;
032import org.apache.uima.resource.ResourceInitializationException;
033import org.cleartk.classifier.CleartkAnnotator;
034import org.cleartk.classifier.Feature;
035import org.cleartk.classifier.Instance;
036import org.cleartk.classifier.feature.extractor.CleartkExtractor;
037import org.cleartk.classifier.feature.extractor.simple.SimpleFeatureExtractor;
038import org.cleartk.timeml.type.Event;
039import org.cleartk.token.type.Sentence;
040import org.uimafit.util.JCasUtil;
041
042/**
043 * <br>
044 * Copyright (c) 2010, Regents of the University of Colorado <br>
045 * All rights reserved.
046 * 
047 * Base class for annotators of TimeML EVENT attributes.
048 * 
049 * @author Steven Bethard
050 */
051public abstract class EventAttributeAnnotator<OUTCOME_TYPE> extends CleartkAnnotator<OUTCOME_TYPE> {
052
053  /**
054   * The list of feature extractors that will be applied directly to the Event annotation.
055   * 
056   * Subclasses should override {@link #initialize(org.apache.uima.UimaContext)} to fill this list.
057   */
058  protected List<SimpleFeatureExtractor> eventFeatureExtractors;
059
060  /**
061   * The list of feature extractors that will be applied to the Event annotation, with a Sentence
062   * window.
063   * 
064   * Subclasses should override {@link #initialize(org.apache.uima.UimaContext)} to fill this list.
065   */
066  protected List<CleartkExtractor> contextExtractors;
067
068  /**
069   * The attribute value that should be considered as a default, e.g. "NONE". When the attribute
070   * value is null in the CAS, this value will be used instead. Additionally, when the classifier
071   * produces this value, no attribute will be added.
072   * 
073   * @return The default attribute value.
074   */
075  protected abstract OUTCOME_TYPE getDefaultValue();
076
077  /**
078   * Get the attribute value from the Event annotation. Typically this will be by calling something
079   * like {@link Event#getTense()}.
080   * 
081   * If this method returns null, {@link #getDefaultValue()} will be called to produce an
082   * appropriate attribute value.
083   * 
084   * @param event
085   *          The Event annotation whose attribute is to be retrieved.
086   * @return The selected attribute value.
087   */
088  protected abstract OUTCOME_TYPE getAttribute(Event event);
089
090  /**
091   * Set the attribute value on the Event annotation. Typically this will be by calling something
092   * like {@link Event#setTense(String)}.
093   * 
094   * This method will not be called if the value is equal to {@link #getDefaultValue()}.
095   * 
096   * @param event
097   *          The Event annotation whose attribute is to be set.
098   * @param value
099   *          The new attribute value.
100   */
101  protected abstract void setAttribute(Event event, OUTCOME_TYPE value);
102
103  @Override
104  public void initialize(UimaContext context) throws ResourceInitializationException {
105    super.initialize(context);
106    this.eventFeatureExtractors = new ArrayList<SimpleFeatureExtractor>();
107    this.contextExtractors = new ArrayList<CleartkExtractor>();
108  }
109
110  @Override
111  public void process(JCas jCas) throws AnalysisEngineProcessException {
112    for (Sentence sentence : JCasUtil.select(jCas, Sentence.class)) {
113      for (Event event : JCasUtil.selectCovered(jCas, Event.class, sentence)) {
114
115        // assemble features
116        List<Feature> features = new ArrayList<Feature>();
117        for (SimpleFeatureExtractor extractor : this.eventFeatureExtractors) {
118          features.addAll(extractor.extract(jCas, event));
119        }
120        for (CleartkExtractor extractor : this.contextExtractors) {
121          features.addAll(extractor.extractWithin(jCas, event, sentence));
122        }
123
124        // if training, determine the attribute value and write the
125        // instance
126        if (this.isTraining()) {
127          OUTCOME_TYPE attribute = this.getAttribute(event);
128          if (attribute == null) {
129            attribute = this.getDefaultValue();
130          }
131          Instance<OUTCOME_TYPE> instance = new Instance<OUTCOME_TYPE>();
132          instance.addAll(features);
133          instance.setOutcome(attribute);
134          this.dataWriter.write(instance);
135        }
136
137        // if predicting, propose an attribute value and modify the
138        // event annotation
139        else {
140          OUTCOME_TYPE label = this.classifier.classify(features);
141          this.setAttribute(event, label);
142        }
143      }
144    }
145  }
146}
