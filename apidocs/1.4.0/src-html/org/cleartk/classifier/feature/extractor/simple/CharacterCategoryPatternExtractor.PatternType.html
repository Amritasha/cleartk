024package org.cleartk.classifier.feature.extractor.simple;
025
026import java.util.Collections;
027import java.util.List;
028
029import org.apache.uima.jcas.JCas;
030import org.apache.uima.jcas.tcas.Annotation;
031import org.cleartk.classifier.Feature;
032import org.cleartk.classifier.feature.extractor.CleartkExtractorException;
033
034/**
035 * A feature extractor that generates a pattern based on the <a
036 * href="http://unicode.org/reports/tr49/">Unicode categories</a> of each of the characters in the
037 * annotation text. For example, "A-z0" is an uppercase letter, followed by a dash, followed by a
038 * lowercase letter, followed by a digit, and so would get the pattern "LuPdLlNd".
039 * 
040 * <br>
041 * Copyright (c) 2011, Regents of the University of Colorado <br>
042 * All rights reserved.
043 * 
044 * @author Steven Bethard
045 */
046public class CharacterCategoryPatternExtractor implements SimpleNamedFeatureExtractor {
047
048  /**
049   * The type of pattern to generate in feature values.
050   */
051  public static enum PatternType {
052    /**
053     * The standard pattern, where one category abbreviation is added to the feature value for each
054     * character in the text.
055     */
056    ONE_PER_CHAR,
057    /**
058     * A simplified pattern, where if the same category appears many times in a row, the category is
059     * added to the feature value only once. For example "XX00" would get the pattern "LuNd" since
060     * there are two uppercase letters followed by two digits.
061     */
062    REPEATS_MERGED,
063    /**
064     * Similar to REPEATS_MERGED, but distinguishes between the same category appearing once and
065     * more than once in a row. If the same category appears twice or more in a row, then we will
066     * mark that category with a Kleene plus '+'. For example "X000" would get the pattern "LuNd+"
067     * since there is a single uppercase letter followed by more than one digit.
068     */
069    REPEATS_AS_KLEENE_PLUS
070  }
071
072  private PatternType patternType;
073
074  private String name;
075
076  /**
077   * Create the standard feature extractor, where one category is added to the feature value for
078   * each character in the text. See {@link PatternType#ONE_PER_CHAR}.
079   */
080  public CharacterCategoryPatternExtractor() {
081    this(PatternType.ONE_PER_CHAR);
082  }
083
084  /**
085   * Create a feature extractor with the specified pattern type. See {@link PatternType} for the
086   * acceptable pattern types.
087   * 
088   * @param patternType
089   *          The type of pattern to generate in feature values.
090   */
091  public CharacterCategoryPatternExtractor(PatternType patternType) {
092    this.patternType = patternType;
093    switch (this.patternType) {
094      case ONE_PER_CHAR:
095        this.name = "CharPattern";
096        break;
097      case REPEATS_MERGED:
098        this.name = "CharPatternRepeatsMerged";
099        break;
100      case REPEATS_AS_KLEENE_PLUS:
101        this.name = "CharPatternRepeatsAsKleenePlus";
102        break;
103    }
104  }
105
106  @Override
107  public String getFeatureName() {
108    return this.name;
109  }
110
111  @Override
112  public List<Feature> extract(JCas view, Annotation focusAnnotation)
113      throws CleartkExtractorException {
114    StringBuilder builder = new StringBuilder();
115    String text = focusAnnotation.getCoveredText();
116    String lastType = null;
117    boolean multipleRepeats = false;
118    for (int i = 0; i < text.length(); i += 1) {
119      char c = text.charAt(i);
120      String type = classifyChar(c);
121      switch (this.patternType) {
122        case ONE_PER_CHAR:
123          builder.append(type);
124          break;
125        case REPEATS_MERGED:
126          if (!type.equals(lastType)) {
127            builder.append(type);
128          }
129          break;
130        case REPEATS_AS_KLEENE_PLUS:
131          if (!type.equals(lastType)) {
132            builder.append(type);
133            multipleRepeats = false;
134          } else if (!multipleRepeats) {
135            builder.append('+');
136            multipleRepeats = true;
137          }
138      }
139      lastType = type;
140    }
141    return Collections.singletonList(new Feature(this.name, builder.toString()));
142  }
143
144  protected String classifyChar(char c) {
145    int typeInt = Character.getType(c);
146    switch (typeInt) {
147      case Character.CONTROL:
148        return "CC";
149      case Character.FORMAT:
150        return "Cf";
151      case Character.UNASSIGNED:
152        return "Cn";
153      case Character.PRIVATE_USE:
154        return "Co";
155      case Character.SURROGATE:
156        return "Cs";
157      case Character.LOWERCASE_LETTER:
158        return "Ll";
159      case Character.MODIFIER_LETTER:
160        return "Lm";
161      case Character.OTHER_LETTER:
162        return "Lo";
163      case Character.TITLECASE_LETTER:
164        return "Lt";
165      case Character.UPPERCASE_LETTER:
166        return "Lu";
167      case Character.COMBINING_SPACING_MARK:
168        return "Mc";
169      case Character.ENCLOSING_MARK:
170        return "Me";
171      case Character.NON_SPACING_MARK:
172        return "Mn";
173      case Character.DECIMAL_DIGIT_NUMBER:
174        return "Nd";
175      case Character.LETTER_NUMBER:
176        return "Nl";
177      case Character.OTHER_NUMBER:
178        return "No";
179      case Character.CONNECTOR_PUNCTUATION:
180        return "Pc";
181      case Character.DASH_PUNCTUATION:
182        return "Pd";
183      case Character.END_PUNCTUATION:
184        return "Pe";
185      case Character.FINAL_QUOTE_PUNCTUATION:
186        return "Pf";
187      case Character.INITIAL_QUOTE_PUNCTUATION:
188        return "Pi";
189      case Character.OTHER_PUNCTUATION:
190        return "Po";
191      case Character.START_PUNCTUATION:
192        return "Ps";
193      case Character.CURRENCY_SYMBOL:
194        return "Sc";
195      case Character.MODIFIER_SYMBOL:
196        return "Sk";
197      case Character.MATH_SYMBOL:
198        return "Sm";
199      case Character.OTHER_SYMBOL:
200        return "So";
201      case Character.LINE_SEPARATOR:
202        return "Zl";
203      case Character.PARAGRAPH_SEPARATOR:
204        return "Zp";
205      case Character.SPACE_SEPARATOR:
206        return "Zs";
207      default:
208        throw new RuntimeException("Unknown character type: " + typeInt);
209    }
210  }
211}
