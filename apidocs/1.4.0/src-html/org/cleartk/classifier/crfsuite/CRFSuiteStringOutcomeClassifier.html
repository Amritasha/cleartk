033package org.cleartk.classifier.crfsuite;
034
035import java.io.File;
036import java.io.IOException;
037import java.util.List;
038
039import org.cleartk.classifier.CleartkProcessingException;
040import org.cleartk.classifier.Feature;
041import org.cleartk.classifier.encoder.features.FeaturesEncoder;
042import org.cleartk.classifier.encoder.features.NameNumber;
043import org.cleartk.classifier.encoder.outcome.OutcomeEncoder;
044import org.cleartk.classifier.jar.SequenceClassifier_ImplBase;
045
046/**
047 * 
048 * 
049 * <br>
050 * Copyright (c) 2011-2012, Technische Universität Darmstadt <br>
051 * All rights reserved.
052 * 
053 * @author Martin Riedl
054 */
055public class CRFSuiteStringOutcomeClassifier extends
056    SequenceClassifier_ImplBase<List<NameNumber>, String, String> {
057
058  private File modelFile;
059
060  CRFSuiteWrapper wrapper;
061
062  public CRFSuiteStringOutcomeClassifier(
063      FeaturesEncoder<List<NameNumber>> featuresEncoder,
064      OutcomeEncoder<String, String> outcomeEncoder,
065      File modelFile) {
066    super(featuresEncoder, outcomeEncoder);
067    this.modelFile = modelFile;
068    this.wrapper = new CRFSuiteWrapper();
069  }
070
071  public List<String> classify(List<List<Feature>> features) throws CleartkProcessingException {
072    List<String> posTags = null;
073
074    try {
075
076      posTags = wrapper.classifyFeatures(features, outcomeEncoder, featuresEncoder, modelFile);
077    } catch (IOException e) {
078      e.printStackTrace();
079    }
080    return posTags;
081  }
082
083}
