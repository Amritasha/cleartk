024package org.cleartk.classifier.opennlp;
025
026import java.util.ArrayList;
027import java.util.Collections;
028import java.util.List;
029
030import opennlp.model.MaxentModel;
031
032import org.cleartk.classifier.CleartkProcessingException;
033import org.cleartk.classifier.Feature;
034import org.cleartk.classifier.ScoredOutcome;
035import org.cleartk.classifier.encoder.CleartkEncoderException;
036import org.cleartk.classifier.encoder.features.FeaturesEncoder;
037import org.cleartk.classifier.encoder.features.NameNumber;
038import org.cleartk.classifier.encoder.outcome.OutcomeEncoder;
039import org.cleartk.classifier.jar.Classifier_ImplBase;
040
041/**
042 * <br>
043 * Copyright (c) 2007-2011, Regents of the University of Colorado <br>
044 * All rights reserved.
045 * 
046 * 
047 * @author Philip Ogren
048 * @author Steven Bethard
049 */
050public abstract class MaxentClassifier_ImplBase<OUTCOME_TYPE> extends
051    Classifier_ImplBase<List<NameNumber>, OUTCOME_TYPE, String> {
052
053  protected MaxentModel model;
054
055  public MaxentClassifier_ImplBase(
056      FeaturesEncoder<List<NameNumber>> featuresEncoder,
057      OutcomeEncoder<OUTCOME_TYPE, String> outcomeEncoder,
058      MaxentModel model) {
059    super(featuresEncoder, outcomeEncoder);
060    this.model = model;
061  }
062
063  public OUTCOME_TYPE classify(List<Feature> features) throws CleartkProcessingException {
064    EvalParams evalParams = convertToEvalParams(features);
065    String encodedOutcome = this.model.getBestOutcome(this.model.eval(
066        evalParams.getContext(),
067        evalParams.getValues()));
068    return outcomeEncoder.decode(encodedOutcome);
069  }
070
071  @Override
072  public List<ScoredOutcome<OUTCOME_TYPE>> score(List<Feature> features, int maxResults)
073      throws CleartkProcessingException {
074    EvalParams evalParams = convertToEvalParams(features);
075    double[] evalResults = this.model.eval(evalParams.getContext(), evalParams.getValues());
076    String[] encodedOutcomes = (String[]) this.model.getDataStructures()[2];
077
078    List<ScoredOutcome<OUTCOME_TYPE>> returnValues = new ArrayList<ScoredOutcome<OUTCOME_TYPE>>();
079
080    if (maxResults == 1) {
081      String bestOutcome = this.model.getBestOutcome(evalResults);
082      OUTCOME_TYPE encodedBestOutcome = outcomeEncoder.decode(bestOutcome);
083      double bestResult = evalResults[this.model.getIndex(bestOutcome)];
084      returnValues.add(new ScoredOutcome<OUTCOME_TYPE>(encodedBestOutcome, bestResult));
085      return returnValues;
086    }
087
088    for (int i = 0; i < evalResults.length; i++) {
089      returnValues.add(new ScoredOutcome<OUTCOME_TYPE>(
090          outcomeEncoder.decode(encodedOutcomes[i]),
091          evalResults[i]));
092    }
093
094    Collections.sort(returnValues);
095    if (returnValues.size() > maxResults) {
096      return returnValues.subList(0, maxResults);
097    }
098
099    return returnValues;
100  }
101
102  protected EvalParams convertToEvalParams(List<Feature> features) throws CleartkEncoderException {
103
104    List<NameNumber> contexts = featuresEncoder.encodeAll(features);
105
106    String[] context = new String[contexts.size()];
107    float[] values = new float[contexts.size()];
108
109    for (int i = 0; i < contexts.size(); i++) {
110      NameNumber contextValue = contexts.get(i);
111      context[i] = contextValue.name;
112      values[i] = contextValue.number.floatValue();
113    }
114
115    return new EvalParams(context, values);
116  }
117
118  public class EvalParams {
119    private String[] context;
120
121    private float[] values;
122
123    public String[] getContext() {
124      return context;
125    }
126
127    public float[] getValues() {
128      return values;
129    }
130
131    public EvalParams(String[] context, float[] values) {
132      this.context = context;
133      this.values = values;
134    }
135
136  }
137
138}
