027import java.io.IOException;

029import java.util.Map;

052/**
053 * <br>
054 * Copyright (c) 2011, Regents of the University of Colorado <br>
055 * All rights reserved.
056 * 
057 * @author Lee Becker
058 * 
059 *         This class provides a framework for handling multiple classifiers within a single
060 *         analysis engine.
061 *         <p>
062 * 
063 *         Use cases that may lend themselves to {@link CleartkMultiAnnotator} over
064 *         {@link CleartkAnnotator} include:
065 *         <ul>
066 *         <li>Explicit control for one-vs-all multi-label classification
067 *         <li>Predicating training/classification on specific conditions. (e.g. if the the word to
068 *         the left is a verb, use the verb classifier, else use the default classifier)
069 *         <li>Voting or ensemble classification
070 *         </ul>
071 */
072public abstract class CleartkMultiAnnotator<OUTCOME_TYPE> extends JCasAnnotator_ImplBase implements
073    Initializable {
074
075  public static final String PARAM_CLASSIFIER_FACTORY_CLASS_NAME = ConfigurationParameterFactory.createConfigurationParameterName(
076      CleartkMultiAnnotator.class,
077      "classifierFactoryClassName");
078
079  private static final String DEFAULT_CLASSIFIER_FACTORY_CLASS_NAME = "org.cleartk.classifier.jar.JarClassifierFactory";
080
081  @ConfigurationParameter(
082      mandatory = false,
083      description = "provides the full name of the ClassifierFactory class to be used.",
084      defaultValue = DEFAULT_CLASSIFIER_FACTORY_CLASS_NAME)
085  private String classifierFactoryClassName;
086
087  public static final String PARAM_DATA_WRITER_FACTORY_CLASS_NAME = ConfigurationParameterFactory.createConfigurationParameterName(
088      CleartkMultiAnnotator.class,
089      "dataWriterFactoryClassName");
090
091  private static final String DEFAULT_DATA_WRITER_FACTORY_CLASS_NAME = "org.cleartk.classifier.jar.DefaultDataWriterFactory";
092
093  @ConfigurationParameter(
094      mandatory = false,
095      description = "provides the full name of the DataWriterFactory class to be used.",
096      defaultValue = DEFAULT_DATA_WRITER_FACTORY_CLASS_NAME)
097  private String dataWriterFactoryClassName;
098
099  public static final String PARAM_IS_TRAINING = ConfigurationParameterFactory.createConfigurationParameterName(
100      CleartkMultiAnnotator.class,
101      "isTraining");
102
103  @ConfigurationParameter(
104      mandatory = false,
105      description = "determines whether this annotator is writing training data or using a classifier to annotate. Normally inferred automatically based on whether or not a DataWriterFactory class has been set.")
106  private Boolean isTraining;
107
108  private boolean primitiveIsTraining;
109
110  protected ClassifierFactory<?> classifierFactory;
111
112  protected DataWriterFactory<?> dataWriterFactory;
113
114  protected Map<String, Classifier<OUTCOME_TYPE>> classifiers;
115
116  protected Map<String, DataWriter<OUTCOME_TYPE>> dataWriters;
117
118  private UimaContext uimaContext;
119
120  protected File outputDirectoryRoot;
121
122  protected File classifierJarPathRoot;
123
124  @Override
125  public void initialize(UimaContext context) throws ResourceInitializationException {
126    super.initialize(context);
127
128    if (this.dataWriterFactoryClassName == null && this.classifierFactoryClassName == null) {
129      CleartkInitializationException.neitherParameterSet(
130          PARAM_DATA_WRITER_FACTORY_CLASS_NAME,
131          this.dataWriterFactoryClassName,
132          PARAM_CLASSIFIER_FACTORY_CLASS_NAME,
133          this.classifierFactoryClassName);
134    }
135
136    // determine whether we start out as training or predicting
137    if (this.isTraining != null) {
138      this.primitiveIsTraining = this.isTraining;
139    } else if (!DEFAULT_DATA_WRITER_FACTORY_CLASS_NAME.equals(this.dataWriterFactoryClassName)) {
140      this.primitiveIsTraining = true;
141    } else if (context.getConfigParameterValue(DirectoryDataWriterFactory.PARAM_OUTPUT_DIRECTORY) != null) {
142      this.primitiveIsTraining = true;
143    } else if (!DEFAULT_CLASSIFIER_FACTORY_CLASS_NAME.equals(this.classifierFactoryClassName)) {
144      this.primitiveIsTraining = false;
145    } else if (context.getConfigParameterValue(GenericJarClassifierFactory.PARAM_CLASSIFIER_JAR_PATH) != null) {
146      this.primitiveIsTraining = false;
147    } else {
148      String message = "Please specify PARAM_IS_TRAINING - unable to infer it from context";
149      throw new IllegalArgumentException(message);
150    }
151
152    this.uimaContext = context;
153    UimaContextAdmin contextAdmin = (UimaContextAdmin) this.uimaContext;
154    ConfigurationManager manager = contextAdmin.getConfigurationManager();
155
156    if (this.isTraining()) {
157      // Create the data writer factory and initialize a Map to hold the data writers
158      // Individual data writers will be created dynamically with the getDataWriter method
159      this.dataWriters = new HashMap<String, DataWriter<OUTCOME_TYPE>>();
160      this.outputDirectoryRoot = (File) manager.getConfigParameterValue(DirectoryDataWriterFactory.PARAM_OUTPUT_DIRECTORY);
161      this.dataWriterFactory = InitializableFactory.create(
162          context,
163          dataWriterFactoryClassName,
164          DataWriterFactory.class);
165    } else {
166
167      // Create the classifier factory and initialize a map to hold the classifiers
168      // Individual classifiers will be created dynamically with the getClassifier method
169      this.classifiers = new HashMap<String, Classifier<OUTCOME_TYPE>>();
170      this.classifierJarPathRoot = (File) manager.getConfigParameterValue(GenericJarClassifierFactory.PARAM_CLASSIFIER_JAR_PATH);
171
172      this.classifierFactory = InitializableFactory.create(
173          context,
174          classifierFactoryClassName,
175          ClassifierFactory.class);
176    }
177  }
178
179  @Override
180  public void collectionProcessComplete() throws AnalysisEngineProcessException {
181    super.collectionProcessComplete();
182    if (this.isTraining()) {
183      try {
184        for (DataWriter<OUTCOME_TYPE> dataWriter : dataWriters.values()) {
185          dataWriter.finish();
186        }
187      } catch (CleartkProcessingException ctke) {
188        throw new AnalysisEngineProcessException(ctke);
189      }
190    }
191  }
192
193  protected boolean isTraining() {
194    return this.primitiveIsTraining;
195  }
196
197  /**
198   * Gets the classifier associated with name. If it does not exist, this method will use the
199   * {@link ClassifierFactory} specified at initialization to create a new one.
200   * 
201   * @param name
202   *          The name of the classifier
203   * @return The classifier associated with the name
204   */
205  protected Classifier<OUTCOME_TYPE> getClassifier(String name)
206      throws ResourceInitializationException {
207    if (classifiers.containsKey(name)) {
208      return classifiers.get(name);
209    }
210
211    File classifierJarPath = new File(this.classifierJarPathRoot, name);
212    UimaContextAdmin contextAdmin = (UimaContextAdmin) this.uimaContext;
213    ConfigurationManager manager = contextAdmin.getConfigurationManager();
214    manager.setConfigParameterValue(contextAdmin.getQualifiedContextName()
215        + GenericJarClassifierFactory.PARAM_CLASSIFIER_JAR_PATH, classifierJarPath.getPath());
216
217    // create the factory and instantiate the classifier
218    ClassifierFactory<?> factory = InitializableFactory.create(
219        uimaContext,
220        classifierFactoryClassName,
221        ClassifierFactory.class);
222    Classifier<?> untypedClassifier;
223    try {
224      untypedClassifier = factory.createClassifier();
225    } catch (IOException e) {
226      throw new ResourceInitializationException(e);
227    }
228
229    Classifier<OUTCOME_TYPE> classifier = ReflectionUtil.uncheckedCast(untypedClassifier);
230    ReflectionUtil.checkTypeParameterIsAssignable(
231        CleartkMultiAnnotator.class,
232        "OUTCOME_TYPE",
233        this,
234        Classifier.class,
235        "OUTCOME_TYPE",
236        classifier);
237    InitializableFactory.initialize(untypedClassifier, this.getContext());
238    this.classifiers.put(name, classifier);
239    return classifier;
240  }
241
242  /**
243   * Gets the {@link DataWriter} associated with name. If it does not exist, this method will use
244   * the {@link DataWriterFactory} specified during initialization to create a dataWriter associated
245   * with the name parameter.
246   * 
247   * @param name
248   *          The name of the {@link DataWriter}
249   * @return The {@link DataWriter} associated with the name
250   */
251  protected DataWriter<OUTCOME_TYPE> getDataWriter(String name)
252      throws ResourceInitializationException {
253    if (dataWriters.containsKey(name)) {
254      return dataWriters.get(name);
255    }
256
257    DataWriter<?> untypedDataWriter;
258    File dataWriterPath = new File(this.outputDirectoryRoot, name);
259    UimaContextAdmin contextAdmin = (UimaContextAdmin) this.uimaContext;
260    ConfigurationManager manager = contextAdmin.getConfigurationManager();
261    manager.setConfigParameterValue(contextAdmin.getQualifiedContextName()
262        + DirectoryDataWriterFactory.PARAM_OUTPUT_DIRECTORY, dataWriterPath);
263
264    try {
265      untypedDataWriter = this.dataWriterFactory.createDataWriter();
266    } catch (IOException e) {
267      throw new ResourceInitializationException(e);
268    }
269
270    InitializableFactory.initialize(untypedDataWriter, uimaContext);
271    DataWriter<OUTCOME_TYPE> dataWriter = ReflectionUtil.uncheckedCast(untypedDataWriter);
272    this.dataWriters.put(name, dataWriter);
273    return dataWriter;
274  }
