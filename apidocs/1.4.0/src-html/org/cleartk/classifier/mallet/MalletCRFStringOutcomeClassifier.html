024package org.cleartk.classifier.mallet;
025
026import java.util.ArrayList;
027import java.util.List;
028
029import org.cleartk.classifier.CleartkProcessingException;
030import org.cleartk.classifier.Feature;
031import org.cleartk.classifier.encoder.CleartkEncoderException;
032import org.cleartk.classifier.encoder.features.FeaturesEncoder;
033import org.cleartk.classifier.encoder.features.NameNumber;
034import org.cleartk.classifier.encoder.outcome.OutcomeEncoder;
035import org.cleartk.classifier.jar.SequenceClassifier_ImplBase;
036import org.cleartk.util.ReflectionUtil;
037
038import cc.mallet.fst.Transducer;
039import cc.mallet.pipe.Pipe;
040import cc.mallet.types.Instance;
041import cc.mallet.types.Sequence;
042
043/**
044 * This classifier provides an interface to the <a
045 * href="http://mallet.cs.umass.edu/index.php/SimpleTagger_example"> Mallet Conditional Random Field
046 * (CRF) tagger</a>. Annotators that use a sequence learner such as this one will need to support
047 * classification of a sequence of instances.
048 * 
049 * <br>
050 * Copyright (c) 2007-2008, Regents of the University of Colorado <br>
051 * All rights reserved.
052 * 
053 * @author Philip Ogren
054 */
055public class MalletCRFStringOutcomeClassifier extends
056    SequenceClassifier_ImplBase<List<NameNumber>, String, String> {
057
058  protected Transducer transducer;
059
060  public MalletCRFStringOutcomeClassifier(
061      FeaturesEncoder<List<NameNumber>> featuresEncoder,
062      OutcomeEncoder<String, String> outcomeEncoder,
063      Transducer transducer) {
064    super(featuresEncoder, outcomeEncoder);
065    this.transducer = transducer;
066  }
067
068  /**
069   * This method classifies several instances at once
070   * 
071   * @param features
072   *          a list of lists of features - each list in the list represents one instance to be
073   *          classified. The list should correspond to some logical sequence of instances to be
074   *          classified (e.g. tokens in a sentence or lines in a document) that corresponds to the
075   *          model that has been built for this classifier.
076   */
077  public List<String> classify(List<List<Feature>> features) throws CleartkProcessingException {
078    String[][] featureStringArray = toStrings(features);
079    Pipe pipe = transducer.getInputPipe();
080
081    Instance instance = new Instance(featureStringArray, null, null, null);
082    instance = pipe.instanceFrom(instance);
083
084    Sequence<?> data = (Sequence<?>) instance.getData();
085    Sequence<?> untypedSequence = transducer.transduce(data);
086    Sequence<String> sequence = ReflectionUtil.uncheckedCast(untypedSequence);
087
088    List<String> returnValues = new ArrayList<String>();
089
090    for (int i = 0; i < sequence.size(); i++) {
091      String encodedOutcome = sequence.get(i);
092      returnValues.add(outcomeEncoder.decode(encodedOutcome));
093    }
094    return returnValues;
095  }
096
097  /**
098   * Converts the features into a 2D string array that Mallet can use. The only thing mildly tricky
099   * about this method is that the length of each string array is one more than the size of each
100   * feature list - i.e. <code>returnValues[0].length == features.get(0).size() + 1</code> where the
101   * last element in each string array is an empty string.
102   * 
103   * @param features
104   *          the features to be converted.
105   * @return a 2D string array that Mallet can use.
106   */
107  private String[][] toStrings(List<List<Feature>> features) throws CleartkEncoderException {
108    List<List<String>> encodedFeatures = new ArrayList<List<String>>(features.size());
109    for (List<Feature> features1 : features) {
110      List<NameNumber> nameNumbers = this.featuresEncoder.encodeAll(features1);
111      List<String> encodedFeatures1 = new ArrayList<String>();
112      for (NameNumber nameNumber : nameNumbers) {
113        encodedFeatures1.add(nameNumber.name);
114      }
115      encodedFeatures.add(encodedFeatures1);
116    }
117
118    String[][] encodedFeaturesArray = new String[encodedFeatures.size()][];
119    for (int i = 0; i < encodedFeatures.size(); i++) {
120      String[] encodedFeaturesArray1 = encodedFeatures.get(i).toArray(new String[0]);
121      encodedFeaturesArray[i] = encodedFeaturesArray1;
122    }
123
124    return encodedFeaturesArray;
125  }
126}
