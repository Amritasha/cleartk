024package org.cleartk.feature.syntax;
025
026import java.util.ArrayList;
027import java.util.List;
028import java.util.ListIterator;
029
030import org.apache.uima.jcas.JCas;
031import org.apache.uima.jcas.tcas.Annotation;
032import org.cleartk.classifier.Feature;
033import org.cleartk.classifier.feature.extractor.CleartkExtractorException;
034import org.cleartk.classifier.feature.extractor.annotationpair.AnnotationPairFeatureExtractor;
035import org.cleartk.classifier.feature.extractor.simple.SimpleFeatureExtractor;
036import org.cleartk.syntax.constituent.type.TreebankNode;
037import org.cleartk.syntax.constituent.type.TreebankNodeUtil;
038
039/**
040 * <br>
041 * Copyright (c) 2007-2008, Regents of the University of Colorado <br>
042 * All rights reserved.
043 * 
044 * 
045 * @author Philipp Wetzler
046 * 
047 */
048public class SyntacticPathExtractor implements AnnotationPairFeatureExtractor {
049  protected final String UP_SEPARATOR = "::";
050
051  protected final String DOWN_SEPARATOR = ";;";
052
053  protected SimpleFeatureExtractor pathMemberExtractor;
054
055  protected String name;
056
057  protected boolean isPartial;
058
059  /**
060   * 
061   * @param pathMemberExtractor
062   *          this extractor will be used to get a feature for every node on the path, which will
063   *          then be combined to form a single string. The extractor should preferably generate
064   *          exactly one <tt>StringFeature</tt>, but must generate at least one
065   *          <tt>StringFeature</tt>, <tt>LongFeature</tt>, <tt>DoubleFeature</tt>, or
066   *          <tt>BooleanFeature</tt>. Only the first feature will then be used and naively
067   *          converted to a string.
068   * @param partial
069   *          if true, generate a partial path only, i.e. from the first node up to the lowest
070   *          common ancestor of the two
071   */
072  public SyntacticPathExtractor(SimpleFeatureExtractor pathMemberExtractor, boolean partial) {
073    this.pathMemberExtractor = pathMemberExtractor;
074    this.isPartial = partial;
075  }
076
077  /**
078   * This constructor defaults to a full rather than a partial path.
079   */
080  public SyntacticPathExtractor(SimpleFeatureExtractor pathMemberExtractor) {
081    this(pathMemberExtractor, false);
082  }
083
084  /**
085   * Extract a string representation of a path feature.
086   * 
087   * @param leftAnnotation
088   *          the first node of the path
089   * 
090   * @param rightAnnotation
091   *          the last node of the path
092   * 
093   * @return List of one <em>StringFeature</em>, which contains a string representation of the path
094   *         between the two nodes.
095   * 
096   */
097  public List<Feature> extract(JCas view, Annotation leftAnnotation, Annotation rightAnnotation)
098      throws CleartkExtractorException {
099    TreebankNode leftConstituent;
100    TreebankNode rightConstituent;
101    Annotation currentAnnotation = leftAnnotation;
102    try {
103      leftConstituent = (TreebankNode) leftAnnotation;
104      currentAnnotation = rightAnnotation;
105      rightConstituent = (TreebankNode) rightAnnotation;
106    } catch (ClassCastException e) {
107      throw CleartkExtractorException.wrongAnnotationType(TreebankNode.class, currentAnnotation);
108    }
109
110    List<TreebankNode> fromStart = TreebankNodeUtil.getPathToRoot(leftConstituent);
111    List<TreebankNode> fromEnd = TreebankNodeUtil.getPathToRoot(rightConstituent);
112    String pathFeatureName = null;
113    String lengthFeatureName = null;
114
115    fromEnd.remove(fromEnd.size() - 1);
116    while (fromStart.size() > 1 && fromEnd.size() > 0
117        && fromStart.get(fromStart.size() - 2) == fromEnd.get(fromEnd.size() - 1)) {
118      fromStart.remove(fromStart.size() - 1);
119      fromEnd.remove(fromEnd.size() - 1);
120    }
121
122    int length = fromStart.size();
123    if (!isPartial)
124      length += fromEnd.size();
125
126    try {
127      ListIterator<TreebankNode> it = fromStart.listIterator();
128      StringBuffer pathBuffer = new StringBuffer();
129      boolean first = true;
130      while (it.hasNext()) {
131        Feature feature = this.pathMemberExtractor.extract(view, it.next()).get(0);
132        if (first) {
133          String s = feature.getName();
134          if (isPartial) {
135            pathFeatureName = Feature.createName(name, "PartialSyntacticPath(" + s + ")");
136            lengthFeatureName = Feature.createName(name, "PartialSyntacticPath", "Length");
137          } else {
138            pathFeatureName = Feature.createName(name, "SyntacticPath(" + s + ")");
139            lengthFeatureName = Feature.createName(name, "SyntacticPath", "Length");
140          }
141          first = false;
142        } else {
143          pathBuffer.append(this.UP_SEPARATOR);
144        }
145        pathBuffer.append(feature.getValue().toString());
146      }
147
148      if (!isPartial) {
149        it = fromEnd.listIterator(fromEnd.size());
150        while (it.hasPrevious()) {
151          Feature feature = this.pathMemberExtractor.extract(view, it.previous()).get(0);
152          pathBuffer.append(this.DOWN_SEPARATOR);
153          pathBuffer.append(feature.getValue().toString());
154        }
155      }
156
157      List<Feature> features = new ArrayList<Feature>(2);
158      features.add(new Feature(pathFeatureName, pathBuffer.toString()));
159      features.add(new Feature(lengthFeatureName, (long) length));
160
161      return features;
162
163    } catch (IndexOutOfBoundsException e) {
164      return new ArrayList<Feature>(0);
165    }
166  }
167}
