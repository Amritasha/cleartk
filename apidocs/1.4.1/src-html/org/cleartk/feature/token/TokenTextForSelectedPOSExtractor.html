024package org.cleartk.feature.token;
025
026import java.util.ArrayList;
027import java.util.Arrays;
028import java.util.Collection;
029import java.util.HashSet;
030import java.util.List;
031import java.util.Set;
032
033import org.apache.uima.jcas.JCas;
034import org.apache.uima.jcas.tcas.Annotation;
035import org.cleartk.classifier.Feature;
036import org.cleartk.classifier.feature.extractor.CleartkExtractorException;
037import org.cleartk.classifier.feature.extractor.simple.CoveredTextExtractor;
038import org.cleartk.classifier.feature.extractor.simple.SimpleNamedFeatureExtractor;
039import org.cleartk.token.type.Token;
040
041/**
042 * <br>
043 * Copyright (c) 2011, Regents of the University of Colorado <br>
044 * All rights reserved.
045 * 
046 * @author Steven Bethard
047 */
048public class TokenTextForSelectedPOSExtractor implements SimpleNamedFeatureExtractor {
049
050  private Set<String> acceptablePOSTags;
051
052  private CoveredTextExtractor extractor;
053
054  public TokenTextForSelectedPOSExtractor(Collection<String> acceptablePOSTags) {
055    this.acceptablePOSTags = new HashSet<String>(acceptablePOSTags);
056    this.extractor = new CoveredTextExtractor();
057  }
058
059  public TokenTextForSelectedPOSExtractor(String... acceptablePOSTags) {
060    this(Arrays.asList(acceptablePOSTags));
061  }
062  
063  @Override
064  public String getFeatureName() {
065    return this.extractor.getFeatureName();
066  }
067
068  @Override
069  public List<Feature> extract(JCas view, Annotation focusAnnotation)
070      throws CleartkExtractorException {
071    List<Feature> features = new ArrayList<Feature>();
072    Token token = (Token) focusAnnotation;
073    String pos = token.getPos();
074    if (pos != null) {
075      if (pos.length() > 2) {
076        pos = pos.substring(0, 2);
077      }
078      if (this.acceptablePOSTags.contains(pos)) {
079        features.addAll(this.extractor.extract(view, token));
080      }
081    }
082    return features;
083  }
084}
