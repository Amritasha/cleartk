057/**
058 * 
059 * This model is based on the SumBasic model described in:
060 * <p>
061 * Nenkova, Vanderwende and McKeown, <i>A compositional context sensitive multi-document summarizer:
062 * exploring the factors that influence summarization</i>, SIGIR 20006 <br>
063 * <p>
064 * and
065 * <p>
066 * Nenkova and Vanderwende <i>The Impact of Frequency on Summarization</i> Microsoft Research
067 * TechReport, MSR-TR-2005-101, 2005 <br>
068 * 
069 * @author Lee Becker
070 * 
071 */
072public class SumBasicModel extends SummarizationModel_ImplBase {

081  public static class Options extends Options_ImplBase {
082
083    @Option(
084        name = "--max-num-sentences",
085        usage = "Specifies the maximum number of sentences to extract in the summary")
086    public int maxNumSentences = 10;
087
088    @Option(name = "--seen-words-prob", usage = "Specify the probability for seen words.")
089    public double seenWordsProbability = 0.0001;
090
091    @Option(
092        name = "--composition-function",
093        usage = "Specifies how word probabilities are combined (AVERAGE|SUM|PRODUCT, default=AVERAGE)")
094    public CompositionFunctionType cfType = CompositionFunctionType.AVERAGE;
095  }

097  // Consider moving this elsewhere if needed by other classes
098  public static class TermFrequencyMap {
099    private Multiset<String> termFrequencies;
100
101    public TermFrequencyMap() {
102      this.termFrequencies = LinkedHashMultiset.create();
103    }
104
105    public void add(String term, int count) {
106      this.termFrequencies.add(term, count);
107    }
108
109    public double getProbability(String term) {
110      return (double) this.termFrequencies.count(term) / (double) this.termFrequencies.size();
111    }
112
113    public void save(URI outputURI) throws IOException {
114      File out = new File(outputURI);
115      BufferedWriter writer = null;
116      writer = new BufferedWriter(new FileWriter(out));
117      for (Multiset.Entry<String> entry : this.termFrequencies.entrySet()) {
118        writer.append(String.format("%s\t%d\n", entry.getElement(), entry.getCount()));
119      }
120      writer.close();
121    }
122
123    public void load(URI inputURI) throws IOException {
124      File in = new File(inputURI);
125      BufferedReader reader = null;
126      reader = new BufferedReader(new FileReader(in));
127
128      // The lines are the term counts divided by a tab
129      String line = null;
130      while ((line = reader.readLine()) != null) {
131        String[] termFreqPair = line.split("\\t");
132        this.termFrequencies.add(termFreqPair[0], Integer.parseInt(termFreqPair[1]));
133      }
134      reader.close();
135    }
136  }

138  /**
139   * 
140   * SumBasic can use three different Composition Functions to combine the probabilities for words
141   * in a sentence.
142   * <ul>
143   * <li>PRODUCT - The product of the individual word probabilities within the sentence</li>
144   * <li>SUM - The sum of the individual word probabilities within the sentence</li>
145   * <li>AVERAGE - The sum of the individual word probabilities within the sentence</li>
146   * </ul>
147   * 
148   * @author Lee Becker
149   * 
150   */
151  public static enum CompositionFunctionType {
152    AVERAGE, PRODUCT, SUM;
153  }
154
155  private static abstract class CompositionFunction implements Function<List<Feature>, Double> {
156    protected Double seenWordProbability;
157
158    protected TermFrequencyMap tfMap;
159
160    protected Set<String> seenWords;
161
162    public CompositionFunction(
163        Double seenWordProbability,
164        TermFrequencyMap tfMap,
165        Set<String> seenWords) {
166      this.seenWordProbability = seenWordProbability;
167      this.tfMap = tfMap;
168      this.seenWords = seenWords;
169    }
170  }
171
172  private static class AverageCF extends CompositionFunction {
173    private SumCF sumcf;
174
175    public AverageCF(Double seenWordProbability, TermFrequencyMap tfMap, Set<String> seenWords) {
176      super(seenWordProbability, tfMap, seenWords);
177      this.sumcf = new SumCF(seenWordProbability, tfMap, seenWords);
178    }
179
180    @Override
181    public Double apply(List<Feature> wordFeatures) {
182      int numWords = wordFeatures.size();
183      return (numWords == 0) ? 0.0 : this.sumcf.apply(wordFeatures) / numWords;
184    }
185  }
186
187  private static class ProductCF extends CompositionFunction {
188
189    public ProductCF(Double seenWordProbability, TermFrequencyMap tfMap, Set<String> seenWords) {
190      super(seenWordProbability, tfMap, seenWords);
191    }
192
193    @Override
194    public Double apply(List<Feature> wordFeatures) {
195      double sentenceWeight = 1.0;
196      for (Feature feature : wordFeatures) {
197        if (this.seenWords.contains(feature.getName())) {
198          sentenceWeight *= this.seenWordProbability;
199        } else {
200          sentenceWeight *= this.tfMap.getProbability(feature.getName());
201        }
202      }
203      return sentenceWeight;
204    }
205  }
206
207  private static class SumCF extends CompositionFunction {
208
209    public SumCF(Double seenWordProbability, TermFrequencyMap tfMap, Set<String> seenWords) {
210      super(seenWordProbability, tfMap, seenWords);
211    }
212
213    @Override
214    public Double apply(List<Feature> wordFeatures) {
215      Double sentenceWeight = 0.0;
216      for (Feature feature : wordFeatures) {
217        if (this.seenWords.contains(feature.getName())) {
218          sentenceWeight += this.seenWordProbability;
219        } else {
220          sentenceWeight += this.tfMap.getProbability(feature.getName());
221        }
222      }
223      return sentenceWeight;
224    }
225  }

227  public static void trainAndWriteModel(File modelDirectory, String... args) throws Exception {
228    // This is where the bulk of the sum basic algorithm resides
229
230    Options options = new Options();
231    options.parseOptions(args);
232    int maxNumSentences = options.maxNumSentences;
233    CompositionFunctionType cfType = options.cfType;
234    double seenWordsProbability = options.seenWordsProbability;
235
236    // Load the serialized instance data
237    Iterable<Instance<Boolean>> instances = InstanceStream.loadFromDirectory(modelDirectory);
238
239    // save off feature data and use it to compute term frequencies
240    List<List<Feature>> sentences = new ArrayList<List<Feature>>();
241    TermFrequencyMap tfMap = new TermFrequencyMap();
242    for (Instance<Boolean> instance : instances) {
243      sentences.add(instance.getFeatures());
244      for (Feature feature : instance.getFeatures()) {
245        tfMap.add(feature.getName(), (Integer) feature.getValue());
246      }
247    }
248
249    // the set of words in the summary (will change over time)
250    final Set<String> seenWords = new HashSet<String>();
251
252    // create scoring function
253    final CompositionFunction getScore;
254    switch (cfType) {
255      case PRODUCT:
256        getScore = new ProductCF(seenWordsProbability, tfMap, seenWords);
257        break;
258      case SUM:
259        getScore = new SumCF(seenWordsProbability, tfMap, seenWords);
260        break;
261      case AVERAGE:
262      default:
263        getScore = new AverageCF(seenWordsProbability, tfMap, seenWords);
264        break;
265    }
266
267    // Use frequency data to select N sentences
268    Map<List<Feature>, Double> selectedSentences = new HashMap<List<Feature>, Double>();
269    for (int i = 0; i < maxNumSentences; i++) {
270
271      // find the sentence with the maximum score
272      List<Feature> selectedSentence = Ordering.natural().onResultOf(getScore).max(sentences);
273      selectedSentences.put(selectedSentence, getScore.apply(selectedSentence));
274
275      // Update the set of seen words with the words in the selected sentence
276      // These are used to discount words that have already made their way into the summary
277      for (Feature feature : selectedSentence) {
278        seenWords.add(feature.getName());
279      }
280    }
281
282    // Save off selected sentences as the model file
283    try {
284      SumBasicModel model = new SumBasicModel(selectedSentences);
285      File outputFile = new File(modelDirectory, model.getModelName());
286      ObjectOutputStream oos = new ObjectOutputStream(new BufferedOutputStream(
287          new FileOutputStream(outputFile)));
288      oos.writeObject(model);
289      oos.close();
290    } catch (FileNotFoundException e) {
291      throw new Exception(e);
292    } catch (IOException e) {
293      throw new Exception(e);
294    }
295  }
