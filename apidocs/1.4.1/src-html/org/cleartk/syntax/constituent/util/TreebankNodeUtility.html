050  public static org.cleartk.syntax.constituent.type.TopTreebankNode convert(
051      TopTreebankNode pojoNode,
052      JCas jCas,
053      boolean addToIndexes) {
054    org.cleartk.syntax.constituent.type.TopTreebankNode uimaNode = new org.cleartk.syntax.constituent.type.TopTreebankNode(
055        jCas,
056        pojoNode.getTextBegin(),
057        pojoNode.getTextEnd());
058    convert(pojoNode, jCas, uimaNode, null, addToIndexes);
059    uimaNode.setTreebankParse(pojoNode.getTreebankParse());
060    initTerminalNodes(uimaNode, jCas);
061    if (addToIndexes)
062      uimaNode.addToIndexes();
063    return uimaNode;
064  }
065
066  public static void initTerminalNodes(
067      org.cleartk.syntax.constituent.type.TopTreebankNode uimaNode,
068      JCas jCas) {
069    List<TerminalTreebankNode> terminals = new ArrayList<org.cleartk.syntax.constituent.type.TerminalTreebankNode>();
070    _initTerminalNodes(uimaNode, terminals);
071
072    for (int i = 0; i < terminals.size(); i++) {
073      TerminalTreebankNode terminal = terminals.get(i);
074      terminal.setIndex(i);
075    }
076
077    FSArray terminalsFSArray = new FSArray(jCas, terminals.size());
078    terminalsFSArray.copyFromArray(
079        terminals.toArray(new FeatureStructure[terminals.size()]),
080        0,
081        0,
082        terminals.size());
083    uimaNode.setTerminals(terminalsFSArray);
084  }
085
086  private static void _initTerminalNodes(
087      org.cleartk.syntax.constituent.type.TreebankNode node,
088      List<TerminalTreebankNode> terminals) {
089    FSArray children = node.getChildren();
090    for (int i = 0; i < children.size(); i++) {
091      org.cleartk.syntax.constituent.type.TreebankNode child = (org.cleartk.syntax.constituent.type.TreebankNode) children.get(i);
092      if (child instanceof TerminalTreebankNode) {
093        terminals.add((TerminalTreebankNode) child);
094      } else
095        _initTerminalNodes(child, terminals);
096    }
097  }

099  public static org.cleartk.syntax.constituent.type.TreebankNode convert(
100      TreebankNode pojoNode,
101      JCas jCas,
102      org.cleartk.syntax.constituent.type.TreebankNode uimaNode,
103      org.cleartk.syntax.constituent.type.TreebankNode parentNode,
104      boolean addToIndexes) {
105    uimaNode.setNodeType(pojoNode.getType());
106    StringArray nodeTags = (StringArray) (FSCollectionFactory.fillArrayFS(new StringArray(
107        jCas,
108        pojoNode.getTags().length), pojoNode.getTags()));
109    uimaNode.setNodeTags(nodeTags);
110    uimaNode.setNodeValue(pojoNode.getValue());
111    uimaNode.setLeaf(pojoNode.isLeaf());
112    uimaNode.setParent(parentNode);
113
114    List<org.cleartk.syntax.constituent.type.TreebankNode> uimaChildren = new ArrayList<org.cleartk.syntax.constituent.type.TreebankNode>();
115    for (TreebankNode child : pojoNode.getChildren()) {
116      org.cleartk.syntax.constituent.type.TreebankNode childNode;
117      if (child.isLeaf()) {
118        childNode = new TerminalTreebankNode(jCas, child.getTextBegin(), child.getTextEnd());
119      } else {
120        childNode = new org.cleartk.syntax.constituent.type.TreebankNode(
121            jCas,
122            child.getTextBegin(),
123            child.getTextEnd());
124      }
125      uimaChildren.add(convert(child, jCas, childNode, uimaNode, addToIndexes));
126      if (addToIndexes)
127        childNode.addToIndexes();
128    }
129    FSArray uimaChildrenFSArray = new FSArray(jCas, uimaChildren.size());
130    uimaChildrenFSArray.copyFromArray(
131        uimaChildren.toArray(new FeatureStructure[uimaChildren.size()]),
132        0,
133        0,
134        uimaChildren.size());
135    uimaNode.setChildren(uimaChildrenFSArray);
136    return uimaNode;
137  }

139  public static org.cleartk.syntax.constituent.type.TopTreebankNode getTopNode(
140      org.cleartk.syntax.constituent.type.TreebankNode node) {
141    if (node instanceof org.cleartk.syntax.constituent.type.TopTreebankNode)
142      return (org.cleartk.syntax.constituent.type.TopTreebankNode) node;
143
144    org.cleartk.syntax.constituent.type.TreebankNode parent = node.getParent();
145    while (parent != null) {
146      if (parent instanceof org.cleartk.syntax.constituent.type.TopTreebankNode)
147        return (org.cleartk.syntax.constituent.type.TopTreebankNode) parent;
148      node = parent;
149      parent = node.getParent();
150    }
151    return null;
152  }

154  /**
155   * A "pretty print" of this node that may be useful for e.g. debugging.
156   */
157  public static void print(PrintStream out, org.cleartk.syntax.constituent.type.TreebankNode node) {
158    out.println(print(node, 0));
159  }
160
161  private static String print(org.cleartk.syntax.constituent.type.TreebankNode node, int tabs) {
162    StringBuffer returnValue = new StringBuffer();
163    String tabString = getTabs(tabs);
164    returnValue.append(tabString + node.getNodeType());
165    if (node.getNodeValue() != null)
166      returnValue.append(":" + node.getNodeValue() + "\n");
167    else {
168      returnValue.append(":" + node.getCoveredText() + "\n");
169    }
170    if (node.getChildren().size() > 0) {
171      Collection<org.cleartk.syntax.constituent.type.TreebankNode> children = JCasUtil.select(
172          node.getChildren(),
173          org.cleartk.syntax.constituent.type.TreebankNode.class);
174      for (org.cleartk.syntax.constituent.type.TreebankNode child : children) {
175        returnValue.append(print(child, (tabs + 1)));
176      }
177    }
178    return returnValue.toString();
179  }
180
181  private static String getTabs(int tabs) {
182    char[] chars = new char[tabs];
183    Arrays.fill(chars, ' ');
184    return new String(chars);
185  }

187  /**
188   * Create a leaf TreebankNode in a JCas.
189   * 
190   * @param jCas
191   *          The JCas which the annotation should be added to.
192   * @param begin
193   *          The begin offset of the node.
194   * @param end
195   *          The end offset of the node.
196   * @param nodeType
197   *          The part of speech tag of the node.
198   * @return The TreebankNode which was added to the JCas.
199   */
200  public static org.cleartk.syntax.constituent.type.TreebankNode newNode(
201      JCas jCas,
202      int begin,
203      int end,
204      String nodeType) {
205    org.cleartk.syntax.constituent.type.TreebankNode node = new org.cleartk.syntax.constituent.type.TreebankNode(
206        jCas,
207        begin,
208        end);
209    node.setNodeType(nodeType);
210    node.setChildren(new FSArray(jCas, 0));
211    node.setLeaf(true);
212    node.addToIndexes();
213    return node;
214  }
215
216  /**
217   * Create a branch TreebankNode in a JCas. The offsets of this node will be determined by its
218   * children.
219   * 
220   * @param jCas
221   *          The JCas which the annotation should be added to.
222   * @param nodeType
223   *          The phrase type tag of the node.
224   * @param children
225   *          The TreebankNode children of the node.
226   * @return The TreebankNode which was added to the JCas.
227   */
228  public static org.cleartk.syntax.constituent.type.TreebankNode newNode(
229      JCas jCas,
230      String nodeType,
231      org.cleartk.syntax.constituent.type.TreebankNode... children) {
232    int begin = children[0].getBegin();
233    int end = children[children.length - 1].getEnd();
234    org.cleartk.syntax.constituent.type.TreebankNode node = new org.cleartk.syntax.constituent.type.TreebankNode(
235        jCas,
236        begin,
237        end);
238    node.setNodeType(nodeType);
239    node.addToIndexes();
240    FSArray fsArray = new FSArray(jCas, children.length);
241    fsArray.copyFromArray(children, 0, 0, children.length);
242    node.setChildren(fsArray);
243    for (org.cleartk.syntax.constituent.type.TreebankNode child : children) {
244      child.setParent(node);
245    }
246    return node;
247  }
