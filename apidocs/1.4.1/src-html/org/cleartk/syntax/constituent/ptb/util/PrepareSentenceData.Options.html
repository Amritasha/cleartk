085  public static void main(String[] args) throws Exception {
086    Options options = new Options();
087    options.parseOptions(args);

089    String inputDirectoryName = options.inputDirectoryName;
090    String outputDirectoryName = options.outputDirectoryName;
091    String sectionsSpecifier = options.sectionsSpecifier;

093    CollectionReader reader = CollectionReaderFactory.createCollectionReader(
094        PennTreebankReader.class,
095        PennTreebankReader.PARAM_CORPUS_DIRECTORY_NAME,
096        inputDirectoryName,
097        PennTreebankReader.PARAM_SECTIONS_SPECIFIER,

099    AnalysisEngine treebankFormatter = AnalysisEngineFactory.createPrimitive(
100        TreebankGoldAnnotator.class,
101        TreebankGoldAnnotator.PARAM_POST_TREES,
102        false);
103    AnalysisEngine xWriter = AnalysisEngineFactory.createPrimitive(
104        XWriter.class,
105        XWriter.PARAM_OUTPUT_DIRECTORY_NAME,
106        outputDirectoryName,
107        XWriter.PARAM_FILE_NAMER_CLASS_NAME,
108        ViewURIFileNamer.class.getName());
109    AnalysisEngine sentencePrinter = AnalysisEngineFactory.createPrimitive(SentencePrinter.class);
110
111    SimplePipeline.runPipeline(reader, treebankFormatter, sentencePrinter, xWriter);
112  }
113
114  public static class SentencePrinter extends JCasAnnotator_ImplBase {
115
116    Map<Character, Integer> charCounts = new HashMap<Character, Integer>();
117
118    int sentenceCount = 0;
119
120    @Override
121    public void process(JCas jCas) throws AnalysisEngineProcessException {
122      for (Sentence sentence : JCasUtil.select(jCas, Sentence.class)) {
123        String sentenceText = sentence.getCoveredText();
124        char lastChar = sentenceText.charAt(sentenceText.length() - 1);
125        int count = 0;
126        if (charCounts.containsKey(lastChar)) {
127          count = charCounts.get(lastChar);
128        }
129        count++;
130        charCounts.put(lastChar, count);
131        sentenceCount++;
132        if (lastChar != '.' && lastChar != '\"' && lastChar != '?') {
133          System.out.println(sentenceText);
134        }
135      }
136    }
137
138    @Override
139    public void collectionProcessComplete() throws AnalysisEngineProcessException {
140      List<Entry<Character, Integer>> entries = new ArrayList<Entry<Character, Integer>>(
141          charCounts.entrySet());
142      Collections.sort(entries, new Comparator<Entry<Character, Integer>>() {
143        public int compare(Entry<Character, Integer> o1, Entry<Character, Integer> o2) {
144          int comparison = o1.getValue().compareTo(o2.getValue());
145          if (comparison == 0) {
146            return o1.getKey().compareTo(o2.getKey());
147          } else {
148            return -comparison;
149          }
150        }
151      });
152
153      for (Entry<Character, Integer> entry : entries) {
154        System.out.println(entry.getKey() + "\t" + entry.getValue());
155      }
156      System.out.println("total number of sentences: " + sentenceCount);
157      super.collectionProcessComplete();
158    }
159
160  }
