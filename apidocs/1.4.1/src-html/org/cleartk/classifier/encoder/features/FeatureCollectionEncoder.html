024package org.cleartk.classifier.encoder.features;
025
026import java.util.ArrayList;
027import java.util.Collections;
028import java.util.List;
029
030import org.cleartk.classifier.Feature;
031import org.cleartk.classifier.encoder.CleartkEncoderException;
032import org.cleartk.classifier.encoder.features.normalizer.NOPNormalizer;
033import org.cleartk.classifier.encoder.features.normalizer.NameNumberNormalizer;
034import org.cleartk.classifier.feature.FeatureCollection;
035
036/**
037 * <br>
038 * Copyright (c) 2007-2008, Regents of the University of Colorado <br>
039 * All rights reserved.
040 * 
041 * @author Philipp Wetzler
042 */
043
044public class FeatureCollectionEncoder implements FeatureEncoder<NameNumber> {
045
046  private static final long serialVersionUID = -7840242678514710238L;
047
048  public FeatureCollectionEncoder(
049      String identifier,
050      FeatureEncoder<NameNumber> subEncoder,
051      NameNumberNormalizer normalizer) {
052    this.identifier = identifier;
053    this.normalizer = normalizer;
054    this.subEncoder = subEncoder;
055  }
056
057  public FeatureCollectionEncoder(
058      FeatureEncoder<NameNumber> subEncoder,
059      NameNumberNormalizer normalizer) {
060    this(null, subEncoder, normalizer);
061  }
062
063  public FeatureCollectionEncoder(String name, FeatureEncoder<NameNumber> subEncoder) {
064    this(name, subEncoder, new NOPNormalizer());
065  }
066
067  public FeatureCollectionEncoder(FeatureEncoder<NameNumber> subEncoder) {
068    this(null, subEncoder, new NOPNormalizer());
069  }
070
071  public List<NameNumber> encode(Feature feature) throws CleartkEncoderException {
072    FeatureCollection fc = (FeatureCollection) feature.getValue();
073    List<NameNumber> fves = new ArrayList<NameNumber>();
074
075    if (identifier != null && !identifier.equals(fc.getIdentifier()))
076      return Collections.emptyList();
077
078    for (Feature f : fc.getFeatures()) {
079      Feature f1 = new Feature(Feature.createName(feature.getName(), f.getName()), f.getValue());
080      fves.addAll(subEncoder.encode(f1));
081    }
082
083    normalizer.normalize(fves);
084
085    return fves;
086  }
087
088  public boolean encodes(Feature feature) {
089    if (feature.getValue() instanceof FeatureCollection) {
090      FeatureCollection f = (FeatureCollection) feature.getValue();
091
092      if (identifier == null || identifier.equals(f.getIdentifier()))
093        return true;
094      else
095        return false;
096    }
097
098    return false;
099  }
100
101  private String identifier;
102
103  private NameNumberNormalizer normalizer = new NOPNormalizer();
104
105  private FeatureEncoder<NameNumber> subEncoder = null;
106}
