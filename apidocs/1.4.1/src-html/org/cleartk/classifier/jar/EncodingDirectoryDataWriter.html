024package org.cleartk.classifier.jar;
025
026import java.io.File;
027import java.io.FileNotFoundException;
028import java.io.PrintWriter;
029
030import org.cleartk.classifier.CleartkProcessingException;
031import org.cleartk.classifier.DataWriter;
032import org.cleartk.classifier.SequenceDataWriter;
033import org.cleartk.classifier.encoder.features.FeaturesEncoder;
034import org.cleartk.classifier.encoder.outcome.OutcomeEncoder;
035
036/**
037 * Superclass for {@link DataWriter} and {@link SequenceDataWriter} implementations that write
038 * training data to a file using a classifier builder with {@link FeaturesEncoder}s and
039 * {@link OutcomeEncoder}s
040 * 
041 * Note that it does not declare that it implements either of the DataWriter interfaces. Subclasses
042 * should do this.
043 * 
044 * <br>
045 * Copyright (c) 2011, Regents of the University of Colorado <br>
046 * All rights reserved.
047 * 
048 * @author Steven Bethard
049 * @author Philip Ogren
050 */
051public abstract class EncodingDirectoryDataWriter<CLASSIFIER_BUILDER_TYPE extends EncodingJarClassifierBuilder<? extends CLASSIFIER_TYPE, ENCODED_FEATURES_TYPE, OUTCOME_TYPE, ENCODED_OUTCOME_TYPE>, CLASSIFIER_TYPE, ENCODED_FEATURES_TYPE, OUTCOME_TYPE, ENCODED_OUTCOME_TYPE>
052    extends DirectoryDataWriter<CLASSIFIER_BUILDER_TYPE, CLASSIFIER_TYPE> {
053
054  public EncodingDirectoryDataWriter(File outputDirectory) throws FileNotFoundException {
055    super(outputDirectory);
056    this.trainingDataFile = this.classifierBuilder.getTrainingDataFile(this.outputDirectory);
057    this.trainingDataWriter = new PrintWriter(this.trainingDataFile);
058  }
059
060  protected File trainingDataFile;
061
062  protected PrintWriter trainingDataWriter;
063
064  public void setFeaturesEncoder(FeaturesEncoder<ENCODED_FEATURES_TYPE> featuresEncoder) {
065    this.classifierBuilder.setFeaturesEncoder(featuresEncoder);
066  }
067
068  public void setOutcomeEncoder(OutcomeEncoder<OUTCOME_TYPE, ENCODED_OUTCOME_TYPE> outcomeEncoder) {
069    this.classifierBuilder.setOutcomeEncoder(outcomeEncoder);
070  }
071
072  @Override
073  public void finish() throws CleartkProcessingException {
074    this.trainingDataWriter.close();
075    super.finish();
076  }
077
078}
