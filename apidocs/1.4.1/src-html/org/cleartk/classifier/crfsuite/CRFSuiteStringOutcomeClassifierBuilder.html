033package org.cleartk.classifier.crfsuite;
034
035import java.io.DataInputStream;
036import java.io.File;
037import java.io.FileOutputStream;
038import java.io.IOException;
039import java.io.InputStream;
040import java.io.OutputStream;
041import java.util.List;
042import java.util.jar.JarInputStream;
043import java.util.jar.JarOutputStream;
044
045import org.apache.uima.UIMAFramework;
046import org.apache.uima.util.Level;
047import org.apache.uima.util.Logger;
048import org.cleartk.classifier.encoder.features.NameNumber;
049import org.cleartk.classifier.jar.JarStreams;
050import org.cleartk.classifier.jar.SequenceClassifierBuilder_ImplBase;
051
052/**
053 * 
054 * 
055 * <br>
056 * Copyright (c) 2011-2012, Technische Universität Darmstadt <br>
057 * All rights reserved.
058 * 
059 * @author Martin Riedl
060 */
061public class CRFSuiteStringOutcomeClassifierBuilder
062    extends
063
064    SequenceClassifierBuilder_ImplBase<CRFSuiteStringOutcomeClassifier, List<NameNumber>, String, String> {
065  public final String MODEL_NAME = "crfsuite.model";
066
067  public final String TRAINING_NAME = "crfsuite.training";
068
069  static Logger logger = UIMAFramework.getLogger(CRFSuiteStringOutcomeClassifierBuilder.class);
070
071  @Override
072  public File getTrainingDataFile(File dir) {
073    return new File(dir, TRAINING_NAME);
074  }
075
076  @Override
077  protected void packageClassifier(File dir, JarOutputStream modelStream) throws IOException {
078    super.packageClassifier(dir, modelStream);
079    JarStreams.putNextJarEntry(modelStream, MODEL_NAME, new File(dir, MODEL_NAME));
080  }
081
082  @Override
083  public void trainClassifier(File dir, String... args) throws Exception {
084    logger.log(Level.FINE, "Start learning CRFsuite classifier");
085    CRFSuiteWrapper wrapper = new CRFSuiteWrapper();
086    String model = new File(dir, MODEL_NAME).getPath();
087    String trainingDataFile = getTrainingDataFile(dir).getPath();
088    wrapper.trainClassifier(model, trainingDataFile, args);
089    logger.log(Level.FINE, "Finished learning CRFsuite classifier");
090  }
091
092  private File modelFile = null;
093
094  /**
095   * As the filename of the model is not known the only solution is to write the model back to a
096   * temporary file
097   */
098  @Override
099  protected void unpackageClassifier(JarInputStream modelStream) throws IOException {
100    super.unpackageClassifier(modelStream);
101    JarStreams.getNextJarEntry(modelStream, MODEL_NAME);
102    this.modelFile = File.createTempFile("model", ".crfsuite");
103    this.modelFile.deleteOnExit();
104    logger.log(Level.FINE, "Start writing model to " + modelFile.getAbsolutePath());
105
106    InputStream inputStream = new DataInputStream(modelStream);
107    OutputStream out = new FileOutputStream(modelFile);
108    byte buf[] = new byte[1024];
109    int len;
110    while ((len = inputStream.read(buf)) > 0) {
111      out.write(buf, 0, len);
112    }
113    out.close();
114    inputStream.close();
115    logger.log(Level.FINE, "Model is written to " + modelFile.getAbsolutePath());
116
117  }
118
119  @Override
120  protected CRFSuiteStringOutcomeClassifier newClassifier() {
121    return new CRFSuiteStringOutcomeClassifier(
122        this.featuresEncoder,
123        this.outcomeEncoder,
124        this.modelFile);
125  }
126
127}
