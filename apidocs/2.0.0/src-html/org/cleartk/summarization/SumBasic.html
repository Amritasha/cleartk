066@Beta
067public class SumBasic extends Summarize_ImplBase<File> {
068
069  private static final int DEFAULT_MAX_NUM_SENTENCES = 10;
070
071  private static final double DEFAULT_SEEN_WORDS_PROB = 0.0001;
072
073  private static final CompositionFunctionType DEFAULT_CF_TYPE = CompositionFunctionType.AVERAGE;
074
075  private static final TokenField DEFAULT_TOKEN_FIELD = TokenField.COVERED_TEXT;
076
077  // public static final String MODEL_NAME = "model.sumbasic";
078
079  private File documentsDirectory;
080
081  private File modelDirectory;
082
083  private File xmiDirectory;
084
085  private Collection<File> items;
086
087  private File stopwordsFile;
088
089  private Double seenWordsProbability = DEFAULT_SEEN_WORDS_PROB;
090
091  private int numSentences = DEFAULT_MAX_NUM_SENTENCES;
092
093  private CompositionFunctionType cfType = DEFAULT_CF_TYPE;
094
095  private TokenField tokenField = DEFAULT_TOKEN_FIELD;
096
097  private boolean outputSentences = false;
098
099  private boolean outputScores = false;
100
101  private File sentencesOutFile;
102
103  public SumBasic(
104      File documentsDirectory,
105      File modelDirectory,
106      File xmiDirectory,
107      File stopwordsFile,
108      SumBasicAnnotator.TokenField tokenField) {
109    this.documentsDirectory = documentsDirectory;
110    this.modelDirectory = modelDirectory;
111    this.xmiDirectory = xmiDirectory;
112
113    this.stopwordsFile = stopwordsFile;
114    this.tokenField = tokenField;
115
116    this.items = FileUtils.listFiles(
117        this.documentsDirectory,
118        new UriCollectionReader.RejectSystemFiles(),
119        new UriCollectionReader.RejectSystemDirectories());
120  }
121
122  @Override
123  protected CollectionReader getCollectionReader(Collection<File> files) throws Exception {
124    return UriCollectionReader.getCollectionReaderFromFiles(files);
125  }
126
127  @Override
128  protected void train() throws Exception {
129    AggregateBuilder builder = this.buildTrainingAggregate();
130
131    // Run preprocessing and tfidf counts analyzer
132    SimplePipeline.runPipeline(
133        this.getCollectionReader(items),
134        builder.createAggregateDescription());
135
136    // For more details on the training arguments refer to SumBasicModel
137    String[] trainingArgs = {
138        "--max-num-sentences",
139        Integer.toString(this.numSentences),
140        "--seen-words-prob",
141        Double.toString(this.seenWordsProbability),
142        "--composition-function",
143        this.cfType.toString() };
144
145    JarClassifierBuilder.trainAndPackage(this.modelDirectory, trainingArgs);
146  }
147
148  @Override
149  public void extract() throws Exception {
150    Collection<File> xmis = FileUtils.listFiles(
151        this.xmiDirectory,
152        new UriCollectionReader.RejectSystemFiles(),
153        new UriCollectionReader.RejectSystemDirectories());
154
155    // This simply runs the model and tags the extracted sentences in the CAS and writes the output
156    // to a file if specified.
157    AggregateBuilder builder = buildExtractAggregate();
158
159    // Run preprocessing and tfidf counts analyzer
160    SimplePipeline.runPipeline(this.getCollectionReader(xmis), builder.createAggregateDescription());
161
162  }
163
164  public AggregateBuilder buildTrainingAggregate() throws ResourceInitializationException {
165
166    AggregateBuilder builder = new AggregateBuilder();
167
168    builder.add(UriToDocumentTextAnnotator.getDescription());
169
170    // NLP pre-processing components
171    builder.add(SentenceAnnotator.getDescription());
172    builder.add(TokenAnnotator.getDescription());
173    builder.add(PosTaggerAnnotator.getDescription());
174    builder.add(DefaultSnowballStemmer.getDescription("English"));
175
176    // This will extract the features for summarization
177    builder.add(AnalysisEngineFactory.createEngineDescription(
178        SumBasicAnnotator.class,
179        DefaultDataWriterFactory.PARAM_DATA_WRITER_CLASS_NAME,
180        SumBasicDataWriter.class.getName(),
181        DirectoryDataWriterFactory.PARAM_OUTPUT_DIRECTORY,
182        this.modelDirectory.getPath(),
183        SumBasicAnnotator.PARAM_TOKEN_FIELD,
184        this.tokenField.name(),
185        SumBasicAnnotator.PARAM_STOPWORDS_URI,
186        stopwordsFile.toURI()));
187
188    // Save off xmis for re-reading
189    builder.add(AnalysisEngineFactory.createEngineDescription(
190        XmiWriter.class,
191        XmiWriter.PARAM_OUTPUT_DIRECTORY,
192        xmiDirectory.getPath()));
193
194    return builder;
195  }
196
197  public AggregateBuilder buildExtractAggregate() throws ResourceInitializationException {
198    AggregateBuilder builder = new AggregateBuilder();
199
200    builder.add(AnalysisEngineFactory.createEngineDescription(
201        XMIReader.class,
202        XMIReader.PARAM_XMI_DIRECTORY,
203        this.xmiDirectory));
204
205    // This will extract the features for summarization
206    builder.add(AnalysisEngineFactory.createEngineDescription(
207        SumBasicAnnotator.class,
208        CleartkAnnotator.PARAM_IS_TRAINING,
209        false,
210        GenericJarClassifierFactory.PARAM_CLASSIFIER_JAR_PATH,
211        JarClassifierBuilder.getModelJarFile(this.modelDirectory),
212        SumBasicAnnotator.PARAM_TOKEN_FIELD,
213        this.tokenField.name(),
214        SumBasicAnnotator.PARAM_STOPWORDS_URI,
215        this.stopwordsFile.toURI()));
216
217    if (this.sentencesOutFile != null && this.outputSentences) {
218      builder.add(SummarySentenceWriterAnnotator.getDescription(sentencesOutFile, this.outputScores));
219    }
220    return builder;
221  }
222
223  public void setSentencesOutFile(File file, boolean writeScores) {
224    this.sentencesOutFile = file;
225    this.outputSentences = true;
226    this.outputScores = writeScores;
227  }

229  public interface Options {
230
231    @Option(
232        longName = "max-num-sentences",
233        description = "Specifies the maximum number of sentences to extract in the summary",
234        defaultValue = "10")
235    public int getMaxNumSentences();
236
237    @Option(
238        longName = "seen-words-prob",
239        description = "Specify the probability for seen words.",
240        defaultValue = "0.0001")
241    public double getSeenWordsProbability();
242
243    @Option(
244        longName = "composition-function",
245        description = "Specifies how word probabilities are combined (AVERAGE|SUM|PRODUCT, default=AVERAGE)",
246        defaultValue = "AVERAGE")
247    public CompositionFunctionType getCFType();
248
249    @Option(
250        longName = "token-field",
251        description = "Specifies what kind of token is used for summarization, (COVERED_TEXT|STEM|LEMMA, default=COVERED_TEXT)",
252        defaultValue = "COVERED_TEXT")
253    public TokenField getTokenField();
254
255    @Option(
256        longName = "stopwords-file",
257        description = "Path to whitespace delimited stopwords text file",
258        defaultValue = "src/main/resources/stopwords.txt")
259    public File getStopwordsFile();
260
261    @Option(
262        longName = "documents-dir",
263        description = "Path to documents to summarize",
264        defaultValue = "src/test/resources/test_documents")
265    public File getDocumentsDir();
266
267    @Option(
268        longName = "model-dir",
269        description = "Path for saving model data",
270        defaultValue = "target/models")
271    public File getModelDir();
272
273    @Option(
274        longName = "xmi-dir",
275        description = "Path for saving intermediate cas xmi files.  Leave unspecified for a temporary directory",
276        defaultToNull = true)
277    public File getXMIDir();
278
279    @Option(
280        longName = "sentencesOutFile",
281        description = "Path to the output file",
282        defaultValue = "target/sentences.out")
283    public File getSentencesOutFile();
284
285    @Option(longName = "outputScores", description = "Path to the output file")
286    public boolean getOutputScores();
287
288  }
289
290  public static void main(String[] args) throws Exception {
291    Options options = CliFactory.parseArguments(Options.class, args);
292    File xmiDir = options.getXMIDir();
293    if (xmiDir == null) {
294      xmiDir = Files.createTempDir();
295    }
296
297    SumBasic summarizer = new SumBasic(
298        options.getDocumentsDir(),
299        options.getModelDir(),
300        xmiDir,
301        options.getStopwordsFile(),
302        SumBasicAnnotator.TokenField.COVERED_TEXT);
303    summarizer.setSentencesOutFile(options.getSentencesOutFile(), options.getOutputScores());
304
305    System.out.println("Training");
306    summarizer.train();
307
308    System.out.println("Extracting sentences");
309    summarizer.extract();
310  }

312  public static class XMIReader extends JCasAnnotator_ImplBase {
313
314    @ConfigurationParameter(name = PARAM_XMI_DIRECTORY, mandatory = true)
315    protected File xmiDirectory;
316
317    public static final String PARAM_XMI_DIRECTORY = "xmiDirectory";
318
319    protected File getFile(JCas jCas) throws AnalysisEngineProcessException {
320      File origFile = new File(ViewUriUtil.getURI(jCas));
321      String ext = Files.getFileExtension(origFile.getName());
322      String xmi = origFile.getName().replaceAll("\\." + ext + "$", "");
323      return new File(this.xmiDirectory, xmi + ".xmi");
324    }
325
326    @Override
327    public void process(JCas jCas) throws AnalysisEngineProcessException {
328      try {
329        FileInputStream stream = new FileInputStream(this.getFile(jCas));
330        try {
331          XmiCasDeserializer.deserialize(stream, jCas.getCas());
332        } finally {
333          stream.close();
334        }
335      } catch (SAXException e) {
336        throw new AnalysisEngineProcessException(e);
337      } catch (IOException e) {
338        throw new AnalysisEngineProcessException(e);
339      }
340    }
341  }
