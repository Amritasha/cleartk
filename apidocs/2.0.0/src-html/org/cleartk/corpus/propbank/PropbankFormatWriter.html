047
048public class PropbankFormatWriter implements AnnotationWriter<Predicate> {
049
050  public void initialize(UimaContext context) throws ResourceInitializationException {
051  }
052
053  public String writeAnnotation(JCas jCas, Predicate predicate)
054      throws AnalysisEngineProcessException {
055    StringBuilder sb = new StringBuilder();
056
057    String uri = new File(ViewUriUtil.getURI(jCas)).getPath();
058    sb.append(uri + "\t");
059
060    int sentenceId = -1;
061    int i = 0;
062    for (Sentence sentence : JCasUtil.select(jCas, Sentence.class)) {
063      if (JCasUtil.selectCovered(jCas, Predicate.class, sentence).contains(predicate)) {
064        sentenceId = i;
065      }
066      i += 1;
067    }
068
069    String treeId = Integer.toString(sentenceId);
070    sb.append(treeId + "\t");
071
072    String frameSetId = predicate.getBaseForm() + "." + predicate.getFrameSet();
073    sb.append(frameSetId + "\t");
074
075    // this part seems like cheating. what we should really do here is calculate the propTxt based
076    // on the other information
077    // in the cas and then compare it to the explicitly stored version and throw an exception if
078    // there is a difference. That way,
079    // we know the data is consistent.
080    Collection<Argument> arguments = JCasUtil.select(predicate.getArguments(), Argument.class);
081    for (Argument argument : arguments) {
082      if (argument instanceof SemanticArgument)
083        sb.append(((SemanticArgument) argument).getPropTxt() + "\t");
084    }
085    return sb.toString();
086  }
087}
