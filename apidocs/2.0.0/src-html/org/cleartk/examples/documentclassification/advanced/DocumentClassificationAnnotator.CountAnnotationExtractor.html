141  public void initialize(UimaContext context) throws ResourceInitializationException {
142    super.initialize(context);
143
144    try {
145      TfidfExtractor<String, DocumentAnnotation> tfIdfExtractor = initTfIdfExtractor();
146      CentroidTfidfSimilarityExtractor<String, DocumentAnnotation> simExtractor = initCentroidTfIdfSimilarityExtractor();
147      ZeroMeanUnitStddevExtractor<String, DocumentAnnotation> zmusExtractor = initZmusExtractor();
148      MinMaxNormalizationExtractor<String, DocumentAnnotation> minmaxExtractor = initMinMaxExtractor();
149      this.extractor = new CombinedExtractor1<DocumentAnnotation>(
150          tfIdfExtractor,
151          simExtractor,
152          zmusExtractor,
153          minmaxExtractor);
154    } catch (IOException e) {
155      throw new ResourceInitializationException(e);
156    }
157  }
158
159  private TfidfExtractor<String, DocumentAnnotation> initTfIdfExtractor() throws IOException {
160    CleartkExtractor<DocumentAnnotation, Token> countsExtractor = new CleartkExtractor<DocumentAnnotation, Token>(
161        Token.class,
162        new CoveredTextExtractor<Token>(),
163        new CleartkExtractor.Count(new CleartkExtractor.Covered()));
164
165    TfidfExtractor<String, DocumentAnnotation> tfIdfExtractor = new TfidfExtractor<String, DocumentAnnotation>(
166        DocumentClassificationAnnotator.TFIDF_EXTRACTOR_KEY,
167        countsExtractor);
168
169    if (this.tfIdfUri != null) {
170      tfIdfExtractor.load(this.tfIdfUri);
171    }
172    return tfIdfExtractor;
173  }
174
175  private CentroidTfidfSimilarityExtractor<String, DocumentAnnotation> initCentroidTfIdfSimilarityExtractor()
176      throws IOException {
177    CleartkExtractor<DocumentAnnotation, Token> countsExtractor = new CleartkExtractor<DocumentAnnotation, Token>(
178        Token.class,
179        new CoveredTextExtractor<Token>(),
180        new CleartkExtractor.Count(new CleartkExtractor.Covered()));
181
182    CentroidTfidfSimilarityExtractor<String, DocumentAnnotation> simExtractor = new CentroidTfidfSimilarityExtractor<String, DocumentAnnotation>(
183        DocumentClassificationAnnotator.CENTROID_TFIDF_SIM_EXTRACTOR_KEY,
184        countsExtractor);
185
186    if (this.tfIdfCentroidSimilarityUri != null) {
187      simExtractor.load(this.tfIdfCentroidSimilarityUri);
188    }
189    return simExtractor;
190  }
191
192  private ZeroMeanUnitStddevExtractor<String, DocumentAnnotation> initZmusExtractor()
193      throws IOException {
194    CombinedExtractor1<DocumentAnnotation> featuresToNormalizeExtractor = new CombinedExtractor1<DocumentAnnotation>(
195        new CountAnnotationExtractor<DocumentAnnotation>(Sentence.class),
196        new CountAnnotationExtractor<DocumentAnnotation>(Token.class));
197
198    ZeroMeanUnitStddevExtractor<String, DocumentAnnotation> zmusExtractor = new ZeroMeanUnitStddevExtractor<String, DocumentAnnotation>(
199        ZMUS_EXTRACTOR_KEY,
200        featuresToNormalizeExtractor);
201
202    if (this.zmusUri != null) {
203      zmusExtractor.load(this.zmusUri);
204    }
205
206    return zmusExtractor;
207  }
208
209  private MinMaxNormalizationExtractor<String, DocumentAnnotation> initMinMaxExtractor()
210      throws IOException {
211    CombinedExtractor1<DocumentAnnotation> featuresToNormalizeExtractor = new CombinedExtractor1<DocumentAnnotation>(
212        new CountAnnotationExtractor<DocumentAnnotation>(Sentence.class),
213        new CountAnnotationExtractor<DocumentAnnotation>(Token.class));
214
215    MinMaxNormalizationExtractor<String, DocumentAnnotation> minmaxExtractor = new MinMaxNormalizationExtractor<String, DocumentAnnotation>(
216        MINMAX_EXTRACTOR_KEY,
217        featuresToNormalizeExtractor);
218
219    if (this.minmaxUri != null) {
220      minmaxExtractor.load(this.minmaxUri);
221    }
222
223    return minmaxExtractor;
224  }

226  public void process(JCas jCas) throws AnalysisEngineProcessException {
227    DocumentAnnotation doc = (DocumentAnnotation) jCas.getDocumentAnnotationFs();
228
229    Instance<String> instance = new Instance<String>();
230    instance.addAll(this.extractor.extract(jCas, doc));
231
232    if (isTraining()) {
233      UsenetDocument document = JCasUtil.selectSingle(jCas, UsenetDocument.class);
234      instance.setOutcome(document.getCategory());
235      this.dataWriter.write(instance);
236    } else {
237      // This is classification, so classify and create UsenetDocument annotation
238      String result = this.classifier.classify(instance.getFeatures());
239      UsenetDocument document = new UsenetDocument(jCas, 0, jCas.getDocumentText().length());
240      document.setCategory(result);
241      document.addToIndexes();
242      // System.out.println("classified " + ViewURIUtil.getURI(jCas) + " as " + result + ".");
243    }
244  }

254  public static class CountAnnotationExtractor<T extends Annotation> implements
255      NamedFeatureExtractor1<T> {
256
257    private Class<? extends Annotation> annotationType;
258
259    private String name;
260
261    public CountAnnotationExtractor(Class<? extends Annotation> annotationType) {
262      this.annotationType = annotationType;
263      this.name = "Count_" + this.annotationType.getName();
264    }
265
266    @Override
267    public String getFeatureName() {
268      return this.name;
269    }
270
271    @Override
272    public List<Feature> extract(JCas view, Annotation focusAnnotation)
273        throws CleartkExtractorException {
274      List<?> annotations = JCasUtil.selectCovered(this.annotationType, focusAnnotation);
275      return Arrays.asList(new Feature(this.name, annotations.size()));
276    }
277  }
