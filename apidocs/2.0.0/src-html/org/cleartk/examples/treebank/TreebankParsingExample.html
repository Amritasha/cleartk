025package org.cleartk.examples.treebank;
026
027import java.io.File;
028import java.util.Collection;
029import java.util.List;
030
031import org.apache.commons.io.FileUtils;
032import org.apache.uima.cas.CAS;
033import org.apache.uima.collection.CollectionReader;
034import org.apache.uima.fit.component.ViewCreatorAnnotator;
035import org.apache.uima.fit.factory.AggregateBuilder;
036import org.apache.uima.fit.factory.AnalysisEngineFactory;
037import org.apache.uima.fit.pipeline.SimplePipeline;
038import org.cleartk.corpus.penntreebank.PennTreebankReader;
039import org.cleartk.corpus.penntreebank.TreebankGoldAnnotator;
040import org.cleartk.eval.EvaluationConstants;
041import org.cleartk.util.ae.UriToDocumentTextAnnotator;
042import org.cleartk.util.ae.XmiWriter;
043import org.cleartk.util.cr.UriCollectionReader;
044
045import com.lexicalscope.jewel.cli.CliFactory;
046import com.lexicalscope.jewel.cli.Option;
047
048/**
049 * <br>
050 * Copyright (c) 2010, Regents of the University of Colorado <br>
051 * All rights reserved.
052 * 
053 * This class takes in treebank files and parses the the PTB-style format and populates a the CAS
054 * with constituent trees and then writes out XMI files. The CAS will contain the raw treebank data
055 * in a view You may find it useful to view the generated XMI files in the CAS Visual Debugger.
056 * 
057 * If you do not have access to PennTreebank data or any other treebank data, then you can run this
058 * script with the following arguments:
059 * <ul>
060 * <li>-td data/pos/treebank</li>
061 * <li>-o data/pos/treebank</li>
062 * <li>-suf tree</li>
063 * </ul>
064 * 
065 * To view the resulting xmi file in the CAS Visual Debugger you can run the eclipse launch
066 * configuration labeled "CVD (cleartk-examples)". When the CVD opens select Menu -> File -> Read
067 * XMI CAS File. Now navigate to the single xmi file located in data/pos/treebank. Once the file is
068 * open in the CVD be sure to examine the different views by selecting different values from the
069 * drop down box labeled "Select View" near the top of the window.
070 * 
071 * @author Philip Ogren
072 * 
073 */
074public class TreebankParsingExample {
075
076  public interface Options {
077    @Option(
078        longName = { "td", "treebankDirectory" },
079        description = "specify the directory containing treebank files")
080    public File getTreebankDirectory();
081
082    @Option(
083        shortName = "o",
084        longName = "outputDirectory",
085        description = "specify the directory to write the XMI files to")
086    public String getOutputDirectory();
087
088    @Option(
089        longName = { "suf", "treebankFileSuffixes" },
090        description = "specify file suffixes of the treebank files in the treebank directory")
091    public List<String> getTreebankFileSuffixes();
092  }
093
094  public static void main(String[] args) throws Exception {
095    Options options = CliFactory.parseArguments(Options.class, args);
096
097    // Loads URIS specified by files into URI view
098    String[] suffixes = options.getTreebankFileSuffixes().toArray(
099        new String[options.getTreebankFileSuffixes().size()]);
100    File treebankDir = options.getTreebankDirectory();
101    Collection<File> files = FileUtils.listFiles(treebankDir, suffixes, false);
102    CollectionReader reader = UriCollectionReader.getCollectionReaderFromFiles(files);
103
104    AggregateBuilder builder = new AggregateBuilder();
105
106    // Reads text into TREEBANK_VIEW
107    builder.add(UriToDocumentTextAnnotator.getDescriptionForView(PennTreebankReader.TREEBANK_VIEW));
108
109    // Ensures GOLD_VIEW is present
110    builder.add(AnalysisEngineFactory.createEngineDescription(
111        ViewCreatorAnnotator.class,
112        ViewCreatorAnnotator.PARAM_VIEW_NAME,
113        EvaluationConstants.GOLD_VIEW));
114
115    // Parses treebank text into gold view (instead of default sofa view)
116    builder.add(
117        TreebankGoldAnnotator.getDescription(),
118        CAS.NAME_DEFAULT_SOFA,
119        EvaluationConstants.GOLD_VIEW);
120
121    // XMI Writer
122    builder.add(XmiWriter.getDescription(new File(options.getOutputDirectory())));
123
124    SimplePipeline.runPipeline(reader, builder.createAggregateDescription());
125
126  }
127}
