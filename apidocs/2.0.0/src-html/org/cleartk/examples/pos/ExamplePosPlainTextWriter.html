024package org.cleartk.examples.pos;
025
026import java.io.File;
027import java.io.FileNotFoundException;
028import java.io.PrintWriter;
029
030import org.apache.uima.UimaContext;
031import org.apache.uima.analysis_engine.AnalysisEngineProcessException;
032import org.apache.uima.jcas.JCas;
033import org.apache.uima.resource.ResourceInitializationException;
034import org.cleartk.token.type.Sentence;
035import org.cleartk.token.type.Token;
036import org.cleartk.util.ViewUriUtil;
037import org.apache.uima.fit.component.JCasAnnotator_ImplBase;
038import org.apache.uima.fit.descriptor.ConfigurationParameter;
039import org.apache.uima.fit.util.JCasUtil;
040
041/**
042 * <br>
043 * Copyright (c) 2007-2008, Regents of the University of Colorado <br>
044 * All rights reserved.
045 * 
046 * @author Steven Bethard
047 */
048public class ExamplePosPlainTextWriter extends JCasAnnotator_ImplBase {
049
050  public static final String DEFAULT_OUTPUT_DIRECTORY = "target/examples/pos";
051
052  public static final String PARAM_OUTPUT_DIRECTORY_NAME = "outputDirectoryName";
053
054  @ConfigurationParameter(
055      name = PARAM_OUTPUT_DIRECTORY_NAME,
056      mandatory = true,
057      defaultValue = DEFAULT_OUTPUT_DIRECTORY,
058      description = "provides the directory where the token/pos text files will be written")
059  private String outputDirectoryName;
060
061  protected File outputDir;
062
063  @Override
064  public void initialize(UimaContext context) throws ResourceInitializationException {
065    super.initialize(context);
066    this.outputDir = new File(outputDirectoryName);
067    if (!this.outputDir.exists()) {
068      this.outputDir.mkdirs();
069    }
070  }
071
072  @Override
073  public void process(JCas jCas) throws AnalysisEngineProcessException {
074    String id = new File(ViewUriUtil.getURI(jCas)).getName();
075    PrintWriter outputWriter;
076    try {
077      outputWriter = new PrintWriter(new File(this.outputDir, id + ".pos"));
078    } catch (FileNotFoundException e) {
079      throw new AnalysisEngineProcessException(e);
080    }
081    for (Sentence sentence : JCasUtil.select(jCas, Sentence.class)) {
082      for (Token token : JCasUtil.selectCovered(jCas, Token.class, sentence)) {
083        outputWriter.print(token.getCoveredText());
084        outputWriter.print('/');
085        outputWriter.print(token.getPos());
086        outputWriter.print(' ');
087      }
088      outputWriter.println();
089    }
090    outputWriter.close();
091  }
092
093  public void setOutputDirectoryName(String outputDirectoryName) {
094    this.outputDirectoryName = outputDirectoryName;
095  }
096
097}
