070public class ExamplePosAnnotator extends CleartkSequenceAnnotator<String> {
071
072  public static final String DEFAULT_OUTPUT_DIRECTORY = "target/examples/pos";
073
074  private FeatureExtractor1<Token> tokenFeatureExtractor;
075
076  private CleartkExtractor<Token, Token> contextFeatureExtractor;
077
078  public void initialize(UimaContext context) throws ResourceInitializationException {
079    super.initialize(context);
080
081    // a feature extractor that creates features corresponding to the word, the word lower cased
082    // the capitalization of the word, the numeric characterization of the word, and character ngram
083    // suffixes of length 2 and 3.
084    this.tokenFeatureExtractor = new FeatureFunctionExtractor<Token>(
085        new CoveredTextExtractor<Token>(),
086        new LowerCaseFeatureFunction(),
087        new CapitalTypeFeatureFunction(),
088        new NumericTypeFeatureFunction(),
089        new CharacterNgramFeatureFunction(Orientation.RIGHT_TO_LEFT, 0, 2),
090        new CharacterNgramFeatureFunction(Orientation.RIGHT_TO_LEFT, 0, 3));
091
092    // a feature extractor that extracts the surrounding token texts (within the same sentence)
093    this.contextFeatureExtractor = new CleartkExtractor<Token, Token>(
094        Token.class,
095        new CoveredTextExtractor<Token>(),
096        new Preceding(2),
097        new Following(2));
098  }
099
100  public void process(JCas jCas) throws AnalysisEngineProcessException {
101    // for each sentence in the document, generate training/classification instances
102    for (Sentence sentence : JCasUtil.select(jCas, Sentence.class)) {
103      List<List<Feature>> tokenFeatureLists = new ArrayList<List<Feature>>();
104      List<String> tokenOutcomes = new ArrayList<String>();
105
106      // for each token, extract features and the outcome
107      List<Token> tokens = JCasUtil.selectCovered(jCas, Token.class, sentence);
108      for (Token token : tokens) {
109
110        // apply the two feature extractors
111        List<Feature> tokenFeatures = new ArrayList<Feature>();
112        tokenFeatures.addAll(this.tokenFeatureExtractor.extract(jCas, token));
113        tokenFeatures.addAll(this.contextFeatureExtractor.extractWithin(jCas, token, sentence));
114        tokenFeatureLists.add(tokenFeatures);
115
116        // add the expected token label from the part of speech
117        if (this.isTraining()) {
118          tokenOutcomes.add(token.getPos());
119        }
120      }
121
122      // for training, write instances to the data write
123      if (this.isTraining()) {
124        this.dataWriter.write(Instances.toInstances(tokenOutcomes, tokenFeatureLists));
125      }
126
127      // for classification, set the token part of speech tags from the classifier outcomes.
128      else {
129        List<String> outcomes = this.classifier.classify(tokenFeatureLists);
130        Iterator<Token> tokensIter = tokens.iterator();
131        for (String outcome : outcomes) {
132          tokensIter.next().setPos(outcome);
133        }
134      }
135    }
136  }
137
138  public static AnalysisEngineDescription getClassifierDescription(String modelFileName)
139      throws ResourceInitializationException {
140    return AnalysisEngineFactory.createEngineDescription(
141        ExamplePosAnnotator.class,
142        GenericJarClassifierFactory.PARAM_CLASSIFIER_JAR_PATH,
143        modelFileName);
144  }
145
146  public static AnalysisEngineDescription getWriterDescription(String outputDirectory)
147      throws ResourceInitializationException {
148    return AnalysisEngineFactory.createEngineDescription(
149        ExamplePosAnnotator.class,
150        CleartkSequenceAnnotator.PARAM_DATA_WRITER_FACTORY_CLASS_NAME,
151        ViterbiDataWriterFactory.class.getName(),
152        DirectoryDataWriterFactory.PARAM_OUTPUT_DIRECTORY,
153        outputDirectory,
154        ViterbiDataWriterFactory.PARAM_DELEGATED_DATA_WRITER_FACTORY_CLASS,
155        DefaultDataWriterFactory.class.getName(),
156        DefaultDataWriterFactory.PARAM_DATA_WRITER_CLASS_NAME,
157        MaxentStringOutcomeDataWriter.class.getName(),
158        ViterbiDataWriterFactory.PARAM_OUTCOME_FEATURE_EXTRACTOR_NAMES,
159        new String[] { DefaultOutcomeFeatureExtractor.class.getName() });
160  }
