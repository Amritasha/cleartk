025package org.cleartk.examples.pos;
026
027import java.io.File;
028import java.util.Collection;
029
030import org.apache.commons.io.FileUtils;
031import org.apache.commons.io.filefilter.FileFilterUtils;
032import org.apache.commons.io.filefilter.IOFileFilter;
033import org.apache.uima.cas.CAS;
034import org.apache.uima.collection.CollectionReader;
035import org.cleartk.corpus.penntreebank.PennTreebankReader;
036import org.cleartk.corpus.penntreebank.TreebankGoldAnnotator;
037import org.cleartk.ml.jar.JarClassifierBuilder;
038import org.cleartk.ml.jar.Train;
039import org.cleartk.util.ae.UriToDocumentTextAnnotator;
040import org.cleartk.util.cr.UriCollectionReader;
041import org.apache.uima.fit.component.ViewCreatorAnnotator;
042import org.apache.uima.fit.factory.AggregateBuilder;
043import org.apache.uima.fit.factory.AnalysisEngineFactory;
044import org.apache.uima.fit.pipeline.SimplePipeline;
045
046/**
047 * <br>
048 * Copyright (c) 2009, Regents of the University of Colorado <br>
049 * All rights reserved.
050 * 
051 * @author Philip Ogren
052 * 
053 *         For examples of using the ExamplePOSAnnotationHandler using different classifiers, please
054 *         see org.cleartk.example.pos.ExamplePosClassifierTest
055 */
056
057public class BuildTestExamplePosModel {
058
059  public static void main(String... args) throws Exception {
060    String outputDirectory = ExamplePosAnnotator.DEFAULT_OUTPUT_DIRECTORY;
061
062    // select all the .tree files in the treebank directory
063    File treebankDirectory = new File("data/pos/treebank");
064    IOFileFilter treeFileFilter = FileFilterUtils.suffixFileFilter(".tree");
065    Collection<File> files = FileUtils.listFiles(treebankDirectory, treeFileFilter, null);
066
067    // A collection reader that creates one CAS per file, containing the file's URI
068    CollectionReader reader = UriCollectionReader.getCollectionReaderFromFiles(files);
069
070    // The pipeline of annotators
071    AggregateBuilder builder = new AggregateBuilder();
072
073    // An annotator that creates an empty treebank view in the CAS
074    builder.add(AnalysisEngineFactory.createEngineDescription(
075        ViewCreatorAnnotator.class,
076        ViewCreatorAnnotator.PARAM_VIEW_NAME,
077        PennTreebankReader.TREEBANK_VIEW));
078
079    // An annotator that reads the treebank-formatted text into the treebank view
080    builder.add(
081        UriToDocumentTextAnnotator.getDescription(),
082        CAS.NAME_DEFAULT_SOFA,
083        PennTreebankReader.TREEBANK_VIEW);
084
085    // An annotator that uses the treebank text to add tokens and POS tags to the CAS
086    builder.add(TreebankGoldAnnotator.getDescriptionPOSTagsOnly());
087
088    // The POS annotator, configured to write training data
089    builder.add(ExamplePosAnnotator.getWriterDescription(outputDirectory));
090
091    // Run the pipeline of annotators on each of the CASes produced by the reader
092    SimplePipeline.runPipeline(reader, builder.createAggregateDescription());
093
094    System.out.println("training data written to " + ExamplePosAnnotator.DEFAULT_OUTPUT_DIRECTORY);
095    System.out.println("training model...");
096
097    // Train a classifier on the training data, and package it into a .jar file
098    Train.main(outputDirectory);
099
100    System.out.println("model written to "
101        + JarClassifierBuilder.getModelJarFile(ExamplePosAnnotator.DEFAULT_OUTPUT_DIRECTORY).getPath());
102
103  }
104}
