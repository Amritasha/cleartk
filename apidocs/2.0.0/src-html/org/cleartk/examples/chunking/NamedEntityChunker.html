024package org.cleartk.examples.chunking;
025
026import java.util.ArrayList;
027import java.util.List;
028
029import org.apache.uima.UimaContext;
030import org.apache.uima.analysis_engine.AnalysisEngineProcessException;
031import org.apache.uima.fit.util.JCasUtil;
032import org.apache.uima.jcas.JCas;
033import org.apache.uima.resource.ResourceInitializationException;
034import org.cleartk.ml.CleartkSequenceAnnotator;
035import org.cleartk.ml.Feature;
036import org.cleartk.ml.Instances;
037import org.cleartk.ml.chunking.BioChunking;
038import org.cleartk.ml.feature.extractor.CleartkExtractor;
039import org.cleartk.ml.feature.extractor.CombinedExtractor1;
040import org.cleartk.ml.feature.extractor.CoveredTextExtractor;
041import org.cleartk.ml.feature.extractor.FeatureExtractor1;
042import org.cleartk.ml.feature.extractor.TypePathExtractor;
043import org.cleartk.ml.feature.extractor.CleartkExtractor.Following;
044import org.cleartk.ml.feature.extractor.CleartkExtractor.Preceding;
045import org.cleartk.ml.feature.function.CharacterCategoryPatternFunction;
046import org.cleartk.ml.feature.function.FeatureFunctionExtractor;
047import org.cleartk.ml.feature.function.CharacterCategoryPatternFunction.PatternType;
048import org.cleartk.ne.type.NamedEntityMention;
049import org.cleartk.token.type.Sentence;
050import org.cleartk.token.type.Token;
051
052/**
053 * This is the most important class in the named entity chunking example -- it demonstrates how to
054 * write a ClearTK annotator that creates NamedEntityMention annotations by classifying Token
055 * annotations.
056 * 
057 * <br>
058 * Copyright (c) 2012, Regents of the University of Colorado <br>
059 * All rights reserved.
060 * 
061 * @author Steven Bethard
062 */
063public class NamedEntityChunker extends CleartkSequenceAnnotator<String> {
064
065  private FeatureExtractor1<Token> extractor;
066
067  private CleartkExtractor<Token, Token> contextExtractor;
068
069  private BioChunking<Token, NamedEntityMention> chunking;
070
071  @Override
072  public void initialize(UimaContext context) throws ResourceInitializationException {
073    super.initialize(context);
074
075    // the token feature extractor: text, char pattern (uppercase, digits, etc.), and part-of-speech
076    this.extractor = new CombinedExtractor1<Token>(
077        new FeatureFunctionExtractor<Token>(
078            new CoveredTextExtractor<Token>(),
079            new CharacterCategoryPatternFunction<Token>(PatternType.REPEATS_MERGED)),
080        new TypePathExtractor<Token>(Token.class, "pos"));
081
082    // the context feature extractor: the features above for the 3 preceding and 3 following tokens
083    this.contextExtractor = new CleartkExtractor<Token, Token>(
084        Token.class,
085        this.extractor,
086        new Preceding(3),
087        new Following(3));
088
089    // the chunking definition: Tokens will be combined to form NamedEntityMentions, with labels
090    // from the "mentionType" attribute so that we get B-location, I-person, etc.
091    this.chunking = new BioChunking<Token, NamedEntityMention>(
092        Token.class,
093        NamedEntityMention.class,
094        "mentionType");
095  }
096
097  @Override
098  public void process(JCas jCas) throws AnalysisEngineProcessException {
099    for (Sentence sentence : JCasUtil.select(jCas, Sentence.class)) {
100
101      // extract features for each token in the sentence
102      List<Token> tokens = JCasUtil.selectCovered(jCas, Token.class, sentence);
103      List<List<Feature>> featureLists = new ArrayList<List<Feature>>();
104      for (Token token : tokens) {
105        List<Feature> features = new ArrayList<Feature>();
106        features.addAll(this.extractor.extract(jCas, token));
107        features.addAll(this.contextExtractor.extract(jCas, token));
108        featureLists.add(features);
109      }
110
111      // during training, convert NamedEntityMentions in the CAS into expected classifier outcomes
112      if (this.isTraining()) {
113
114        // extract the gold (human annotated) NamedEntityMention annotations
115        List<NamedEntityMention> namedEntityMentions = JCasUtil.selectCovered(
116            jCas,
117            NamedEntityMention.class,
118            sentence);
119
120        // convert the NamedEntityMention annotations into token-level BIO outcome labels
121        List<String> outcomes = this.chunking.createOutcomes(jCas, tokens, namedEntityMentions);
122
123        // write the features and outcomes as training instances
124        this.dataWriter.write(Instances.toInstances(outcomes, featureLists));
125      }
126
127      // during classification, convert classifier outcomes into NamedEntityMentions in the CAS
128      else {
129
130        // get the predicted BIO outcome labels from the classifier
131        List<String> outcomes = this.classifier.classify(featureLists);
132
133        // create the NamedEntityMention annotations in the CAS
134        this.chunking.createChunks(jCas, tokens, outcomes);
135      }
136    }
137  }
138}
