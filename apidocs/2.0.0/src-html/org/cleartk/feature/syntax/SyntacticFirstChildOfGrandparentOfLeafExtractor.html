024package org.cleartk.feature.syntax;
025
026import java.util.ArrayList;
027import java.util.List;
028
029import org.apache.uima.jcas.JCas;
030import org.apache.uima.jcas.tcas.Annotation;
031import org.cleartk.ml.Feature;
032import org.cleartk.ml.feature.extractor.NamedFeatureExtractor1;
033import org.cleartk.syntax.constituent.type.TreebankNode;
034import org.cleartk.syntax.constituent.type.TreebankNodeUtil;
035
036/**
037 * <br>
038 * Copyright (c) 2011, Regents of the University of Colorado <br>
039 * All rights reserved.
040 * 
041 * @author Steven Bethard
042 */
043public class SyntacticFirstChildOfGrandparentOfLeafExtractor<T extends Annotation> implements NamedFeatureExtractor1<T> {
044  
045  private String featureName = "FirstChildOfGrandparent";
046  
047  @Override
048  public String getFeatureName() {
049    return this.featureName;
050  }
051
052  @Override
053  public List<Feature> extract(JCas view, T focusAnnotation) {
054    List<Feature> features = new ArrayList<Feature>();
055    TreebankNode node = TreebankNodeUtil.selectMatchingLeaf(view, focusAnnotation);
056    if (node != null) {
057      node = node.getParent();
058    }
059    if (node != null) {
060      node = node.getParent();
061    }
062    if (node != null) {
063      node = node.getChildren(0);
064    }
065    if (node != null && node.getLeaf()) {
066      features.add(new Feature(this.featureName, node.getCoveredText()));
067    }
068    return features;
069  }
070}
