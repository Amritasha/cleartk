024package org.cleartk.ml.svmlight;
025
026import java.util.List;
027import java.util.Map;
028
029import org.cleartk.ml.CleartkProcessingException;
030import org.cleartk.ml.Feature;
031import org.cleartk.ml.encoder.features.FeaturesEncoder;
032import org.cleartk.ml.encoder.outcome.OutcomeEncoder;
033import org.cleartk.ml.jar.Classifier_ImplBase;
034import org.cleartk.ml.sigmoid.Sigmoid;
035import org.cleartk.ml.svmlight.model.SvmLightModel;
036import org.cleartk.ml.util.featurevector.FeatureVector;
037
038import com.google.common.collect.Maps;
039
040/**
041 * <br>
042 * Copyright (c) 2007-2008, Regents of the University of Colorado <br>
043 * All rights reserved.
044 */
045public class SvmLightStringOutcomeClassifier extends Classifier_ImplBase<FeatureVector, String, Integer> {
046
047  Map<Integer, SvmLightModel> models;
048
049  Map<Integer, Sigmoid> sigmoids;
050
051  public SvmLightStringOutcomeClassifier(
052      FeaturesEncoder<FeatureVector> featuresEncoder,
053      OutcomeEncoder<String, Integer> outcomeEncoder,
054      Map<Integer, SvmLightModel> models,
055      Map<Integer, Sigmoid> sigmoids) {
056    super(featuresEncoder, outcomeEncoder);
057    this.models = models;
058    this.sigmoids = sigmoids;
059  }
060
061  public String classify(List<Feature> features) throws CleartkProcessingException {
062    FeatureVector featureVector = this.featuresEncoder.encodeAll(features);
063
064    int maxScoredIndex = 0;
065    double maxScore = 0;
066    boolean first = true;
067    for (int i : models.keySet()) {
068      double score = score(featureVector, i);
069      if (first || score > maxScore) {
070        first = false;
071        maxScore = score;
072        maxScoredIndex = i;
073      }
074    }
075
076    return outcomeEncoder.decode(maxScoredIndex);
077  }
078
079  @Override
080  public Map<String, Double> score(List<Feature> features) throws CleartkProcessingException {
081    FeatureVector featureVector = this.featuresEncoder.encodeAll(features);
082
083    Map<String, Double> results = Maps.newHashMap();
084    for (int i : models.keySet()) {
085      double score = score(featureVector, i);
086      String name = outcomeEncoder.decode(i);
087
088      results.put(name, score);
089    }
090
091    return results;
092  }
093
094  private double score(FeatureVector fv, int i) {
095    return sigmoids.get(i).evaluate(models.get(i).evaluate(fv));
096  }
097}
