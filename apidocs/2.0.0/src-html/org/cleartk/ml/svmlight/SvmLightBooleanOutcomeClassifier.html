024package org.cleartk.ml.svmlight;
025
026import java.util.List;
027import java.util.Map;
028
029import org.cleartk.ml.CleartkProcessingException;
030import org.cleartk.ml.Feature;
031import org.cleartk.ml.encoder.features.FeaturesEncoder;
032import org.cleartk.ml.encoder.outcome.OutcomeEncoder;
033import org.cleartk.ml.sigmoid.Sigmoid;
034import org.cleartk.ml.svmlight.model.SvmLightModel;
035import org.cleartk.ml.util.featurevector.FeatureVector;
036
037import com.google.common.collect.Maps;
038
039/**
040 * <br>
041 * Copyright (c) 2007-2011, Regents of the University of Colorado <br>
042 * All rights reserved.
043 * 
044 * @author Steven Bethard
045 */
046public class SvmLightBooleanOutcomeClassifier extends SvmLightClassifier_ImplBase<Boolean, Boolean> {
047
048  protected Sigmoid sigmoid;
049
050  public SvmLightBooleanOutcomeClassifier(
051      FeaturesEncoder<FeatureVector> featuresEncoder,
052      OutcomeEncoder<Boolean, Boolean> outcomeEncoder,
053      SvmLightModel model,
054      Sigmoid sigmoid) {
055    super(featuresEncoder, outcomeEncoder, model);
056    this.sigmoid = sigmoid;
057  }
058
059  @Override
060  protected Boolean predictionToOutcome(double prediction) {
061    return this.sigmoid.evaluate(prediction) > 0.5;
062  }
063
064  @Override
065  public Map<Boolean, Double> score(List<Feature> features) throws CleartkProcessingException {
066    FeatureVector featureVector = featuresEncoder.encodeAll(features);
067
068    double prediction = sigmoid.evaluate(model.evaluate(featureVector));
069    Map<Boolean, Double> scores = Maps.newHashMap();
070    scores.put(true, prediction);
071    scores.put(false, 1 - prediction);
072    return scores;
073  }
074}
