024package org.cleartk.ml.svmlight.rank;
025
026import java.io.File;
027import java.io.IOException;
028import java.io.InputStream;
029import java.io.PrintStream;
030
031import org.apache.uima.UIMAFramework;
032import org.apache.uima.util.Level;
033import org.apache.uima.util.Logger;
034import org.cleartk.ml.svmlight.SvmLightClassifierBuilder_ImplBase;
035
036import com.google.common.base.Joiner;
037
038/**
039 * <br>
040 * Copyright (c) 2011, Regents of the University of Colorado <br>
041 * All rights reserved.
042 * 
043 * @author Lee Becker
044 * 
045 *         Trains an svm_light ranking model.
046 */
047public class SvmLightRankBuilder extends
048    SvmLightClassifierBuilder_ImplBase<SvmLightRank, Double, Double> {
049
050  static Logger logger = UIMAFramework.getLogger(SvmLightClassifierBuilder_ImplBase.class);
051
052  @Override
053  public void trainClassifier(File dir, String... args) throws Exception {
054    File trainingDataFile = getTrainingDataFile(dir);
055    this.trainClassifier(dir, trainingDataFile, args);
056  }
057
058  public void trainClassifier(File dir, File trainingDataFile, String... args) throws Exception {
059    String executable = "svm_rank_learn";
060    if (args.length > 0 && args[0].equals(COMMAND_ARGUMENT)) {
061      executable = args[1];
062      String[] tempArgs = new String[args.length - 2];
063      System.arraycopy(args, 2, tempArgs, 0, tempArgs.length);
064      args = tempArgs;
065    }
066
067    String[] command = new String[args.length + 3];
068    command[0] = executable;
069    System.arraycopy(args, 0, command, 1, args.length);
070    command[command.length - 2] = trainingDataFile.getPath();
071    command[command.length - 1] = trainingDataFile.getPath() + ".model";
072
073    logger.log(Level.FINE, "training svmlight rank using the following command: "
074        + Joiner.on(" ").join(command));
075    Process process = Runtime.getRuntime().exec(command);
076    process.getOutputStream().close();
077    output(process.getInputStream(), System.out);
078    output(process.getErrorStream(), System.err);
079    process.waitFor();
080  }
081
082  @Override
083  protected SvmLightRank newClassifier() {
084    return new SvmLightRank(this.featuresEncoder, this.outcomeEncoder, this.model);
085  }
086
087  private static void output(InputStream input, PrintStream output) throws IOException {
088    byte[] buffer = new byte[128];
089    int count = input.read(buffer);
090    while (count != -1) {
091      output.write(buffer, 0, count);
092      count = input.read(buffer);
093    }
094  }
095
096}
