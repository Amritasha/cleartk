024package org.cleartk.ml.baseline;
025
026import java.io.File;
027import java.util.HashMap;
028import java.util.Map;
029
030import com.google.common.base.Charsets;
031import com.google.common.io.Files;
032
033/**
034 * 
035 * <br>
036 * Copyright (c) 2011, Regents of the University of Colorado <br>
037 * All rights reserved.
038 * 
039 * @author Steven Bethard
040 */
041public abstract class MostFrequentValueClassifierBuilder<OUTCOME_TYPE> extends
042    SingleOutcomeClassifierBuilder<OUTCOME_TYPE> {
043
044  @Override
045  public void trainClassifier(File dir, String... args) throws Exception {
046    Map<OUTCOME_TYPE, Integer> counts = new HashMap<OUTCOME_TYPE, Integer>();
047    for (String outcomeString : Files.readLines(this.getTrainingDataFile(dir), Charsets.US_ASCII)) {
048      OUTCOME_TYPE outcome = this.parseOutcome(outcomeString);
049      Integer count = counts.get(outcome);
050      counts.put(outcome, count == null ? 0 : count + 1);
051    }
052    OUTCOME_TYPE maxOutcome = null;
053    int maxCount = Integer.MIN_VALUE;
054    for (OUTCOME_TYPE outcome : counts.keySet()) {
055      int count = counts.get(outcome);
056      if (count > maxCount) {
057        maxCount = count;
058        maxOutcome = outcome;
059      }
060    }
061    this.writeOutcome(dir, maxOutcome);
062  }
063}
