033package org.cleartk.ml.crfsuite;
034
035import java.io.File;
036import java.io.FileNotFoundException;
037import java.util.List;
038
039import org.cleartk.ml.encoder.features.BooleanEncoder;
040import org.cleartk.ml.encoder.features.NameNumber;
041import org.cleartk.ml.encoder.features.NameNumberFeaturesEncoder;
042import org.cleartk.ml.encoder.features.NumberEncoder;
043import org.cleartk.ml.encoder.features.StringEncoder;
044import org.cleartk.ml.encoder.outcome.StringToStringOutcomeEncoder;
045import org.cleartk.ml.jar.SequenceDataWriter_ImplBase;
046
047/**
048 * 
049 * 
050 * <br>
051 * Copyright (c) 2011-2012, Technische Universität Darmstadt <br>
052 * All rights reserved.
053 * 
054 * @author Martin Riedl
055 */
056public class CrfSuiteStringOutcomeDataWriter
057    extends
058    SequenceDataWriter_ImplBase<CrfSuiteStringOutcomeClassifierBuilder, List<NameNumber>, String, String> {
059
060  private String featureSeparator = "\t";
061
062  public CrfSuiteStringOutcomeDataWriter(File outputDirectory) throws FileNotFoundException {
063    super(outputDirectory);
064    NameNumberFeaturesEncoder fe = new NameNumberFeaturesEncoder(false, false);
065    fe.addEncoder(new NumberEncoder());
066    fe.addEncoder(new BooleanEncoder());
067    fe.addEncoder(new StringEncoder());
068    this.setFeaturesEncoder(fe);
069    this.setOutcomeEncoder(new StringToStringOutcomeEncoder());
070  }
071
072  @Override
073  public void writeEncoded(List<NameNumber> features, String outcome) {
074    this.trainingDataWriter.print(outcome);
075    for (NameNumber nameNumber : features) {
076      this.trainingDataWriter.print(featureSeparator);
077      this.trainingDataWriter.print(nameNumber.name);
078    }
079    this.trainingDataWriter.println();
080  }
081
082  @Override
083  public void writeEndSequence() {
084    this.trainingDataWriter.println();
085  }
086
087  @Override
088  protected CrfSuiteStringOutcomeClassifierBuilder newClassifierBuilder() {
089    return new CrfSuiteStringOutcomeClassifierBuilder();
090  }
091
092}
