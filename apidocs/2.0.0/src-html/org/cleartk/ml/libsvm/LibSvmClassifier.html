024package org.cleartk.ml.libsvm;
025
026import java.util.ArrayList;
027import java.util.List;
028import java.util.Map;
029
030import org.cleartk.ml.CleartkProcessingException;
031import org.cleartk.ml.Feature;
032import org.cleartk.ml.encoder.features.FeaturesEncoder;
033import org.cleartk.ml.encoder.outcome.OutcomeEncoder;
034import org.cleartk.ml.jar.Classifier_ImplBase;
035import org.cleartk.ml.util.featurevector.FeatureVector;
036
037import com.google.common.collect.Maps;
038
039/**

047 */
048public abstract class LibSvmClassifier<OUTCOME_TYPE, ENCODED_OUTCOME_TYPE> extends
049    Classifier_ImplBase<FeatureVector, OUTCOME_TYPE, ENCODED_OUTCOME_TYPE> {
050
051  public LibSvmClassifier(
052      FeaturesEncoder<FeatureVector> featuresEncoder,
053      OutcomeEncoder<OUTCOME_TYPE, ENCODED_OUTCOME_TYPE> outcomeEncoder,
054      libsvm.svm_model model) {
055    super(featuresEncoder, outcomeEncoder);
056    this.model = model;
057  }
058
059  public OUTCOME_TYPE classify(List<Feature> features) throws CleartkProcessingException {
060    FeatureVector featureVector = this.featuresEncoder.encodeAll(features);
061
062    ENCODED_OUTCOME_TYPE encodedOutcome = decodePrediction(libsvm.svm.svm_predict(
063        this.model,
064        convertToLIBSVM(featureVector)));
065
066    return outcomeEncoder.decode(encodedOutcome);
067  }
068
069  @Override
070  public Map<OUTCOME_TYPE, Double> score(List<Feature> features) throws CleartkProcessingException {
071    FeatureVector featureVector = this.featuresEncoder.encodeAll(features);
072    double[] decisionValues = new double[this.model.nr_class];
073    libsvm.svm.svm_predict_probability(this.model, convertToLIBSVM(featureVector), decisionValues);
074    Map<OUTCOME_TYPE, Double> results = Maps.newHashMap();
075    for (int i = 0; i < this.model.nr_class; ++i) {
076      int intLabel = this.model.label[i];
077      OUTCOME_TYPE outcome = this.outcomeEncoder.decode(this.decodePrediction(intLabel));
078      results.put(outcome, decisionValues[i]);
079    }
080    return results;
081  }
082
083  protected static libsvm.svm_node[] convertToLIBSVM(FeatureVector featureVector) {
084    List<libsvm.svm_node> nodes = new ArrayList<libsvm.svm_node>();
085
086    for (FeatureVector.Entry entry : featureVector) {
087      libsvm.svm_node node = new libsvm.svm_node();
088      node.index = entry.index;
089      node.value = entry.value;
090      nodes.add(node);
091    }
092
093    return nodes.toArray(new libsvm.svm_node[nodes.size()]);
094  }
095
096  protected abstract ENCODED_OUTCOME_TYPE decodePrediction(double prediction);
097
098  protected libsvm.svm_model model;
099}
