024package org.cleartk.ml.tksvmlight;
025
026import java.io.File;
027import java.io.IOException;
028import java.util.jar.JarInputStream;
029import java.util.jar.JarOutputStream;
030
031import org.apache.uima.UIMAFramework;
032import org.apache.uima.util.Level;
033import org.apache.uima.util.Logger;
034import org.cleartk.ml.jar.JarStreams;
035import org.cleartk.ml.tksvmlight.model.TreeKernelSvmModel;
036
037import com.google.common.annotations.Beta;
038import com.google.common.base.Joiner;
039import com.google.common.io.ByteStreams;
040
041/**
042 * A class that provided interfaces to train, package and unpackage a
043 * {@link TreeKernelSvmBooleanOutcomeClassifier} into a jar file.
044 * 
045 * <br>
046 * Copyright (c) 2007-2008, Regents of the University of Colorado <br>
047 * All rights reserved.
048 * 
049 * @author Daryl Lonnon
050 */
051@Beta
052public class TkSvmLightBooleanOutcomeClassifierBuilder
053    extends
054    TreeKernelSvmBooleanOutcomeClassifierBuilder<TreeKernelSvmBooleanOutcomeClassifier>{
055
056  static Logger logger = UIMAFramework.getLogger(TkSvmLightBooleanOutcomeClassifierBuilder.class);
057
058  public static String COMMAND_ARGUMENT = "--executable";
059
060  /**
061   * Train a SVMTK classifier. Assumes the executable name is "tk_svm_learn". Note: this is public
062   * so the One verse All TK SVMlight classifier may take advantage of it, casual users probably
063   * shouldn't use this interface.
064   * 
065   * @param args
066   *          The arguments to be used by the tk_svm_classify command. Note: -t 5 is used to specify
067   *          the use of Tree Kernels.
068   */
069  public static void train(String filePath, String[] args) throws Exception {
070    String executable = "tk_svm_learn";
071    if (args.length > 0 && args[0].equals(COMMAND_ARGUMENT)) {
072      executable = args[1];
073      String[] tempArgs = new String[args.length - 2];
074      System.arraycopy(args, 2, tempArgs, 0, tempArgs.length);
075      args = tempArgs;
076    }
077
078    String[] command = new String[args.length + 3];
079    command[0] = executable;
080    System.arraycopy(args, 0, command, 1, args.length);
081    command[command.length - 2] = new File(filePath).getPath();
082    command[command.length - 1] = new File(filePath + ".model").getPath();
083
084    logger.log(Level.FINE, "training with tree kernel svmlight using the following command: "
085        + Joiner.on(" ").join(command));
086    Process process = Runtime.getRuntime().exec(command);
087    process.getOutputStream().close();
088    ByteStreams.copy(process.getInputStream(), System.out);
089    ByteStreams.copy(process.getErrorStream(), System.err);
090    process.waitFor();
091  }
092
093  public File getTrainingDataFile(File dir) {
094    return new File(dir, "training-data.svmlight");
095  }
096
097  private File getModelFile(File dir) {
098    return new File(dir, "training-data.svmlight.model");
099  }
100
101  /**
102   * Train this SVMTK classifier.
103   * 
104   * @param dir
105   *          The directory where the training data has been written.
106   * @param args
107   *          The arguments to be used by the tk_svm_classify command. Note: -t 5 is used to specify
108   *          the use of Tree Kernels.
109   */
110  public void trainClassifier(File dir, String... args) throws Exception {
111    File trainingDataFile = (dir.isDirectory() ? getTrainingDataFile(dir) : dir);
112    train(trainingDataFile.getPath(), args);
113  }
114
115  /**
116   * package the classifier found in dir into the a Jar file.
117   */
118  @Override
119  protected void packageClassifier(File dir, JarOutputStream modelStream) throws IOException {
120    super.packageClassifier(dir, modelStream);
121    JarStreams.putNextJarEntry(modelStream, "model.svmlight", getModelFile(dir));
122  }
123
124  private TreeKernelSvmModel model;
125
126  /**
127   * unpackage the model files found in a JarInputStream.
128   */
129  @Override
130  protected void unpackageClassifier(JarInputStream modelStream) throws IOException {
131    super.unpackageClassifier(modelStream);
132    JarStreams.getNextJarEntry(modelStream, "model.svmlight");
133    model = TreeKernelSvmModel.fromInputStream(modelStream);
134  }
135
136  /**
137   * Create a TKSVMlightClassifier.
138   */
139  @Override
140  protected TreeKernelSvmBooleanOutcomeClassifier newClassifier() {
141    return new TreeKernelSvmBooleanOutcomeClassifier(
142        this.featuresEncoder,
143        this.outcomeEncoder,
144        this.model);
145  }
146}
