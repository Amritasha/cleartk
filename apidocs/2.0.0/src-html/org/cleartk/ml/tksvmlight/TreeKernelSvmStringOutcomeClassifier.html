024package org.cleartk.ml.tksvmlight;
025
026import java.util.List;
027import java.util.Map;
028
029import org.cleartk.ml.CleartkProcessingException;
030import org.cleartk.ml.Feature;
031import org.cleartk.ml.encoder.features.FeaturesEncoder;
032import org.cleartk.ml.encoder.outcome.OutcomeEncoder;
033import org.cleartk.ml.jar.Classifier_ImplBase;
034import org.cleartk.ml.tksvmlight.model.TreeKernelSvmModel;
035
036import com.google.common.annotations.Beta;
037import com.google.common.base.Functions;
038import com.google.common.collect.Maps;
039import com.google.common.collect.Ordering;
040
041/**
042 * <br>
043 * Copyright (c) 2007-2013, Regents of the University of Colorado <br>

046 * @author Daryl Lonnon

048 */
049@Beta
050public class TreeKernelSvmStringOutcomeClassifier extends
051    Classifier_ImplBase<TreeFeatureVector, String, Integer> {
052
053  Map<Integer, TreeKernelSvmModel> models;
054
055  /**
056   * Constructor
057   * 
058   * @param featuresEncoder
059   *          The features encoder used by this classifier.
060   * @param outcomeEncoder
061   *          The outcome encoder used by this classifier.
062   * @param models
063   *          The files for the models used by this classifier.
064   */
065  public TreeKernelSvmStringOutcomeClassifier(
066      FeaturesEncoder<TreeFeatureVector> featuresEncoder,
067      OutcomeEncoder<String, Integer> outcomeEncoder,
068      Map<Integer, TreeKernelSvmModel> models) {
069    super(featuresEncoder, outcomeEncoder);
070    this.models = models;
071  }
072
073  /**
074   * Classify a features list.
075   * 
076   * @param features
077   *          The feature list to classify.
078   * @return A String of the most likely classification.
079   */
080  public String classify(List<Feature> features) throws CleartkProcessingException {
081    Map<String, Double> scores = this.score(features);
082    Ordering<String> byScore = Ordering.natural().onResultOf(Functions.forMap(scores));
083    return byScore.max(scores.keySet());
084  }
085
086  /**
087   * Score a list of features against the various models used by the One Verse All SVM classifier.
088   * 
089   * @param features
090   *          The features to classify
091   * @return A map from String outcomes to likelihood.
092   */
093  @Override
094  public Map<String, Double> score(List<Feature> features)
095      throws CleartkProcessingException {
096    TreeFeatureVector featureVector = this.featuresEncoder.encodeAll(features);
097
098    Map<String, Double> results = Maps.newHashMap();
099    for (int i : models.keySet()) {
100      double score = predict(featureVector, i);
101      String name = outcomeEncoder.decode(i);
102
103      results.put(name, score);
104    }
105    return results;
106  }
107
108  private double predict(TreeFeatureVector featureVector, int i) {
109    return this.models.get(i).evaluate(featureVector);
110  }
111 }
