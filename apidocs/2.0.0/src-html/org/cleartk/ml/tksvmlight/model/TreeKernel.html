085  public TreeKernel(

100  public void initExponents(){
101    for(int i = 0; i < lambdaPowers.length; i++){
102      lambdaPowers[i] = Math.pow(this.lambda, i);
103    }
104  }

106  public double evaluate(TreeFeatureVector fv1, TreeFeatureVector fv2) {
107    double sim = 0.0;
108    if (sumMethod == ForestSumMethod.SEQUENTIAL) {
109      List<String> fv1Trees = new ArrayList<String>(fv1.getTrees().values());
110      List<String> fv2Trees = new ArrayList<String>(fv2.getTrees().values());
111      for (int i = 0; i < fv1Trees.size(); i++) {
112        String tree1Str = fv1Trees.get(i);
113        String tree2Str = fv2Trees.get(i);
114        if (kernelType == KernelType.SUBSET) {
115          sim += sst(tree1Str, tree2Str);
116        } else if (kernelType == KernelType.PARTIAL) {
117          sim += ptk(tree1Str, tree2Str);
118        } else {
119          throw new NotImplementedException("The only kernel types implemented are SST and PTK!");
120        }
121      }
122    } else {
123      throw new NotImplementedException("The only summation method implemented is Sequential!");
124    }
125    return sim;
126  }

128  private double sst(String tree1Str, String tree2Str) {
129
130    TopTreebankNode node1 = null;
131    if (!trees.containsKey(tree1Str)) {
132      node1 = TreebankFormatParser.parse(tree1Str);
133      trees.put(tree1Str, node1);
134    } else
135      node1 = trees.get(tree1Str);
136
137    TopTreebankNode node2 = null;
138    if (!trees.containsKey(tree2Str)) {
139      node2 = TreebankFormatParser.parse(tree2Str);
140      trees.put(tree2Str, node2);
141    } else
142      node2 = trees.get(tree2Str);
143
144    double norm1 = 0.0;
145    double norm2 = 0.0;
146    if (normalize) {
147      if (!normalizers.containsKey(tree1Str)) {
148        double norm = sim(node1, node1);
149        normalizers.put(tree1Str, norm);
150      }
151      if (!normalizers.containsKey(tree2Str)) {
152        double norm = sim(node2, node2);
153        normalizers.put(tree2Str, norm);
154      }
155
156      norm1 = normalizers.get(tree1Str);
157      norm2 = normalizers.get(tree2Str);
158    }
159    if (normalize) {
160      return (sim(node1, node2) / Math.sqrt(norm1 * norm2));
161    } else {
162      return sim(node1, node2);
163    }
164  }
165
166  private double sim(TreebankNode node1, TreebankNode node2) {
167    double sim = 0.0;
168    List<TreebankNode> N1 = getNodeList(node1);
169    List<TreebankNode> N2 = getNodeList(node2);
170    for (TreebankNode n1 : N1) {
171      for (TreebankNode n2 : N2) {
172        sim += numCommonSubtrees(n1, n2);
173      }
174    }
175    return sim;
176  }
177
178  private double numCommonSubtrees(TreebankNode n1, TreebankNode n2) {
179    double retVal = 1.0;
180    List<TreebankNode> children1 = n1.getChildren();
181    List<TreebankNode> children2 = n2.getChildren();
182    int c1size = children1.size();
183    int c2size = children2.size();
184    if (c1size != c2size) {
185      retVal = 0;
186    } else if (!n1.getType().equals(n2.getType())) {
187      retVal = 0;
188    } else if (n1.isLeaf() && n2.isLeaf()) {
189      // both are preterminals, and we know they have the same type, need to check value (word)
190      // Collins & Duffy tech report says lambdaSquared, but Nips 02 paper uses lambda
191      // Moschitti's papers also use lambda
192      // retVal = lambdaSquared;
193      if (n1.getValue().equals(n2.getValue())) {
194        retVal = lambda;
195      } else {
196        retVal = 0;
197      }
198    } else {
199      // At this point they have the same label and same # children. Check if children the same.
200      boolean sameProd = true;
201      for (int i = 0; i < c1size; i++) {
202        String l1 = children1.get(i).getType();
203        String l2 = children2.get(i).getType();
204        if (!l1.equals(l2)) {
205          sameProd = false;
206          break;
207        }
208      }
209      if (sameProd == true) {
210        for (int i = 0; i < c1size; i++) {
211          retVal *= (1 + numCommonSubtrees(children1.get(i), children2.get(i)));
212        }
213        // again, some disagreement in the literature, with Collins and Duffy saying
214        // lambdaSquared here in tech report and lambda here in nips 02. We'll stick with
215        // lambda b/c that's what moschitti's code (which was presumably used for model-building)
216        // uses.
217        retVal = lambda * retVal;
218      } else {
219        retVal = 0;
220      }
221    }
222    return retVal;
223  }

225  private double ptk(String tree1Str, String tree2Str){
226    SimpleDepTree node1 = null;
227    if (!depTrees.containsKey(tree1Str)) {
228      node1 = SimpleDepTree.fromString(tree1Str);
229      depTrees.put(tree1Str, node1);
230    } else
231      node1 = depTrees.get(tree1Str);
232
233    SimpleDepTree node2 = null;
234    if (!depTrees.containsKey(tree2Str)) {
235      node2 = SimpleDepTree.fromString(tree2Str);
236      depTrees.put(tree2Str, node2);
237    } else
238      node2 = depTrees.get(tree2Str);
239    
240    double norm1 = 0.0;
241    double norm2 = 0.0;
242    if (normalize) {
243      if (!normalizers.containsKey(tree1Str)) {
244        double norm = ptkSim(node1, node1);
245        normalizers.put(tree1Str, norm);
246      }
247      if (!normalizers.containsKey(tree2Str)) {
248        double norm = ptkSim(node2, node2);
249        normalizers.put(tree2Str, norm);
250      }
251
252      norm1 = normalizers.get(tree1Str);
253      norm2 = normalizers.get(tree2Str);
254      return (ptkSim(node1, node2) / Math.sqrt(norm1 * norm2));
255    } else {
256      return ptkSim(node1, node2);
257    }
258  }

260  private double ptkSim(SimpleDepTree t1, SimpleDepTree t2){
261    double sim = 0.0;
262    List<SimpleDepTree> t1Nodes = getDepNodeList(t1);
263    List<SimpleDepTree> t2Nodes = getDepNodeList(t2);
264    
265    for(int i = 0; i < t1Nodes.size(); i++){
266      SimpleDepTree t1Node = t1Nodes.get(i);
267      for(int j = 0; j < t2Nodes.size(); j++){
268        SimpleDepTree t2Node = t2Nodes.get(j);
269        double nodeSim = 0.0;
270        if(t1Node.cat.equals(t2Node.cat)){
271          if(t1Node.isLeaf()){  // && t2Node.isLeaf() is unnecessary since we already know they are the same category
272            if(t1Node.cat.equals(t2Node.cat)){
273              nodeSim += mu * lambdaSquared;
274            }
275          }else{
276            nodeSim = ptkDelta(t1Node, t2Node);            
277          }
278        }
279        sim += nodeSim;
280      }
281    }
282    return sim;
283  }
284  
285  private double ptkDelta(SimpleDepTree node1, SimpleDepTree node2){
286    double delta = 0.0;// lambdaSquared;
287    
288    if(!node1.cat.equals(node2.cat)) return 0.0;
289    
290    if(useCache && cache.containsKey(node1) && cache.get(node1).containsKey(node2)){
291//      cacheHits++;
292      return cache.get(node1).get(node2);
293    }
294    
295    int l1 = node1.children.size();
296    int l2 = node2.children.size();
297
298    delta = 1.0;
299    for(int p = 1; p <= Math.min(Math.min(l1,  l2), MAX_CHILDREN); p++){
300
301      double contrP = ptkDeltaP(node1.children, node2.children, p);
302      delta += contrP;
303    }
304    
305    double score = mu * lambdaSquared * delta;
306    
307    if(useCache){
308      if(!cache.containsKey(node1)){
309        cache.put(node1, new HashMap<SimpleDepTree,Double>());
310      }
311      cache.get(node1).put(node2, score);
312    }
313    
314    return score;
315  }
316
317  private double ptkDeltaP(List<SimpleDepTree> c1, List<SimpleDepTree> c2, int p){
318    return ptkDeltaP(c1, c2, c1.size()-1, c2.size()-1, p);
319  }
320  
321  private double ptkDeltaP(List<SimpleDepTree> c1, List<SimpleDepTree> c2, int s1len, int s2len, int p){
322    double delta = ptkDelta(c1.get(s1len), c2.get(s2len));
323    int exp;
324    double lambdaPow;
325    
326    for(int i = 0; i < s1len; i++){
327      for(int r = 0; r < s2len; r++){
328        // FYI -- sublist second argument is exclusive -- so sublist (i, i) gets you 0 elements,
329        // sublist(i, i+1) gets you 1 element.
330        exp = s1len-i+s2len-r;
331        if(exp < lambdaPowers.length-1){
332          lambdaPow = lambdaPowers[exp];
333        }else{
334          lambdaPow = Math.pow(lambda, exp);
335        }
336        delta +=  lambdaPow * //ptkDeltaP(c1.subList(0, i+1), c2.subList(0, r+1), p-1);
337                ptkDeltaP(c1, c2, i, r, p-1);
338      }
339    }
340    return delta;
341  }

343  private static final List<TreebankNode> getNodeList(TreebankNode tree) {
344    ArrayList<TreebankNode> list = new ArrayList<TreebankNode>();
345    list.add(tree);
346    for (int i = 0; i < list.size(); ++i) {
347      list.addAll(list.get(i).getChildren());
348    }
349    return list;
350  }
351  
352  private static final List<SimpleDepTree> getDepNodeList(SimpleDepTree tree){
353    ArrayList<SimpleDepTree> list = Lists.newArrayList();
354    list.add(tree);
355    for(int i = 0; i < list.size(); i++){
356      list.addAll(list.get(i).children);
357    }
358    return list;
359  }

370class SimpleDepTree {
371  public String cat;
372  public ArrayList<SimpleDepTree> children;
373  public SimpleDepTree parent = null;
374  static Pattern ptPatt = Pattern.compile("\\(([^ (]+) +([^ )]+)\\)");
375  static Pattern orphanPatt = Pattern.compile("\\(([^ (]+) \\)");
376  
377  public SimpleDepTree(String s){
378    this(s, null);
379  }
380  
381  public SimpleDepTree(String s, SimpleDepTree p){
382    cat = escapeCat(s);
383    children = new ArrayList<SimpleDepTree>();
384    parent = p;
385  }
386  
387  public static String escapeCat(String c) {
388    c = c.replaceAll("\\(", "LPAREN");
389    c = c.replaceAll("\\)", "RPAREN");
390    return c;
391  }
392
393  public void addChild(SimpleDepTree t){
394    children.add(t);
395  }
396  
397  private static String[] splitChildren(String s){
398    ArrayList<String> children = new ArrayList<String>();
399    char[] chars = s.toCharArray();
400    int numParens = 0;
401    int startIndex = 0;
402    for(int i = 0; i < chars.length; i++){
403      if(chars[i] == '('){
404        numParens++;
405        if(numParens == 1){
406          startIndex = i;
407        }
408      }else if(chars[i] == ')'){
409        numParens--;
410        if(numParens == 0){
411          children.add(s.substring(startIndex, i+1));
412        }else if(numParens < 0){
413          break;
414        }
415      }
416    }
417    return children.toArray(new String[]{});
418  }
419
420  static SimpleDepTree fromString(String string){
421    SimpleDepTree tree = null;
422    
423    // pre-terminal case is the base case:
424    Matcher pretermMatcher = ptPatt.matcher(string);
425    Matcher orphanMatcher = orphanPatt.matcher(string);
426    
427    if(pretermMatcher.matches()){
428      tree = new SimpleDepTree(pretermMatcher.group(1));
429      SimpleDepTree leaf = new SimpleDepTree(pretermMatcher.group(2));
430      tree.addChild(leaf);
431      leaf.parent = tree;
432    }else if(orphanMatcher.matches()){
433      tree = new SimpleDepTree(orphanMatcher.group(1));
434    }else{
435      int firstWS = string.indexOf(' ');
436      tree = new SimpleDepTree(string.substring(1, firstWS));
437      String[] childStrings = splitChildren(string.substring(firstWS+1, string.length()-1));
438      for(int i = 0; i < childStrings.length; i++){
439        SimpleDepTree child = fromString(childStrings[i]);
440        child.parent = tree;
441        tree.addChild(child);
442      }
443    }
444    return tree;
445  }
446  
447  public final boolean isLeaf(){
448    return children.size() == 0;
449  }
