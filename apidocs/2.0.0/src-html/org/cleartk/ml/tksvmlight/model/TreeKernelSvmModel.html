058@Beta
059public class TreeKernelSvmModel {
060  String version;
061
062  CompositeKernel kernel = null;
063
064  double bias;
065
066  TkSupportVector[] supportVectors = null;
067
068  public double evaluate(TreeFeatureVector fv) {
069    double result = -bias;
070
071    for (TkSupportVector sv : supportVectors) {
072      result += sv.getAlpha() * kernel.evaluate(fv, sv.tfv);
073    }
074
075    return result;
076  }
077
078  public static TreeKernelSvmModel fromFile(File modelFile) throws IOException {
079    InputStream modelStream = new FileInputStream(modelFile);
080    TreeKernelSvmModel model = fromInputStream(modelStream);
081    modelStream.close();
082    return model;
083  }
084
085  public static TreeKernelSvmModel fromInputStream(InputStream modelStream) throws IOException {
086    TreeKernelSvmModel model = new TreeKernelSvmModel();
087    TkSvmLightReader in = new TkSvmLightReader(modelStream);
088    Kernel featKernel = null;
089    TreeKernel treeKernel = null;
090    String buffer;
091
092    // Model File Version
093    buffer = in.readLine();
094    model.version = buffer;
095
096    // kernel type
097    int ktype = Integer.parseInt(in.readLine());
098    // kernel parameter -d (parameter degree in polynomial kernel)
099    int kpar_degree = Integer.parseInt(in.readLine());
100    // kernel parameter -g (parameter gamma in rbf kernel)
101    double kpar_gamma = Double.parseDouble(in.readLine());
102    // kernel parameter -s (parameter s in sigmoid/poly kernel)
103    double kpar_s = Double.parseDouble(in.readLine());
104    // kernel parameter -r (parameter c in sigmoid/poly kernel)
105    double kpar_c = Double.parseDouble(in.readLine());
106    // kernel parameter -u (parameter of user defined kernel)
107    in.readLine();
108    // kernel parameter -L (decay factor in tree kernel)
109    double lambda = Double.parseDouble(in.readLine());
110    // kernel parameter -T (multiplicative constant for the contribution of tree kernels when -C =
111    // '+')
112    double multiplier = Double.parseDouble(in.readLine());
113    // kernel parameter -C (combine kernel types: {+, *, T, V} for add, mult, tree only, vector
114    // only)
115    String comboOperator = in.readLine();
116    // kernel parameter -F (???)
117    in.readLine();
118    // kernel parameter -S (kernel to be used with vectors)
119    int fkType = Integer.parseInt(in.readLine());
120    // kernel parameter -D (0, SubTree kernel or 1, SubSet Tree kernels)
121    int tkType = (int) Double.parseDouble(in.readLine());
122    // kernel parameter -N (0 = no normalization, 1 = tree, 2 = vector and 3 = both)
123    int norm = Integer.parseInt(in.readLine());
124    // kernel parameter -V (kernel applied to 'S' (sequence of trees), or 'A' (all tree pairs))
125    in.readLine();
126    // kernel parameter -W (kernel applied to 'S' (sequence of trees), or 'A' (all tree pairs))
127    String treeSeqMethod = in.readLine();
128
129    for (int kCode : new int[] { ktype, fkType }) {
130      if (featKernel != null) {
131        // no need to look at the fkType if the ktype is present - not sure what that would contain
132        break;
133      }
134      switch (kCode) {
135        case 0:
136          // linear kernel
137          featKernel = new LinearKernel();
138          break;
139        case 1:
140          // polynomial kernel
141          featKernel = new PolynomialKernel(kpar_s, kpar_c, kpar_degree);
142          break;
143        case 2:
144          // rbf kernel
145          featKernel = new RbfKernel(kpar_gamma);
146          break;
147        case 3:
148          // sigmoid kernel
149          featKernel = new SigmoidKernel(kpar_s, kpar_c);
150          break;
151        case 5:
152          break;
153        default:
154          throw new UnsupportedKernelError();
155      }
156    }
157
158    ComboOperator operator = null;
159    // check for code for TK
160    if (ktype == 5) {
161      ForestSumMethod sumMethod;
162      if (treeSeqMethod.equals("S")) {
163        sumMethod = ForestSumMethod.SEQUENTIAL;
164      } else {
165        sumMethod = ForestSumMethod.ALL_PAIRS;
166      }
167
168      TreeKernel.KernelType kernelType;
169      switch (tkType) {
170        case 0:
171          kernelType = KernelType.SUBTREE;
172          break;
173        case 1:
174          kernelType = KernelType.SUBSET;
175          break;
176        case 2:
177          kernelType = KernelType.SUBSET_BOW;
178          break;
179        case 3:
180          kernelType = KernelType.PARTIAL;
181          break;
182        default:
183          throw new UnsupportedKernelError();
184      }
185      boolean normalize = norm == 1 || norm == 3;
186      treeKernel = new TreeKernel(lambda, sumMethod, kernelType, normalize);
187      if (comboOperator.equals("+")) {
188        operator = ComboOperator.SUM;
189      } else if (comboOperator.equals("*")) {
190        operator = ComboOperator.PRODUCT;
191      } else if (comboOperator.equals("T")) {
192        operator = ComboOperator.TREE_ONLY;
193      } else if (comboOperator.equals("V")) {
194        operator = ComboOperator.VECTOR_ONLY;
195      } else {
196        throw new UnsupportedKernelError();
197      }
198    } else {
199      operator = ComboOperator.VECTOR_ONLY;
200    }
201
202    Normalize normal = Normalize.BOTH;
203    switch (norm) {
204      case 0:
205        normal = Normalize.NEITHER;
206        break;
207      case 1:
208        normal = Normalize.TREE;
209        break;
210      case 2:
211        normal = Normalize.VECTOR;
212        break;
213      case 3:
214        normal = Normalize.BOTH;
215        break;
216      default:
217        throw new UnsupportedKernelError();
218    }
219    model.kernel = new CompositeKernel(featKernel, treeKernel, operator, multiplier, normal);
220
221    in.readLine(); // feature index
222    in.readLine(); // num documents
223    int numSVs = Integer.valueOf(in.readLine()) - 1; // number of support vectors
224    model.bias = Double.valueOf(in.readLine());
225
226    in.inSVs = true;
227    // read support vectors
228    model.supportVectors = new TkSupportVector[numSVs];
229    for (int i = 0; i < numSVs; i++) {
230      model.supportVectors[i] = readSV(in);
231    }
232
233    return model;
234  }

236  private static TkSupportVector readSV(TkSvmLightReader in) throws IOException {
237    String line = in.readLine();
238    int firstWhitespace = line.indexOf(' ');
239    double alpha_y = Double.parseDouble(line.substring(0, firstWhitespace));
240    String treesAndVecs = line.substring(firstWhitespace).trim();
241
242    String treeSect = null;
243    String vectSect = null;
244
245    int etInd = treesAndVecs.indexOf("|ET|");
246    if (etInd >= 0) {
247      // there are at least trees here.
248      treeSect = treesAndVecs.substring(0, etInd - 1).trim();
249      if (treesAndVecs.substring(etInd).contains("|EV|")) {
250        vectSect = treesAndVecs.substring(etInd + 4); // skip over symbol
251      }
252    } else {
253      vectSect = treesAndVecs;
254    }
255
256    LinkedHashMap<String, String> trees = new LinkedHashMap<String, String>();
257    if (treeSect != null) {
258      String[] subtrees = treeSect.substring(5).split("\\|BT\\|"); // skip over "|BT| " at start
259      for (int i = 0; i < subtrees.length; i++) {
260        trees.put("TK_" + i, subtrees[i].replaceAll("\\((\\S+)\\)", "$1"));
261      }
262    }
263
264    FeatureVector fv = new SparseFeatureVector();
265
266    if (vectSect != null) {
267      String[] features = vectSect.trim().split(" +");
268      for (int i = 0; i < features.length - 1; i++) {
269        String[] parts = features[i].split(":");
270        int featureIndex = Integer.valueOf(parts[0]);
271        double featureValue = Double.valueOf(parts[1]);
272        try {
273          fv.set(featureIndex, featureValue);
274        } catch (InvalidFeatureVectorValueException e) {
275          throw new IOException(e);
276        }
277      }
278    }
279
280    TreeFeatureVector tfv = new TreeFeatureVector();
281    tfv.setFeatures(fv);
282    tfv.setTrees(trees);
283    return new TkSupportVector(alpha_y, tfv);
284  }
285
286  private static class TkSvmLightReader {
287    BufferedReader reader;
288
289    boolean inSVs = false;
290
291    TkSvmLightReader(InputStream modelStream) {
292      this.reader = new BufferedReader(new InputStreamReader(modelStream));
293    }
294
295    String readLine() throws IOException {
296      String line = reader.readLine().trim();
297      // need a special case here because there can be "comment" symbols (#) in the
298      // tree as part of the sentence, don't want to burden upstream processing by
299      // requiring escaping in svmlight-tk code.
300      if (inSVs) {
301        return line;
302      } else {
303        return line.split("#")[0].trim();
304      }
305    }
306  }
