024package org.cleartk.ml.util.featurevector;
025
026import java.util.Map;
027import java.util.SortedMap;
028import java.util.TreeMap;
029
030/**
031 * <br>
032 * Copyright (c) 2007-2008, Regents of the University of Colorado <br>
033 * All rights reserved.
034 * 
035 * 
036 * 
037 * @author Philipp Wetzler
038 * 
039 */
040public class SparseFeatureVector extends FeatureVector {
041
042  SortedMap<Integer, Double> values;
043
044  public SparseFeatureVector() {
045    values = new TreeMap<Integer, Double>();
046  }
047
048  public SparseFeatureVector(FeatureVector fv) throws InvalidFeatureVectorValueException {
049    this();
050    for (FeatureVector.Entry entry : fv) {
051      this.set(entry.index, entry.value);
052    }
053  }
054
055  public double get(int index) {
056    Double returnValue = this.values.get(index);
057
058    if (returnValue != null)
059      return returnValue;
060    else
061      return 0.0;
062  }
063
064  public java.util.Iterator<Entry> iterator() {
065    return new Iterator(this.values);
066  }
067
068  public void set(int index, double value) throws InvalidFeatureVectorValueException {
069    if (Double.isInfinite(value) || Double.isNaN(value))
070      throw new InvalidFeatureVectorValueException(index, value);
071
072    if (value != 0.0)
073      this.values.put(index, value);
074    else {
075      if (this.values.containsKey(index))
076        this.values.remove(index);
077    }
078  }
079
080  class Iterator implements java.util.Iterator<FeatureVector.Entry> {
081
082    java.util.Iterator<Map.Entry<Integer, Double>> subIterator;
083
084    SortedMap<Integer, Double> itValues;
085
086    Iterator(SortedMap<Integer, Double> values) {
087      this.itValues = values;
088      this.subIterator = values.entrySet().iterator();
089    }
090
091    public boolean hasNext() {
092      return this.subIterator.hasNext();
093    }
094
095    public FeatureVector.Entry next() {
096      Map.Entry<Integer, Double> nextEntry = this.subIterator.next();
097
098      int key = nextEntry.getKey();
099      double value = nextEntry.getValue();
100
101      return new FeatureVector.Entry(key, value);
102    }
103
104    public void remove() {
105      throw new UnsupportedOperationException();
106    }
107
108  }
109
110  public double innerProduct(FeatureVector other) {
111    double result = 0.0;
112
113    for (FeatureVector.Entry entry : this) {
114      result += entry.value * other.get(entry.index);
115    }
116
117    return result;
118  }
119}
