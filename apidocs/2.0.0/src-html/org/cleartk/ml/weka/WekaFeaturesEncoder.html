019package org.cleartk.ml.weka;

021import java.io.File;
022import java.util.ArrayList;
023import java.util.HashMap;
024import java.util.Map;
025
026import org.cleartk.ml.Feature;
027import org.cleartk.ml.encoder.features.FeaturesEncoder;
028
029import com.google.common.annotations.Beta;
030
031import weka.core.Attribute;
032import weka.core.Utils;
033
034/**
035 * Copyright (c) 2012, Regents of the University of Colorado <br>
036 * All rights reserved.

038 * @author Philip Ogren
039 * 
040 */
041@Beta
042public class WekaFeaturesEncoder implements FeaturesEncoder<Iterable<Feature>> {
043
044  private static final long serialVersionUID = 1L;
045
046  private ArrayList<Attribute> attributes;
047
048  private Map<String, Attribute> attributeMap;
049
050  public WekaFeaturesEncoder() {
051    attributes = new ArrayList<Attribute>();
052    attributeMap = new HashMap<String, Attribute>();
053  }
054
055  public Iterable<Feature> encodeAll(Iterable<Feature> features) {
056    for (Feature feature : features) {
057      featureToAttribute(feature);
058    }
059    return features;
060  }
061
062  /**
063   * @param feature
064   * @return
065   */
066  private Attribute featureToAttribute(Feature feature) {
067    String name = feature.getName();
068    Attribute attribute = attributeMap.get(name);
069    if (attribute == null) {
070      attribute = featureToAttribute(feature, attributes.size());
071      attributes.add(attribute);
072      attributeMap.put(name, attribute);
073    }
074    return attribute;
075  }
076
077  public static Attribute featureToAttribute(Feature feature, int attributeIndex) {
078    String name = Utils.quote(feature.getName());
079    Object value = feature.getValue();
080    Attribute attribute;
081    // if value is a number then create a numeric attribute
082    if (value instanceof Number) {
083      attribute = new Attribute(name);
084    }// if value is a boolean then create a numeric attribute
085    else if (value instanceof Boolean) {
086      attribute = new Attribute(name);
087    }
088    // if value is an Enum thene create a nominal attribute
089    else if (value instanceof Enum) {
090      Object[] enumConstants = value.getClass().getEnumConstants();
091      ArrayList<String> attributeValues = new ArrayList<String>(enumConstants.length);
092      for (Object enumConstant : enumConstants) {
093        attributeValues.add(enumConstant.toString());
094      }
095      attribute = new Attribute(name, attributeValues);
096    }
097    // if value is not a number, boolean, or enum, then we will create a
098    // string attribute
099    else {
100      attribute = new Attribute(name, (ArrayList<String>) null);
101    }
102    return attribute;
103  }
104
105  public void finalizeFeatureSet(File outputDirectory) {
106  }
107
108  public ArrayList<Attribute> getWekaAttributes() {
109    return attributes;
110  }
111
112  public Map<String, Attribute> getWekaAttributeMap() {
113    return attributeMap;
114  }
115
116}
