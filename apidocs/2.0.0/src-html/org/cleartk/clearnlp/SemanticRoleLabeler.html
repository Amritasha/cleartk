024package org.cleartk.clearnlp;
025
026
027import org.apache.uima.analysis_engine.AnalysisEngineDescription;
028import org.apache.uima.fit.descriptor.TypeCapability;
029import org.apache.uima.fit.factory.AnalysisEngineFactory;
030import org.apache.uima.resource.ResourceInitializationException;
031import org.cleartk.srl.type.Predicate;
032import org.cleartk.srl.type.SemanticArgument;
033import org.cleartk.syntax.dependency.type.DependencyNode;
034import org.cleartk.syntax.dependency.type.DependencyRelation;
035import org.cleartk.syntax.dependency.type.TopDependencyNode;
036import org.cleartk.token.type.Sentence;
037import org.cleartk.token.type.Token;
038
039/**
040 * <br>
041 * Copyright (c) 2012, Regents of the University of Colorado <br>
042 * All rights reserved.
043 * <p>
044 * This class provides a UIMA/ClearTK wrapper for the ClearNLP semantic role labeler. A typical
045 * pipeline preceding this analysis engine would consist of a tokenizer, sentence segmenter,
046 * POS tagger, lemmatizer (mp analyzer), and dependency parser.
047 * <p>
048 * The ClearNLP labeler is available here:
049 * <p>
050 * http://clearnlp.googlecode.com
051 * <p>
052* 
053 * @author Lee Becker
054 * 
055 */
056@TypeCapability(
057    inputs = { 
058        "org.cleartk.token.type.Sentence", 
059        "org.cleartk.token.type.Token:pos", 
060        "org.cleartk.token.type.Token:lemma",
061        "org.cleartk.syntax.dependency.type.TopDependencyNode",
062        "org.cleartk.syntax.dependency.type.DependencyNode", 
063        "org.cleartk.syntax.dependency.type.DependencyNode"},
064    outputs = {
065        "org.cleartk.srl.type.Predicate",
066        "org.cleartk.srl.type.SemanticArgument"} )
067  
068public class SemanticRoleLabeler extends SemanticRoleLabeler_ImplBase<Sentence, Token, DependencyNode, TopDependencyNode, DependencyRelation, SemanticArgument, Predicate> {
069  
070  public static AnalysisEngineDescription getDescription() throws ResourceInitializationException {
071    return AnalysisEngineFactory.createEngineDescription(SemanticRoleLabeler.class);
072  }
073  
074  public static AnalysisEngineDescription getDescription(String languageCode, String predIdModelPath, String rolesetModelPath, String srlModelPath) throws ResourceInitializationException {
075    return AnalysisEngineFactory.createEngineDescription(SemanticRoleLabeler.class, 
076      SemanticRoleLabeler_ImplBase.PARAM_LANGUAGE_CODE,
077      languageCode,
078      SemanticRoleLabeler_ImplBase.PARAM_PRED_ID_MODEL_PATH,
079      predIdModelPath,
080      SemanticRoleLabeler_ImplBase.PARAM_ROLESET_MODEL_PATH,
081      rolesetModelPath,
082      SemanticRoleLabeler_ImplBase.PARAM_SRL_MODEL_PATH,
083      srlModelPath);
084  }
085
086   public SemanticRoleLabeler() {
087    super(new CleartkTokenOps(), new CleartkDependencyOps(), new CleartkSemanticRoleOps());
088  }
089}
