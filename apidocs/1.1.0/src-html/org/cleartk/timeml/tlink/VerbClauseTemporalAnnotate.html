031    import org.cleartk.syntax.opennlp.PosTaggerAnnotator;
032    import org.cleartk.syntax.opennlp.SentenceAnnotator;
033    import org.cleartk.timeml.corpus.TimeMLWriter;
034    import org.cleartk.token.stem.snowball.DefaultSnowballStemmer;
035    import org.cleartk.token.tokenizer.TokenAnnotator;
036    import org.cleartk.util.cr.FilesCollectionReader;
037    import org.uimafit.factory.UimaContextFactory;
038    import org.uimafit.pipeline.SimplePipeline;
039    
040    /**
041     * <br>

043     * All rights reserved.
044     * 
045     * @author Steven Bethard
046     */
047    public class VerbClauseTemporalAnnotate {
048    
049      private static void error(String message) throws Exception {
050        Logger logger = UimaContextFactory.createUimaContext().getLogger();
051        logger.log(Level.SEVERE, String.format("%s\nusage: "
052            + "VerbClauseTemporalAnnotate input-file-or-dir [output-dir]", message));
053        System.exit(1);
054      }
055    
056      public static void main(String[] args) throws Exception {
057        // check arguments
058        if (args.length != 1 && args.length != 2) {
059          error("wrong number of arguments");
060        } else if (!new File(args[0]).exists()) {
061          error("file or directory not found: " + args[0]);
062        }
063    
064        // parse arguments
065        String inputFileOrDir = args[0];
066        File outputDir;
067        if (args.length == 2) {
068          outputDir = new File(args[1]);
069        } else {
070          outputDir = new File(".");
071        }
072        if (!outputDir.exists()) {
073          outputDir.mkdirs();
074        }
075    
076        // run the components on the selected documents
077        SimplePipeline.runPipeline(
078            FilesCollectionReader.getCollectionReader(inputFileOrDir),
079            SentenceAnnotator.getDescription(),
080            TokenAnnotator.getDescription(),
081            PosTaggerAnnotator.getDescription(),
082            DefaultSnowballStemmer.getDescription("English"),
083            ParserAnnotator.getDescription(),
084            VerbClauseTemporalAnnotator.FACTORY.getAnnotatorDescription(),
085            TimeMLWriter.getDescription(outputDir.getPath()));
086      }
087    }
