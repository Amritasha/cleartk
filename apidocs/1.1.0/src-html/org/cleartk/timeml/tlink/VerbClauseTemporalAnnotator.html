133      public VerbClauseTemporalAnnotator() {
134        this.eventID = 1;
135    
136        SimpleFeatureExtractor precedingAuxiliaries = new ContextExtractor<Token>(
137            Token.class,
138            new TokenTextForSelectedPOSExtractor("MD", "TO", "IN", "VB", "RB"),
139            new Preceding(3));
140        SimpleFeatureExtractor tokenStemExtractor = new TypePathExtractor(Token.class, "stem");
141        SimpleFeatureExtractor tokenPOSExtractor = new TypePathExtractor(Token.class, "pos");
142    
143        this.sourceFeatureExtractors = new ArrayList<SimpleFeatureExtractor>();
144        this.sourceFeatureExtractors.add(new NamingExtractor("Source", new SpannedTextExtractor()));
145        this.sourceFeatureExtractors.add(new NamingExtractor("Source", tokenPOSExtractor));
146        this.sourceFeatureExtractors.add(new NamingExtractor("Source", tokenStemExtractor));
147        this.sourceFeatureExtractors.add(new NamingExtractor("Source", precedingAuxiliaries));
148    
149        this.targetFeatureExtractors = new ArrayList<SimpleFeatureExtractor>();
150        this.targetFeatureExtractors.add(new NamingExtractor("Target", new SpannedTextExtractor()));
151        this.targetFeatureExtractors.add(new NamingExtractor("Target", tokenPOSExtractor));
152        this.targetFeatureExtractors.add(new NamingExtractor("Target", tokenStemExtractor));
153        this.targetFeatureExtractors.add(new NamingExtractor("Target", precedingAuxiliaries));
154    
155        this.betweenAnchorsFeatureExtractors = new ArrayList<SimpleFeatureExtractor>();
156        this.betweenAnchorsFeatureExtractors.add(new NamingExtractor("WordsBetween", new BagExtractor(
157            Token.class,
158            new SpannedTextExtractor())));
159        this.pathExtractor = new TargetPathExtractor();
160      }

162      public void process(JCas jCas) throws AnalysisEngineProcessException {
163        int docEnd = jCas.getDocumentText().length();
164    
165        // collect TLINKs if necessary
166        Map<String, TemporalLink> tlinks = null;
167        if (this.isTraining()) {
168          tlinks = this.getTemporalLinks(jCas);
169        }
170    
171        // look for verb-clause pairs in each sentence in the document
172        for (Sentence sentence : JCasUtil.select(jCas, Sentence.class)) {
173          TopTreebankNode tree = AnnotationUtil.selectFirstMatching(
174              jCas,
175              TopTreebankNode.class,
176              sentence);
177          if (tree == null) {
178            String fmt = "missing syntactic parse for sentence: %s";
179            String msg = String.format(fmt, sentence.getCoveredText());
180            this.getContext().getLogger().log(Level.WARNING, msg);
181            continue;
182          }
183    
184          // iterate over all verb-clause pairs
185          List<TreebankNodeLink> links = new ArrayList<TreebankNodeLink>();
186          this.collectVerbClausePairs(tree, links);
187          for (TreebankNodeLink link : links) {
188    
189            Token sourceToken = JCasUtil.selectCovered(jCas, Token.class, link.source).get(0);
190            Token targetToken = JCasUtil.selectCovered(jCas, Token.class, link.target).get(0);
191            int firstEnd = Math.min(sourceToken.getEnd(), targetToken.getEnd());
192            int lastBegin = Math.max(sourceToken.getBegin(), targetToken.getBegin());
193    
194            // create an instance and populate it with features
195            Instance<String> instance = new Instance<String>();
196            for (SimpleFeatureExtractor extractor : this.sourceFeatureExtractors) {
197              instance.addAll(extractor.extract(jCas, sourceToken));
198            }
199            for (SimpleFeatureExtractor extractor : this.targetFeatureExtractors) {
200              instance.addAll(extractor.extract(jCas, targetToken));
201            }
202            Annotation windowAnnotation = new Annotation(jCas, firstEnd, lastBegin);
203            for (SimpleFeatureExtractor extractor : this.betweenAnchorsFeatureExtractors) {
204              instance.addAll(extractor.extract(jCas, windowAnnotation));
205            }
206            instance.addAll(this.pathExtractor.extract(jCas, link.source, link.target));
207    
208            // find source and target anchors if they're available
209            Anchor source = AnnotationUtil.selectFirstMatching(jCas, Anchor.class, link.source);
210            Anchor target = AnnotationUtil.selectFirstMatching(jCas, Anchor.class, link.target);
211    
212            // if we're building training data, get the relation type from a
213            // TLINK
214            if (this.isTraining()) {
215              if (source != null && target != null) {
216                String key = String.format("%s:%s", source.getId(), target.getId());
217                TemporalLink tlink = tlinks.remove(key);
218                if (tlink != null) {
219                  instance.setOutcome(tlink.getRelationType());
220                  this.dataWriter.write(instance);
221                }
222              }
223            }
224    
225            // if we're classifying create new TLINKs from the
226            // classification outcomes
227            else {
228              source = this.getOrCreateEvent(jCas, source, link.source);
229              target = this.getOrCreateEvent(jCas, target, link.target);
230              // only create TLINKs for events that exist (or were created, if requested)
231              if (source != null && target != null) {
232                String relationType = this.classifier.classify(instance.getFeatures());
233                TemporalLink tlink = new TemporalLink(jCas, docEnd, docEnd);
234                tlink.setSource(source);
235                tlink.setTarget(target);
236                tlink.setRelationType(relationType);
237                tlink.addToIndexes();
238              }
239            }
240          }
241        }
242      }
243    
244      private Event getOrCreateEvent(JCas jCas, Anchor anchor, TreebankNode node) {
245        if (anchor != null && anchor instanceof Event) {
246          return (Event) anchor;
247        } else if (this.createEvents) {
248          Event event = new Event(jCas, node.getBegin(), node.getEnd());
249          event.setId("e" + this.eventID);
250          this.eventID++;
251          event.addToIndexes();
252          return event;
253        } else {
254          return null;
255        }
256      }
257    
258      private Map<String, TemporalLink> getTemporalLinks(JCas jCas) {
259        Map<String, TemporalLink> tlinks = new HashMap<String, TemporalLink>();
260        for (TemporalLink tlink : JCasUtil.select(jCas, TemporalLink.class)) {
261          String sourceID = tlink.getSource().getId();
262          String targetID = tlink.getTarget().getId();
263          String key = String.format("%s:%s", sourceID, targetID);
264          tlinks.put(key, tlink);
265        }
266        return tlinks;
267      }

269      private void collectVerbClausePairs(TreebankNode node, List<TreebankNodeLink> links) {
270        if (this.isVerbPhrase(node)) {
271          List<TreebankNode> sources = new ArrayList<TreebankNode>();
272          List<TreebankNode> targets = new ArrayList<TreebankNode>();
273          this.collectHeads(node, sources);
274    
275          // look for clauses in descendants
276          for (int i = 0; i < node.getChildren().size(); i++) {
277            TreebankNode child = node.getChildren(i);
278            if (this.isClause(child)) {
279    
280              // pair the verb phrase heads with the clause heads
281              targets.clear();
282              this.collectHeads(child, targets);
283              for (TreebankNode source : sources) {
284                for (TreebankNode target : targets) {
285    
286                  // skip pairs where the head of the VP is inside the
287                  // clause
288                  if (!this.contains(child, source)) {
289                    links.add(new TreebankNodeLink(source, target));
290                  }
291                }
292              }
293            }
294          }
295        }
296        // look for verb phrases in descendants
297        for (int i = 0; i < node.getChildren().size(); i++) {
298          TreebankNode child = node.getChildren(i);
299          this.collectVerbClausePairs(child, links);
300        }
301      }
302    
303      private void collectHeads(TreebankNode node, List<TreebankNode> heads) {
304        if (node.getLeaf()) {
305          heads.add(node);
306        }
307        String[] headTypes = VerbClauseTemporalAnnotator.headMap.get(node.getNodeType());
308        if (headTypes != null) {
309          for (String headType : headTypes) {
310            boolean foundChildWithHeadType = false;
311            for (int i = 0; i < node.getChildren().size(); i++) {
312              TreebankNode child = node.getChildren(i);
313              if (child.getNodeType().equals(headType)) {
314                String text = child.getCoveredText();
315                if (!VerbClauseTemporalAnnotator.stopWords.contains(text)) {
316                  this.collectHeads(child, heads);
317                  foundChildWithHeadType = true;
318                }
319              }
320            }
321            if (foundChildWithHeadType) {
322              break;
323            }
324          }
325        }
326      }
327    
328      private boolean contains(TreebankNode node, TreebankNode descendant) {
329        if (node == descendant) {
330          return true;
331        }
332        for (int i = 0; i < node.getChildren().size(); i++) {
333          boolean result = this.contains(node.getChildren(i), descendant);
334          if (result) {
335            return true;
336          }
337        }
338        return false;
339      }
340    
341      private boolean isVerbPhrase(TreebankNode node) {
342        return node.getNodeType().startsWith("VP");
343      }
344    
345      private boolean isClause(TreebankNode node) {
346        return node.getNodeType().startsWith("S");
347      }
348    
349      private class TreebankNodeLink {
350        public TreebankNode source;
351    
352        public TreebankNode target;
353    
354        public TreebankNodeLink(TreebankNode source, TreebankNode target) {
355          this.source = source;
356          this.target = target;
357        }
358      }
