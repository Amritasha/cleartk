024    package org.cleartk.timeml.eval;
025    
026    import java.io.File;
027    import java.util.Arrays;
028    
029    import org.cleartk.eval.provider.AnnotationEvaluator;
030    import org.cleartk.eval.provider.BatchBasedEvaluationPipelineProvider;
031    import org.cleartk.eval.provider.CleartkPipelineProvider;
032    import org.cleartk.eval.provider.EvaluationPipelineProvider;
033    import org.cleartk.syntax.opennlp.PosTaggerAnnotator;
034    import org.cleartk.timeml.corpus.TempEval2010GoldAnnotator;
035    import org.cleartk.timeml.corpus.TempEval2010Writer;
036    import org.cleartk.timeml.time.TimeTypeAnnotator;
037    import org.cleartk.timeml.type.Time;
038    import org.cleartk.token.stem.snowball.DefaultSnowballStemmer;
039    import org.uimafit.factory.AnalysisEngineFactory;
040    
041    /**
042     * TempEval 2010 task A: time attributes
043     * 
044     * Best reported accuracies in TempEval 2010:
045     * <ul>
046     * <li>type: 0.98</li>
047     * <li>value: 0.85</li>
048     * </ul>
049     * 
050     * <br>
051     * Copyright (c) 2011, Regents of the University of Colorado <br>
052     * All rights reserved.
053     * 
054     * @author Steven Bethard
055     */
056    public class TempEval2010TaskAAttributes extends TempEval2010Main {
057    
058      @Override
059      protected CleartkPipelineProvider getCleartkPipelineProvider(
060          File modelDirectory,
061          File xmiDirectory) throws Exception {
062        return new TempEval2010PipelineProvider(
063            modelDirectory,
064            xmiDirectory,
065            Arrays.asList(
066                TempEval2010GoldAnnotator.PARAM_TEXT_VIEWS,
067                TempEval2010GoldAnnotator.PARAM_TIME_EXTENT_VIEWS),
068            TempEval2010GoldAnnotator.PARAM_TIME_ATTRIBUTE_VIEWS,
069            Arrays.asList(
070                DefaultSnowballStemmer.getDescription("English"),
071                PosTaggerAnnotator.getDescription()),
072            Arrays.asList(TimeTypeAnnotator.FACTORY));
073      }
074    
075      @Override
076      protected EvaluationPipelineProvider getEvaluationPipelineProvider(File evalDirectory)
077          throws Exception {
078        return new BatchBasedEvaluationPipelineProvider(Arrays.asList(
079            AnalysisEngineFactory.createPrimitive(
080                TempEval2010Writer.getDescription(),
081                TempEval2010Writer.PARAM_OUTPUT_DIRECTORY,
082                evalDirectory.getPath(),
083                TempEval2010Writer.PARAM_TEXT_VIEW,
084                TempEval2010PipelineProvider.SYSTEM_VIEW_NAME,
085                TempEval2010Writer.PARAM_TIME_ATTRIBUTE_VIEW,
086                TempEval2010PipelineProvider.SYSTEM_VIEW_NAME),
087            AnalysisEngineFactory.createPrimitive(
088                AnnotationEvaluator.class,
089                AnnotationEvaluator.PARAM_ANNOTATION_CLASS_NAME,
090                Time.class.getName(),
091                AnnotationEvaluator.PARAM_ANNOTATION_ATTRIBUTE_NAME,
092                "timeType",
093                AnnotationEvaluator.PARAM_GOLD_VIEW_NAME,
094                TempEval2010PipelineProvider.GOLD_VIEW_NAME,
095                AnnotationEvaluator.PARAM_SYSTEM_VIEW_NAME,
096                TempEval2010PipelineProvider.SYSTEM_VIEW_NAME,
097                AnnotationEvaluator.PARAM_IGNORE_SYSTEM_SPANS_NOT_IN_GOLD,
098                true)));
099      }
100    
101      public static void main(String[] args) throws Exception {
102        new TempEval2010TaskAAttributes().runCommand(args);
103      }
104    }
