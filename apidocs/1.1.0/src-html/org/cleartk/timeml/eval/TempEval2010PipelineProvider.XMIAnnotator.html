078      public static final String GOLD_VIEW_NAME = "GoldView";
079    
080      public static final String SYSTEM_VIEW_NAME = CAS.NAME_DEFAULT_SOFA;

082      private File modelDirectory;

084      private File xmiDirectory;

087

091

094      private AnalysisEngine xmiAnnotator;
095    
096      protected Map<CleartkInternalModelFactory, AnalysisEngine> systemAnnotators;
097    
098      private List<AnalysisEngine> pipeline;

100      public TempEval2010PipelineProvider(
101          File modelDirectory,
102          File xmiDirectory,
103          Collection<String> requiredViewParams,
104          String annotatedViewParam,
105          Collection<AnalysisEngineDescription> preprocessingAnnotators,
106          List<CleartkInternalModelFactory> systemAnnotatorFactories)
107          throws ResourceInitializationException {
108        this.modelDirectory = modelDirectory;
109        int dotIndex = Math.max(0, annotatedViewParam.lastIndexOf('.'));
110        String xmiDirName = annotatedViewParam.substring(dotIndex + 1);
111        this.xmiDirectory = new File(xmiDirectory, xmiDirName);
112    
113        // set up TempEval2010GoldAnnotator view parameters
114        this.annotatedViewParam = annotatedViewParam;
115        this.requiredViewParam = new HashSet<String>(requiredViewParams);
116        this.viewParams = Arrays.asList(
117            TempEval2010GoldAnnotator.PARAM_TEXT_VIEWS,
118            TempEval2010GoldAnnotator.PARAM_DOCUMENT_CREATION_TIME_VIEWS,
119            TempEval2010GoldAnnotator.PARAM_TIME_EXTENT_VIEWS,
120            TempEval2010GoldAnnotator.PARAM_TIME_ATTRIBUTE_VIEWS,
121            TempEval2010GoldAnnotator.PARAM_EVENT_EXTENT_VIEWS,
122            TempEval2010GoldAnnotator.PARAM_EVENT_ATTRIBUTE_VIEWS,
123            TempEval2010GoldAnnotator.PARAM_TEMPORAL_LINK_EVENT_TO_DOCUMENT_CREATION_TIME_VIEWS,
124            TempEval2010GoldAnnotator.PARAM_TEMPORAL_LINK_EVENT_TO_SAME_SENTENCE_TIME_VIEWS,
125            TempEval2010GoldAnnotator.PARAM_TEMPORAL_LINK_EVENT_TO_SUBORDINATED_EVENT_VIEWS,
126            TempEval2010GoldAnnotator.PARAM_TEMPORAL_LINK_MAIN_EVENT_TO_NEXT_SENTENCE_MAIN_EVENT_VIEWS);
127    
128        // set up the pipeline, loading from XMI files when possible
129        this.goldAnnotator = AnalysisEngineFactory.createPrimitive(TempEval2010GoldAnnotator.getDescription());
130        this.systemAnnotators = new HashMap<CleartkInternalModelFactory, AnalysisEngine>();
131        for (CleartkInternalModelFactory factory : systemAnnotatorFactories) {
132          File outputDirectory = this.getOutputDirectory("", factory);
133          AnalysisEngineDescription desc = factory.getWriterDescription(outputDirectory);
134          AnalysisEngine engine = AnalysisEngineFactory.createPrimitive(desc);
135          this.systemAnnotators.put(factory, engine);
136        }
137        this.pipeline = new ArrayList<AnalysisEngine>();
138        if (!this.xmiDirectory.exists()) {
139          this.xmiAnnotator = AnalysisEngineFactory.createPrimitive(
140              XMIWriter.class,
141              XMIAnnotator.PARAM_XMI_DIRECTORY,
142              this.xmiDirectory.getPath());
143          this.pipeline.add(this.goldAnnotator);
144          for (AnalysisEngineDescription desc : preprocessingAnnotators) {
145            this.pipeline.add(AnalysisEngineFactory.createPrimitive(desc));
146          }
147          this.pipeline.add(this.xmiAnnotator);
148        } else {
149          this.xmiAnnotator = AnalysisEngineFactory.createPrimitive(
150              XMIReader.class,
151              XMIAnnotator.PARAM_XMI_DIRECTORY,
152              this.xmiDirectory.getPath());
153          this.pipeline.add(this.xmiAnnotator);
154        }
155        this.pipeline.addAll(this.systemAnnotators.values());
156      }
157    
158      @Override
159      public List<AnalysisEngine> getTrainingPipeline(String runName) throws UIMAException {
160        // reconfigure xmi annotator to use training CASes
161        String dir = new File(this.xmiDirectory, "training").getPath();
162        this.xmiAnnotator.setConfigParameterValue(XMIAnnotator.PARAM_XMI_DIRECTORY, dir);
163        this.xmiAnnotator.reconfigure();
164    
165        // reconfigure gold annotator to read annotations into system view
166        for (String viewParam : this.viewParams) {
167          if (viewParam.equals(this.annotatedViewParam) || this.requiredViewParam.contains(viewParam)) {
168            this.goldAnnotator.setConfigParameterValue(viewParam, new String[] { SYSTEM_VIEW_NAME });
169          } else {
170            this.goldAnnotator.setConfigParameterValue(viewParam, new String[] {});
171          }
172        }
173        this.goldAnnotator.reconfigure();
174    
175        // reconfigure system annotators to write training data
176        for (CleartkInternalModelFactory factory : this.systemAnnotators.keySet()) {
177          AnalysisEngine systemAnnotator = this.systemAnnotators.get(factory);
178          systemAnnotator.setConfigParameterValue(CleartkAnnotator.PARAM_IS_TRAINING, true);
179          systemAnnotator.setConfigParameterValue(CleartkSequenceAnnotator.PARAM_IS_TRAINING, true);
180          systemAnnotator.setConfigParameterValue(
181              DirectoryDataWriterFactory.PARAM_OUTPUT_DIRECTORY,
182              this.getOutputDirectory(runName, factory).getPath());
183          systemAnnotator.reconfigure();
184        }
185    
186        // return the configured pipeline
187        return this.pipeline;
188      }
189    
190      @Override
191      public void train(String runName, String... trainingArguments) throws Exception {
192        Logger.getLogger("cc.mallet").setLevel(Level.WARNING);
193        Logger.getLogger("opennlp").setLevel(Level.WARNING);
194        // don't use HideOutput - overriding System.err screws up the logging for some reason
195        PrintStream oldOut = System.out;
196        System.setOut(new PrintStream(new OutputStream() {
197          @Override
198          public void write(int b) throws IOException {
199            // write nothing
200          }
201        }));
202        try {
203          for (CleartkInternalModelFactory factory : this.systemAnnotators.keySet()) {
204            File modelDir = this.getOutputDirectory(runName, factory);
205            JarClassifierBuilder.trainAndPackage(modelDir, trainingArguments);
206          }
207        } finally {
208          System.setOut(oldOut);
209        }
210        // if building to the pre-defined training directory, clean up non-model files
211        for (CleartkInternalModelFactory factory : this.systemAnnotators.keySet()) {
212          File modelDir = factory.getTrainingDirectory();
213          if (modelDir.exists()) {
214            for (File file : modelDir.listFiles()) {
215              if (!file.isDirectory() && !file.getName().equals("model.jar")) {
216                file.delete();
217              }
218            }
219          }
220        }
221      }

223      @Override
224      public List<AnalysisEngine> getClassifyingPipeline(String runName) throws UIMAException {
225        // reconfigure xmi annotator to use classifying CASes
226        String dir = new File(this.xmiDirectory, "classifying").getPath();
227        this.xmiAnnotator.setConfigParameterValue(XMIAnnotator.PARAM_XMI_DIRECTORY, dir);
228        this.xmiAnnotator.reconfigure();
229    
230        // reconfigure gold annotator to read annotations into gold view
231        for (String viewParam : this.viewParams) {
232          if (viewParam.equals(this.annotatedViewParam)) {
233            this.goldAnnotator.setConfigParameterValue(viewParam, new String[] { GOLD_VIEW_NAME });
234          } else if (this.requiredViewParam.contains(viewParam)) {
235            this.goldAnnotator.setConfigParameterValue(viewParam, new String[] {
236                SYSTEM_VIEW_NAME,
237                GOLD_VIEW_NAME });
238          } else {
239            this.goldAnnotator.setConfigParameterValue(viewParam, new String[] {});
240          }
241        }
242        this.goldAnnotator.reconfigure();
243    
244        // reconfigure system annotators to load model files
245        for (CleartkInternalModelFactory factory : this.systemAnnotators.keySet()) {
246          AnalysisEngine systemAnnotator = this.systemAnnotators.get(factory);
247          File modelDir = this.getOutputDirectory(runName, factory);
248          systemAnnotator.setConfigParameterValue(CleartkAnnotator.PARAM_IS_TRAINING, false);
249          systemAnnotator.setConfigParameterValue(CleartkSequenceAnnotator.PARAM_IS_TRAINING, false);
250          File modelFile;
251          try {
252            modelFile = JarClassifierBuilder.fromTrainingDirectory(modelDir).getModelJarFile(modelDir);
253          } catch (IOException e) {
254            throw new UIMAException(e);
255          }
256          systemAnnotator.setConfigParameterValue(
257              GenericJarClassifierFactory.PARAM_CLASSIFIER_JAR_PATH,
258              modelFile.getPath());
259          systemAnnotator.reconfigure();
260        }
261    
262        // return the configured pipeline
263        return this.pipeline;
264      }
265    
266      private File getOutputDirectory(String runName, CleartkInternalModelFactory factory) {
267        if (this.modelDirectory == null) {
268          return factory.getTrainingDirectory();
269        } else {
270          String annotatorName = factory.getAnnotatorClass().getName();
271          return new File(new File(this.modelDirectory, annotatorName), runName);
272        }
273      }

275      public static abstract class XMIAnnotator extends JCasAnnotator_ImplBase {
276    
277        @ConfigurationParameter(mandatory = true)
278        protected File xmiDirectory;
279    
280        public static final String PARAM_XMI_DIRECTORY = ConfigurationParameterFactory.createConfigurationParameterName(
281            XMIAnnotator.class,
282            "xmiDirectory");
283    
284        protected File getFile(JCas jCas) throws AnalysisEngineProcessException {
285          return new File(this.xmiDirectory, ViewURIUtil.getURI(jCas).getFragment() + ".xmi");
286        }
287    
288      }

290      public static class XMIReader extends XMIAnnotator {
291    
292        @Override
293        public void process(JCas jCas) throws AnalysisEngineProcessException {
294          try {
295            FileInputStream stream = new FileInputStream(this.getFile(jCas));
296            try {
297              XmiCasDeserializer.deserialize(stream, jCas.getCas());
298            } finally {
299              stream.close();
300            }
301          } catch (SAXException e) {
302            throw new AnalysisEngineProcessException(e);
303          } catch (IOException e) {
304            throw new AnalysisEngineProcessException(e);
305          }
306        }
307      }
308    
309      public static class XMIWriter extends XMIAnnotator {
310    
311        @Override
312        public void initialize(UimaContext context) throws ResourceInitializationException {
313          super.initialize(context);
314          if (!this.xmiDirectory.exists()) {
315            this.xmiDirectory.mkdirs();
316          }
317        }
318    
319        @Override
320        public void process(JCas jCas) throws AnalysisEngineProcessException {
321          XmiCasSerializer ser = new XmiCasSerializer(jCas.getTypeSystem());
322          try {
323            FileOutputStream stream = new FileOutputStream(this.getFile(jCas));
324            try {
325              ser.serialize(jCas.getCas(), new XMLSerializer(stream, false).getContentHandler());
326            } finally {
327              stream.close();
328            }
329          } catch (SAXException e) {
330            throw new AnalysisEngineProcessException(e);
331          } catch (IOException e) {
332            throw new AnalysisEngineProcessException(e);
333          }
334        }
335      }
