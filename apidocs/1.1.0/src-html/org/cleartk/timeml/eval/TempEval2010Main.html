024    package org.cleartk.timeml.eval;
025    
026    import java.io.File;
027    import java.util.Arrays;
028    
029    import org.apache.uima.UIMAFramework;
030    import org.apache.uima.util.Level;
031    import org.cleartk.eval.Evaluation;
032    import org.cleartk.eval.provider.CleartkPipelineProvider;
033    import org.cleartk.eval.provider.CorpusReaderProvider;
034    import org.cleartk.eval.provider.EvaluationPipelineProvider;
035    
036    /**
037     * <br>
038     * Copyright (c) 2011, Regents of the University of Colorado <br>
039     * All rights reserved.
040     * 
041     * @author Steven Bethard
042     */
043    public abstract class TempEval2010Main {
044    
045      enum Command {
046        CV, TEST, TRAIN
047      }
048    
049      private File tempEvalTrainingDir;
050    
051      private File tempEvalTestDir;
052    
053      private File outputDirectory;
054    
055      private CorpusReaderProvider getCorpusReaderProvider() throws Exception {
056        return new TempEval2010ReaderProvider(this.tempEvalTrainingDir, this.tempEvalTestDir);
057      }
058    
059      public void runCrossValidation() throws Exception {
060        File modelDir = new File(this.outputDirectory, "cv/model");
061        File xmiDir = new File(this.outputDirectory, "cv/xmi");
062        File evalDir = new File(this.outputDirectory, "cv");
063        CorpusReaderProvider corpus = this.getCorpusReaderProvider();
064        CleartkPipelineProvider pipeline = this.getCleartkPipelineProvider(modelDir, xmiDir);
065        EvaluationPipelineProvider evaluator = this.getEvaluationPipelineProvider(evalDir);
066        Evaluation evaluation = new Evaluation();
067        corpus.setNumberOfFolds(5);
068        evaluation.runCrossValidation(corpus, pipeline, evaluator, this.getTrainingArguments());
069      }
070    
071      public void runHoldoutEvaluation() throws Exception {
072        File modelDir = new File(this.outputDirectory, "test/model");
073        File xmiDir = new File(this.outputDirectory, "test/xmi");
074        File evalDir = new File(this.outputDirectory, "test");
075        CorpusReaderProvider corpus = this.getCorpusReaderProvider();
076        CleartkPipelineProvider pipeline = this.getCleartkPipelineProvider(modelDir, xmiDir);
077        EvaluationPipelineProvider evaluator = this.getEvaluationPipelineProvider(evalDir);
078        Evaluation evaluation = new Evaluation();
079        evaluation.runHoldoutEvaluation(corpus, pipeline, evaluator, this.getTrainingArguments());
080      }
081    
082      public void buildCorpusModel() throws Exception {
083        File modelDir = null; // build to annotator-defined training directory
084        File xmiDir = new File(this.outputDirectory, "train/xmi");
085        CorpusReaderProvider corpus = this.getCorpusReaderProvider();
086        CleartkPipelineProvider pipeline = this.getCleartkPipelineProvider(modelDir, xmiDir);
087        Evaluation evaluation = new Evaluation();
088        evaluation.buildCorpusModel("model", corpus, pipeline, this.getTrainingArguments());
089    
090      }
091    
092      protected String[] getTrainingArguments() {
093        return new String[0];
094      }
095    
096      protected abstract CleartkPipelineProvider getCleartkPipelineProvider(
097          File modelDirectory,
098          File xmiDirectory) throws Exception;
099    
100      protected abstract EvaluationPipelineProvider getEvaluationPipelineProvider(File evalDirectory)
101          throws Exception;
102    
103      public void runCommand(String[] args) throws Exception {
104        String className = this.getClass().getName();
105        if (args.length != 4) {
106          System.err.printf(
107              "usage: java %s {cv,test,train} tempeval-training-dir tempeval-test-dir output-dir",
108              className);
109          System.exit(1);
110        }
111    
112        Command command = Command.valueOf(args[0].toUpperCase());
113        char[] lineChars = new char[className.length()];
114        Arrays.fill(lineChars, '=');
115        String line = new String(lineChars);
116        String message = String.format("\n%s\n%s\n%s", line, className, line);
117        UIMAFramework.getLogger(this.getClass()).log(Level.INFO, message);
118        this.tempEvalTrainingDir = new File(args[1]);
119        this.tempEvalTestDir = new File(args[2]);
120        this.outputDirectory = new File(args[3]);
121        switch (command) {
122          case CV:
123            this.runCrossValidation();
124            break;
125          case TEST:
126            this.runHoldoutEvaluation();
127            break;
128          case TRAIN:
129            this.buildCorpusModel();
130            break;
131        }
132      }
133    
134      public static void main(String[] args) throws Exception {
135        for (TempEval2010Main main : Arrays.asList(
136            new TempEval2010TaskAAttributes(),
137            new TempEval2010TaskAExtents(),
138            new TempEval2010TaskBAttributes(),
139            new TempEval2010TaskBExtents(),
140            new TempEval2010TaskC(),
141            new TempEval2010TaskD(),
142            new TempEval2010TaskE(),
143            new TempEval2010TaskF())) {
144          main.runCommand(args);
145        }
146      }
147    }
