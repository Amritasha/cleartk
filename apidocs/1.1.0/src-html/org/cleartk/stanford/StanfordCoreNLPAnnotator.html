093      @Override
094      public void initialize(UimaContext context) throws ResourceInitializationException {
095        super.initialize(context);
096    
097        Properties properties = new Properties();
098        properties.put("annotators", "tokenize, ssplit, pos, lemma, ner, parse, dcoref");
099    
100        this.processor = new StanfordCoreNLP(properties);
101      }

103      @Override
104      public void process(JCas jCas) throws AnalysisEngineProcessException {
105        Annotation document = this.processor.process(jCas.getDocumentText());
106    
107        String lastNETag = "O";
108        int lastNEBegin = -1;
109        int lastNEEnd = -1;
110        for (CoreMap tokenAnn : document.get(TokensAnnotation.class)) {
111    
112          // create the token annotation
113          int begin = tokenAnn.get(CharacterOffsetBeginAnnotation.class);
114          int end = tokenAnn.get(CharacterOffsetEndAnnotation.class);
115          String pos = tokenAnn.get(PartOfSpeechAnnotation.class);
116          String lemma = tokenAnn.get(LemmaAnnotation.class);
117          Token token = new Token(jCas, begin, end);
118          token.setPos(pos);
119          token.setLemma(lemma);
120          token.addToIndexes();
121    
122          // hackery to convert token-level named entity tag into phrase-level tag
123          String neTag = tokenAnn.get(NamedEntityTagAnnotation.class);
124          if (neTag.equals("O") && !lastNETag.equals("O")) {
125            NamedEntityMention ne = new NamedEntityMention(jCas, lastNEBegin, lastNEEnd);
126            ne.setMentionType(lastNETag);
127            ne.addToIndexes();
128          } else {
129            if (lastNETag.equals("O")) {
130              lastNEBegin = begin;
131            } else if (lastNETag.equals(neTag)) {
132              // do nothing - begin was already set
133            } else {
134              NamedEntityMention ne = new NamedEntityMention(jCas, lastNEBegin, lastNEEnd);
135              ne.setMentionType(lastNETag);
136              ne.addToIndexes();
137              lastNEBegin = begin;
138            }
139            lastNEEnd = end;
140          }
141          lastNETag = neTag;
142        }
143        if (!lastNETag.equals("O")) {
144          NamedEntityMention ne = new NamedEntityMention(jCas, lastNEBegin, lastNEEnd);
145          ne.setMentionType(lastNETag);
146          ne.addToIndexes();
147        }
148    
149        // add sentences and trees
150        for (CoreMap sentenceAnn : document.get(SentencesAnnotation.class)) {
151    
152          // add the sentence annotation
153          int sentBegin = sentenceAnn.get(CharacterOffsetBeginAnnotation.class);
154          int sentEnd = sentenceAnn.get(CharacterOffsetEndAnnotation.class);
155          Sentence sentence = new Sentence(jCas, sentBegin, sentEnd);
156          sentence.addToIndexes();
157    
158          // add the syntactic tree annotation
159          List<CoreLabel> tokenAnns = sentenceAnn.get(TokensAnnotation.class);
160          Tree tree = sentenceAnn.get(TreeAnnotation.class);
161          if (tree.children().length != 1) {
162            throw new RuntimeException("Expected single root node, found " + tree);
163          }
164          tree = tree.firstChild();
165          tree.indexSpans(0);
166          TopTreebankNode root = new TopTreebankNode(jCas);
167          root.setTreebankParse(tree.toString());
168          // TODO: root.setTerminals(v)
169          this.addTreebankNodeToIndexes(root, jCas, tree, tokenAnns);
170    
171          // get the dependencies
172          SemanticGraph dependencies = sentenceAnn.get(CollapsedCCProcessedDependenciesAnnotation.class);
173    
174          // convert Stanford nodes to UIMA annotations
175          List<Token> tokens = JCasUtil.selectCovered(jCas, Token.class, sentence);
176          Map<IndexedWord, DependencyNode> stanfordToUima = new HashMap<IndexedWord, DependencyNode>();
177          for (IndexedWord stanfordNode : dependencies.vertexSet()) {
178            int indexBegin = stanfordNode.get(BeginIndexAnnotation.class);
179            int indexEnd = stanfordNode.get(EndIndexAnnotation.class);
180            int tokenBegin = tokens.get(indexBegin).getBegin();
181            int tokenEnd = tokens.get(indexEnd - 1).getEnd();
182            DependencyNode node;
183            if (dependencies.getRoots().contains(stanfordNode)) {
184              node = new TopDependencyNode(jCas, tokenBegin, tokenEnd);
185            } else {
186              node = new DependencyNode(jCas, tokenBegin, tokenEnd);
187            }
188            stanfordToUima.put(stanfordNode, node);
189          }
190    
191          // create relation annotations for each Stanford dependency
192          ArrayListMultimap<DependencyNode, DependencyRelation> headRelations = ArrayListMultimap.create();
193          ArrayListMultimap<DependencyNode, DependencyRelation> childRelations = ArrayListMultimap.create();
194          for (SemanticGraphEdge stanfordEdge : dependencies.edgeIterable()) {
195            DependencyRelation relation = new DependencyRelation(jCas);
196            DependencyNode head = stanfordToUima.get(stanfordEdge.getGovernor());
197            DependencyNode child = stanfordToUima.get(stanfordEdge.getDependent());
198            String relationType = stanfordEdge.getRelation().toString();
199            if (head == null || child == null || relationType == null) {
200              throw new RuntimeException(String.format(
201                  "null elements not allowed in relation:\nrelation=%s\nchild=%s\nhead=%s\n",
202                  relation,
203                  child,
204                  head));
205            }
206            relation.setHead(head);
207            relation.setChild(child);
208            relation.setRelation(relationType);
209            relation.addToIndexes();
210            headRelations.put(child, relation);
211            childRelations.put(head, relation);
212          }
213    
214          // set the relations for each node annotation
215          for (DependencyNode node : stanfordToUima.values()) {
216            node.setHeadRelations(UIMAUtil.toFSArray(jCas, headRelations.get(node)));
217            node.setChildRelations(UIMAUtil.toFSArray(jCas, childRelations.get(node)));
218            node.addToIndexes();
219          }
220        }
221    
222        // map from spans to named entity mentions
223        Map<Span, NamedEntityMention> spanMentionMap = new HashMap<Span, NamedEntityMention>();
224        for (NamedEntityMention mention : JCasUtil.select(jCas, NamedEntityMention.class)) {
225          spanMentionMap.put(new Span(mention.getBegin(), mention.getEnd()), mention);
226        }
227    
228        // add mentions for all entities identified by the coreference system
229        List<NamedEntity> entities = new ArrayList<NamedEntity>();
230        List<List<Token>> sentenceTokens = new ArrayList<List<Token>>();
231        for (Sentence sentence : JCasUtil.select(jCas, Sentence.class)) {
232          sentenceTokens.add(JCasUtil.selectCovered(jCas, Token.class, sentence));
233        }
234        Map<Integer, CorefChain> corefChains = document.get(CorefChainAnnotation.class);
235        for (CorefChain chain : corefChains.values()) {
236          List<NamedEntityMention> mentions = new ArrayList<NamedEntityMention>();
237          for (CorefMention corefMention : chain.getCorefMentions()) {
238    
239            // figure out the character span of the token
240            List<Token> tokens = sentenceTokens.get(corefMention.sentNum - 1);
241            int begin = tokens.get(corefMention.startIndex - 1).getBegin();
242            int end = tokens.get(corefMention.endIndex - 2).getEnd();
243    
244            // use an existing named entity mention when possible; otherwise create a new one
245            NamedEntityMention mention = spanMentionMap.get(new Span(begin, end));
246            if (mention == null) {
247              mention = new NamedEntityMention(jCas, begin, end);
248              mention.addToIndexes();
249            }
250            mentions.add(mention);
251          }
252    
253          // create an entity for the mentions
254          Collections.sort(mentions, new Comparator<NamedEntityMention>() {
255            @Override
256            public int compare(NamedEntityMention m1, NamedEntityMention m2) {
257              return m1.getBegin() - m2.getBegin();
258            }
259          });
260    
261          // create mentions and add them to entity
262          NamedEntity entity = new NamedEntity(jCas);
263          entity.setMentions(new FSArray(jCas, mentions.size()));
264          int index = 0;
265          for (NamedEntityMention mention : mentions) {
266            mention.setMentionedEntity(entity);
267            entity.setMentions(index, mention);
268            index += 1;
269          }
270          entities.add(entity);
271        }
272    
273        // add singleton entities for any named entities not picked up by coreference system
274        for (NamedEntityMention mention : JCasUtil.select(jCas, NamedEntityMention.class)) {
275          if (mention.getMentionedEntity() == null) {
276            NamedEntity entity = new NamedEntity(jCas);
277            entity.setMentions(new FSArray(jCas, 1));
278            entity.setMentions(0, mention);
279            mention.setMentionedEntity(entity);
280            entity.getMentions();
281            entities.add(entity);
282          }
283        }
284    
285        // sort entities by document order
286        Collections.sort(entities, new Comparator<NamedEntity>() {
287          @Override
288          public int compare(NamedEntity o1, NamedEntity o2) {
289            return getFirstBegin(o1) - getFirstBegin(o2);
290          }
291    
292          private int getFirstBegin(NamedEntity entity) {
293            int min = Integer.MAX_VALUE;
294            for (NamedEntityMention mention : JCasUtil.select(
295                entity.getMentions(),
296                NamedEntityMention.class)) {
297              if (mention.getBegin() < min) {
298                min = mention.getBegin();
299              }
300            }
301            return min;
302          }
303        });
304    
305        // add entities to document
306        for (NamedEntity entity : entities) {
307          entity.addToIndexes();
308        }
309    
310      }

312      private FSArray addTreebankNodeChildrenToIndexes(
313          TreebankNode parent,
314          JCas jCas,
315          List<CoreLabel> tokenAnns,
316          Tree tree) {
317        Tree[] childTrees = tree.children();
318    
319        // collect all children (except leaves, which are just the words - POS tags are pre-terminals in
320        // a Stanford tree)
321        List<TreebankNode> childNodes = new ArrayList<TreebankNode>();
322        for (Tree child : childTrees) {
323          if (!child.isLeaf()) {
324    
325            // set node attributes and add children (mutual recursion)
326            TreebankNode node = new TreebankNode(jCas);
327            node.setParent(parent);
328            this.addTreebankNodeToIndexes(node, jCas, child, tokenAnns);
329            childNodes.add(node);
330          }
331        }
332    
333        // convert the child list into an FSArray
334        FSArray childNodeArray = new FSArray(jCas, childNodes.size());
335        for (int i = 0; i < childNodes.size(); ++i) {
336          childNodeArray.set(i, childNodes.get(i));
337        }
338        return childNodeArray;
339      }
340    
341      private void addTreebankNodeToIndexes(
342          TreebankNode node,
343          JCas jCas,
344          Tree tree,
345          List<CoreLabel> tokenAnns) {
346        // figure out begin and end character offsets
347        CoreMap label = (CoreMap) tree.label();
348        CoreMap beginToken = tokenAnns.get(label.get(BeginIndexAnnotation.class));
349        CoreMap endToken = tokenAnns.get(label.get(EndIndexAnnotation.class) - 1);
350        int nodeBegin = beginToken.get(CharacterOffsetBeginAnnotation.class);
351        int nodeEnd = endToken.get(CharacterOffsetEndAnnotation.class);
352    
353        // set span, node type, children (mutual recursion), and add it to the JCas
354        node.setBegin(nodeBegin);
355        node.setEnd(nodeEnd);
356        node.setNodeType(tree.value());
357        node.setChildren(this.addTreebankNodeChildrenToIndexes(node, jCas, tokenAnns, tree));
358        node.setLeaf(node.getChildren().size() == 0);
359        node.addToIndexes();
360      }
361    
362      private static class Span {
363        public int begin;
364    
365        public int end;
366    
367        public Span(int begin, int end) {
368          this.begin = begin;
369          this.end = end;
370        }
371    
372        public boolean equals(Object object) {
373          if (object instanceof Span) {
374            Span that = (Span) object;
375            return this.begin == that.begin && this.end == that.end;
376          } else {
377            return false;
378          }
379        }
380    
381        public int hashCode() {
382          return Arrays.hashCode(new int[] { this.begin, this.end });
383        }
384      }
