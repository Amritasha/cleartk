052    public abstract class ChunkLabeler_ImplBase implements ChunkLabeler, Initializable {
053    
054      public static final String PARAM_CHUNK_ANNOTATION_CLASS_NAME = ConfigurationParameterFactory.createConfigurationParameterName(
055          ChunkLabeler_ImplBase.class,
056          "chunkAnnotationClassName");
057    
058      @ConfigurationParameter(
059          mandatory = true,
060          description = "names the class of the type system chunk annotation type. An example value might be something like: 'org.cleartk.type.ne.NamedEntityMention'")
061      private String chunkAnnotationClassName;
062    
063      @ConfigurationParameter(
064          name = Chunker.PARAM_LABELED_ANNOTATION_CLASS_NAME,
065          mandatory = true,
066          description = "names the class of the type system type used to associate B, I, and O (for example) labels with.  An example value might be 'org.cleartk.type.Token'")
067      private String labeledAnnotationClassName;
068    
069      public static final String BEGIN_PREFIX = "B";
070    
071      public static final String INSIDE_PREFIX = "I";
072    
073      public static final String OUTSIDE_LABEL = "O";
074    
075      public static final String SEPARATOR = "-";
076    
077      protected Class<? extends Annotation> chunkAnnotationClass;
078    
079      protected Type chunkAnnotationType;
080    
081      protected Class<? extends Annotation> labeledAnnotationClass;
082    
083      protected Type labeledAnnotationType;
084    
085      protected boolean typesInitialized = false;
086    
087      protected Map<Annotation, String> annotationLabels;
088    
089      public void initialize(UimaContext context) throws ResourceInitializationException {
090        ConfigurationParameterInitializer.initialize(this, context);
091        labeledAnnotationClass = InitializableFactory.getClass(
092            labeledAnnotationClassName,
093            Annotation.class);
094        chunkAnnotationClass = InitializableFactory.getClass(chunkAnnotationClassName, Annotation.class);
095        annotationLabels = new HashMap<Annotation, String>();
096      }
097    
098      public abstract String getChunkLabel(JCas jCas, Annotation chunkAnnotation)
099          throws AnalysisEngineProcessException;
100    
101      public abstract Annotation createChunk(
102          JCas jCas,
103          List<? extends Annotation> labeledAnnotations,
104          String label) throws AnalysisEngineProcessException;
105    
106      protected void initializeTypes(JCas jCas) throws AnalysisEngineProcessException {
107        try {
108          chunkAnnotationType = UIMAUtil.getCasType(jCas, chunkAnnotationClass);
109          labeledAnnotationType = UIMAUtil.getCasType(jCas, labeledAnnotationClass);
110        } catch (Exception e) {
111          throw new AnalysisEngineProcessException(e);
112        }
113        typesInitialized = true;
114      }
115    
116      public void chunks2Labels(JCas jCas, Annotation sequence) throws AnalysisEngineProcessException {
117        if (!typesInitialized)
118          initializeTypes(jCas);
119    
120        annotationLabels.clear();
121    
122        FSIterator<Annotation> chunkAnnotations = jCas.getAnnotationIndex(chunkAnnotationType).subiterator(
123            sequence);
124        while (chunkAnnotations.hasNext()) {
125          Annotation chunkAnnotation = chunkAnnotations.next();
126          String labelBase = getChunkLabel(jCas, chunkAnnotation);
127          String label = labelBase;
128    
129          List<? extends Annotation> labeledAnnotations = JCasUtil.selectCovered(
130              jCas,
131              labeledAnnotationClass,
132              chunkAnnotation);
133    
134          boolean begin = true;
135          for (Annotation labelAnnotation : labeledAnnotations) {
136            if (begin)
137              label = BEGIN_PREFIX + SEPARATOR + labelBase;
138            else
139              label = INSIDE_PREFIX + SEPARATOR + labelBase;
140            begin = false;
141            setLabel(labelAnnotation, label);
142          }
143        }
144      }
145    
146      public List<Annotation> labels2Chunks(JCas jCas, Annotation sequence)
147          throws AnalysisEngineProcessException {
148        if (!typesInitialized)
149          initializeTypes(jCas);
150    
151        List<Annotation> returnValues = new ArrayList<Annotation>();
152        FSIterator<Annotation> labeledAnnotations = jCas.getAnnotationIndex(labeledAnnotationType).subiterator(
153            sequence);
154    
155        String currentLabelValue = null;
156        List<Annotation> currentLabeledAnnotations = new ArrayList<Annotation>();
157    
158        while (labeledAnnotations.hasNext()) {
159          Annotation labeledAnnotation = labeledAnnotations.next();
160          String label = getLabel(labeledAnnotation);
161          if (label == null)
162            label = OUTSIDE_LABEL;
163    
164          String labelPrefix = label.equals(OUTSIDE_LABEL) ? null : label.substring(0, 1);
165          String labelValue = label.equals(OUTSIDE_LABEL) ? null : label.substring(2);
166    
167          if (labelPrefix == null || labelPrefix.equals(BEGIN_PREFIX)
168              || !labelValue.equals(currentLabelValue)) {
169            if (currentLabeledAnnotations.size() > 0) {
170              Annotation chunk = createChunk(jCas, currentLabeledAnnotations, currentLabelValue);
171              if (chunk != null)
172                returnValues.add(chunk);
173            }
174            currentLabeledAnnotations.clear();
175            currentLabelValue = labelValue;
176          }
177          if (labelValue != null)
178            currentLabeledAnnotations.add(labeledAnnotation);
179        }
180    
181        if (currentLabeledAnnotations.size() > 0) {
182          Annotation chunk = createChunk(jCas, currentLabeledAnnotations, currentLabelValue);
183          if (chunk != null)
184            returnValues.add(chunk);
185        }
186        return returnValues;
187      }
188    
189      public String getLabel(Annotation labeledAnnotation) throws AnalysisEngineProcessException {
190        if (annotationLabels.containsKey(labeledAnnotation))
191          return annotationLabels.get(labeledAnnotation);
192        return OUTSIDE_LABEL;
193      }
194    
195      public void setLabel(Annotation labeledAnnotation, String label)
196          throws AnalysisEngineProcessException {
197        annotationLabels.put(labeledAnnotation, label);
198      }
199    
200      public void setChunkAnnotationClassName(String chunkAnnotationClassName) {
201        this.chunkAnnotationClassName = chunkAnnotationClassName;
202      }
203    
204      public void setLabeledAnnotationClassName(String labeledAnnotationClassName) {
205        this.labeledAnnotationClassName = labeledAnnotationClassName;
206      }
