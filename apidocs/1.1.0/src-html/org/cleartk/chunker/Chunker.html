048    public class Chunker extends CleartkSequenceAnnotator<String> {
049    
050      // The name is hard coded here as a string because it needs to be referenced as a constant in
051      // ChunkLabeler_ImplBase
052      public static final String PARAM_LABELED_ANNOTATION_CLASS_NAME = "org.cleartk.chunker.Chunker.labeledAnnotationClassName";
053    
054      @ConfigurationParameter(mandatory = true, description = "names the class of the type system type used to associate B, I, and O (for example) labels with.  An example value might be 'org.cleartk.type.Token'")
055      protected String labeledAnnotationClassName;
056    
057      public static final String PARAM_SEQUENCE_CLASS_NAME = ConfigurationParameterFactory.createConfigurationParameterName(
058          Chunker.class,
059          "sequenceClassName");
060    
061      @ConfigurationParameter(mandatory = true, description = "names the class of the type system type that specifies a 'sequence' of labels.  An example might be something like 'org.cleartk.type.Sentence'")
062      protected String sequenceClassName;
063    
064      public static final String PARAM_CHUNK_LABELER_CLASS_NAME = ConfigurationParameterFactory.createConfigurationParameterName(
065          Chunker.class,
066          "chunkLabelerClassName");
067    
068      @ConfigurationParameter(mandatory = true, description = "provides the class name of a class that extends org.cleartk.chunk.ChunkLabeler.")
069      protected String chunkLabelerClassName;
070    
071      public static final String PARAM_CHUNKER_FEATURE_EXTRACTOR_CLASS_NAME = ConfigurationParameterFactory.createConfigurationParameterName(
072          Chunker.class,
073          "chunkerFeatureExtractorClassName");
074    
075      @ConfigurationParameter(mandatory = true, description = "provides the class name of a class that extends org.cleartk.chunk.ChunkFeatureExtractor.")
076      protected String chunkerFeatureExtractorClassName;
077    
078      protected Class<? extends Annotation> labeledAnnotationClass;
079    
080      protected Class<? extends Annotation> sequenceClass;
081    
082      protected ChunkLabeler chunkLabeler;
083    
084      protected ChunkerFeatureExtractor featureExtractor;
085    
086      @Override
087      public void initialize(UimaContext context) throws ResourceInitializationException {
088        super.initialize(context);
089        labeledAnnotationClass = InitializableFactory.getClass(
090            labeledAnnotationClassName,
091            Annotation.class);
092        sequenceClass = InitializableFactory.getClass(sequenceClassName, Annotation.class);
093        chunkLabeler = InitializableFactory.create(context, chunkLabelerClassName, ChunkLabeler.class);
094        featureExtractor = InitializableFactory.create(
095            context,
096            chunkerFeatureExtractorClassName,
097            ChunkerFeatureExtractor.class);
098      }
099    
100      @Override
101      public void process(JCas jCas) throws AnalysisEngineProcessException {
102    
103        List<Instance<String>> instances = new ArrayList<Instance<String>>();
104        Instance<String> instance;
105    
106        List<Annotation> labeledAnnotationList = new ArrayList<Annotation>();
107    
108        for (Annotation sequence : JCasUtil.select(jCas, this.sequenceClass)) {
109          if (this.isTraining()) {
110            chunkLabeler.chunks2Labels(jCas, sequence);
111          }
112          instances.clear();
113          labeledAnnotationList.clear();
114    
115          // select the sub-annotations within this sequence
116          Collection<? extends Annotation> labeledAnnotations;
117          labeledAnnotations = JCasUtil.selectCovered(jCas, this.labeledAnnotationClass, sequence);
118          if (labeledAnnotations.size() == 0) {
119            throw new RuntimeException(String.format(
120                "Found %s containing no %s: \"%s\" %s",
121                this.sequenceClassName,
122                this.labeledAnnotationClassName,
123                sequence.getCoveredText(),
124                sequence));
125          }
126    
127          // create an instance for each sub-annotation
128          for (Annotation labeledAnnotation : labeledAnnotations) {
129            labeledAnnotationList.add(labeledAnnotation);
130            instance = featureExtractor.extractFeatures(jCas, labeledAnnotation, sequence);
131            String label = chunkLabeler.getLabel(labeledAnnotation);
132            instance.setOutcome(label);
133            instances.add(instance);
134          }
135    
136          // write data while training
137          if (this.isTraining()) {
138            this.dataWriter.write(instances);
139          }
140    
141          // set labels during classification
142          else {
143            List<String> results = this.classify(instances);
144            for (int i = 0; i < results.size(); i++) {
145              Annotation labeledAnnotation = labeledAnnotationList.get(i);
146              String label = results.get(i);
147              chunkLabeler.setLabel(labeledAnnotation, label);
148            }
149            chunkLabeler.labels2Chunks(jCas, sequence);
150          }
151        }
152    
153      }
154    
155      public void setLabeledAnnotationClassName(String labeledAnnotationClassName) {
156        this.labeledAnnotationClassName = labeledAnnotationClassName;
157      }
158    
159      public void setSequenceClassName(String sequenceClassName) {
160        this.sequenceClassName = sequenceClassName;
161      }
162    
163      public void setChunkLabelerClassName(String chunkLabelerClassName) {
164        this.chunkLabelerClassName = chunkLabelerClassName;
165      }
166    
167      public void setChunkerFeatureExtractorClassName(String chunkerFeatureExtractorClassName) {
168        this.chunkerFeatureExtractorClassName = chunkerFeatureExtractorClassName;
169      }
