061          Conll2005Writer.class,
062          "outputFile");

064      private PrintWriter output;
065    
066      private boolean first;
067    
068      @Override
069      public void initialize(UimaContext context) throws ResourceInitializationException {
070        super.initialize(context);
071        try {
072          this.output = new PrintWriter(outputFile);
073          this.first = true;
074        } catch (FileNotFoundException e) {
075          throw new ResourceInitializationException(e);
076        }
077      }
078    
079      @Override
080      public void process(JCas jCas) throws AnalysisEngineProcessException {
081        Collection<Sentence> sentences = JCasUtil.select(jCas, Sentence.class);
082        for (Sentence sentence : sentences) {
083          if (first)
084            first = false;
085          else
086            output.println();
087    
088          List<PredicateWriter> predicateWriters = new ArrayList<PredicateWriter>();
089          for (Predicate predicate : JCasUtil.selectCovered(jCas, Predicate.class, sentence)) {
090            predicateWriters.add(new PredicateWriter(jCas, predicate));
091          }
092    
093          for (Token token : JCasUtil.selectCovered(jCas, Token.class, sentence)) {
094            CoNLL05Line line = new CoNLL05Line();
095    
096            // line.setLexeme(token.getCoveredText());
097            // line.setPos(token.getPos());
098    
099            for (PredicateWriter predicateWriter : predicateWriters) {
100              predicateWriter.write(token, line);
101            }
102    
103            output.println(line.evalString());
104          }
105    
106        }
107        output.flush();
108      }
109    
110      @Override
111      public void collectionProcessComplete() throws AnalysisEngineProcessException {
112        output.close();
113        super.collectionProcessComplete();
114      }

195      private static class PredicateWriter {
196        String baseform;
197    
198        Integer frameset;
199    
200        Token token;
201    
202        List<ArgumentWriter> argumentWriters;
203    
204        PredicateWriter(JCas jCas, Predicate predicate) {
205          this.token = JCasUtil.selectCovered(jCas, Token.class, predicate.getAnnotation()).get(0);
206          this.baseform = predicate.getBaseForm();
207          this.frameset = 1;
208    
209          List<Argument> allArgs = UIMAUtil.toList(predicate.getArguments(), Argument.class);
210          this.argumentWriters = new ArrayList<ArgumentWriter>();
211          for (Argument arg : allArgs) {
212            if (arg instanceof SemanticArgument) {
213              this.argumentWriters.add(new ArgumentWriter(jCas, (SemanticArgument) arg));
214            }
215          }
216        }
217    
218        void write(Token tok, CoNLL05Line line) {
219          if (this.token.equals(tok)) {
220            line.setPredicateBaseform(this.baseform);
221            line.setPredicateFrameset(this.frameset);
222          }
223    
224          line.addArgumentRole(getArgumentsString(tok));
225        }
226    
227        String getArgumentsString(Token tok) {
228          StringBuffer buffer = new StringBuffer();
229    
230          for (ArgumentWriter argumentWriter : this.argumentWriters) {
231            buffer.append(argumentWriter.getStartString(tok));
232          }
233    
234          buffer.append("*");
235    
236          for (ArgumentWriter argumentWriter : this.argumentWriters) {
237            buffer.append(argumentWriter.getEndString(tok));
238          }
239    
240          return buffer.toString();
241        }
242      }
243    
244      private static class ArgumentWriter {
245        String label;
246    
247        String feature;
248    
249        String preposition;
250    
251        List<Token> tokens;
252    
253        ArgumentWriter(JCas jCas, SemanticArgument arg) {
254          this.label = arg.getLabel();
255          this.feature = arg.getFeature();
256          this.preposition = arg.getPreposition();
257          this.tokens = JCasUtil.selectCovered(jCas, Token.class, arg.getAnnotation());
258        }
259    
260        String getStartString(Token token) {
261          if (token == this.tokens.get(0))
262            return "(" + getFullLabel();
263          else
264            return "";
265        }
266    
267        String getEndString(Token token) {
268          if (token == this.tokens.get(this.tokens.size() - 1))
269            return ")";
270          else
271            return "";
272        }
273    
274        String getFullLabel() {
275          StringBuffer buffer = new StringBuffer();
276    
277          buffer.append(this.label);
278          if (this.feature != null)
279            buffer.append("-" + this.feature);
280          if (this.preposition != null)
281            buffer.append("-" + this.preposition);
282    
283          return buffer.toString();
284        }
