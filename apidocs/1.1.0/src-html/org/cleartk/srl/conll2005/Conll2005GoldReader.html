024    package org.cleartk.srl.conll2005;

027    import java.io.File;
028    import java.io.FileInputStream;

030    import java.io.InputStream;

032    import java.util.zip.GZIPInputStream;

055    @SofaCapability(outputSofas = { Conll2005Constants.CONLL_2005_VIEW, ViewURIUtil.URI })
056    public class Conll2005GoldReader extends JCasCollectionReader_ImplBase {
057    
058      public static CollectionReader getCollectionReader(String conll2005DataFile)
059          throws ResourceInitializationException {
060        return CollectionReaderFactory.createCollectionReader(
061            Conll2005GoldReader.class,
062            SrlComponents.TYPE_SYSTEM_DESCRIPTION,
063            PARAM_CONLL2005_DATA_FILE,
064            conll2005DataFile);
065      }
066    
067      @ConfigurationParameter(mandatory = true, description = "the path of the CoNLL 2005 data file")
068      private File conll2005DataFile;
069    
070      public static final String PARAM_CONLL2005_DATA_FILE = ConfigurationParameterFactory
071          .createConfigurationParameterName(Conll2005GoldReader.class, "conll2005DataFile");
072    
073      private BufferedReader reader;
074    
075      private boolean finished = false;
076    
077      private int documentNumber;
078    
079      private int totalDocuments;
080    
081      @Override
082      public void initialize(UimaContext context) throws ResourceInitializationException {
083        try {
084          this.reader = this.getBufferedReader();
085          String line;
086          this.totalDocuments = 0;
087          do {
088            line = this.reader.readLine();
089            while (line != null && line.trim().length() == 0) {
090              line = this.reader.readLine();
091            }
092            if (line == null) {
093              break;
094            }
095            this.totalDocuments += 1;
096            while (line != null && line.trim().length() > 0) {
097              line = this.reader.readLine();
098            }
099          } while (line != null);
100          this.reader.close();
101    
102          this.reader = this.getBufferedReader();
103          documentNumber = 0;
104    
105        } catch (IOException e) {
106          throw new ResourceInitializationException(e);
107        }
108      }
109    
110      private BufferedReader getBufferedReader() throws IOException {
111        InputStream in;
112        if (this.conll2005DataFile.getName().endsWith(".gz")) {
113          in = new GZIPInputStream(new FileInputStream(this.conll2005DataFile));
114        } else {
115          in = new FileInputStream(this.conll2005DataFile);
116        }
117        return new BufferedReader(new InputStreamReader(in));
118      }
119    
120      public void getNext(JCas jCas) throws IOException, CollectionException {
121        try {
122          JCas conllView = jCas.createView(Conll2005Constants.CONLL_2005_VIEW);
123    
124          String lineBuffer;
125          StringBuffer docBuffer = new StringBuffer();
126    
127          lineBuffer = reader.readLine();
128          while (lineBuffer != null && lineBuffer.trim().length() == 0) {
129            lineBuffer = reader.readLine();
130          }
131    
132          if (lineBuffer == null) {
133            throw new CollectionException("unexpected end of input", null);
134          }
135    
136          while (lineBuffer != null && lineBuffer.trim().length() != 0) {
137            docBuffer.append(lineBuffer.trim());
138            docBuffer.append("\n");
139            lineBuffer = reader.readLine();
140          }
141    
142          documentNumber += 1;
143    
144          if (documentNumber == totalDocuments) {
145            finished = true;
146          }
147    
148          conllView.setSofaDataString(docBuffer.toString(), "text/plain");
149          ViewURIUtil.setURI(jCas, new File(String.valueOf(documentNumber)).toURI());
150        } catch (CASException e) {
151          throw new CollectionException(e);
152        }
153      }
154    
155      public void close() throws IOException {
156        reader.close();
157      }
158    
159      public Progress[] getProgress() {
160        return new Progress[] { new ProgressImpl(documentNumber, totalDocuments, Progress.ENTITIES) };
161      }
162    
163      public boolean hasNext() throws IOException, CollectionException {
164        return !finished;
165      }
166    }
