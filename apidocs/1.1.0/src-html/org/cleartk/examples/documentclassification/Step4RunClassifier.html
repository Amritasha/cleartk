052    public class Step4RunClassifier {
053    
054      public static class Args {
055        @Option(name = "-d", aliases = "--documentDirectory", usage = "specify the directory containing the test documents")
056        public String documentDirectory = "../ClearTK Data/data/20newsgroups/20news-bydate-test/";
057    
058        @Option(name = "-m", aliases = "--modelFileName", usage = "specify the file name of the model file")
059        public String modelFileName = "example/documentclassification/libsvm/model.jar";
060    
061        public static Args parseArguments(String[] stringArgs) {
062          Args args = new Args();
063          CmdLineParser parser = new CmdLineParser(args);
064          try {
065            parser.parseArgument(stringArgs);
066          } catch (CmdLineException e) {
067            e.printStackTrace();
068            parser.printUsage(System.err);
069            System.exit(1);
070          }
071          return args;
072        }
073      }
074    
075      public static void main(String[] stringArgs) throws UIMAException, IOException {
076    
077        Args args = Args.parseArguments(stringArgs);
078        String documentDirectory = args.documentDirectory;
079        String modelFileName = args.modelFileName;
080    
081        AnalysisEngineDescription documentClassificationAnnotatorDescription = AnalysisEngineFactory
082            .createPrimitiveDescription(
083                DocumentClassificationAnnotator.class,
084                ExampleComponents.TYPE_SYSTEM_DESCRIPTION,
085                GenericJarClassifierFactory.PARAM_CLASSIFIER_JAR_PATH,
086                modelFileName);
087    
088        System.out.println("classifying documents located in '" + documentDirectory + "'");
089        SimplePipeline.runPipeline(
090            FilesCollectionReader.getCollectionReader(documentDirectory),
091            SentenceAnnotator.getDescription(),
092            TokenAnnotator.getDescription(),
093            DefaultSnowballStemmer.getDescription("English"),
094            AnalysisEngineFactory.createPrimitiveDescription(
095                GoldAnnotator.class,
096                ExampleComponents.TYPE_SYSTEM_DESCRIPTION),
097            documentClassificationAnnotatorDescription,
098            AnalysisEngineFactory.createPrimitiveDescription(
099                EvaluationAnnotator.class,
100                ExampleComponents.TYPE_SYSTEM_DESCRIPTION));
101      }
102    }
