053    public class Step1BuildIDFMap {
054    
055      public static class Args {
056        @Option(name = "-d", aliases = "--documentDirectory", usage = "specify the directory containing the training documents")
057        public String documentDirectory = "../ClearTK Data/data/20newsgroups/20news-bydate-train/";
058    
059        @Option(name = "-map", aliases = "--idfmapFileName", usage = "specify the directory name where the IDFMap file will be written")
060        public String idfmapDirectoryName = "example/documentclassification";
061    
062        public static Args parseArguments(String[] stringArgs) {
063          Args args = new Args();
064          CmdLineParser parser = new CmdLineParser(args);
065          try {
066            parser.parseArgument(stringArgs);
067          } catch (CmdLineException e) {
068            e.printStackTrace();
069            parser.printUsage(System.err);
070            System.exit(1);
071          }
072          return args;
073        }
074      }
075    
076      public static void main(String[] stringArgs) throws UIMAException, IOException {
077    
078        Args args = Args.parseArguments(stringArgs);
079        String documentDirectory = args.documentDirectory;
080        File idfmapDirectory = new File(args.idfmapDirectoryName);
081        if (!idfmapDirectory.exists()) {
082          idfmapDirectory.mkdirs();
083        }
084    
085        AnalysisEngineDescription documentClassificationAnnotatorDescription = AnalysisEngineFactory
086            .createPrimitiveDescription(
087                DocumentClassificationAnnotator.class,
088                ExampleComponents.TYPE_SYSTEM_DESCRIPTION,
089                CleartkAnnotator.PARAM_DATA_WRITER_FACTORY_CLASS_NAME,
090                IDFMapWriterFactory.class.getName(),
091                IDFMapWriterFactory.PARAM_OUTPUT_DIRECTORY,
092                idfmapDirectory.getPath());
093    
094        SimplePipeline.runPipeline(
095            FilesCollectionReader.getCollectionReader(documentDirectory),
096            SentenceAnnotator.getDescription(),
097            TokenAnnotator.getDescription(),
098            DefaultSnowballStemmer.getDescription("English"),
099            AnalysisEngineFactory.createPrimitiveDescription(
100                GoldAnnotator.class,
101                ExampleComponents.TYPE_SYSTEM_DESCRIPTION),
102            documentClassificationAnnotatorDescription);
103    
104        System.out.println("IDF map file written to: " + idfmapDirectory.getPath());
105      }
106    }
