025    package org.cleartk.token.stem.snowball;
026    
027    import org.apache.uima.UimaContext;
028    import org.apache.uima.analysis_engine.AnalysisEngineDescription;
029    import org.apache.uima.resource.ResourceInitializationException;
030    import org.apache.uima.resource.metadata.TypeSystemDescription;
031    import org.cleartk.token.type.Token;
032    import org.uimafit.factory.AnalysisEngineFactory;
033    import org.uimafit.factory.TypeSystemDescriptionFactory;
034    
035    /**
036     * <br>
037     * Copyright (c) 2009, Regents of the University of Colorado <br>
038     * All rights reserved.
039     * 
040     * 
041     * @author Philip Ogren
042     * 
043     * 
044     *         This class borrows from org.apache.lucene.analysis.snowball.SnowballFilter
045     * @see org.apache.lucene.analysis.snowball.SnowballFilter
046     */
047    
048    public class DefaultSnowballStemmer extends SnowballStemmer<Token> {
049    
050      public static TypeSystemDescription TYPE_SYSTEM_DESCRIPTION = TypeSystemDescriptionFactory.createTypeSystemDescription("org.cleartk.token.TypeSystem");
051    
052      public static AnalysisEngineDescription getDescription(String language)
053          throws ResourceInitializationException {
054        return AnalysisEngineFactory.createPrimitiveDescription(
055            DefaultSnowballStemmer.class,
056            TYPE_SYSTEM_DESCRIPTION,
057            SnowballStemmer.PARAM_STEMMER_NAME,
058            language);
059      }
060    
061      @Override
062      public void initialize(UimaContext context) throws ResourceInitializationException {
063        super.initialize(context);
064      }
065    
066      @Override
067      public void setStem(Token token, String stem) {
068        token.setStem(stem);
069      }
070    
071    }
