025    package org.cleartk.token.breakit;
026    
027    import java.util.Locale;
028    
029    import org.apache.uima.analysis_engine.AnalysisEngineDescription;
030    import org.apache.uima.resource.ResourceInitializationException;
031    import org.cleartk.token.TokenComponents;
032    import org.cleartk.token.type.Sentence;
033    import org.cleartk.token.type.Token;
034    import org.uimafit.factory.AnalysisEngineFactory;
035    
036    /**
037     * <br>
038     * Copyright (c) 2010, Regents of the University of Colorado <br>
039     * All rights reserved.
040     * <p>
041     * 
042     * @author Philip Ogren
043     */
044    
045    public class BreakIteratorAnnotatorFactory {
046    
047      public static AnalysisEngineDescription createSentenceAnnotator(Locale locale)
048          throws ResourceInitializationException {
049        return AnalysisEngineFactory.createPrimitiveDescription(
050            BreakIteratorAnnotator.class,
051            TokenComponents.TYPE_SYSTEM_DESCRIPTION,
052            BreakIteratorAnnotator.PARAM_ANNOTATION_TYPE_NAME,
053            Sentence.class.getName(),
054            BreakIteratorAnnotator.PARAM_LOCALE,
055            locale.toString());
056      }
057    
058      public static AnalysisEngineDescription createTokenAnnotator(Locale locale)
059          throws ResourceInitializationException {
060        return AnalysisEngineFactory.createPrimitiveDescription(
061            BreakIteratorAnnotator.class,
062            TokenComponents.TYPE_SYSTEM_DESCRIPTION,
063            BreakIteratorAnnotator.PARAM_BREAK_ITERATOR_TYPE,
064            "WORD",
065            BreakIteratorAnnotator.PARAM_ANNOTATION_TYPE_NAME,
066            Token.class.getName(),
067            BreakIteratorAnnotator.PARAM_LOCALE,
068            locale.toString());
069      }
070    
071    }
