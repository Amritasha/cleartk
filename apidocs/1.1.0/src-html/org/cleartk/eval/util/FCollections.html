024    package org.cleartk.eval.util;
025    
026    import java.util.ArrayList;
027    import java.util.Collections;
028    import java.util.HashSet;
029    import java.util.List;
030    import java.util.Set;
031    
032    /**
033     * This data structure is convenient if you want to stratify evaluation across different dimensions.
034     * For example, if you wanted to see how good a part-of-speech tagger works for different tags you
035     * might declare a FCollection of type String. Another example is a named-entity recognizer for
036     * which you want to measure performance for different lengths of named entities (in tokens). For
037     * this example, you could declare a FCollection of type Integer.
038     * 
039     * <br>
040     * Copyright (c) 2011, Regents of the University of Colorado <br>
041     * All rights reserved.
042     * 
043     * @author Philip Ogren
044     * 
045     * @param <T>
046     */
047    
048    public class FCollections<T extends Comparable<T>> {
049    
050      protected CountCollection<T> truePositives;
051    
052      protected CountCollection<T> falsePositives;
053    
054      protected CountCollection<T> falseNegatives;
055    
056      protected int totalTruePositives = 0;
057    
058      protected int totalFalsePositives = 0;
059    
060      protected int totalFalseNegatives = 0;
061    
062      protected Set<T> objects;
063    
064      public FCollections() {
065        truePositives = new CountCollection<T>();
066        falsePositives = new CountCollection<T>();
067        falseNegatives = new CountCollection<T>();
068        objects = new HashSet<T>();
069      }
070    
071      public void addTruePositive(T truePositive) {
072        truePositives.add(truePositive);
073        objects.add(truePositive);
074        totalTruePositives++;
075      }
076    
077      public void addTruePositives(T truePositive, int countIncrement) {
078        truePositives.add(truePositive, countIncrement);
079        objects.add(truePositive);
080        totalTruePositives += countIncrement;
081      }
082    
083      public int getTruePositivesCount(T object) {
084        return truePositives.getCount(object);
085      }
086    
087      public int getTruePositivesCount() {
088        return totalTruePositives;
089      }
090    
091      public void addFalsePositive(T falsePositive) {
092        falsePositives.add(falsePositive);
093        objects.add(falsePositive);
094        totalFalsePositives++;
095      }
096    
097      public void addFalsePositives(T falsePositive, int countIncrement) {
098        falsePositives.add(falsePositive, countIncrement);
099        objects.add(falsePositive);
100        totalFalsePositives += countIncrement;
101      }
102    
103      public int getFalsePositivesCount(T object) {
104        return falsePositives.getCount(object);
105      }
106    
107      public int getFalsePositivesCount() {
108        return totalFalsePositives;
109      }
110    
111      public void addFalseNegative(T falseNegative) {
112        falseNegatives.add(falseNegative);
113        objects.add(falseNegative);
114        totalFalseNegatives++;
115      }
116    
117      public void addFalseNegatives(T falseNegative, int countIncrement) {
118        falseNegatives.add(falseNegative, countIncrement);
119        objects.add(falseNegative);
120        totalFalseNegatives += countIncrement;
121      }
122    
123      public int getFalseNegativesCount(T object) {
124        return falseNegatives.getCount(object);
125      }
126    
127      public int getFalseNegativesCount() {
128        return totalFalseNegatives;
129      }
130    
131      public List<T> getSortedObjects() {
132        List<T> sortedObjects = new ArrayList<T>(objects);
133        Collections.sort(sortedObjects);
134        return sortedObjects;
135      }
136    
137      public double getPrecision(T object) {
138        return precision(getTruePositivesCount(object), getFalsePositivesCount(object));
139      }
140    
141      public double getPrecision() {
142        return precision(getTruePositivesCount(), getFalsePositivesCount());
143      }
144    
145      public static double precision(int tps, int fps) {
146        return (double) tps / (tps + fps);
147      }
148    
149      public double getRecall(T object) {
150        return recall(getTruePositivesCount(object), getFalseNegativesCount(object));
151      }
152    
153      public double getRecall() {
154        return recall(getTruePositivesCount(), getFalseNegativesCount());
155      }
156    
157      public static double recall(int tps, int fns) {
158        return (double) tps / (tps + fns);
159      }
160    
161      public double getF(T object) {
162        return F(
163            getTruePositivesCount(object),
164            getFalsePositivesCount(object),
165            getFalseNegativesCount(object));
166      }
167    
168      public double getF() {
169        return F(getTruePositivesCount(), getFalsePositivesCount(), getFalseNegativesCount());
170      }
171    
172      /**
173       * When the beta weight is .5, F-measure works out to be a very simple ratio if you work out the
174       * algebra (which I did once.) This result is published in the literature here:
175       * 
176       * http://www.ncbi.nlm.nih.gov/pmc/articles/PMC1090460/
177       * 
178       * See equation 3.
179       * 
180       * @param object
181       * @return
182       */
183    
184      public static double F(int tps, int fps, int fns) {
185        return (double) (2 * tps) / (2 * tps + fps + fns);
186      }
187    
188      public static double F(int tps, int fps, int fns, double B) {
189        double precision = precision(tps, fps);
190        double recall = recall(tps, fns);
191        double numerator = (1.0d + B * B) * recall * precision;
192        double denominator = ((B * B) * precision) + recall;
193        return numerator / denominator;
194      }
195    
196      public double getF(T object, double B) {
197        return F(
198            getTruePositivesCount(object),
199            getFalsePositivesCount(object),
200            getFalseNegativesCount(object),
201            B);
202      }
203    
204    }
