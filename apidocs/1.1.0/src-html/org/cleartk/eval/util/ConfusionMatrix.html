024    package org.cleartk.eval.util;
025    
026    import java.util.HashMap;
027    import java.util.Map;
028    import java.util.SortedSet;
029    import java.util.TreeSet;
030    
031    /**
032     * This data structure provides an easy way to build and output a confusion matrix. A confusion
033     * matrix is a two dimensional table with a row and table for each class. Each element in the matrix
034     * shows the number of test examples for which the actual class is the row and the predicted class
035     * is the column. Display of this matrix is useful for identifying when a system is confusing two
036     * classes
037     * 
038     * <br>
039     * For more info @see <a href="http://en.wikipedia.org/wiki/Confusion_matrix">The wikipedia page on
040     * Confusion Matrices</a>
041     * 
042     * <br>
043     * Copyright (c) 2011, Regents of the University of Colorado <br>
044     * All rights reserved.
045     * 
046     * @author Lee Becker
047     * 
048     * @param <T>
049     *          The data type used to represent the class labels
050     */
051    
052    public class ConfusionMatrix<T extends Comparable<T>> {
053      private Map<T, CountCollection<T>> matrix;
054    
055      private SortedSet<T> classes;
056    
057      /**
058       * Creates an empty confusion Matrix
059       */
060      public ConfusionMatrix() {
061        this.matrix = new HashMap<T, CountCollection<T>>();
062        this.classes = new TreeSet<T>();
063      }
064    
065      /**
066       * Creates a new ConfusionMatrix initialized with the contents of another ConfusionMatrix.
067       * 
068       * @param other
069       */
070      public ConfusionMatrix(ConfusionMatrix<T> other) {
071        this();
072        this.add(other);
073      }
074    
075      /**
076       * Increments the entry specified by actual and predicted by one.
077       * 
078       * @param actual
079       * @param predicted
080       */
081      public void add(T actual, T predicted) {
082        add(actual, predicted, 1);
083      }
084    
085      /**
086       * Increments the entry specified by actual and predicted by count.
087       * 
088       * @param actual
089       * @param predicted
090       * @param count
091       */
092      public void add(T actual, T predicted, int count) {
093        if (matrix.containsKey(actual)) {
094          matrix.get(actual).add(predicted, count);
095        } else {
096          CountCollection<T> counts = new CountCollection<T>();
097          counts.add(predicted, count);
098          matrix.put(actual, counts);
099        }
100    
101        classes.add(actual);
102        classes.add(predicted);
103      }
104    
105      /**
106       * Adds the entries from another confusion matrix to this one.
107       * 
108       * @param other
109       */
110      public void add(ConfusionMatrix<T> other) {
111        for (T actual : other.matrix.keySet()) {
112          CountCollection<T> counts = other.matrix.get(actual);
113          for (T predicted : counts.getCountedObjects()) {
114            int count = counts.getCount(predicted);
115            this.add(actual, predicted, count);
116          }
117        }
118      }
119    
120      /**
121       * Gives the set of all classes in the confusion matrix.
122       * 
123       * @return
124       */
125      public SortedSet<T> getClasses() {
126        return classes;
127      }
128    
129      /**
130       * Gives the count of the number of times the "predicted" class was predicted for the "actual"
131       * class.
132       * 
133       * @param actual
134       * @param predicted
135       * @return
136       */
137      public int getCount(T actual, T predicted) {
138        if (!matrix.containsKey(actual)) {
139          return 0;
140        } else {
141          return matrix.get(actual).getCount(predicted);
142        }
143      }
144    
145      /**
146       * Computes the total number of times the class was predicted by the classifier.
147       * 
148       * @param predicted
149       * @return
150       */
151      public int getPredictedTotal(T predicted) {
152        int total = 0;
153        for (T actual : classes) {
154          total += getCount(actual, predicted);
155        }
156        return total;
157      }
158    
159      /**
160       * Computes the total number of times the class actually appeared in the data.
161       * 
162       * @param actual
163       * @return
164       */
165      public int getActualTotal(T actual) {
166        if (!matrix.containsKey(actual)) {
167          return 0;
168        } else {
169          int total = 0;
170          for (CountObject<T> countObj : matrix.get(actual).getCounts()) {
171            total += countObj.getCount();
172          }
173          return total;
174        }
175      }
176    
177      /**
178       * Outputs the ConfusionMatrix as comma-separated values for easy import into spreadsheets
179       * 
180       * @return
181       */
182      public String toCSV() {
183        StringBuilder builder = new StringBuilder();
184    
185        // Header Row
186        builder.append(",,Predicted Class,\n");
187    
188        // Predicted Classes Header Row
189        builder.append(",,");
190        for (T predicted : classes) {
191          builder.append(String.format("%s,", predicted.toString()));
192        }
193        builder.append("Total\n");
194    
195        // Data Rows
196        String firstColumnLabel = "Actual Class,";
197        for (T actual : classes) {
198          builder.append(firstColumnLabel);
199          firstColumnLabel = ",";
200          builder.append(String.format("%s,", actual.toString()));
201    
202          for (T predicted : classes) {
203            builder.append(getCount(actual, predicted));
204            builder.append(",");
205          }
206          // Actual Class Totals Column
207          builder.append(getActualTotal(actual));
208          builder.append("\n");
209        }
210    
211        // Predicted Class Totals Row
212        builder.append(",Total,");
213        for (T predicted : classes) {
214          builder.append(getPredictedTotal(predicted));
215          builder.append(",");
216        }
217        builder.append("\n");
218    
219        return builder.toString();
220      }
221    
222      /**
223       * Outputs Confusion Matrix in an HTML table. Cascading Style Sheets (CSS) can control the table's
224       * appearance by defining the empty-space, actual-count-header, predicted-class-header, and
225       * count-element classes. For example
226       * 
227       * @return html string
228       */
229      public String toHTML() {
230        StringBuilder builder = new StringBuilder();
231    
232        int numClasses = classes.size();
233        // Header Row
234        builder.append("<table>\n");
235        builder.append("<tr><th class=\"empty-space\" colspan=\"2\" rowspan=\"2\">");
236        builder.append(String.format(
237            "<th class=\"predicted-class-header\" colspan=\"%d\">Predicted Class</th></tr>\n",
238            numClasses + 1));
239    
240        // Predicted Classes Header Row
241        builder.append("<tr>");
242        // builder.append("<th></th><th></th>");
243        for (T predicted : classes) {
244          builder.append("<th class=\"predicted-class-header\">");
245          builder.append(predicted.toString());
246          builder.append("</th>");
247        }
248        builder.append("<th class=\"predicted-class-header\">Total</th>");
249        builder.append("</tr>\n");
250    
251        // Data Rows
252        String firstColumnLabel = String.format(
253            "<tr><th class=\"actual-class-header\" rowspan=\"%d\">Actual Class</th>",
254            numClasses + 1);
255        for (T actual : classes) {
256          builder.append(firstColumnLabel);
257          firstColumnLabel = "<tr>";
258          builder
259              .append(String.format("<th class=\"actual-class-header\" >%s</th>", actual.toString()));
260    
261          for (T predicted : classes) {
262            builder.append("<td class=\"count-element\">");
263            builder.append(getCount(actual, predicted));
264            builder.append("</td>");
265          }
266    
267          // Actual Class Totals Column
268          builder.append("<td class=\"count-element\">");
269          builder.append(getActualTotal(actual));
270          builder.append("</td>");
271          builder.append("</tr>\n");
272        }
273    
274        // Predicted Class Totals Row
275        builder.append("<tr><th class=\"actual-class-header\">Total</th>");
276        for (T predicted : classes) {
277          builder.append("<td class=\"count-element\">");
278          builder.append(getPredictedTotal(predicted));
279          builder.append("</td>");
280        }
281        builder.append("<td class=\"empty-space\"></td>\n");
282        builder.append("</tr>\n");
283        builder.append("</table>\n");
284    
285        return builder.toString();
286      }
287    
288      public static void main(String[] args) {
289        ConfusionMatrix<String> confusionMatrix = new ConfusionMatrix<String>();
290    
291        confusionMatrix.add("a", "a", 88);
292        confusionMatrix.add("a", "b", 10);
293        // confusionMatrix.add("a", "c", 2);
294        confusionMatrix.add("b", "a", 14);
295        confusionMatrix.add("b", "b", 40);
296        confusionMatrix.add("b", "c", 6);
297        confusionMatrix.add("c", "a", 18);
298        confusionMatrix.add("c", "b", 10);
299        confusionMatrix.add("c", "c", 12);
300    
301        ConfusionMatrix<String> confusionMatrix2 = new ConfusionMatrix<String>(confusionMatrix);
302        confusionMatrix2.add(confusionMatrix);
303        System.out.println(confusionMatrix2.toHTML());
304        System.out.println(confusionMatrix2.toCSV());
305      }
