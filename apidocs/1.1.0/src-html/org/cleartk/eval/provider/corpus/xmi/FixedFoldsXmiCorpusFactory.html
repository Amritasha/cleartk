025    package org.cleartk.eval.provider.corpus.xmi;
026    
027    import java.io.File;
028    import java.io.IOException;
029    import java.util.ArrayList;
030    import java.util.Arrays;
031    import java.util.List;
032    
033    import org.apache.uima.pear.util.FileUtil;
034    import org.apache.uima.resource.ResourceInitializationException;
035    import org.apache.uima.resource.metadata.TypeSystemDescription;
036    import org.cleartk.eval.provider.CorpusReaderProvider;
037    
038    /**
039     * Copyright (c) 2011, Regents of the University of Colorado <br>
040     * All rights reserved. <br>
041     * This extension of {@link XmiCorpusFactory} assumes that the folds for a corpus are fixed. The
042     * folds are specified files that contain lists of file names (one file name per line.) So, for
043     * example, {@link #getTrainFile()} might return a file named "<your-path>/train.txt" that contains
044     * a list of those XMI files that belong in the training data set.
045     * <p>
046     * Please see {@link CorpusReaderProvider} for a description, broadly speaking, of how you should
047     * set up your corpus. Here, this is implemented by simply creating files that contain file names
048     * for the different sets and folds. The names of the files that you should create might be:
049     * <ul>
050     * <li>fold-1.txt - consists of the xmi file names for the test set for fold 1</li>
051     * <li>fold-2.txt</li>
052     * <li>...</li>
053     * <li>fold-n.txt</li>
054     * <li>test.txt - consists of the xmi file names for the test (i.e. holdout evaluation) set.</li>
055     * <li>train.txt - consists of the xmi file names for the training set</li>
056     * </ul>
057     * 
058     * For the list above, the method {@link #getFoldFiles()} would return the paths for the files
059     * "fold-1.txt" ... "fold-n.txt", {@link #getTrainFile()} would return the path to the file
060     * "train.txt" and {@link #getTestFile()} would return the path to the file "test.txt".
061     * 
062     * @author Philip Ogren
063     * 
064     */
065    
066    public abstract class FixedFoldsXmiCorpusFactory extends XmiCorpusFactory {
067    
068      public FixedFoldsXmiCorpusFactory(
069          TypeSystemDescription typeSystemDescription,
070          String xmiDirectoryName) {
071        super(typeSystemDescription, xmiDirectoryName);
072      }
073    
074      /**
075       * 
076       * @return a list of path names to the files that contain lists of file names for each fold.
077       */
078      public abstract List<String> getFoldFiles();
079    
080      /**
081       * @return a path name to the file that contains a list of the file names for the training set.
082       */
083      public abstract String getTrainFile();
084    
085      /**
086       * @return a path name to the file that contains a list of the file names for the test set.
087       */
088      public abstract String getTestFile();
089    
090      @Override
091      public int getNumberOfFolds() {
092        return getFoldFiles().size();
093      }
094    
095      @Override
096      public void setNumberOfFolds(int numberOfFolds) {
097        throw new UnsupportedOperationException(
098            "the the number of folds for this corpus factory is fixed to " + getFoldFiles().size());
099      }
100    
101      @Override
102      public String[] getTrainNames(int fold) throws ResourceInitializationException {
103        verifyFoldValue(fold);
104        List<String> foldFiles = new ArrayList<String>(getFoldFiles());
105        foldFiles.remove(fold);
106    
107        try {
108    
109          List<String> fileNames = new ArrayList<String>();
110          for (String foldFile : foldFiles) {
111            String[] names;
112            names = FileUtil.loadListOfStrings(new File(foldFile));
113            fileNames.addAll(Arrays.asList(names));
114          }
115          return fileNames.toArray(new String[fileNames.size()]);
116    
117        } catch (IOException ioe) {
118          throw new ResourceInitializationException(ioe);
119        }
120    
121      }
122    
123      @Override
124      public String[] getTestNames(int fold) throws ResourceInitializationException {
125        verifyFoldValue(fold);
126        return loadNames(getFoldFiles().get(fold));
127      }
128    
129      @Override
130      public String[] getTrainNames() throws ResourceInitializationException {
131        return loadNames(getTrainFile());
132      }
133    
134      @Override
135      public String[] getTestNames() throws ResourceInitializationException {
136        return loadNames(getTestFile());
137      }
138    
139      private String[] loadNames(String fileName) throws ResourceInitializationException {
140        try {
141          return FileUtil.loadListOfStrings(new File(fileName));
142        } catch (IOException ioe) {
143          throw new ResourceInitializationException(ioe);
144        }
145    
146      }
147    }
