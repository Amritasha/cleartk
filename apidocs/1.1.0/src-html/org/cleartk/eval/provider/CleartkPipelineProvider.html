024    package org.cleartk.eval.provider;
025    
026    import java.util.List;
027    
028    import org.apache.uima.UIMAException;
029    import org.apache.uima.analysis_engine.AnalysisEngine;
030    import org.cleartk.classifier.ClassifierFactory;
031    import org.cleartk.classifier.CleartkAnnotator;
032    import org.cleartk.classifier.DataWriterFactory;
033    
034    /**
035     * <br>
036     * Copyright (c) 2011, Regents of the University of Colorado <br>
037     * All rights reserved. <br>
038     * 
039     * @author Philip Ogren
040     * @author Steven Bethard
041     */
042    public interface CleartkPipelineProvider {
043    
044      /**
045       * This method creates a pipeline of analysis engines that will create a training data for the
046       * learner being used. This will generally include a {@link CleartkAnnotator} that is initialized
047       * with a {@link DataWriterFactory}. However, a number of other analysis engines may also be
048       * required depending on what data from the gold-standard corpus is provided and what needs to be
049       * generated. This aggregate analysis engine should generally operate on the
050       * {@link ViewNames#GOLD_VIEW}.
051       * 
052       * @param name
053       *          will generally be a value such as 'fold-01' or 'holdout'. It provides a name for this
054       *          method to provide to the pipeline it returns that serves as a key for that pipeline.
055       *          One expected use of the name is for naming a subdirectory that a data writer could use
056       *          to write training data to.
057       */
058      public List<AnalysisEngine> getTrainingPipeline(String name) throws UIMAException;
059    
060      public void trainingPipelineComplete(String name, List<AnalysisEngine> engines)
061          throws UIMAException;
062    
063      public void trainingComplete() throws UIMAException;
064    
065      /**
066       * This method trains a model(s) for a given training pipeline (identified by name) using whatever
067       * training data was generated as a result of running the training pipeline. If your task involves
068       * only a single {@link CleartkAnnotator} which runs off of a single classification model, then
069       * this method will generally consist of a single line of code. However, it is not uncommmon for a
070       * task to require many models. In such cases, this method needs to know how to train each model
071       * that needs to be trained.
072       * 
073       * @param name
074       *          will generally be a value such as 'fold-01' or 'holdout'. It provides a name for this
075       *          method for the pipeline whose data we are training a model(s) for. One expected use of
076       *          the name is to identify a subdirectory where training data is located.
077       * @param trainingArguments
078       *          arguments that are passed on to the classifiers model trainer.
079       */
080      public void train(String name, String... trainingArguments) throws Exception;
081    
082      /**
083       * This method creates a pipeline of analysis engines that will run the task using the model(s)
084       * generated by {@link #train(String, String...)}. This will generally include a
085       * {@link CleartkAnnotator} that is initialized with a {@link ClassifierFactory}. However, a
086       * number of other analysis engines may also be required depending on what data from the
087       * gold-standard corpus is provided and what needs to be generated. This pipeline of analysis
088       * engines should generally operate on the {@link ViewNames#SYSTEM_VIEW}.
089       */
090      public List<AnalysisEngine> getClassifyingPipeline(String name) throws UIMAException;
091    
092      public void classifyingPipelineComplete(String name, List<AnalysisEngine> engines)
093          throws UIMAException;
094    
095      public void classifyingComplete() throws UIMAException;
096    }
