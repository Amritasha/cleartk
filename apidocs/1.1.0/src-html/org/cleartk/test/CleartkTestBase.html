062      /**
063       * The name of the property that can be set to skip some tests. The property value will be parsed
064       * as a comma-separated list of values, where each value can designate one type of test to skip,
065       * e.g. {@value #LONG_TESTS_PROPERTY_VALUE} or {@value #BIG_MEMORY_TESTS_PROPERTY_VALUE}. Example
066       * usage: <br/>
067       * <code>-Dcleartk.skipTests=long,bigMem</code>
068       */
069      public static final String SKIP_TESTS_PROPERTY = "cleartk.skipTests";

071      /**

132      private boolean skipLongTests;
133    
134      private boolean skipBigMemoryTests;
135    
136      private boolean skipTkSvmLightTests;
137    
138      private boolean skipSvmLightTests;
139    
140      private boolean skipLibsvmTests;
141    
142      private boolean skipLiblinearTests;
143    
144      public CleartkTestBase() {
145        super();
146        this.logger = Logger.getLogger(this.getClass().getName());
147        String skipTests = System.getProperty(SKIP_TESTS_PROPERTY);
148        if (skipTests != null) {
149          for (String skip : skipTests.split("\\s*[,]\\s*")) {
150            if (skip.equals(LONG_TESTS_PROPERTY_VALUE)) {
151              this.skipLongTests = true;
152            } else if (skip.equals(BIG_MEMORY_TESTS_PROPERTY_VALUE)) {
153              this.skipBigMemoryTests = true;
154            } else if (skip.equals(TK_SVMLIGHT_TESTS_PROPERTY_VALUE)) {
155              this.skipTkSvmLightTests = true;
156            } else if (skip.equals(SVMLIGHT_TESTS_PROPERTY_VALUE)) {
157              this.skipSvmLightTests = true;
158            } else if(skip.equals(LIBSVM_TESTS_PROPERTY_VALUE)) {
159              this.skipLibsvmTests = true;
160            } else if(skip.equals(LIBLINEAR_TESTS_PROPERTY_VALUE)) {
161              this.skipLiblinearTests = true;
162            }
163            else if (skip.equals(COMMON_TESTS_PROPERTY_VALUE)) {
164              this.skipLongTests = true;
165              this.skipBigMemoryTests = true;
166              this.skipTkSvmLightTests = true;
167            } 
168            else if (skip.equals(ALL_TESTS_PROPERTY_VALUE)) {
169              this.skipLongTests = true;
170              this.skipBigMemoryTests = true;
171              this.skipTkSvmLightTests = true;
172              this.skipSvmLightTests = true;
173              this.skipLibsvmTests = true;
174              this.skipLiblinearTests = true;
175            } 
176            else {
177              throw new IllegalArgumentException(String.format(
178                  "expected %s to be one of [%s, %s, %s, %s, %s, %s, %s, %s], found %s",
179                  SKIP_TESTS_PROPERTY,
180                  LONG_TESTS_PROPERTY_VALUE,
181                  BIG_MEMORY_TESTS_PROPERTY_VALUE,
182                  TK_SVMLIGHT_TESTS_PROPERTY_VALUE,
183                  SVMLIGHT_TESTS_PROPERTY_VALUE,
184                  LIBSVM_TESTS_PROPERTY_VALUE,
185                  LIBLINEAR_TESTS_PROPERTY_VALUE,
186                  COMMON_TESTS_PROPERTY_VALUE,
187                  ALL_TESTS_PROPERTY_VALUE,
188                  skip));
189            }
190          }
191        }
192      }

194      /**
195       * Subclasses should call this method at the beginning of a test that will take a long time to
196       * run. Immediately after calling this method, they should also call
197       * <code>this.logger.info({@link #LONG_TEST_MESSAGE})</code>.
198       */
199      protected void assumeLongTestsEnabled() {
200        // note that we can't log the message here as well, or it the log will display the wrong method
201        Assume.assumeTrue(!this.skipLongTests);
202      }

300      public int getCollectionReaderCount(CollectionReader reader) throws UIMAException, IOException {
301    
302        AnalysisEngine aeAdapter = AnalysisEngineFactory.createPrimitive(
303            NoOpAnnotator.class,
304            typeSystemDescription);
305    
306        int count = 0;
307        JCasIterable jCases = new JCasIterable(reader, aeAdapter);
308        for (@SuppressWarnings("unused")
309        JCas jcs : jCases) {
310          count++;
311        }
312        return count;
313      }
314    
315      public void testCollectionReaderCount(CollectionReader reader, int expectedCount)
316          throws UIMAException, IOException {
317        assertEquals(expectedCount, getCollectionReaderCount(reader));
318      }
