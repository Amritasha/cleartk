053    public class ParentheticalAnnotator extends JCasAnnotator_ImplBase {
054    
055      public static final String PARAM_WINDOW_TYPE_NAME = ConfigurationParameterFactory.createConfigurationParameterName(
056          ParentheticalAnnotator.class,
057          "windowTypeName");
058    
059      private static final String WINDOW_TYPE_DESCRIPTION = "specifies the class type of annotations that will be tokenized. "
060          + "If no value is given, then the entire document will be tokenized at once. ";
061    
062      // do not set the default value to 'org.cleartk.token.type.Sentence'. If you do, then unit tests
063      // will break. The symptom will be a tokenizer that doesn't generate any tokens (because there
064      // are no sentences to iterate over.
065      @ConfigurationParameter(description = WINDOW_TYPE_DESCRIPTION)
066      private String windowTypeName;
067    
068      public static final String PARAM_PARENTHETICAL_TYPE_NAME = ConfigurationParameterFactory.createConfigurationParameterName(
069          ParentheticalAnnotator.class,
070          "parentheticalTypeName");
071    
072      @ConfigurationParameter(
073          description = "class name of the annotations that are created by this annotator.",
074          defaultValue = "org.cleartk.util.type.Parenthetical",
075          mandatory = true)
076      private String parentheticalTypeName;
077    
078      public static final String PARAM_LEFT_PARENTHESIS = ConfigurationParameterFactory.createConfigurationParameterName(
079          ParentheticalAnnotator.class,
080          "leftParenthesis");
081    
082      @ConfigurationParameter(defaultValue = "(", mandatory = true)
083      private String leftParenthesis;
084    
085      private char leftParen;
086    
087      public static final String PARAM_RIGHT_PARENTHESIS = ConfigurationParameterFactory.createConfigurationParameterName(
088          ParentheticalAnnotator.class,
089          "rightParenthesis");
090    
091      @ConfigurationParameter(defaultValue = ")", mandatory = true)
092      private String rightParenthesis;
093    
094      private char rightParen;
095    
096      private Class<? extends Annotation> windowClass;
097    
098      private Constructor<? extends Annotation> parentheticalConstructor;
099    
100      public void initialize(UimaContext uimaContext) throws ResourceInitializationException {
101        super.initialize(uimaContext);
102        if (windowTypeName != null)
103          windowClass = InitializableFactory.getClass(windowTypeName, Annotation.class);
104    
105        if (leftParenthesis.length() != 1) {
106          throw CleartkInitializationException.notSingleCharacter(
107              PARAM_LEFT_PARENTHESIS,
108              leftParenthesis);
109        }
110        leftParen = leftParenthesis.charAt(0);
111    
112        if (rightParenthesis.length() != 1) {
113          throw CleartkInitializationException.notSingleCharacter(
114              PARAM_RIGHT_PARENTHESIS,
115              rightParenthesis);
116        }
117        rightParen = rightParenthesis.charAt(0);
118    
119        Class<? extends Annotation> parentheticalClass = InitializableFactory.getClass(
120            parentheticalTypeName,
121            Annotation.class);
122    
123        try {
124          parentheticalConstructor = parentheticalClass.getConstructor(new Class[] {
125              JCas.class,
126              Integer.TYPE,
127              Integer.TYPE });
128        } catch (Exception e) {
129          throw new ResourceInitializationException(e);
130        }
131      }
132    
133      @Override
134      public void process(JCas jCas) throws AnalysisEngineProcessException {
135        if (windowClass != null) {
136          for (Annotation window : JCasUtil.select(jCas, windowClass)) {
137            String text = window.getCoveredText();
138            createParentheticals(jCas, text, window.getBegin());
139          }
140        } else {
141          String text = jCas.getDocumentText();
142          createParentheticals(jCas, text, 0);
143        }
144      }
145    
146      private void createParentheticals(JCas jCas, String text, int offset)
147          throws AnalysisEngineProcessException {
148        Stack<Integer> leftRoundedParens = new Stack<Integer>();
149        leftRoundedParens.clear();
150        for (int ci = 0; ci < text.length(); ci++) {
151          char c = text.charAt(ci);
152          if (c == leftParen) {
153            leftRoundedParens.push(ci);
154          }
155          if (c == rightParen && !leftRoundedParens.isEmpty()) {
156            int leftOffset = leftRoundedParens.pop();
157            Annotation ann;
158            try {
159              ann = parentheticalConstructor.newInstance(jCas, offset + leftOffset, offset + ci + 1);
160            } catch (Exception e) {
161              throw new AnalysisEngineProcessException(e);
162            }
163            ann.addToIndexes();
164          }
165        }
166      }
167    
168      public static AnalysisEngineDescription getDescription(TypeSystemDescription typeSystemDescription)
169          throws ResourceInitializationException {
170        return getDescription(typeSystemDescription, null);
171      }
172    
173      public static AnalysisEngineDescription getDescription(
174          TypeSystemDescription typeSystemDescription,
175          Class<? extends Annotation> windowClass) throws ResourceInitializationException {
176        return getDescription(typeSystemDescription, windowClass, '(', ')');
177      }
178    
179      public static AnalysisEngineDescription getDescription(
180          TypeSystemDescription typeSystemDescription,
181          Class<? extends Annotation> windowClass,
182          char leftParen,
183          char rightParen) throws ResourceInitializationException {
184        AnalysisEngineDescription aed = AnalysisEngineFactory.createPrimitiveDescription(
185            ParentheticalAnnotator.class,
186            typeSystemDescription,
187            PARAM_LEFT_PARENTHESIS,
188            "" + leftParen,
189            PARAM_RIGHT_PARENTHESIS,
190            "" + rightParen);
191    
192        if (windowClass != null) {
193          ConfigurationParameterFactory.addConfigurationParameters(
194              aed,
195              PARAM_WINDOW_TYPE_NAME,
196              windowClass.getName());
197        }
198    
199        return aed;
200      }
