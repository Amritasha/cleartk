070      public WindowExtractor(
071          Class<? extends Annotation> featureClass,
072          SimpleFeatureExtractor featureExtractor,
073          String windowOrientation,
074          int windowStart,
075          int windowEnd) {
076        this.featureClass = featureClass;
077        this.featureExtractor = featureExtractor;
078        this.windowOrientation = windowOrientation;
079        this.windowStart = windowStart;
080        this.windowEnd = windowEnd;
081      }
082    
083      /**
084       * This feature extractor provides a mechanism for getting features such as:
085       * <ul>
086       * <li>the three words to the left or right of a token</li>
087       * <li>get the three words to the left and right of a constituent bounded by a sentence.</li>
088       * <li>get the words "inside" the constituent</li>
089       * <li>the three pos tags of the three words to the left of a token</li>
090       * <li>the concept ids of the three normalized named entities to the left of a consituent inside a
091       * paragraph/section annotation.</li>
092       * </ul>
093       * 
094       * An example call to this constructor might look something like this: <br>
095       * <code>
096       * <pre>
097       *     new WindowExtractor(Token.class, 
098       *                                                                             new SpannedTextExtractor(),
099       *                                                                             WindowExtractor.ORIENTATION_LEFT,
100       *                                                                             0,3)
101       * 
102       * &#064;param name the name of the extractor
103       * &#064;param featureClass the type of the annotations that we are going to extract a feature from
104       * &#064;param featureExtractor a feature extractor that will extract features from the annotations of type featureType
105       * &#064;param windowOrientation one of ORIENTATION_LEFT, ORIENTATION_RIGHT, ORIENTATION_MIDDLE, ORIENTATION_MIDDLE_REVERSE
106       * &#064;param windowStart the index (inclusive) of the first annotation of type featureType to examine
107       * &#064;param windowEnd the index (exclusive) of the last annotation to examine. 
108       * 
109       * &#064;see WindowFeature
110       * 
111       */
112      public WindowExtractor(
113          String name,
114          Class<? extends Annotation> featureClass,
115          SimpleFeatureExtractor featureExtractor,
116          String windowOrientation,
117          int windowStart,
118          int windowEnd) {
119        this(featureClass, featureExtractor, windowOrientation, windowStart, windowEnd);
120        this.name = name;
121      }
122    
123      /**
124       * VERY SLOW! DO NOT USE!
125       */
126      public List<Feature> extract(
127          JCas jCas,
128          Annotation focusAnnotation,
129          Class<? extends Annotation> cls) throws CleartkExtractorException {
130        int annBegin = focusAnnotation.getBegin();
131        int annEnd = focusAnnotation.getEnd();
132        List<? extends Annotation> covering = JCasUtil.selectCovering(jCas, cls, annBegin, annEnd);
133        Annotation windowAnnotation = covering.size() > 0 ? covering.get(0) : null;
134        return extract(jCas, focusAnnotation, windowAnnotation);
135      }
136    
137      public List<Feature> extract(JCas jCas, Annotation focusAnnotation, Annotation windowAnnotation)
138          throws CleartkExtractorException {
139        if (this.featureType == null)
140          this.featureType = UIMAUtil.getCasType(jCas, this.featureClass);
141    
142        List<Feature> returnValues = new ArrayList<Feature>();
143    
144        Annotation startAnnotation = getStartAnnotation(jCas, focusAnnotation);
145    
146        FSIterator<Annotation> featureAnnotationIterator = jCas.getAnnotationIndex(featureType).iterator();
147    
148        if (startAnnotation != null) {
149          featureAnnotationIterator.moveTo(startAnnotation);
150        }
151    
152        int outOfBoundsDistance = 0;
153    
154        for (int i = 0; i < windowEnd; i++) {
155          // if start annotation is null, then we will mistakenly create a
156          // featureAnnotation for whatever
157          // annotation the iterator starts at. Therefore, we check for null
158          // here.
159          if (outOfBoundsDistance == 0 && startAnnotation != null
160              && featureAnnotationIterator.isValid()) {
161            Annotation featureAnnotation = featureAnnotationIterator.get();
162            if (isWithinBoundaries(featureAnnotation, focusAnnotation, windowAnnotation)) {
163              if (i >= windowStart) {
164                returnValues.addAll(extractWindowedFeatures(jCas, i, featureAnnotation));
165              }
166              moveIterator(featureAnnotationIterator);
167              continue;
168            }
169          }
170          outOfBoundsDistance++;
171          if (i >= windowStart) {
172            Feature feature = new WindowFeature(
173                this.name,
174                null,
175                windowOrientation,
176                i,
177                outOfBoundsDistance);
178            returnValues.add(feature);
179          }
180    
181        }
182        return returnValues;
183      }
184    
185      private boolean isWithinBoundaries(
186          Annotation featureAnnotation,
187          Annotation focusAnnotation,
188          Annotation windowAnnotation) {
189        if (windowOrientation.equals(WindowFeature.ORIENTATION_LEFT)
190            || windowOrientation.equals(WindowFeature.ORIENTATION_RIGHT)) {
191          return AnnotationUtil.contains(windowAnnotation, featureAnnotation);
192        } else
193          return AnnotationUtil.contains(windowAnnotation, featureAnnotation)
194              && AnnotationUtil.contains(focusAnnotation, featureAnnotation);
195      }
196    
197      private void moveIterator(FSIterator<Annotation> windowIterator) {
198        if (windowOrientation.equals(WindowFeature.ORIENTATION_LEFT)
199            || windowOrientation.equals(WindowFeature.ORIENTATION_MIDDLE_REVERSE))
200          windowIterator.moveToPrevious();
201        else if (windowOrientation.equals(WindowFeature.ORIENTATION_RIGHT)
202            || windowOrientation.equals(WindowFeature.ORIENTATION_MIDDLE))
203          windowIterator.moveToNext();
204      }
205    
206      private List<Feature> extractWindowedFeatures(JCas jCas, int i, Annotation annotation)
207          throws CleartkExtractorException {
208        List<Feature> windowedFeatures = featureExtractor.extract(jCas, annotation);
209        List<Feature> returnValues = new ArrayList<Feature>();
210        if (windowedFeatures != null && windowedFeatures.size() > 0) {
211          for (Feature windowedFeature : windowedFeatures) {
212            Feature feature = new WindowFeature(
213                this.name,
214                windowedFeature.getValue(),
215                windowOrientation,
216                i,
217                windowedFeature,
218                0);
219            returnValues.add(feature);
220          }
221          return returnValues;
222        }
223    
224        Feature feature = new WindowFeature(this.name, null, windowOrientation, i, 0);
225        return Collections.singletonList(feature);
226      }

228      /**
229       * This method returns the annotation at index 0 w.r.t. the orientation of the extractor, the
230       * annotation we are extracting features for and the window the provides the boundaries for
231       * feature extraction.
232       * 
233       * @param annotation
234       *          the annotation of focus for which we are extracting features for
235       * @return the annotation at index 0:
236       *         <ul>
237       *         <li>for ORIENTATION_LEFT return the annotation adjacent and to the left of annotation
238       *         <li>for ORIENTATION_RIGHT return the annotation adjacent and to the right of annotation
239       *         <li>for ORIENTATION_MIDDLE return the left most annotation completely contained in
240       *         annotation, null if does not exist
241       *         <li>for ORIENTATION_MIDDLE_REVERSE return the right most annotation completely
242       *         contained in annotation, null if does not exist
243       *         </ul>
244       */
245      public Annotation getStartAnnotation(JCas jCas, Annotation annotation) {
246        return WindowFeature.getStartAnnotation(jCas, featureClass, annotation, windowOrientation);
247      }
