024    package org.cleartk.classifier.feature.proliferate;
025    
026    import java.util.Collections;
027    import java.util.List;
028    
029    import org.cleartk.classifier.Feature;
030    import org.cleartk.classifier.feature.util.CaseUtil;
031    
032    /**
033     * <br>
034     * Copyright (c) 2007-2008, Regents of the University of Colorado <br>
035     * All rights reserved.
036     * 
037     * 
038     * @author Philip Ogren
039     * 
040     */
041    public class CapitalTypeProliferator extends FeatureProliferator {
042    
043      public static final String DEFAULT_NAME = "CapitalType";
044    
045      public static final String ALL_UPPERCASE = "ALL_UPPERCASE";
046    
047      public static final String ALL_LOWERCASE = "ALL_LOWERCASE";
048    
049      public static final String INITIAL_UPPERCASE = "INITIAL_UPPERCASE";
050    
051      public static final String MIXED_CASE = "MIXED_CASE";
052    
053      public CapitalTypeProliferator() {
054        super(CapitalTypeProliferator.DEFAULT_NAME);
055      }
056    
057      public CapitalTypeProliferator(String featureName) {
058        super(featureName);
059      }
060    
061      /**
062       * If the value of the feature is a StringValue and is determined to be one of ALL_UPPERCASE,
063       * ALL_LOWERCASE, INITIAL_UPPERCASE, or MIXED_CASE, then a new feature containing one of those
064       * four values is returned. If the value of the feature cannot be characterized by one of these
065       * four values, then the empty list is returned (e.g. the value is an empty string, contains only
066       * white space, or contains only digits, etc.)
067       * 
068       * <P>
069       * This method was inspired by CapitalizationTypeTagger.py written by Steven Bethard.
070       * 
071       * @return a feature that has a value that is one of ALL_UPPERCASE, ALL_LOWERCASE,
072       *         INITIAL_UPPERCASE, or MIXED_CASE. Otherwise the empty list is returned.
073       */
074      public List<Feature> proliferate(Feature feature) {
075        String featureName = Feature.createName(this.getFeatureName(), feature.getName());
076        Object featureValue = feature.getValue();
077        if (featureValue == null)
078          return Collections.emptyList();
079        else if (featureValue instanceof String) {
080          String value = featureValue.toString();
081          if (value == null || value.length() == 0)
082            return Collections.emptyList();
083    
084          String lowerCaseValue = value.toLowerCase();
085          String upperCaseValue = value.toUpperCase();
086          if (lowerCaseValue.equals(upperCaseValue))
087            return Collections.emptyList();
088    
089          if (value.equals(value.toLowerCase())) {
090            return Collections.singletonList(new Feature(featureName, ALL_LOWERCASE));
091          } else if (value.equals(value.toUpperCase())) {
092            return Collections.singletonList(new Feature(featureName, ALL_UPPERCASE));
093          }
094    
095          if (CaseUtil.isInitialUppercase(value)) {
096            return Collections.singletonList(new Feature(featureName, INITIAL_UPPERCASE));
097          }
098    
099          return Collections.singletonList(new Feature(featureName, MIXED_CASE));
100        } else
101          return Collections.emptyList();
102      }
103    
104    }
