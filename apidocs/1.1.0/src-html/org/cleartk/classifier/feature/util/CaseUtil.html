038    public class CaseUtil {
039    
040      public static boolean isInitialUppercase(String word) {
041        return isInitialUppercase(word, Locale.getDefault());
042      }
043    
044      public static boolean isInitialUppercase(String word, Locale locale) {
045        if (word.length() < 2) {
046          return false;
047        }
048        String firstLetter = word.substring(0, 1);
049        String rest = word.substring(1);
050        // the first condition checks that capitalization matters at all to the first character - e.g.
051        // if the first letter is 1 then the first condition will be true.
052        // the third condition checks that capitalization matters at all to the rest of the word - e.g.
053        // if the rest of the word is 1234 then the first condition will be true.
054        return !firstLetter.equals(firstLetter.toLowerCase(locale))
055            && firstLetter.equals(firstLetter.toUpperCase(locale))
056            && !rest.equals(rest.toUpperCase(locale)) && rest.equals(rest.toLowerCase(locale));
057      }
058    
059      public static boolean isAllUppercase(String word) {
060        return isAllUppercase(word, Locale.getDefault());
061      }
062    
063      public static boolean isAllUppercase(String word, Locale locale) {
064        return !word.equals(word.toLowerCase(locale)) && word.equals(word.toUpperCase(locale));
065      }
066    
067    }
