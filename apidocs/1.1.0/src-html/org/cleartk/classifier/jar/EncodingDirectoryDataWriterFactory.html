024    package org.cleartk.classifier.jar;
025    
026    import java.io.File;
027    import java.io.FileInputStream;
028    import java.io.ObjectInputStream;
029    
030    import org.apache.uima.UimaContext;
031    import org.apache.uima.resource.ResourceInitializationException;
032    import org.cleartk.classifier.DataWriterFactory;
033    import org.cleartk.classifier.SequenceDataWriterFactory;
034    import org.cleartk.classifier.encoder.features.FeaturesEncoder;
035    import org.cleartk.classifier.encoder.features.FeaturesEncoder_ImplBase;
036    import org.cleartk.classifier.encoder.outcome.OutcomeEncoder;
037    import org.cleartk.util.CleartkInitializationException;
038    import org.cleartk.util.ReflectionUtil;
039    import org.uimafit.component.initialize.ConfigurationParameterInitializer;
040    import org.uimafit.descriptor.ConfigurationParameter;
041    import org.uimafit.factory.ConfigurationParameterFactory;
042    import org.uimafit.factory.initializable.Initializable;
043    
044    /**
045     * Superclass for {@link DataWriterFactory} and {@link SequenceDataWriterFactory} implementations
046     * that use {@link FeaturesEncoder}s and {@link OutcomeEncoder}s.
047     * 
048     * Note that it does not declare that it implements either of the DataWriterFactory interfaces.
049     * Subclasses should do this.
050     * 
051     * <br>
052     * Copyright (c) 2011, Regents of the University of Colorado <br>
053     * All rights reserved.
054     * 
055     * @author Steven Bethard
056     * @author Philip Ogren
057     */
058    public abstract class EncodingDirectoryDataWriterFactory<ENCODED_FEATURES_TYPE, OUTCOME_TYPE, ENCODED_OUTCOME_TYPE>
059        extends DirectoryDataWriterFactory implements Initializable {
060    
061      public static final String PARAM_LOAD_ENCODERS_FROM_FILE_SYSTEM = ConfigurationParameterFactory.createConfigurationParameterName(
062          EncodingDirectoryDataWriterFactory.class,
063          "loadEncodersFromFileSystem");
064    
065      @ConfigurationParameter(
066          mandatory = false,
067          description = "when true indicates that the FeaturesEncoder and "
068              + "OutcomeEncoder should be loaded from the file system "
069              + "instead of being created by the DataWriterFactory",
070          defaultValue = "false")
071      private boolean loadEncodersFromFileSystem = false;
072    
073      public void initialize(UimaContext context) throws ResourceInitializationException {
074        super.initialize(context);
075        ConfigurationParameterInitializer.initialize(this, context);
076        if (loadEncodersFromFileSystem) {
077          try {
078            File encoderFile = new File(outputDirectory, FeaturesEncoder_ImplBase.ENCODERS_FILE_NAME);
079    
080            if (!encoderFile.exists()) {
081              throw CleartkInitializationException.fileNotFound(encoderFile);
082            }
083    
084            ObjectInputStream is = new ObjectInputStream(new FileInputStream(encoderFile));
085    
086            // read the FeaturesEncoder and check the types
087            FeaturesEncoder<?> untypedFeaturesEncoder = FeaturesEncoder.class.cast(is.readObject());
088            ReflectionUtil.checkTypeParameterIsAssignable(
089                FeaturesEncoder.class,
090                "ENCODED_FEATURES_TYPE",
091                untypedFeaturesEncoder,
092                EncodingDirectoryDataWriterFactory.class,
093                "ENCODED_FEATURES_TYPE",
094                this);
095    
096            // read the OutcomeEncoder and check the types
097            OutcomeEncoder<?, ?> untypedOutcomeEncoder = OutcomeEncoder.class.cast(is.readObject());
098            ReflectionUtil.checkTypeParameterIsAssignable(
099                OutcomeEncoder.class,
100                "OUTCOME_TYPE",
101                untypedOutcomeEncoder,
102                EncodingDirectoryDataWriterFactory.class,
103                "OUTCOME_TYPE",
104                this);
105            ReflectionUtil.checkTypeParameterIsAssignable(
106                OutcomeEncoder.class,
107                "ENCODED_OUTCOME_TYPE",
108                untypedOutcomeEncoder,
109                EncodingDirectoryDataWriterFactory.class,
110                "ENCODED_OUTCOME_TYPE",
111                this);
112    
113            // assign the encoders to the instance variables
114            this.featuresEncoder = ReflectionUtil.uncheckedCast(untypedFeaturesEncoder);
115            this.outcomeEncoder = ReflectionUtil.uncheckedCast(untypedOutcomeEncoder);
116            is.close();
117          } catch (Exception e) {
118            throw new ResourceInitializationException(e);
119          }
120        }
121      }
122    
123      protected boolean setEncodersFromFileSystem(
124          EncodingDirectoryDataWriter<?, ?, ENCODED_FEATURES_TYPE, OUTCOME_TYPE, ENCODED_OUTCOME_TYPE> dataWriter) {
125        if (this.featuresEncoder != null && this.outcomeEncoder != null) {
126          dataWriter.setFeaturesEncoder(this.featuresEncoder);
127          dataWriter.setOutcomeEncoder(this.outcomeEncoder);
128          return true;
129        }
130        return false;
131      }
132    
133      protected FeaturesEncoder<ENCODED_FEATURES_TYPE> featuresEncoder = null;
134    
135      protected OutcomeEncoder<OUTCOME_TYPE, ENCODED_OUTCOME_TYPE> outcomeEncoder = null;
136    
137    }
