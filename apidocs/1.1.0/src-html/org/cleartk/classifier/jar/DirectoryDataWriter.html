024    package org.cleartk.classifier.jar;
025    
026    import java.io.File;
027    import java.io.IOException;
028    
029    import org.cleartk.classifier.CleartkProcessingException;
030    import org.cleartk.classifier.DataWriter;
031    import org.cleartk.classifier.SequenceDataWriter;
032    
033    /**
034     * Superclass for {@link DataWriter} and {@link SequenceDataWriter} implementations that saves files
035     * to a training directory using a {@link JarClassifierBuilder}.
036     * 
037     * Note that it does not declare that it implements either of the DataWriter interfaces. Subclasses
038     * should do this.
039     * 
040     * <br>
041     * Copyright (c) 2011, Regents of the University of Colorado <br>
042     * All rights reserved.
043     * 
044     * @author Steven Bethard
045     * @author Philip Ogren
046     */
047    public abstract class DirectoryDataWriter<CLASSIFIER_BUILDER extends JarClassifierBuilder<? extends CLASSIFIER_TYPE>, CLASSIFIER_TYPE> {
048    
049      protected File outputDirectory;
050    
051      protected CLASSIFIER_BUILDER classifierBuilder;
052    
053      public DirectoryDataWriter(File outputDirectory) {
054        this.outputDirectory = outputDirectory;
055        if (!this.outputDirectory.exists()) {
056          this.outputDirectory.mkdirs();
057        }
058        this.classifierBuilder = this.newClassifierBuilder();
059      }
060    
061      /**
062       * Constructs a new {@link JarClassifierBuilder} that will be set as the
063       * {@link #classifierBuilder} during object construction.
064       */
065      protected abstract CLASSIFIER_BUILDER newClassifierBuilder();
066    
067      /**
068       * Get the {@link JarClassifierBuilder} associated with this DataWriter.
069       * 
070       * @return The classifier builder.
071       */
072      public CLASSIFIER_BUILDER getClassifierBuilder() {
073        return classifierBuilder;
074      }
075    
076      /**
077       * Basic implementation of {@link DataWriter#finish()} and {@link SequenceDataWriter#finish()}
078       * that calls {@link ClassifierBuilder#saveToTrainingDirectory(File)}
079       */
080      public void finish() throws CleartkProcessingException {
081        try {
082          this.classifierBuilder.saveToTrainingDirectory(this.outputDirectory);
083        } catch (IOException e) {
084          throw new CleartkProcessingException(e);
085        }
086      }
087    }
