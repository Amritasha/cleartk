051    public class JarMultiClassifierFactory<OUTCOME_TYPE> implements
052        MultiClassifierFactory<OUTCOME_TYPE>, Initializable {
053    
054      public static final String PARAM_CLASSIFIER_JAR_DIR = ConfigurationParameterFactory
055          .createConfigurationParameterName(JarMultiClassifierFactory.class, "classifierJarDir");
056    
057      @ConfigurationParameter(mandatory = true, description = "the directory where the jar files to instantiate the classifier reside.")
058      private String classifierJarDir;
059    
060      public void setClassifierJarDir(String classifierJarDir) {
061        this.classifierJarDir = classifierJarDir;
062      }
063    
064      public void initialize(UimaContext context) throws ResourceInitializationException {
065        ConfigurationParameterInitializer.initialize(this, context);
066      }
067    
068      @SuppressWarnings("unchecked")
069      public Classifier<OUTCOME_TYPE> createClassifier(String name) throws IOException {
070        return this.createUntypedClassifier(Classifier.class, name);
071      }
072    
073      private <CLASSIFIER_TYPE> CLASSIFIER_TYPE createUntypedClassifier(
074          Class<CLASSIFIER_TYPE> superClass,
075          String name) throws IOException {
076        InputStream stream;
077        // Build a path to the classifier jar file -- this is the same convention used by the dataWriter
078        File dir = new File(this.classifierJarDir, name);
079        File classifierJarPath = new File(dir.getPath(), "model.jar"); // FIXME: see issue #227
080        try {
081          stream = new URL(classifierJarPath.getPath()).openStream();
082        } catch (MalformedURLException e) {
083          stream = new FileInputStream(classifierJarPath.getPath());
084        }
085        stream = new BufferedInputStream(stream);
086        JarInputStream modelStream = new JarInputStream(stream);
087        JarClassifierBuilder<?> builder = JarClassifierBuilder.fromManifest(modelStream.getManifest());
088        try {
089          return superClass.cast(builder.loadClassifier(modelStream));
090        } finally {
091          stream.close();
092        }
093      }
094    }
