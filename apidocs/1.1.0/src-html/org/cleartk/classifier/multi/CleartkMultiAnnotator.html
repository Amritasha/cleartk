024    package org.cleartk.classifier.multi;
025    
026    import java.io.IOException;
027    import java.util.HashMap;
028    import java.util.Map;
029    
030    import org.apache.uima.UimaContext;
031    import org.apache.uima.analysis_engine.AnalysisEngineProcessException;
032    import org.apache.uima.resource.ResourceInitializationException;
033    import org.cleartk.classifier.Classifier;
034    import org.cleartk.classifier.CleartkAnnotator;
035    import org.cleartk.classifier.CleartkProcessingException;
036    import org.cleartk.classifier.DataWriter;
037    import org.cleartk.util.CleartkInitializationException;
038    import org.cleartk.util.ReflectionUtil;
039    import org.uimafit.component.JCasAnnotator_ImplBase;
040    import org.uimafit.descriptor.ConfigurationParameter;
041    import org.uimafit.factory.ConfigurationParameterFactory;
042    import org.uimafit.factory.initializable.Initializable;
043    import org.uimafit.factory.initializable.InitializableFactory;
044    
045    /**
046     * <br>
047     * Copyright (c) 2011, Regents of the University of Colorado <br>
048     * All rights reserved.
049     * 
050     * @author Lee Becker
051     * 
052     *         This class provides a framework for handling multiple classifiers within a single
053     *         analysis engine.
054     *         <p>
055     * 
056     *         Use cases that may lend themselves to {@link CleartkMultiAnnotator} over
057     *         {@link CleartkAnnotator} include:
058     *         <ul>
059     *         <li>Explicit control for one-vs-all multi-label classification
060     *         <li>Predicating training/classification on specific conditions. (e.g. if the the word to
061     *         the left is a verb, use the verb classifier, else use the default classifier)
062     *         <li>Voting or ensemble classification
063     *         </ul>
064     */
065    public abstract class CleartkMultiAnnotator<OUTCOME_TYPE> extends JCasAnnotator_ImplBase implements
066        Initializable {
067    
068      public static final String PARAM_MULTI_CLASSIFIER_FACTORY_CLASS_NAME = ConfigurationParameterFactory
069          .createConfigurationParameterName(
070              CleartkMultiAnnotator.class,
071              "multiClassifierFactoryClassName");
072    
073      @ConfigurationParameter(mandatory = false, description = "provides the full name of the MultiClassifierFactory class to be used.", defaultValue = "org.cleartk.classifier.multi.jar.JarMultiClassifierFactory")
074      protected String multiClassifierFactoryClassName;
075    
076      public static final String PARAM_MULTI_DATA_WRITER_FACTORY_CLASS_NAME = ConfigurationParameterFactory
077          .createConfigurationParameterName(
078              CleartkMultiAnnotator.class,
079              "multiDataWriterFactoryClassName");
080    
081      @ConfigurationParameter(mandatory = false, description = "provides the full name of the MultiDataWriterFactory class to be used.")
082      protected String multiDataWriterFactoryClassName;
083    
084      public static final String PARAM_IS_TRAINING = ConfigurationParameterFactory
085          .createConfigurationParameterName(CleartkMultiAnnotator.class, "isTraining");
086    
087      @ConfigurationParameter(mandatory = false, description = "determines whether this annotator is writing training data or using a classifier to annotate. Normally inferred automatically based on whether or not a DataWriterFactory class has been set.")
088      private Boolean isTraining;
089    
090      private boolean primitiveIsTraining;
091    
092      protected MultiDataWriterFactory<?> multiDataWriterFactory;
093    
094      protected MultiClassifierFactory<?> multiClassifierFactory;
095    
096      protected Map<String, Classifier<OUTCOME_TYPE>> classifiers;
097    
098      protected Map<String, DataWriter<OUTCOME_TYPE>> dataWriters;
099    
100      @Override
101      public void initialize(UimaContext context) throws ResourceInitializationException {
102        super.initialize(context);
103    
104        if (multiDataWriterFactoryClassName == null && multiClassifierFactoryClassName == null) {
105          CleartkInitializationException.neitherParameterSet(
106              PARAM_MULTI_DATA_WRITER_FACTORY_CLASS_NAME,
107              multiDataWriterFactoryClassName,
108              PARAM_MULTI_CLASSIFIER_FACTORY_CLASS_NAME,
109              multiClassifierFactoryClassName);
110        }
111    
112        // determine whether we start out as training or predicting
113        if (this.isTraining == null) {
114          this.primitiveIsTraining = multiDataWriterFactoryClassName != null;
115        } else {
116          this.primitiveIsTraining = this.isTraining;
117        }
118    
119        if (this.isTraining()) {
120          // create the multiDataWriter factory and initialize a Map to hold the data writers
121          this.dataWriters = new HashMap<String, DataWriter<OUTCOME_TYPE>>();
122    
123          // create the factory and instantiate the data writer
124          this.multiDataWriterFactory = InitializableFactory.create(
125              context,
126              multiDataWriterFactoryClassName,
127              MultiDataWriterFactory.class);
128        }
129    
130        // create the multi classifier factory and initialize a map to hold the classifiers
131        // While this may be superfluous in some cases, this is done in all instances because some
132        // MultiAnnotators utilize other classifiers during training
133        this.classifiers = new HashMap<String, Classifier<OUTCOME_TYPE>>();
134        // create the factory and instantiate the classifier
135        multiClassifierFactory = InitializableFactory.create(
136            context,
137            multiClassifierFactoryClassName,
138            MultiClassifierFactory.class);
139      }
140    
141      @Override
142      public void collectionProcessComplete() throws AnalysisEngineProcessException {
143        super.collectionProcessComplete();
144        if (this.isTraining()) {
145          try {
146            for (DataWriter<OUTCOME_TYPE> dataWriter : dataWriters.values()) {
147              dataWriter.finish();
148            }
149          } catch (CleartkProcessingException ctke) {
150            throw new AnalysisEngineProcessException(ctke);
151          }
152        }
153      }
154    
155      protected boolean isTraining() {
156        return this.primitiveIsTraining;
157      }
158    
159      /**
160       * Gets the classifier associated with name. If it does not exist, this method will use the
161       * {@link MultiClassifierFactory} specified at initialization to create a new one.
162       * 
163       * @param name
164       * @return the classifier associated with name
165       * @throws ResourceInitializationException
166       */
167      protected Classifier<OUTCOME_TYPE> getClassifier(String name)
168          throws ResourceInitializationException {
169        if (classifiers.containsKey(name)) {
170          return classifiers.get(name);
171        }
172    
173        Classifier<?> untypedClassifier;
174        try {
175          untypedClassifier = multiClassifierFactory.createClassifier(name);
176        } catch (IOException e) {
177          throw new ResourceInitializationException(e);
178        }
179    
180        Classifier<OUTCOME_TYPE> classifier = ReflectionUtil.uncheckedCast(untypedClassifier);
181        ReflectionUtil.checkTypeParameterIsAssignable(
182            CleartkMultiAnnotator.class,
183            "OUTCOME_TYPE",
184            this,
185            Classifier.class,
186            "OUTCOME_TYPE",
187            classifier);
188        InitializableFactory.initialize(untypedClassifier, this.getContext());
189        this.classifiers.put(name, classifier);
190        return classifier;
191    
192      }
193    
194      /**
195       * Gets the dataWriter associated with name. If it does not exist, this method will use the
196       * {@link MultiDataWriterFactory} specified during initialization to create a dataWriter
197       * associated with the name parameter.
198       * 
199       * @param name
200       * @return the data writer associated with name
201       * @throws ResourceInitializationException
202       */
203      protected DataWriter<OUTCOME_TYPE> getDataWriter(String name)
204          throws ResourceInitializationException {
205        if (dataWriters.containsKey(name)) {
206          return dataWriters.get(name);
207        }
208    
209        DataWriter<?> untypedDataWriter;
210        try {
211          untypedDataWriter = multiDataWriterFactory.createDataWriter(name);
212          DataWriter<OUTCOME_TYPE> dataWriter = ReflectionUtil.uncheckedCast(untypedDataWriter);
213          this.dataWriters.put(name, dataWriter);
214          return dataWriter;
215    
216        } catch (IOException e) {
217          throw new ResourceInitializationException(e);
218        }
219      }
220    
221    }
