025    package org.cleartk.classifier.multi;
026    
027    import java.io.File;
028    import java.io.IOException;
029    
030    import org.apache.uima.UimaContext;
031    import org.apache.uima.resource.ResourceInitializationException;
032    import org.cleartk.classifier.DataWriter;
033    import org.cleartk.classifier.DataWriterFactory;
034    import org.cleartk.classifier.jar.DirectoryDataWriterFactory;
035    import org.cleartk.util.ReflectionUtil;
036    import org.uimafit.component.initialize.ConfigurationParameterInitializer;
037    import org.uimafit.descriptor.ConfigurationParameter;
038    import org.uimafit.factory.ConfigurationParameterFactory;
039    import org.uimafit.factory.initializable.Initializable;
040    import org.uimafit.factory.initializable.InitializableFactory;
041    
042    /**
043     * This factory delegates creation of {@link DataWriter} creation to a {@link DataWriterFactory} of
044     * class {@link DirectoryDataWriterFactory} or one of its subclasses.
045     * 
046     * <br>
047     * Copyright (c) 2011, Regents of the University of Colorado <br>
048     * All rights reserved.
049     * 
050     * @author Lee Becker
051     */
052    public class MultiDirectoryDataWriterFactory<OUTCOME_TYPE> implements
053        MultiDataWriterFactory<OUTCOME_TYPE>, Initializable {
054    
055      public static final String PARAM_OUTPUT_DIRECTORY = ConfigurationParameterFactory
056          .createConfigurationParameterName(MultiDirectoryDataWriterFactory.class, "outputDirectory");
057    
058      @ConfigurationParameter(mandatory = false, description = "provides the name of the directory where the training data will be written.  If you do not set this parameter, then you must call setOutputDirectory directly.")
059      protected File outputDirectory;
060    
061      public static final String PARAM_DATA_WRITER_FACTORY_CLASS_NAME = ConfigurationParameterFactory
062          .createConfigurationParameterName(
063              MultiDirectoryDataWriterFactory.class,
064              "dataWriterFactoryClassName");
065    
066      @ConfigurationParameter(mandatory = false, description = "Provides the full name of the DataWriterFactory class to be used.  CleartkMultiAnnotator assumes this is a DirectoryDataWriter or subclass")
067      private String dataWriterFactoryClassName;
068    
069      private DataWriterFactory<?> dataWriterFactory;
070    
071      private UimaContext uimaContext;
072    
073      @Override
074      public void initialize(UimaContext context) throws ResourceInitializationException {
075        // Initialize configuration parameters
076        ConfigurationParameterInitializer.initialize(this, context);
077    
078        // create the factory and instantiate the data writer
079        this.uimaContext = context;
080        this.dataWriterFactory = InitializableFactory.create(
081            context,
082            dataWriterFactoryClassName,
083            DataWriterFactory.class);
084    
085      }
086    
087      @Override
088      public DataWriter<OUTCOME_TYPE> createDataWriter(String name) throws IOException {
089    
090        File dataWriterPath = new File(this.outputDirectory, name);
091        DataWriter<?> untypedDataWriter;
092        try {
093          // Set path for dataWriterFactory
094          // The following code assumes the factory is a DirectoryDataWriterFactory or subclass
095          DirectoryDataWriterFactory dirDataWriterFactory = ReflectionUtil
096              .uncheckedCast(dataWriterFactory);
097          dirDataWriterFactory.setOutputDirectory(dataWriterPath);
098          // / Create the dataWriter
099          untypedDataWriter = dataWriterFactory.createDataWriter();
100    
101          InitializableFactory.initialize(untypedDataWriter, this.uimaContext);
102          DataWriter<OUTCOME_TYPE> dataWriter = ReflectionUtil.uncheckedCast(untypedDataWriter);
103    
104          return dataWriter;
105        } catch (ResourceInitializationException e) {
106          throw new IOException(e);
107        }
108    
109      }
110    
111      public File getOutputDirectory() {
112        return outputDirectory;
113      }
114    
115      public void setOutputDirectory(File outputDirectory) {
116        this.outputDirectory = outputDirectory;
117      }
118    
119    }
