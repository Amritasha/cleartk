024    package org.cleartk.classifier.baseline;

026    import java.io.File;
027    import java.io.IOException;
028    import java.io.InputStreamReader;
029    import java.util.jar.JarInputStream;
030    import java.util.jar.JarOutputStream;
031    
032    import org.cleartk.classifier.jar.JarClassifierBuilder;
033    import org.cleartk.classifier.jar.JarStreams;
034    
035    import com.google.common.base.Charsets;
036    import com.google.common.io.CharStreams;
037    import com.google.common.io.Files;
038

046     */
047    public abstract class SingleOutcomeClassifierBuilder<OUTCOME_TYPE> extends
048        JarClassifierBuilder<SingleOutcomeClassifier<OUTCOME_TYPE>> {
049    
050      private static final String MODEL_NAME = "model.singlevalue";
051    
052      public File getTrainingDataFile(File dir) {
053        return new File(dir, "training-data.outcomes");
054      }
055    
056      protected abstract OUTCOME_TYPE parseOutcome(String outcome);
057    
058      protected void writeOutcome(File dir, OUTCOME_TYPE outcome) throws IOException {
059        Files.write(outcome.toString(), new File(dir, MODEL_NAME), Charsets.US_ASCII);
060      }
061    
062      @Override
063      protected void packageClassifier(File dir, JarOutputStream modelStream) throws IOException {
064        super.packageClassifier(dir, modelStream);
065        JarStreams.putNextJarEntry(modelStream, MODEL_NAME, new File(dir, MODEL_NAME));
066      }
067    
068      protected OUTCOME_TYPE value;
069    
070      @Override
071      protected void unpackageClassifier(JarInputStream modelStream) throws IOException {
072        super.unpackageClassifier(modelStream);
073        JarStreams.getNextJarEntry(modelStream, MODEL_NAME);
074        InputStreamReader reader = new InputStreamReader(modelStream, Charsets.US_ASCII);
075        this.value = this.parseOutcome(CharStreams.toString(reader));
076      }
077    
078      @Override
079      protected SingleOutcomeClassifier<OUTCOME_TYPE> newClassifier() {
080        return new SingleOutcomeClassifier<OUTCOME_TYPE>(this.value);
081      }
082    }
