051     * A class that provided interfaces to train, package and unpackage a SVMTKClassifier into a jar file.
052     */
053    
054    public class TKSVMlightClassifierBuilder extends
055        ClassifierBuilder_ImplBase<TKSVMlightClassifier, TreeFeatureVector, Boolean, Boolean> {
056    
057      static Logger logger = UIMAFramework.getLogger(TKSVMlightClassifierBuilder.class);
058    
059      public static String COMMAND_ARGUMENT = "--executable";
060    
061      /**
062       * Train a SVMTK classifier. Assumes the executable name is "tk_svm_learn". Note: this is public so
063       * the One verse All TK SVMlight classifier may take advantage of it, casual users probably shouldn't use this
064       * interface.
065       * @param dir The directory where the training data has been written.
066       * @param args The arguments to be used by the tk_svm_classify command. Note: -t 5 is used to specify
067       * the use of Tree Kernels.
068       */
069      public static void train(String filePath, String[] args) throws Exception {
070        String executable = "tk_svm_learn";
071        if (args.length > 0 && args[0].equals(COMMAND_ARGUMENT)) {
072          executable = args[1];
073          String[] tempArgs = new String[args.length - 2];
074          System.arraycopy(args, 2, tempArgs, 0, tempArgs.length);
075          args = tempArgs;
076        }
077    
078        String[] command = new String[args.length + 3];
079        command[0] = executable;
080        System.arraycopy(args, 0, command, 1, args.length);
081        command[command.length - 2] = new File(filePath).getPath();
082        command[command.length - 1] = new File(filePath + ".model").getPath();
083    
084        logger.log(Level.INFO, "training with tree kernel svmlight using the following command: "
085            + toString(command));
086        logger
087            .log(
088                Level.INFO,
089                "if the tree kernel svmlight learner does not seem to be working correctly, then try running the above command directly to see if e.g. svm_learn or svm_perf_learn gives a useful error message.");
090        Process process = Runtime.getRuntime().exec(command);
091        output(process.getInputStream(), System.out);
092        output(process.getErrorStream(), System.err);
093        process.waitFor();
094      }
095    
096      public File getTrainingDataFile(File dir) {
097        return new File(dir, "training-data.svmlight");
098      }
099    
100      private File getModelFile(File dir) {
101        return new File(dir, "training-data.svmlight.model");
102      }
103    
104      /**
105       * Train this SVMTK classifier.
106       * @param dir The directory where the training data has been written.
107       * @param args The arguments to be used by the tk_svm_classify command. Note: -t 5 is used to specify
108       * the use of Tree Kernels.
109       */
110      public void trainClassifier(File dir, String... args) throws Exception {
111        File trainingDataFile = getTrainingDataFile(dir);
112        train(trainingDataFile.getPath(), args);
113      }
114    
115      /**
116       * package the classifier found in dir into the a Jar file.
117       */
118      @Override
119      protected void packageClassifier(File dir, JarOutputStream modelStream) throws IOException {
120        super.packageClassifier(dir, modelStream);
121        JarStreams.putNextJarEntry(modelStream, "model.svmlight", getModelFile(dir));
122      }
123    
124      private File modelFile;
125    
126      /**
127       * unpackage the model files found in a JarInputStream.
128       */
129      @Override
130      protected void unpackageClassifier(JarInputStream modelStream) throws IOException {
131        super.unpackageClassifier(modelStream);
132        JarStreams.getNextJarEntry(modelStream, "model.svmlight");
133        this.modelFile = File.createTempFile("model", ".svmlight");
134        BufferedWriter out = new BufferedWriter(new FileWriter(this.modelFile));
135        BufferedReader in = new BufferedReader(new InputStreamReader(modelStream));
136        String line;
137        while ((line = in.readLine()) != null) {
138          out.append(line);
139          out.append("\n");
140        }
141        out.close();    
142      }
143    
144      /**
145       * Create a TKSVMlightClassifier.
146       */
147      @Override
148      protected TKSVMlightClassifier newClassifier() {
149        return new TKSVMlightClassifier(
150            this.featuresEncoder,
151            this.outcomeEncoder,
152            this.modelFile);
153      }
154    
155      private static String toString(String[] command) {
156        StringBuilder sb = new StringBuilder();
157        for (String cmmnd : command) {
158          sb.append(cmmnd + " ");
159        }
160        return sb.toString();
161      }
162    
163      private static void output(InputStream input, PrintStream output) throws IOException {
164        byte[] buffer = new byte[128];
165        int count = input.read(buffer);
166        while (count != -1) {
167          output.write(buffer, 0, count);
168          count = input.read(buffer);
169        }
170      }
