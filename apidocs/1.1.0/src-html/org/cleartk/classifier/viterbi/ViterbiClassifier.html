055    public class ViterbiClassifier<OUTCOME_TYPE> implements SequenceClassifier<OUTCOME_TYPE>,

062      public static final String PARAM_STACK_SIZE = ConfigurationParameterFactory
063          .createConfigurationParameterName(ViterbiClassifier.class, "stackSize");
064    
065      @ConfigurationParameter(description = "specifies the maximum number of candidate paths to "
066          + "keep track of. In general, this number should be higher than the number "
067          + "of possible classifications at any given point in the sequence. This "
068          + "guarantees that highest-possible scoring sequence will be returned. If, "
069          + "however, the number of possible classifications is quite high and/or you "
070          + "are concerned about throughput performance, then you may want to reduce the number "
071          + "of candidate paths to maintain.  If Classifier.score is not implemented for the given delegated classifier, then "
072          + "the value of this parameter must be 1. ", defaultValue = "1")
073      protected int stackSize;
074    
075      public static final String PARAM_ADD_SCORES = ConfigurationParameterFactory
076          .createConfigurationParameterName(ViterbiClassifier.class, "addScores");
077    
078      @ConfigurationParameter(description = "specifies whether the scores of candidate sequence classifications should be "
079          + "calculated by summing classfication scores for each member of the sequence or by multiplying them. A value of "
080          + "true means that the scores will be summed. A value of false means that the scores will be multiplied. ", defaultValue = "false")
081      protected boolean addScores = false;
082    
083      public ViterbiClassifier(
084          Classifier<OUTCOME_TYPE> delegatedClassifier,
085          OutcomeFeatureExtractor[] outcomeFeatureExtractors) {
086        this.delegatedClassifier = delegatedClassifier;
087        this.outcomeFeatureExtractors = outcomeFeatureExtractors;
088      }
089    
090      public void initialize(UimaContext context) throws ResourceInitializationException {
091        ConfigurationParameterInitializer.initialize(this, context);
092        if (stackSize < 1) {
093          throw CleartkInitializationException.parameterLessThan(PARAM_STACK_SIZE, 1, stackSize);
094        }
095      }
096    
097      public List<OUTCOME_TYPE> classify(List<List<Feature>> features)
098          throws CleartkProcessingException {
099        if (stackSize == 1) {
100          List<Object> outcomes = new ArrayList<Object>();
101          List<OUTCOME_TYPE> returnValues = new ArrayList<OUTCOME_TYPE>();
102          for (List<Feature> instanceFeatures : features) {
103            for (OutcomeFeatureExtractor outcomeFeatureExtractor : outcomeFeatureExtractors) {
104              instanceFeatures.addAll(outcomeFeatureExtractor.extractFeatures(outcomes));
105            }
106            OUTCOME_TYPE outcome = delegatedClassifier.classify(instanceFeatures);
107            outcomes.add(outcome);
108            returnValues.add(outcome);
109          }
110          return returnValues;
111        } else {
112          try {
113            return viterbi(features);
114          } catch (UnsupportedOperationException uoe) {
115            throw CleartkProcessingException.unsupportedOperationSetParameter(
116                delegatedClassifier,
117                "score",
118                PARAM_STACK_SIZE,
119                1);
120          }
121        }
122    
123      }
124    
125      /**
126       * This implementation of Viterbi requires at most stackSize * sequenceLength calls to the
127       * classifier. If this proves to be too expensive, then consider using a smaller stack size.
128       * 
129       * @param features
130       *          a sequence-worth of features. Each List<Feature> in features should corresond to all
131       *          of the features for a given element in a sequence to be classified.
132       * @return a list of outcomes (classifications) - one classification for each member of the
133       *         sequence.
134       * @see #PARAM_STACK_SIZE
135       * @see OutcomeFeatureExtractor
136       */
137      public List<OUTCOME_TYPE> viterbi(List<List<Feature>> features) throws CleartkProcessingException {
138    
139        List<ScoredOutcome<List<OUTCOME_TYPE>>> nbestSequences = new ArrayList<ScoredOutcome<List<OUTCOME_TYPE>>>();
140    
141        if (features == null || features.size() == 0) {
142          return Collections.emptyList();
143        }
144    
145        List<ScoredOutcome<OUTCOME_TYPE>> scoredOutcomes = delegatedClassifier.score(
146            features.get(0),
147            stackSize);
148        for (ScoredOutcome<OUTCOME_TYPE> scoredOutcome : scoredOutcomes) {
149          double score = scoredOutcome.getScore();
150          List<OUTCOME_TYPE> sequence = new ArrayList<OUTCOME_TYPE>();
151          sequence.add(scoredOutcome.getOutcome());
152          nbestSequences.add(new ScoredOutcome<List<OUTCOME_TYPE>>(sequence, score));
153        }
154    
155        Map<OUTCOME_TYPE, Double> l = new HashMap<OUTCOME_TYPE, Double>();
156        Map<OUTCOME_TYPE, List<OUTCOME_TYPE>> m = new HashMap<OUTCOME_TYPE, List<OUTCOME_TYPE>>();
157    
158        for (int i = 1; i < features.size(); i++) {
159    
160          List<Feature> instanceFeatures = features.get(i);
161          l.clear();
162          m.clear();
163          for (ScoredOutcome<List<OUTCOME_TYPE>> scoredSequence : nbestSequences) {
164            // add features from previous outcomes from each scoredSequence
165            // in returnValues
166            int outcomeFeaturesCount = 0;
167            List<Object> previousOutcomes = new ArrayList<Object>(scoredSequence.getOutcome());
168            for (OutcomeFeatureExtractor outcomeFeatureExtractor : outcomeFeatureExtractors) {
169              List<Feature> outcomeFeatures = outcomeFeatureExtractor.extractFeatures(previousOutcomes);
170              instanceFeatures.addAll(outcomeFeatures);
171              outcomeFeaturesCount += outcomeFeatures.size();
172            }
173            // score the instance features using the features added by the
174            // outcomeFeatureExtractors
175            scoredOutcomes = delegatedClassifier.score(instanceFeatures, stackSize);
176            // remove the added features from previous outcomes for this
177            // scoredSequence
178            instanceFeatures = instanceFeatures.subList(0, instanceFeatures.size()
179                - outcomeFeaturesCount);
180    
181            for (ScoredOutcome<OUTCOME_TYPE> scoredOutcome : scoredOutcomes) {
182    
183              if (!l.containsKey(scoredOutcome.getOutcome())) {
184                double score = scoredSequence.getScore();
185                if (addScores) {
186                  score = score + scoredOutcome.getScore();
187                } else {
188                  score = score * scoredOutcome.getScore();
189                }
190                l.put(scoredOutcome.getOutcome(), score);
191                m.put(
192                    scoredOutcome.getOutcome(),
193                    new ArrayList<OUTCOME_TYPE>(scoredSequence.getOutcome()));
194              } else {
195                double newScore = scoredSequence.getScore();
196                if (addScores) {
197                  newScore = newScore + scoredOutcome.getScore();
198                } else {
199                  newScore = newScore * scoredOutcome.getScore();
200                }
201                double bestScore = l.get(scoredOutcome.getOutcome());
202    
203                if (newScore > bestScore) {
204                  l.put(scoredOutcome.getOutcome(), newScore);
205                  m.put(
206                      scoredOutcome.getOutcome(),
207                      new ArrayList<OUTCOME_TYPE>(scoredSequence.getOutcome()));
208                }
209              }
210            }
211          }
212    
213          nbestSequences.clear();
214          for (OUTCOME_TYPE outcome : l.keySet()) {
215            List<OUTCOME_TYPE> outcomeSequence = m.get(outcome);
216            outcomeSequence.add(outcome);
217            double score = l.get(outcome);
218            ScoredOutcome<List<OUTCOME_TYPE>> returnValue = new ScoredOutcome<List<OUTCOME_TYPE>>(
219                outcomeSequence,
220                score);
221            nbestSequences.add(returnValue);
222          }
223    
224          Collections.sort(nbestSequences);
225        }
226    
227        Collections.sort(nbestSequences);
228        if (nbestSequences.size() > 0) {
229          return nbestSequences.get(0).getOutcome();
230        }
231    
232        return null;
233      }
