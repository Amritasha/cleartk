024    package org.cleartk.classifier.svmlight.rank;
025    
026    import java.io.File;
027    import java.io.IOException;
028    import java.util.HashMap;
029    import java.util.Locale;
030    import java.util.Map;
031    
032    import org.cleartk.classifier.CleartkProcessingException;
033    import org.cleartk.classifier.Instance;
034    import org.cleartk.classifier.encoder.CleartkEncoderException;
035    import org.cleartk.classifier.svmlight.SVMlightDataWriter_ImplBase;
036    import org.cleartk.classifier.util.featurevector.FeatureVector;
037    
038    /**
039     * <br>
040     * Copyright (c) 2011, Regents of the University of Colorado <br>
041     * All rights reserved.
042     * 
043     * @author Lee Becker
044     * 
045     * Writes the training file for SVMlight's ranking classifier.
046     * Because ranking requires an additional item/query id (qid),
047     * instances passed into the write() method should be of type
048     * QidInstance.
049     * 
050     */
051    public class SVMlightRankDataWriter extends
052        SVMlightDataWriter_ImplBase<SVMlightRankBuilder, Double, Double> {
053      
054      public SVMlightRankDataWriter(File outputDirectory) throws IOException {
055        super(outputDirectory);
056        this.instQidToEncodedQidMap = new HashMap<String, Integer>();
057        this.lastEncodedQid = 0;
058      }
059    
060      @Override
061      protected String outcomeToString(Double outcome) {
062        return outcome.toString();
063      }
064    
065     @Override
066      public void write(Instance<Double> instance) throws CleartkProcessingException {
067         if (!(instance instanceof QidInstance)) {
068           throw new CleartkProcessingException("", "Unable to write non-QidInstance");
069         }
070         String qid = ((QidInstance<Double>) instance).getQid();
071    
072         writeEncoded(
073             this.getEncodedQid(qid),
074             this.classifierBuilder.getFeaturesEncoder().encodeAll(instance.getFeatures()),
075             this.classifierBuilder.getOutcomeEncoder().encode(instance.getOutcome()));
076     }
077      
078      @Override
079      protected SVMlightRankBuilder newClassifierBuilder() {
080        return new SVMlightRankBuilder();
081      }
082      
083      public void writeEncoded(int qid, FeatureVector features, Double outcome)
084          throws CleartkProcessingException {
085        
086        StringBuffer output = new StringBuffer();
087    
088        // The following two lines should be the only difference between this and
089        // SVMLightDataWriter_ImplBase.writeEncoded()
090        output.append(this.outcomeToString(outcome));
091        output.append(String.format(Locale.US, " qid:%d", qid));
092    
093        for (FeatureVector.Entry entry : features) {
094          if (Double.isInfinite(entry.value) || Double.isNaN(entry.value))
095            throw CleartkEncoderException.invalidFeatureVectorValue(entry.index, entry.value);
096          output.append(String.format(Locale.US, " %d:%.7f", entry.index, entry.value));
097        }
098    
099        this.trainingDataWriter.println(output); 
100        
101      }
102      
103      private int getEncodedQid(String qid) {
104          if (this.instQidToEncodedQidMap.containsKey(qid)) {
105              return this.instQidToEncodedQidMap.get(qid);
106          } else {
107              this.lastEncodedQid++;
108              this.instQidToEncodedQidMap.put(qid, lastEncodedQid);
109              return lastEncodedQid;
110          }
111      }
112      
113      private Map<String, Integer> instQidToEncodedQidMap;
114      private int lastEncodedQid;
115      
116    }
