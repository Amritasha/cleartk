024    package org.cleartk.classifier.encoder.features;
025    
026    import java.io.File;
027    import java.io.IOException;
028    import java.io.PrintWriter;
029    import java.util.ArrayList;
030    import java.util.List;
031    
032    import org.cleartk.classifier.Feature;
033    import org.cleartk.classifier.encoder.CleartkEncoderException;
034    import org.cleartk.classifier.encoder.FeatureEncoderUtil;
035    import org.cleartk.util.collection.CompressedStringBiMap;
036    
037    /**

044    
045    public class NameNumberFeaturesEncoder extends
046        FeaturesEncoder_ImplBase<List<NameNumber>, NameNumber> {
047    
048      private static final long serialVersionUID = 7508330794260661987L;
049    
050      public static final String LOOKUP_FILE_NAME = "names-lookup.txt";
051    
052      private CompressedStringBiMap csbm;
053    
054      private boolean compressFeatures;
055    
056      public boolean isCompressFeatures() {
057        return compressFeatures;
058      }
059    
060      private boolean allowNewFeatures = true;
061    
062      private boolean sortNameLookup;
063    
064      public NameNumberFeaturesEncoder(boolean compressFeatures, boolean sortNameLookup) {
065        super();
066        this.compressFeatures = compressFeatures;
067        this.sortNameLookup = sortNameLookup;
068        if (compressFeatures) {
069          csbm = new CompressedStringBiMap();
070        }
071      }
072    
073      @Override
074      public List<NameNumber> encodeAll(Iterable<Feature> features) throws CleartkEncoderException {
075        List<NameNumber> returnValues = new ArrayList<NameNumber>();
076    
077        for (Feature feature : features) {
078          for (NameNumber nameNumber : this.encode(feature)) {
079            nameNumber.name = compress(escape(nameNumber.name));
080            if (nameNumber.name != null) {
081              returnValues.add(nameNumber);
082            }
083          }
084        }
085        return returnValues;
086      }
087    
088      @Override
089      public void finalizeFeatureSet(File outputDirectory) throws IOException {
090        this.allowNewFeatures = false;
091    
092        if (compressFeatures) {
093          File lookupFile = new File(outputDirectory, LOOKUP_FILE_NAME);
094          PrintWriter writer = new PrintWriter(lookupFile);
095          csbm.write(writer, sortNameLookup);
096          writer.close();
097        }
098      }
099    
100      private String escape(String string) {
101        return FeatureEncoderUtil.escape(string, new char[] { '=' });
102      }
103    
104      private String compress(String featureString) {
105        if (compressFeatures) {
106          if (allowNewFeatures) {
107            return csbm.getOrGenerateKey(featureString);
108          } else {
109            return csbm.inverse().get(featureString);
110          }
111        }
112        return featureString;
113      }
114    }
