024    package org.cleartk.ne.term.util;
025    
026    import java.util.ArrayList;
027    import java.util.HashMap;
028    import java.util.LinkedList;
029    import java.util.List;
030    import java.util.ListIterator;
031    import java.util.Map;
032    
033    import org.cleartk.token.tokenizer.PennTreebankTokenizer;
034    import org.cleartk.token.tokenizer.Token;
035    import org.cleartk.token.tokenizer.Tokenizer;
036    
037    /**
038     * <br>
039     * Copyright (c) 2007-2008, Regents of the University of Colorado <br>
040     * All rights reserved.
041     * 
042     * <p>
043     * 
044     * This class provides a very simple and fast term finder. It performs a simple left to right term
045     * matching doing a single pass through the text token by token. It finds only exact matches where
046     * the terms are found in the text exactly as they appear (with the option of case insensitive
047     * matching) in the term list.
048     * 
049     * I performed a very simple test with throughput and got the following results:
050     * <ul>
051     * <li>loaded 2 million terms in 10.182 seconds using TermList.loadSimpleFile()
052     * <li>initialize instance of SimpleTermFinder with list by calling addTermList(). This took 96.416
053     * seconds.
054     * <li>A file containing 495,503 tokens was read in line by line (about 4000 lines) and tokenized.
055     * Time spent tokenizing was 8.567 seconds.
056     * <li>Each line was passed into the term finder getMatches() method one after the next after being
057     * tokenized. 8,633 term matches were found in the 495,503 tokens in .243 seconds.
058     * </ul>
059     * This experiment was run on a dual-core 1.99GHz Pentium. The code for this experiment can be found
060     * in SimpleTermFinderTests.testTime()
061     * 
062     * @author Philip Ogren
063     * 
064     */
065    public class SimpleTermFinder implements TermFinder {
066    
067      Tokenizer tokenizer;
068    
069      boolean caseSensitive = true;
070    
071      Node topNode;
072    
073      public SimpleTermFinder(boolean caseSensitive, Tokenizer tokenizer) {
074    
075        this.caseSensitive = caseSensitive;
076        this.tokenizer = tokenizer;
077    
078        topNode = new Node();
079      }
080    
081      public SimpleTermFinder() {
082        this(true, new PennTreebankTokenizer());
083      }
084    
085      public void addTermList(TermList termList) {
086        for (Term entry : termList.getTerms()) {
087    
088          List<Token> tokens = tokenizer.getTokens(entry.getTermText());
089    
090          if (!caseSensitive) {
091            List<Token> lowerTokens = new ArrayList<Token>(tokens.size());
092            for (Token token : tokens) {
093              lowerTokens.add(new Token(token.getBegin(), token.getEnd(), token
094                  .getTokenText()
095                  .toLowerCase()));
096            }
097            tokens = lowerTokens;
098          }
099    
100          Node node = topNode;
101          for (Token token : tokens) {
102            if (!node.containsChild(token)) {
103              node.addChild(token);
104            }
105            node = node.getChild(token);
106          }
107          node.addEntry(entry);
108        }
109      }
110    
111      private List<TermMatch> getMatches(Token token, LinkedList<Candidate> candidates) {
112        List<TermMatch> returnValues = new ArrayList<TermMatch>();
113    
114        String tokenText = token.getTokenText();
115        if (!caseSensitive)
116          tokenText = tokenText.toLowerCase();
117    
118        if (topNode.containsChild(token)) {
119          candidates.add(new Candidate());
120        }
121    
122        ListIterator<Candidate> candidatesIterator = candidates.listIterator();
123        while (candidatesIterator.hasNext()) {
124          Candidate candidate = candidatesIterator.next();
125          if (candidate.node.containsChild(token)) {
126            candidate.addToken(token);
127            Node childNode = candidate.node.getChild(token);
128    
129            List<Term> entries = childNode.entries;
130            if (entries != null) {
131              for (Term entry : entries) {
132                returnValues.add(createTermMatch(candidate.tokens, entry));
133              }
134            }
135            if (childNode.hasChildren())
136              candidate.node = childNode;
137            else
138              candidatesIterator.remove();
139          } else {
140            candidatesIterator.remove();
141          }
142        }
143        return returnValues;
144      }
145    
146      /**
147       * @param tokens
148       *          It is the responsibility of the caller of this method to be sure that the same
149       *          tokenizer is used to create the tokens passed in that is passed into the constructor.
150       * @see #SimpleTermFinder(boolean, Tokenizer)
151       */
152      public List<TermMatch> getMatches(List<Token> tokens) {
153        LinkedList<Candidate> candidates = new LinkedList<Candidate>();
154        List<TermMatch> returnValues = new ArrayList<TermMatch>();
155    
156        for (Token token : tokens) {
157          returnValues.addAll(getMatches(token, candidates));
158        }
159        return returnValues;
160      }
161    
162      private TermMatch createTermMatch(List<Token> tokens, Term entry) {
163    
164        if (tokens == null || tokens.size() == 0)
165          return null;
166    
167        int begin = tokens.get(0).getBegin();
168        int end = tokens.get(tokens.size() - 1).getEnd();
169        return new TermMatch(begin, end, entry);
170      }
171    
172      private class Candidate {
173        List<Token> tokens;
174    
175        Node node;
176    
177        Candidate() {
178          tokens = new ArrayList<Token>();
179          node = topNode;
180        }
181    
182        public void addToken(Token token) {
183          tokens.add(token);
184        }
185      }
186    
187      class Node {
188        List<Term> entries;
189    
190        Map<String, Node> children;
191    
192        public boolean containsChild(Token child) {
193          if (children == null) {
194            return false;
195          } else
196            return children.containsKey(child.getTokenText());
197        }
198    
199        public boolean hasChildren() {
200          if (children == null)
201            return false;
202          else if (children.size() == 0)
203            return false;
204          else
205            return true;
206        }
207    
208        public void addChild(Token child) {
209          if (children == null) {
210            children = new HashMap<String, Node>();
211          }
212          children.put(child.getTokenText(), new Node());
213        }
214    
215        public Node getChild(Token child) {
216          if (children == null || !children.containsKey(child.getTokenText())) {
217            return null;
218          }
219          return children.get(child.getTokenText());
220        }
221    
222        public void addEntry(Term entry) {
223          if (entries == null)
224            entries = new ArrayList<Term>();
225          entries.add(entry);
226        }
227    
228      }
229    }
