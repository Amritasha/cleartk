064    public class TermFinderAnnotator extends JCasAnnotator_ImplBase {
065    
066      public static final String PARAM_TERM_LIST_FILE_NAMES_FILE_NAME = ConfigurationParameterFactory
067          .createConfigurationParameterName(TermFinderAnnotator.class, "termListFileNamesFileName");
068    
069      public static final String TERM_LIST_FILE_NAMES_FILE_NAME_DESCRIPTION = "Provides the name of a file that contains file names of term lists that are to be loaded. "
070          + "Each line of the file should contain the name of a term list followed by the name of the file that contains the terms, a boolean ('true' or 'false')  "
071          + "that indicates whether the file should be treated as case sensitive followed optionally by separator string to be used to separate  "
072          + "an id from a term if the file contains ids. The values on each line should be tab delimited. ";
073    
074      @ConfigurationParameter(mandatory = true, description = TERM_LIST_FILE_NAMES_FILE_NAME_DESCRIPTION)
075      public String termListFileNamesFileName;
076    
077      public static final String PARAM_WINDOW_CLASS_NAME = ConfigurationParameterFactory
078          .createConfigurationParameterName(TermFinderAnnotator.class, "windowClassName");
079    
080      @ConfigurationParameter(description = "names the class of the type system type from which to extract tokens. "
081          + "Any annotation that contains tokens can be used (e.g. sentence, paragraph, document).  "
082          + "If no value is given for this parameter, then all tokens will be searched. An example value might be 'org.cleartk.type.Sentence'")
083      private String windowClassName;
084    
085      public static final String PARAM_TOKEN_CLASS_NAME = ConfigurationParameterFactory
086          .createConfigurationParameterName(TermFinderAnnotator.class, "tokenClassName");
087    
088      @ConfigurationParameter(mandatory = true, defaultValue = "org.cleartk.type.Token", description = "names the class of the type system type corresponding to tokens. ")
089      private String tokenClassName;
090    
091      public static final String PARAM_TERM_MATCH_ANNOTATION_CREATOR_CLASS_NAME = ConfigurationParameterFactory
092          .createConfigurationParameterName(
093              TermFinderAnnotator.class,
094              "termMatchAnnotationCreatorClassName");
095    
096      @ConfigurationParameter(description = "provides the class name of a class that extends org.cleartk.ne.term.TermMatchAnnotationCreator. If this parameter is "
097          + "not given a value, then the parameter 'termMatchAnnotationClassName'  must be given a value.")
098      private String termMatchAnnotationCreatorClassName;
099    
100      public static final String PARAM_TERM_MATCH_ANNOTATION_CLASS_NAME = ConfigurationParameterFactory
101          .createConfigurationParameterName(TermFinderAnnotator.class, "termMatchAnnotationClassName");
102    
103      @ConfigurationParameter(defaultValue = "org.cleartk.ne.type.NamedEntityMention", description = "names the class of the type system type that specifies the annotations "
104          + "created of found term matches. One annotation is created for each term "
105          + "match found of the given type specified by this parameter. This parameter is ignored if 'termMatchAnnotationCreatorClassName' is given a value.")
106      private String termMatchAnnotationClassName;
107    
108      TermFinder caseSensitiveTermFinder;
109    
110      TermFinder caseInsensitiveTermFinder;
111    
112      boolean allTokens = true;
113    
114      boolean typesInitialized = false;
115    
116      protected Class<? extends Annotation> sentenceClass;
117    
118      protected Type sentenceType;
119    
120      protected Class<? extends Annotation> tokenClass;
121    
122      protected Type tokenType;
123    
124      TermMatchAnnotationCreator annotationCreator;
125    
126      Constructor<? extends Annotation> annotationConstructor;
127    
128      @Override
129      public void initialize(UimaContext context) throws ResourceInitializationException {
130        try {
131          super.initialize(context);
132    
133          // load the term lists
134          BufferedReader input = new BufferedReader(new FileReader(termListFileNamesFileName));
135          String line;
136          while ((line = input.readLine()) != null) {
137            line = line.trim();
138            String[] columns = line.split("\t");
139            String termListName = columns[0];
140            String fileName = columns[1];
141            boolean caseSensitive = Boolean.parseBoolean(columns[2]);
142            String separator = columns.length == 4 ? columns[3] : null;
143    
144            if (caseSensitive && caseSensitiveTermFinder == null)
145              caseSensitiveTermFinder = new SimpleTermFinder(true, new PennTreebankTokenizer());
146            if (!caseSensitive && caseInsensitiveTermFinder == null)
147              caseInsensitiveTermFinder = new SimpleTermFinder(false, new PennTreebankTokenizer());
148    
149            TermList termList = TermList.loadSimpleFile(termListName, new File(fileName), separator);
150            if (caseSensitive)
151              caseSensitiveTermFinder.addTermList(termList);
152            else
153              caseInsensitiveTermFinder.addTermList(termList);
154          }
155    
156          if (windowClassName != null) {
157            allTokens = false;
158            sentenceClass = InitializableFactory.getClass(windowClassName, Annotation.class);
159          }
160    
161          tokenClass = InitializableFactory.getClass(tokenClassName, Annotation.class);
162    
163          if (termMatchAnnotationCreatorClassName != null
164              && !termMatchAnnotationCreatorClassName.equals("")) {
165            annotationCreator = InitializableFactory.create(
166                context,
167                termMatchAnnotationCreatorClassName,
168                TermMatchAnnotationCreator.class);
169          } else {
170            Class<? extends Annotation> annotationClass = InitializableFactory.getClass(
171                termMatchAnnotationClassName,
172                Annotation.class);
173            annotationConstructor = annotationClass.getConstructor(new Class[] {
174                JCas.class,
175                java.lang.Integer.TYPE,
176                java.lang.Integer.TYPE });
177          }
178        } catch (Exception e) {
179          throw new ResourceInitializationException(e);
180        }
181      }
182    
183      private void initializeTypes(JCas jCas) {
184        if (!allTokens)
185          sentenceType = UIMAUtil.getCasType(jCas, sentenceClass);
186        tokenType = UIMAUtil.getCasType(jCas, tokenClass);
187        typesInitialized = true;
188      }
189    
190      private Token createToken(Annotation annotation) {
191        return new Token(annotation.getBegin(), annotation.getEnd(), annotation.getCoveredText());
192      }
193    
194      @Override
195      public void process(JCas jCas) throws AnalysisEngineProcessException {
196        if (!typesInitialized)
197          initializeTypes(jCas);
198    
199        List<Token> tokens = new ArrayList<Token>();
200        if (allTokens) {
201          FSIterator<Annotation> tokenAnnotations = jCas.getAnnotationIndex(tokenType).iterator();
202          while (tokenAnnotations.hasNext()) {
203            Annotation tokenAnnotation = tokenAnnotations.next();
204            tokens.add(createToken(tokenAnnotation));
205          }
206          findTerms(jCas, tokens);
207        } else {
208          FSIterator<Annotation> sentenceAnnotations = jCas.getAnnotationIndex(sentenceType).iterator();
209          while (sentenceAnnotations.hasNext()) {
210            tokens.clear();
211            Annotation sentenceAnnotation = sentenceAnnotations.next();
212            FSIterator<Annotation> tokenAnnotations = jCas.getAnnotationIndex(tokenType).subiterator(
213                sentenceAnnotation);
214            while (tokenAnnotations.hasNext()) {
215              Annotation tokenAnnotation = tokenAnnotations.next();
216              tokens.add(createToken(tokenAnnotation));
217            }
218            findTerms(jCas, tokens);
219          }
220        }
221      }
222    
223      public void findTerms(JCas jCas, List<Token> tokens) throws AnalysisEngineProcessException {
224    
225        if (caseSensitiveTermFinder != null) {
226          findTerms(jCas, tokens, caseSensitiveTermFinder);
227        }
228        if (caseInsensitiveTermFinder != null) {
229          findTerms(jCas, tokens, caseInsensitiveTermFinder);
230        }
231      }
232    
233      public void findTerms(JCas jCas, List<Token> tokens, TermFinder termFinder)
234          throws AnalysisEngineProcessException {
235    
236        List<TermMatch> termMatches = termFinder.getMatches(tokens);
237        for (TermMatch termMatch : termMatches) {
238          if (annotationCreator != null) {
239            annotationCreator.createTermMatchAnnotation(jCas, termMatch);
240          } else {
241            try {
242              int begin = termMatch.getBegin();
243              int end = termMatch.getEnd();
244              Annotation annotation = annotationConstructor
245                  .newInstance(new Object[] { jCas, begin, end });
246              annotation.addToIndexes();
247            } catch (Exception e) {
248              throw new AnalysisEngineProcessException(e);
249            }
250          }
251    
252        }
253      }
