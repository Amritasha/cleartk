059     * 
060     * @author Philip Ogren
061     * 
062     * 
063     *         This collection reader reads in the CoNLL 2003 named entity data. The data can be
064     *         retrieved from http://www.cnts.ua.ac.be/conll2003/ner/
065     * 
066     */
067    @SofaCapability(outputSofas = ViewURIUtil.URI)
068    public class Conll2003GoldReader extends JCasCollectionReader_ImplBase {
069      public static final String PARAM_DATA_FILE_NAME = ConfigurationParameterFactory
070          .createConfigurationParameterName(Conll2003GoldReader.class, "dataFileName");
071    
072      @ConfigurationParameter(mandatory = true, description = "Points to CoNLL data (e.g. ner/eng.train).")
073      private String dataFileName;
074    
075      public static final String PARAM_LOAD_NAMED_ENTITIES = ConfigurationParameterFactory
076          .createConfigurationParameterName(Conll2003GoldReader.class, "loadNamedEntities");
077    
078      @ConfigurationParameter(mandatory = true, description = "determines if the named entities are loaded (i.e. named entity mention annotations are created) or if just plain text from the files is loaded.", defaultValue = "true")
079      private boolean loadNamedEntities;
080    
081      public static final String DOCSTART = "-DOCSTART-";
082    
083      BufferedReader reader;
084    
085      boolean hasNext = true;
086    
087      int documentIndex = 0;
088    
089      int entityIdIndex = 0;
090    
091      public void initialize(UimaContext context) throws ResourceInitializationException {
092    
093        try {
094          File conllFile = new File(dataFileName);
095          reader = new BufferedReader(new FileReader(conllFile));
096          // advance the reader past the first occurrence of a document start and
097          // blank line.
098          String line;
099          while ((line = reader.readLine()) != null) {
100            if (line.trim().startsWith(DOCSTART)) {
101              reader.readLine(); // read the blank line
102              break;
103            }
104          }
105    
106          sentenceTokens = new ArrayList<Token>();
107          sentenceChunks = new ArrayList<Chunk>();
108          chunkTokens = new ArrayList<Token>();
109          namedEntityTokens = new ArrayList<Token>();
110        } catch (FileNotFoundException fnfe) {
111          throw new ResourceInitializationException(fnfe);
112        } catch (IOException ioe) {
113          throw new ResourceInitializationException(ioe);
114        }
115      }
116    
117      List<String> documentData; // contains every line from for the current document
118    
119      StringBuffer documentText; // collects the text of the tokens and then we set the document text
120                                 // with contents
121    
122      int sentenceStart; // the start (character offset) of the current sentence
123    
124      List<Token> sentenceTokens; // the tokens for the current sentence
125    
126      List<Chunk> sentenceChunks; // the chunks for the current sentence
127    
128      int tokenPosition; // the index of the current token
129    
130      int chunkStart; // the start (char offset) of the current chunk
131    
132      String currentChunkType; // the type of the current chunk (including the I- or B- prefix)
133    
134      List<Token> chunkTokens; // the tokens for the current chunk
135    
136      int namedEntityStart; // the start (char offset) of the current named entity
137    
138      String currentNamedEntityType; // the type of the current named entity (including the I- or B-
139                                     // prefix)
140    
141      List<Token> namedEntityTokens; // the tokens for the current named entity
142    
143      public void getNext(JCas jCas) throws IOException, CollectionException {
144        // read in the data for the next document from the reader
145        documentData = new ArrayList<String>();
146        String line;
147        while ((line = reader.readLine()) != null && !line.startsWith(DOCSTART)) {
148          documentData.add(line.trim());
149        }
150    
151        if (line == null)
152          hasNext = false;
153        else
154          line = reader.readLine().trim(); // advance past the blank line that follows
155                                           // "-DOCSTART- -X- O O"
156        // (we don't want an empty sentence on the front end!)
157        documentText = new StringBuffer();
158    
159        initSentence();
160        tokenPosition = 0;
161        chunkStart = 0;
162        currentChunkType = "";
163        chunkTokens.clear();
164        namedEntityStart = 0;
165        currentNamedEntityType = "";
166        namedEntityTokens.clear();
167    
168        for (String dataLine : documentData) {
169          if (dataLine.trim().equals("")) {
170            createChunk(jCas);
171            currentChunkType = "";
172            createNamedEntity(jCas);
173            currentNamedEntityType = "";
174    
175            Sentence sentence = new Sentence(jCas, sentenceStart, documentText.length());
176            sentence.addToIndexes();
177    
178            initSentence();
179          } else {
180            String[] dataPieces = dataLine.split(" ");
181            String tok = dataPieces[0];
182            String pos = dataPieces[1];
183            String chunkType = dataPieces[2];
184            if (currentChunkType.equals(""))
185              initChunk(chunkType);
186            String namedEntityType = dataPieces[3];
187            if (currentNamedEntityType.equals(""))
188              initNamedEntity(namedEntityType);
189    
190            Token token = new Token(jCas, documentText.length(), documentText.length() + tok.length());
191            token.setPos(pos);
192            token.addToIndexes();
193    
194            boolean chunkStartsWithB = startsWithB(currentChunkType, chunkType);
195            if (!chunkType.equals(currentChunkType) && !chunkStartsWithB) {
196              createChunk(jCas);
197              initChunk(chunkType);
198            }
199    
200            boolean namedEntityStartsWithB = startsWithB(currentNamedEntityType, namedEntityType);
201    
202            if (!namedEntityType.equals(currentNamedEntityType) && !namedEntityStartsWithB) {
203              createNamedEntity(jCas);
204              initNamedEntity(namedEntityType);
205            }
206    
207            sentenceTokens.add(token);
208            chunkTokens.add(token);
209            namedEntityTokens.add(token);
210            documentText.append(tok + " ");
211          }
212        }
213    
214        jCas.setDocumentText(documentText.toString());
215    
216        String identifier = String.format("%s#%s", dataFileName, documentIndex);
217        ViewURIUtil.setURI(jCas, new File(identifier).toURI());
218        ++documentIndex;
219    
220      }
221    
222      private void initSentence() {
223        sentenceStart = documentText.length();
224        sentenceTokens.clear();
225        sentenceChunks.clear();
226      }
227    
228      private void createChunk(JCas jCas) {
229        if (!currentChunkType.equals("O")) {
230          Chunk chunk = new Chunk(jCas, chunkStart, documentText.length() - 1);
231          chunk.setChunkType(currentChunkType.substring(2));
232          chunk.addToIndexes();
233          sentenceChunks.add(chunk);
234        }
235      }
236    
237      private void initChunk(String chunkType) {
238        chunkStart = documentText.length();
239        chunkTokens.clear();
240        currentChunkType = chunkType;
241      }
242    
243      private void createNamedEntity(JCas jCas) {
244        if (!currentNamedEntityType.equals("O") && loadNamedEntities) {
245          NamedEntity ne = new NamedEntity(jCas);
246          ne.setEntityClass("SPC");
247          ne.setEntityId("" + entityIdIndex++);
248          ne.setEntityType(currentNamedEntityType.substring(2));
249          ne.setEntitySubtype(currentNamedEntityType.substring(2));
250          ne.addToIndexes();
251    
252          NamedEntityMention nem = new NamedEntityMention(
253              jCas,
254              namedEntityStart,
255              documentText.length() - 1);
256          nem.setMentionType("NAM");
257          Annotation annotation = new Annotation(jCas, namedEntityStart, documentText.length() - 1);
258          annotation.addToIndexes();
259          // Chunk chunk = new Chunk(jCas, namedEntityStart, documentText.length()-1);
260          // chunk.setTokens(UIMAUtil.toFSArray(jCas, namedEntityTokens));
261          // chunk.setChunkType("CoNLL NEM annotation");
262          // chunk.addToIndexes();
263          nem.setAnnotation(annotation);
264          nem.setHead(annotation);
265          nem.setMentionedEntity(ne);
266          nem.addToIndexes();
267    
268          ne.setMentions(UIMAUtil.toFSArray(jCas, Collections.singletonList(nem)));
269        }
270      }
271    
272      private void initNamedEntity(String namedEntityType) {
273        namedEntityStart = documentText.length();
274        namedEntityTokens.clear();
275        currentNamedEntityType = namedEntityType;
276      }
277    
278      /**
279       * Determines if the read type is the same as the current type with the only difference being that
280       * the current type (bType) starts with "B-" and the read type starts with "I-".
281       * 
282       * @param bType
283       *          - the current type for the chunk or named entity
284       * @param iType
285       *          - the read chunk or named entity type for the token being examined.
286       * @return true if we should consder iType to be the same as bType so we know not to make a new
287       *         chunk or named entity.
288       */
289      private boolean startsWithB(String bType, String iType) {
290        if (bType.startsWith("B") && iType.startsWith("I")
291            && iType.substring(1).equals(bType.substring(1))) {
292          return true;
293        }
294        return false;
295      }
296    
297      public void close() throws IOException {
298        reader.close();
299      }
300    
301      public Progress[] getProgress() {
302        return new Progress[] { new ProgressImpl(documentIndex, 5000, Progress.ENTITIES) };
303      }
304    
305      public boolean hasNext() throws IOException, CollectionException {
306        return hasNext;
307      }
308    
309      public void setDataFileName(String dataFileName) {
310        this.dataFileName = dataFileName;
311      }
312    
313      public void setLoadNamedEntities(boolean loadNamedEntities) {
314        this.loadNamedEntities = loadNamedEntities;
315      }
