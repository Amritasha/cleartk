025package org.cleartk.util.ae;
026
027import java.io.InputStream;
028import java.net.URI;
029
030import org.apache.uima.analysis_engine.AnalysisEngineDescription;
031import org.apache.uima.analysis_engine.AnalysisEngineProcessException;
032import org.apache.uima.cas.CAS;
033import org.apache.uima.cas.impl.XCASDeserializer;
034import org.apache.uima.cas.impl.XCASSerializer;
035import org.apache.uima.cas.impl.XmiCasDeserializer;
036import org.apache.uima.cas.impl.XmiCasSerializer;
037import org.apache.uima.jcas.JCas;
038import org.apache.uima.resource.ResourceInitializationException;
039import org.cleartk.util.ViewURIUtil;
040import org.uimafit.component.JCasAnnotator_ImplBase;
041import org.uimafit.component.ViewCreatorAnnotator;
042import org.uimafit.descriptor.ConfigurationParameter;
043import org.uimafit.factory.AggregateBuilder;
044import org.uimafit.factory.AnalysisEngineFactory;
045import org.uimafit.factory.ConfigurationParameterFactory;
046
047/**
048 * <br>
049 * Copyright (c) 2012, Regents of the University of Colorado <br>
050 * All rights reserved.
051 * 
052 * Reads contents of URI in URIView and deserializes it into the CAS.
053 * 
054 * @see XmiCasSerializer
055 * @see XCASSerializer
056 * 
057 * @author Lee Becker
058 * 
059 */
060public class UriToXmiCasAnnotator extends JCasAnnotator_ImplBase {
061
062  public static AnalysisEngineDescription getDescription() throws ResourceInitializationException {
063    return AnalysisEngineFactory.createPrimitiveDescription(UriToXmiCasAnnotator.class);
064  }
065
066  /**
067   * Use this aggregate description if UriToDocumentTextAnnotator will be writing to a mapped view.
068   */
069  public static AnalysisEngineDescription getCreateViewAggregateDescription(String targetViewName)
070      throws ResourceInitializationException {
071    AggregateBuilder builder = new AggregateBuilder();
072    builder.add(AnalysisEngineFactory.createPrimitiveDescription(
073        ViewCreatorAnnotator.class,
074        ViewCreatorAnnotator.PARAM_VIEW_NAME,
075        targetViewName));
076    builder.add(UriToXmiCasAnnotator.getDescription(), CAS.NAME_DEFAULT_SOFA, targetViewName);
077    return builder.createAggregateDescription();
078  }
079
080  public static enum XmlScheme {
081    XMI, XCAS
082  }
083
084  public static final String PARAM_XML_SCHEME = ConfigurationParameterFactory.createConfigurationParameterName(
085      UriToXmiCasAnnotator.class,
086      "xmlScheme");
087
088  @ConfigurationParameter(
089      defaultValue = "XMI",
090      description = "specifies the UIMA XML serialization scheme that should be used. Valid values for this parameter are 'XMI' and 'XCAS'")
091  private XmlScheme xmlScheme;
092
093  @Override
094  public void process(JCas jCas) throws AnalysisEngineProcessException {
095    URI uri = ViewURIUtil.getURI(jCas);
096
097    InputStream inputStream = null;
098    try {
099      inputStream = uri.toURL().openStream();
100      switch (this.xmlScheme) {
101        case XMI:
102          XmiCasDeserializer.deserialize(inputStream, jCas.getCas());
103          break;
104        case XCAS:
105          XCASDeserializer.deserialize(inputStream, jCas.getCas());
106          break;
107      }
108      inputStream.close();
109    } catch (Exception e) {
110      throw new AnalysisEngineProcessException(e);
111    }
112
113  }
114
115}
