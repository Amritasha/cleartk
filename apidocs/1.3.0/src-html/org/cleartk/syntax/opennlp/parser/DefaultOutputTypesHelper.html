050public class DefaultOutputTypesHelper<TOKEN_TYPE extends Annotation, SENTENCE_TYPE extends Annotation>
051    implements OutputTypesHelper<TOKEN_TYPE, SENTENCE_TYPE, Parse, TreebankNode> {
052
053  @Override
054  public TreebankNode addParse(
055      JCas jCas,
056      Parse parse,
057      SENTENCE_TYPE sentence,
058      List<TOKEN_TYPE> tokens) {
059
060    TopTreebankNode node = new TopTreebankNode(jCas);
061    this.setAttributes(node, parse, null, jCas);
062    node.addToIndexes();
063
064    StringBuffer sb = new StringBuffer();
065    parse.show(sb);
066    node.setTreebankParse(sb.toString());
067    node.setTerminals(UIMAUtil.toFSArray(jCas, getTerminals(node)));
068    return node;
069  }
070
071  private void setAttributes(TreebankNode node, Parse parse, TreebankNode parent, JCas jCas) {
072    node.setParent(parent);
073    node.setNodeType(parse.getType());
074    node.setBegin(parse.getSpan().getStart());
075    node.setEnd(parse.getSpan().getEnd());
076
077    // leaf node
078    if (isLeaf(parse)) {
079      node.setLeaf(true);
080      node.setNodeValue(parse.getChildren()[0].toString());
081      node.setChildren(new FSArray(jCas, 0));
082    }
083
084    // branch node
085    else {
086      node.setLeaf(false);
087      node.setNodeValue(null);
088      List<TreebankNode> childNodes = new ArrayList<TreebankNode>();
089      for (Parse childParse : parse.getChildren()) {
090        TreebankNode childNode = isLeaf(childParse)
091            ? new TerminalTreebankNode(jCas)
092            : new TreebankNode(jCas);
093        this.setAttributes(childNode, childParse, node, jCas);
094        childNode.addToIndexes();
095        childNodes.add(childNode);
096      }
097      node.setChildren(UIMAUtil.toFSArray(jCas, childNodes));
098    }
099  }
100
101  protected List<TreebankNode> getTerminals(TreebankNode node) {
102    List<TreebankNode> tList = new ArrayList<TreebankNode>();
103    int nChildren = node.getChildren().size();
104    if (nChildren == 0) {
105      tList.add(node);
106    }
107    for (int i = 0; i < nChildren; ++i) {
108      tList.addAll(getTerminals(node.getChildren(i)));
109    }
110    return tList;
111  }
112
113  private boolean isLeaf(Parse parse) {
114    Parse[] childParses = parse.getChildren();
115    return childParses.length == 1 && childParses[0].getType() == AbstractBottomUpParser.TOK_NODE;
116  }
117}
