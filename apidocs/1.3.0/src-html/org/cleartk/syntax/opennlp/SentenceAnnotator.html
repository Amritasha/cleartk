061 * @author Philip Ogren
062 * @author Lee Becker
063 * 
064 *         This sentence segmenter is a simple wrapper around the OpenNLP SentenceDetector with
065 *         additional sentence detection added that handles multiple newlines (i.e. if multiple
066 *         newlines appear (separated only by whitespace) together, then this is treated as a
067 *         sentence delimiter.
068 * 
069 * @see SentenceDetector
070 */

072@TypeCapability(outputs = "org.cleartk.token.type.Sentence")
073public class SentenceAnnotator extends JCasAnnotator_ImplBase {
074
075  public static AnalysisEngineDescription getDescription() throws ResourceInitializationException {
076    return AnalysisEngineFactory.createPrimitiveDescription(
077        SentenceAnnotator.class,
078        PARAM_SENTENCE_MODEL_PATH,
079        ParamUtil.getParameterValue(PARAM_SENTENCE_MODEL_PATH, "/models/en-sent.bin"),
080        PARAM_WINDOW_CLASS_NAMES,
081        ParamUtil.getParameterValue(PARAM_WINDOW_CLASS_NAMES, null));
082  }
083
084  public static final String PARAM_SENTENCE_MODEL_PATH = ConfigurationParameterFactory.createConfigurationParameterName(
085      SentenceAnnotator.class,
086      "sentenceModelPath");
087
088  @ConfigurationParameter(
089      mandatory = true,
090      description = "provides the path of the OpenNLP sentence segmenter model file")
091  private String sentenceModelPath;
092
093  public static final String PARAM_SENTENCE_TYPE_NAME = ConfigurationParameterFactory.createConfigurationParameterName(
094      SentenceAnnotator.class,
095      "sentenceTypeName");
096
097  public static final String PARAM_WINDOW_CLASS_NAMES = ConfigurationParameterFactory.createConfigurationParameterName(
098      SentenceAnnotator.class,
099      "windowClassNames");
100
101  @ConfigurationParameter(
102      mandatory = false,
103      description = "provides an array of the annotation types that will be processed by this sentence annotator.  If the parameter is not filled, then SentenceAnnotator will process on the contents of jCas.getDocumentText().  It us up to the caller to ensure annotations do not overlap.")
104  private String[] windowClassNames;
105
106  @ConfigurationParameter(
107      description = "class type of the sentences that are created by this annotator. If this parameter is not filled, then sentencesof type org.cleartk.type.Sentence will be created.",
108      defaultValue = "org.cleartk.token.type.Sentence")
109  private String sentenceTypeName;
110
111  Class<? extends Annotation> sentenceClass;
112
113  protected List<Class<? extends Annotation>> windowClasses;
114
115  Constructor<? extends Annotation> sentenceConstructor;
116
117  public static final String multipleNewlinesRegex = "\\s*\\n\\s*\\n\\s*";
118
119  SentenceDetector sentenceDetector;
120
121  Pattern multipleNewlinesPattern;
122
123  Pattern leadingWhitespacePattern;
124
125  Pattern trailingWhitespacePattern;
126
127  @Override
128  public void initialize(UimaContext uimaContext) throws ResourceInitializationException {
129    super.initialize(uimaContext);
130
131    try {
132      sentenceClass = InitializableFactory.getClass(sentenceTypeName, Annotation.class);
133      sentenceConstructor = sentenceClass.getConstructor(new Class[] {
134          JCas.class,
135          Integer.TYPE,
136          Integer.TYPE });
137
138      if (windowClassNames != null && windowClassNames.length > 0) {
139        windowClasses = new ArrayList<Class<? extends Annotation>>();
140        for (String windowClassName : windowClassNames) {
141          windowClasses.add(InitializableFactory.getClass(windowClassName, Annotation.class));
142        }
143      }
144
145      InputStream modelInputStream = IOUtil.getInputStream(
146          SentenceAnnotator.class,
147          sentenceModelPath);
148      SentenceModel model = new SentenceModel(modelInputStream);
149      sentenceDetector = new SentenceDetectorME(model);
150      multipleNewlinesPattern = Pattern.compile(multipleNewlinesRegex, Pattern.MULTILINE
151          | Pattern.DOTALL);
152      leadingWhitespacePattern = Pattern.compile("^\\s+");
153      trailingWhitespacePattern = Pattern.compile("\\s+$");
154    } catch (Exception e) {
155      throw new ResourceInitializationException(e);
156    }
157  }
158
159  @Override
160  public void process(JCas jCas) throws AnalysisEngineProcessException {
161    if (windowClasses == null) {
162      // No window class names are set, operate on the entirety of the CAS' document text
163      String text = jCas.getDocumentText();
164      processText(jCas, text, 0);
165    } else {
166      // Window class names are specified, iterate over all annotations of the specified classes
167      for (Class<? extends Annotation> windowClass : windowClasses) {
168        // make list copy to avoid concurrent modification
169        for (Annotation window : new ArrayList<Annotation>(JCasUtil.select(jCas, windowClass))) {
170          String text = window.getCoveredText();
171          processText(jCas, text, window.getBegin());
172        }
173      }
174    }
175  }
176
177  protected void processText(JCas jCas, String text, int textOffset)
178      throws AnalysisEngineProcessException {
179    List<Integer> sentenceOffsets = getSentenceOffsets(text);
180
181    int begin = 0;
182    int end = 0;
183
184    // advance the first sentence to first non-whitespace char
185    Matcher matcher;
186    matcher = leadingWhitespacePattern.matcher(text);
187    if (matcher.find()) {
188      begin += matcher.group().length();
189    }
190    try {
191      for (Integer offset : sentenceOffsets) {
192        end = offset; // offset is really the beginning of the next sentence so we may adjust this
193                      // below.
194        String sentenceText = text.substring(begin, end);
195        // it is possible that there is a duplicate offset or that getSentenceOffsets returned the
196        // first sentence offset
197        if (sentenceText.trim().length() > 0) {
198          matcher = trailingWhitespacePattern.matcher(sentenceText);
199          if (matcher.find()) {
200            end -= matcher.group().length();
201          }
202          sentenceConstructor.newInstance(jCas, textOffset + begin, textOffset + end).addToIndexes();
203        }
204        begin = offset; // we need to advance begin regardless of whether a sentence was created.
205      }
206      // take the remaining text if there is any and add it to a sentence.
207      // this code will not execute if the text ends with a sentence detected by
208      // SentenceDetector because it actually returns an offset corresponding to the end
209      // of the last sentence (see note on getSentenceOffsets)
210      if (begin < text.length()) {
211        String sentenceText = text.substring(begin, text.length());
212        end = text.length();
213        if (sentenceText.trim().length() > 0) {
214          matcher = trailingWhitespacePattern.matcher(sentenceText);
215          if (matcher.find()) {
216            end -= matcher.group().length();
217          }
218          sentenceConstructor.newInstance(jCas, textOffset + begin, textOffset + end).addToIndexes();
219        }
220      }
221    } catch (Exception e) {
222      throw new AnalysisEngineProcessException(e);
223    }
224  }
225
226  /**
227   * returns a list of the beginnings of sentences - except (possibly) the first sentence - from
228   * both the OpenNLP sentence detector and the multiple newlines regex. It is possible for this
229   * method to return duplicate values. The sentence detector will return an offset corresponding to
230   * the end of the text if the last non-whitespace character was classified an end of sentence
231   * character.
232   */
233  private List<Integer> getSentenceOffsets(String text) {
234    Matcher matcher = multipleNewlinesPattern.matcher(text);
235    List<Integer> offsets = new ArrayList<Integer>();
236    while (matcher.find()) {
237      offsets.add(matcher.end());
238    }
239
240    Span[] sentenceOffsetsML = sentenceDetector.sentPosDetect(text);
241    for (int i = 0; i < sentenceOffsetsML.length; i++) {
242      offsets.add(sentenceOffsetsML[i].getStart());
243    }
244    Collections.sort(offsets);
245    return offsets;
246  }
