042public class TreebankNodeUtil {
043
044  /**
045   * Selects a single TreebankNode leaf that has the same span as the given annotation.
046   * 
047   * @param jCas
048   *          The JCas containing the TreebankNodes.
049   * @param annotation
050   *          The Annotation whose span should match a TreebankNode leaf.
051   * @return The single TreebankNode leaf that matches the annotation, or null if no such annotation
052   *         exists.
053   */
054  public static TreebankNode selectMatchingLeaf(JCas jCas, Annotation annotation) {
055    TreebankNode leaf = null;
056    for (TreebankNode node : JCasUtil.selectCovered(jCas, TreebankNode.class, annotation)) {
057      if (node.getLeaf() && node.getBegin() == annotation.getBegin()
058          && node.getEnd() == annotation.getEnd()) {
059        if (leaf == null) {
060          leaf = node;
061        } else {
062          throw new IllegalArgumentException(String.format(
063              "expected one leaf matching annotation %s, found %s",
064              annotation,
065              Arrays.asList(leaf, node)));
066        }
067      }
068    }
069    return leaf;
070  }
071
072  /**
073   * Selects the highest TreebankNode in the parse tree that has the same span as the given
074   * annotation.
075   * 
076   * @param jCas
077   *          The JCas containing the TreebankNodes.
078   * @param annotation
079   *          The Annotation whose span should be matched.
080   * @return The highest TreebankNode matching the given span, or null if no such annotation exists.
081   */
082  public static TreebankNode selectHighestMatchingTreebankNode(JCas jCas, Annotation annotation) {
083    TreebankNode highestNode = null;
084    int smallestDepth = Integer.MAX_VALUE;
085    for (TreebankNode node : JCasUtil.selectCovered(jCas, TreebankNode.class, annotation)) {
086      if (node.getBegin() == annotation.getBegin() && node.getEnd() == annotation.getEnd()) {
087        int depth = getDepth(node);
088        if (depth < smallestDepth) {
089          highestNode = node;
090          smallestDepth = depth;
091        }
092      }
093    }
094    return highestNode;
095  }
096
097  /**
098   * Selects the highest TreebankNode in the parse tree that is at least partially covered by the
099   * given annotation.
100   * 
101   * @param jCas
102   *          The JCas containing the TreebankNodes.
103   * @param annotation
104   *          The Annotation whose span should be matched.
105   * @return The highest TreebankNode at least partially covered by the given span, or null if no
106   *         such annotation exists.
107   */
108  public static TreebankNode selectHighestCoveredTreebankNode(JCas jCas, Annotation annotation) {
109    TreebankNode highestNode = null;
110    int smallestDepth = Integer.MAX_VALUE;
111    for (TreebankNode node : JCasUtil.selectCovered(jCas, TreebankNode.class, annotation)) {
112      if (annotation.getBegin() <= node.getBegin() && node.getEnd() <= annotation.getEnd()) {
113        int depth = getDepth(node);
114        if (depth < smallestDepth) {
115          highestNode = node;
116          smallestDepth = depth;
117        }
118      }
119    }
120    return highestNode;
121  }
122
123  /**
124   * Calculates the depth of the TreebankNode. The root node has depth 0, children of the root node
125   * have depth 1, etc.
126   * 
127   * @param node
128   *          The TreebankNode whose depth is to be calculated.
129   * @return The depth of the TreebankNode.
130   */
131  public static int getDepth(TreebankNode node) {
132    int depth = -1;
133    while (node != null) {
134      depth += 1;
135      node = node.getParent();
136    }
137    return depth;
138  }
139
140  /**
141   * Find the path from a TreebankNode to the root of the tree it belongs to.
142   * 
143   * @param startNode
144   *          The start node of the path
145   * 
146   * @return A list of TreebankNodes that make up the path from <b>startNode</b> to the root of the
147   *         tree
148   */
149  public static List<TreebankNode> getPathToRoot(TreebankNode startNode) {
150    List<TreebankNode> nlist = new ArrayList<TreebankNode>(20);
151    TreebankNode cursorNode = startNode;
152
153    while (cursorNode != null) {
154      nlist.add(cursorNode);
155      cursorNode = cursorNode.getParent();
156    }
157
158    return nlist;
159  }
160
161  /**
162   * Representation of a path from one TreebankNode to another, via a common ancestor in the tree.
163   */
164  public static class TreebankNodePath {
165    private List<TreebankNode> sourceToAncestor;
166
167    private TreebankNode commonAncestor;
168
169    private List<TreebankNode> targetToAncestor;
170
171    public TreebankNodePath(
172        TreebankNode commonAncestor,
173        List<TreebankNode> sourceToAncestor,
174        List<TreebankNode> targetToAncestor) {
175      this.commonAncestor = commonAncestor;
176      this.sourceToAncestor = sourceToAncestor;
177      this.targetToAncestor = targetToAncestor;
178    }
179
180    public TreebankNode getCommonAncestor() {
181      return this.commonAncestor;
182    }
183
184    public List<TreebankNode> getSourceToAncestorPath() {
185      return this.sourceToAncestor;
186    }
187
188    public List<TreebankNode> getTargetToAncestorPath() {
189      return this.targetToAncestor;
190    }
191  }
192
193  /**
194   * Get the path from the source TreebankNode to the target TreebankNode via the least common
195   * ancestor.
196   * 
197   * @param source
198   *          The TreebankNode where the path should start.
199   * @param target
200   *          The TreebankNode where the path should end.
201   * @return The path from the source node to the target node.
202   */
203  public static TreebankNodePath getPath(TreebankNode source, TreebankNode target) {
204    List<TreebankNode> sourceToRoot = getPathToRoot(source);
205    List<TreebankNode> targetToRoot = getPathToRoot(target);
206
207    TreebankNode ancestor = null;
208    while (sourceToRoot.size() > 0 && targetToRoot.size() > 0
209        && sourceToRoot.get(sourceToRoot.size() - 1) == targetToRoot.get(targetToRoot.size() - 1)) {
210      ancestor = sourceToRoot.remove(sourceToRoot.size() - 1);
211      ancestor = targetToRoot.remove(targetToRoot.size() - 1);
212    }
213
214    return new TreebankNodePath(ancestor, sourceToRoot, targetToRoot);
215  }
216
217  /**
218   * Format the TreebankNode as a Penn-Treebank-style parenthesized string.
219   * 
220   * @param node
221   *          The TreebankNode to be formatted.
222   * @return A parenthesized Penn-Treebank-style string.
223   */
224  public static String toTreebankString(TreebankNode node) {
225    StringBuilder builder = new StringBuilder();
226    builder.append('(').append(node.getNodeType());
227    if (node.getLeaf()) {
228      builder.append(' ').append(node.getCoveredText());
229    } else {
230      for (TreebankNode child : UIMAUtil.toList(node.getChildren(), TreebankNode.class)) {
231        builder.append(' ').append(toTreebankString(child));
232      }
233    }
234    builder.append(')');
235    return builder.toString();
236  }
237
238  public static TreebankNode getParent(TreebankNode node) {
239    if (node != null) {
240      node = node.getParent();
241    }
242    return node;
243  }
244
245  public static TreebankNode getAncestorWithType(TreebankNode node, String type) {
246    while (node != null && !node.getNodeType().equals(type)) {
247      node = node.getParent();
248    }
249    return node;
250  }
