024package org.cleartk.ne.term.util;
025
026import java.io.BufferedReader;
027import java.io.File;
028import java.io.FileReader;
029import java.io.IOException;
030import java.util.ArrayList;
031import java.util.Collection;
032import java.util.Collections;
033import java.util.List;
034import java.util.regex.Pattern;
035
036/**
037 * <br>
038 * Copyright (c) 2007-2008, Regents of the University of Colorado <br>
039 * All rights reserved.
040 * 
041 * <p>
042 * 
043 * @author Philip Ogren
044 * 
045 *         This class provides a very simple data structure for a list of terms. There is no real
046 *         expectation that a TermFinder implementation will make any meaningful use of this class
047 *         other than to obtain the terms as a collection. A term finder should determine how to
048 *         optimally access the contents of a list with its own data structures.
049 */
050public class TermList {
051  private String name;
052
053  private List<Term> terms;
054
055  public TermList(String name) {
056    super();
057    this.name = name;
058    this.terms = new ArrayList<Term>();
059  }
060
061  public String getName() {
062    return name;
063  }
064
065  /**
066   * A term finder is not expected to automatically update its internal representation of a term
067   * list when this method is called. This method should be called when populating the term list
068   * before a term finder calls getTerms().
069   */
070  public void add(Term term) {
071    this.terms.add(term);
072  }
073
074  public Collection<Term> getTerms() {
075    return Collections.unmodifiableCollection(terms);
076  }
077
078  public int size() {
079    return terms.size();
080  }
081
082  /**
083   * Calls loadSimpleFile(listName, file, null).
084   * 
085   * @see #loadSimpleFile(String, File, String)
086   */
087  public static TermList loadSimpleFile(String listName, File file) throws IOException {
088    return loadSimpleFile(listName, file, null);
089  }
090
091  /**
092   * This method expects a very simple format. Each line should contain an id and a term delimited
093   * by some separator. e.g.:
094   * <p>
095   * 1|Aixas <br>
096   * 2|Aixirivali <br>
097   * ... <br>
098   * 
099   * Ids can be omitted in which case there should be no separator and the id of each term in the
100   * list will correspond to the line number that the term appears on.
101   * 
102   * @param listName
103   *          the name of the list
104   * @param file
105   *          a file that contains one term per line as described above.
106   * @param columnSeparator
107   *          the string that separates the id from the term. If null is passed in then the id of
108   *          each term will be the line number as it appears in the file.
109   * @return a term list populated with the data in the provided file
110   */
111  public static TermList loadSimpleFile(String listName, File file, String columnSeparator)
112      throws IOException {
113
114    TermList termList = new TermList(listName);
115
116    BufferedReader input = new BufferedReader(new FileReader(file));
117    String line;
118    int i = 1;
119    try {
120      while ((line = input.readLine()) != null) {
121        line = line.trim();
122        String id;
123        String termText;
124        if (columnSeparator == null) {
125          id = "" + i++;
126          termText = line.trim();
127        } else {
128          String[] columns = line.split(Pattern.quote(columnSeparator));
129          id = columns[0];
130          termText = columns[1].trim();
131        }
132        Term term = new Term(id, termText, termList);
133        termList.add(term);
134        // if (i % 100000 == 0) System.out.println("loaded " + i + " terms from term list: " +
135        // file.getName() + ".");
136      }
137      // System.out.println("loaded " + (i - 1) + " terms from term list: " + file.getName() + ".");
138    } finally {
139      input.close();
140    }
141    return termList;
142  }
143}
