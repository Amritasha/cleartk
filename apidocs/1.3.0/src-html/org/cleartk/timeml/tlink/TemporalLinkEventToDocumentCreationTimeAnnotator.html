051 * @author Steven Bethard
052 */
053public class TemporalLinkEventToDocumentCreationTimeAnnotator extends
054    TemporalLinkAnnotator_ImplBase<Event, DocumentCreationTime> {
055
056  public static final CleartkInternalModelFactory FACTORY = new CleartkInternalModelFactory() {
057    @Override
058    public Class<?> getAnnotatorClass() {
059      return TemporalLinkEventToDocumentCreationTimeAnnotator.class;
060    }
061
062    @Override
063    public Class<?> getDataWriterClass() {
064      return MaxentStringOutcomeDataWriter.class;
065    }
066
067    @Override
068    public AnalysisEngineDescription getBaseDescription() throws ResourceInitializationException {
069      return AnalysisEngineFactory.createPrimitiveDescription(TemporalLinkEventToDocumentCreationTimeAnnotator.class);
070    }
071  };
072
073  public TemporalLinkEventToDocumentCreationTimeAnnotator() {
074    super(Event.class, DocumentCreationTime.class, "BEFORE", "OVERLAP", "AFTER");
075  }
076
077  @Override
078  public void initialize(UimaContext context) throws ResourceInitializationException {
079    super.initialize(context);
080
081    // I explored a ton of features here, and the following were the only ones that worked
082    // The only feature that I didn't try that seems like it might still have some promise
083    // would be to find any times within, say, 5 tokens, and do the time value comparison
084    // to see whether the nearby time is before, overlapping with or after the DCT
085    this.setSourceExtractors(Arrays.asList(
086        new TypePathExtractor(Event.class, "tense"),
087        new TypePathExtractor(Event.class, "aspect"),
088        new TypePathExtractor(Event.class, "eventClass"),
089        new TypePathExtractor(Event.class, "polarity"),
090        new TypePathExtractor(Event.class, "modality"),
091        // the word, but only if it's an aspectual event
092        new FilteringExtractor<Event>(Event.class, new CoveredTextExtractor()) {
093          @Override
094          protected boolean accept(Event event) {
095            return event.getEventClass().equals("ASPECTUAL");
096          }
097        }));
098  }
099
100  @Override
101  public void process(JCas jCas) throws AnalysisEngineProcessException {
102    Map<Event, Map<DocumentCreationTime, TemporalLink>> links = this.getLinks(jCas);
103
104    Collection<DocumentCreationTime> dcts = JCasUtil.select(jCas, DocumentCreationTime.class);
105    if (dcts.size() != 1) {
106      throw new RuntimeException("expected 1 DocumentCreationTime, found " + dcts.size());
107    }
108    DocumentCreationTime dct = dcts.iterator().next();
109
110    for (Event event : JCasUtil.select(jCas, Event.class)) {
111      this.processLink(event, dct, links, jCas);
112    }
113
114    this.logSkippedLinks(jCas, links);
115  }
116}
