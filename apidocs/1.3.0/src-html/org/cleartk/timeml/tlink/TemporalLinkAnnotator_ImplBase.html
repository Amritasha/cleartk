059public abstract class TemporalLinkAnnotator_ImplBase<SOURCE extends Anchor, TARGET extends Anchor>
060    extends CleartkAnnotator<String> {
061
062  private Class<SOURCE> sourceClass;
063
064  private Class<TARGET> targetClass;
065
066  private Set<String> trainingRelationTypes;
067
068  protected List<SimpleFeatureExtractor> sourceExtractors;
069
070  protected List<SimpleFeatureExtractor> targetExtractors;
071
072  protected List<BetweenAnnotationsFeatureExtractor> betweenExtractors;
073
074  public TemporalLinkAnnotator_ImplBase(
075      Class<SOURCE> sourceClass,
076      Class<TARGET> targetClass,
077      String... trainingRelationTypes) {
078    this.sourceClass = sourceClass;
079    this.targetClass = targetClass;
080    this.trainingRelationTypes = new HashSet<String>(Arrays.asList(trainingRelationTypes));
081    this.sourceExtractors = new ArrayList<SimpleFeatureExtractor>();
082    this.targetExtractors = new ArrayList<SimpleFeatureExtractor>();
083    this.betweenExtractors = new ArrayList<BetweenAnnotationsFeatureExtractor>();
084  }
085
086  protected void setSourceExtractors(List<SimpleFeatureExtractor> extractors) {
087    this.sourceExtractors = new ArrayList<SimpleFeatureExtractor>();
088    for (SimpleFeatureExtractor extractor : extractors) {
089      this.sourceExtractors.add(new NamingExtractor("Source", extractor));
090    }
091  }
092
093  protected void setTargetExtractors(List<SimpleFeatureExtractor> extractors) {
094    this.targetExtractors = new ArrayList<SimpleFeatureExtractor>();
095    for (SimpleFeatureExtractor extractor : extractors) {
096      this.targetExtractors.add(new NamingExtractor("Target", extractor));
097    }
098  }
099
100  protected void setBetweenExtractors(List<BetweenAnnotationsFeatureExtractor> extractors) {
101    this.betweenExtractors = extractors;
102  }
103
104  protected Map<SOURCE, Map<TARGET, TemporalLink>> getLinks(JCas jCas) {
105    Map<SOURCE, Map<TARGET, TemporalLink>> links = new HashMap<SOURCE, Map<TARGET, TemporalLink>>();
106    if (this.isTraining()) {
107      for (TemporalLink tlink : JCasUtil.select(jCas, TemporalLink.class)) {
108        SOURCE source;
109        try {
110          source = this.sourceClass.cast(tlink.getSource());
111        } catch (ClassCastException e) {
112          String message = "Expected all sources to be of type %s, found %s";
113          throw new RuntimeException(String.format(message, this.sourceClass, tlink.getSource()));
114        }
115        TARGET target;
116        try {
117          target = this.targetClass.cast(tlink.getTarget());
118        } catch (ClassCastException e) {
119          String message = "Expected all targets to be of type %s, found %s";
120          throw new RuntimeException(String.format(message, this.targetClass, tlink.getSource()));
121        }
122        if (!links.containsKey(source)) {
123          links.put(source, new HashMap<TARGET, TemporalLink>());
124        }
125        if (links.get(source).containsKey(target)) {
126          String message = "Duplicate link between source %s and target %s";
127          throw new RuntimeException(String.format(message, source.getId(), target.getId()));
128        }
129        links.get(source).put(target, tlink);
130      }
131    }
132    return links;
133  }
134
135  protected void processLink(
136      SOURCE source,
137      TARGET target,
138      Map<SOURCE, Map<TARGET, TemporalLink>> links,
139      JCas jCas) throws CleartkProcessingException {
140
141    List<Feature> features = new ArrayList<Feature>();
142    for (SimpleFeatureExtractor extractor : this.sourceExtractors) {
143      features.addAll(extractor.extract(jCas, source));
144    }
145    for (SimpleFeatureExtractor extractor : this.targetExtractors) {
146      features.addAll(extractor.extract(jCas, target));
147    }
148    for (BetweenAnnotationsFeatureExtractor extractor : this.betweenExtractors) {
149      features.addAll(extractor.extractBetween(jCas, source, target));
150    }
151    if (this.isTraining()) {
152      if (links.containsKey(source) && links.get(source).containsKey(target)) {
153        TemporalLink link = null;
154        Map<TARGET, TemporalLink> targetLinks = links.get(source);
155        if (targetLinks != null) {
156          link = targetLinks.remove(target);
157          if (targetLinks.isEmpty()) {
158            links.remove(source);
159          }
160        }
161        if (link != null) {
162          String relation = link.getRelationType();
163          if (this.trainingRelationTypes.contains(relation)) {
164            this.dataWriter.write(new Instance<String>(relation, features));
165          }
166        }
167      }
168    } else {
169      String relation = this.classifier.classify(features);
170      int offset = jCas.getDocumentText().length();
171      TemporalLink tlink = new TemporalLink(jCas, offset, offset);
172      tlink.setSource(source);
173      tlink.setTarget(target);
174      tlink.setRelationType(relation);
175      tlink.addToIndexes();
176    }
177  }
178
179  protected void logSkippedLinks(JCas jCas, Map<SOURCE, Map<TARGET, TemporalLink>> links) {
180    if (!links.isEmpty()) {
181      int count = 0;
182      for (Map<TARGET, TemporalLink> targetLinks : links.values()) {
183        count += targetLinks.size();
184      }
185      Map<Anchor, Sentence> sentences = new HashMap<Anchor, Sentence>();
186      Map<Sentence, Integer> sentenceIndexes = new HashMap<Sentence, Integer>();
187      int index = 0;
188      for (Sentence sentence : JCasUtil.select(jCas, Sentence.class)) {
189        sentenceIndexes.put(sentence, index);
190        index += 1;
191        for (SOURCE source : JCasUtil.selectCovered(jCas, this.sourceClass, sentence)) {
192          sentences.put(source, sentence);
193        }
194        for (TARGET target : JCasUtil.selectCovered(jCas, this.targetClass, sentence)) {
195          sentences.put(target, sentence);
196        }
197      }
198      StringBuilder errorBuilder = new StringBuilder();
199      errorBuilder.append("Missed ").append(count).append(" TLINK(s)\n");
200      List<SOURCE> sources = new ArrayList<SOURCE>(links.keySet());
201      Collections.sort(sources, new Comparator<SOURCE>() {
202        @Override
203        public int compare(SOURCE source1, SOURCE source2) {
204          return source1.getBegin() - source2.getBegin();
205        }
206      });
207      for (SOURCE source : sources) {
208        for (TARGET target : links.get(source).keySet()) {
209          TemporalLink link = links.get(source).get(target);
210          Sentence sent1 = sentences.get(source);
211          Sentence sent2 = sentences.get(target);
212          errorBuilder.append(String.format(
213              "%s(%s, %s)\n%d: %s\n%d: %s\n",
214              link.getRelationType(),
215              source.getCoveredText(),
216              target.getCoveredText(),
217              sentenceIndexes.get(sent1),
218              sent1.getCoveredText(),
219              sentenceIndexes.get(sent2),
220              sent2.getCoveredText()));
221        }
222      }
223      this.getContext().getLogger().log(Level.FINE, errorBuilder.toString());
224    }
225  }

227  protected static String noLeavesPath(TreebankNodePath path) {
228    if (path.getCommonAncestor() == null) {
229      return null;
230    }
231    List<String> sourceTypes = new ArrayList<String>();
232    for (TreebankNode node : path.getSourceToAncestorPath()) {
233      if (!node.getLeaf()) {
234        sourceTypes.add(node.getNodeType());
235      }
236    }
237    List<String> targetTypes = new ArrayList<String>();
238    for (TreebankNode node : path.getTargetToAncestorPath()) {
239      if (!node.getLeaf()) {
240        targetTypes.add(node.getNodeType());
241      }
242    }
243    Collections.reverse(targetTypes);
244    StringBuilder builder = new StringBuilder();
245    for (String type : sourceTypes) {
246      builder.append(type).append('>');
247    }
248    builder.append(path.getCommonAncestor().getNodeType());
249    for (String type : targetTypes) {
250      builder.append('<').append(type);
251    }
252    return builder.toString();
253  }
