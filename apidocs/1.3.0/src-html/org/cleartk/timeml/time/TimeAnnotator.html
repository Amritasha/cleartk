056/**
057 * <br>
058 * Copyright (c) 2011, Regents of the University of Colorado <br>
059 * All rights reserved.
060 * 
061 * @author Steven Bethard
062 */
063public class TimeAnnotator extends CleartkSequenceAnnotator<String> {
064
065  public static final CleartkInternalModelFactory FACTORY = new CleartkInternalModelFactory() {
066
067    @Override
068    public Class<?> getAnnotatorClass() {
069      return TimeAnnotator.class;
070    }
071
072    @Override
073    public Class<?> getDataWriterClass() {
074      return MalletCRFStringOutcomeDataWriter.class;
075    }
076
077    @Override
078    public AnalysisEngineDescription getBaseDescription() throws ResourceInitializationException {
079      return AnalysisEngineFactory.createPrimitiveDescription(TimeAnnotator.class);
080    }
081  };
082
083  private List<SimpleFeatureExtractor> tokenFeatureExtractors;
084
085  private List<CleartkExtractor> contextFeatureExtractors;
086
087  private BIOChunking<Token, Time> chunking;
088
089  @Override
090  public void initialize(UimaContext context) throws ResourceInitializationException {
091    super.initialize(context);
092
093    // define chunking type
094    this.chunking = new BIOChunking<Token, Time>(Token.class, Time.class);
095
096    // add features: word, character pattern, stem, pos
097    this.tokenFeatureExtractors = Arrays.asList(
098        new CoveredTextExtractor(),
099        new CharacterCategoryPatternExtractor(PatternType.REPEATS_MERGED),
100        new TimeWordsExtractor(),
101        new TypePathExtractor(Token.class, "stem"),
102        new TypePathExtractor(Token.class, "pos"));
103
104    // add window of features before and after
105    this.contextFeatureExtractors = new ArrayList<CleartkExtractor>();
106    for (SimpleFeatureExtractor extractor : this.tokenFeatureExtractors) {
107      this.contextFeatureExtractors.add(new CleartkExtractor(Token.class, extractor, new Preceding(
108          3), new Following(3)));
109    }
110  }
111
112  @Override
113  public void process(JCas jCas) throws AnalysisEngineProcessException {
114
115    // classify tokens within each sentence
116    for (Sentence sentence : JCasUtil.select(jCas, Sentence.class)) {
117      List<Token> tokens = JCasUtil.selectCovered(jCas, Token.class, sentence);
118
119      // extract features for all tokens
120      List<List<Feature>> featureLists = new ArrayList<List<Feature>>();
121      for (Token token : tokens) {
122        List<Feature> features = new ArrayList<Feature>();
123        for (SimpleFeatureExtractor extractor : this.tokenFeatureExtractors) {
124          features.addAll(extractor.extract(jCas, token));
125        }
126        for (CleartkExtractor extractor : this.contextFeatureExtractors) {
127          features.addAll(extractor.extractWithin(jCas, token, sentence));
128        }
129        featureLists.add(features);
130      }
131
132      // during training, convert Times to chunk labels and write the training Instances
133      if (this.isTraining()) {
134        List<Time> times = JCasUtil.selectCovered(jCas, Time.class, sentence);
135        List<String> outcomes = this.chunking.createOutcomes(jCas, tokens, times);
136        this.dataWriter.write(Instances.toInstances(outcomes, featureLists));
137      }
138
139      // during prediction, convert chunk labels to Times and add them to the CAS
140      else {
141        List<String> outcomes = this.classifier.classify(featureLists);
142        this.chunking.createChunks(jCas, tokens, outcomes);
143      }
144    }
145
146    // add IDs to all Times
147    int timeIndex = 1;
148    for (Time time : JCasUtil.select(jCas, Time.class)) {
149      String id = String.format("t%d", timeIndex);
150      time.setId(id);
151      timeIndex += 1;
152    }
153  }
154}
