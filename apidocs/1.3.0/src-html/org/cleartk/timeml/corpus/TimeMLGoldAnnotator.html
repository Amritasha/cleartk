079  @ConfigurationParameter(
080      description = "when false indicates that annotation should not be created for TLINKs (though annotations will still be created for TIMEX3s, EVENTs, etc.).",
081      defaultValue = "true")
082  private boolean loadTlinks;

101  @Override
102  public void process(JCas jCas) throws AnalysisEngineProcessException {
103    JCas timemlView;
104    JCas initialView;
105    try {
106      timemlView = jCas.getView(TimeMLViewName.TIMEML);
107      initialView = jCas.getView(CAS.NAME_DEFAULT_SOFA);
108    } catch (CASException e) {
109      throw new AnalysisEngineProcessException(e);
110    }
111
112    String timeML = timemlView.getDocumentText();
113    SAXBuilder builder = new SAXBuilder();
114    builder.setDTDHandler(null);
115    Element root;
116    try {
117      Document doc = builder.build(new StringReader(timeML));
118      root = doc.getRootElement();
119    } catch (JDOMException e) {
120      getContext().getLogger().log(
121          Level.SEVERE,
122          "problem parsing document: " + ViewURIUtil.getURI(jCas));
123      throw new AnalysisEngineProcessException(e);
124    } catch (IOException e) {
125      throw new AnalysisEngineProcessException(e);
126    }
127
128    // collect the document text, add the Event, Time and TemporalLink annotations,
129    // and collect the necessary information to fill in the cross-annotation links
130    StringBuffer textBuffer = new StringBuffer();
131    Map<String, Anchor> anchors = new HashMap<String, Anchor>();
132    Map<Time, String> anchorTimeIDs = new HashMap<Time, String>();
133    List<Element> makeInstances = new ArrayList<Element>();
134    Map<TemporalLink, String> tlinkSourceIDs = new HashMap<TemporalLink, String>();
135    Map<TemporalLink, String> tlinkTargetIDs = new HashMap<TemporalLink, String>();
136    this.addAnnotations(
137        initialView,
138        root,
139        textBuffer,
140        anchors,
141        anchorTimeIDs,
142        makeInstances,
143        tlinkSourceIDs,
144        tlinkTargetIDs);
145    initialView.setDocumentText(textBuffer.toString());
146
147    // point make-instance IDs to their events, and copy attributes over
148    for (Element makeInstance : makeInstances) {
149      String eventID = makeInstance.getAttributeValue("eventID");
150      String eventInstanceID = makeInstance.getAttributeValue("eiid");
151      Event event = (Event) this.getAnchor(jCas, anchors, eventID);
152      anchors.put(eventInstanceID, event);
153      eventInstanceID = event.getEventInstanceID();
154      if (eventInstanceID == null) {
155        TimeMLUtil.copyAttributes(makeInstance, event, jCas);
156      } else {
157        TimeMLUtil.removeInconsistentAttributes(makeInstance, event, jCas);
158        event.setId(eventID);
159        event.setEventInstanceID(eventInstanceID);
160      }
161    }
162
163    // set anchor times
164    for (Time time : anchorTimeIDs.keySet()) {
165      Time anchorTime = (Time) this.getAnchor(jCas, anchors, anchorTimeIDs.get(time));
166      time.setAnchorTime(anchorTime);
167    }
168
169    // set tlink sources and targets
170    for (TemporalLink tlink : tlinkSourceIDs.keySet()) {
171      tlink.setSource(this.getAnchor(jCas, anchors, tlinkSourceIDs.get(tlink)));
172      tlink.setTarget(this.getAnchor(jCas, anchors, tlinkTargetIDs.get(tlink)));
173    }
174  }
175
176  private void addAnnotations(
177      JCas jCas,
178      Element element,
179      StringBuffer textBuffer,
180      Map<String, Anchor> anchors,
181      Map<Time, String> anchorTimeIDs,
182      List<Element> makeInstances,
183      Map<TemporalLink, String> tlinkSourceIDs,
184      Map<TemporalLink, String> tlinkTargetIDs) throws AnalysisEngineProcessException {
185    int startOffset = textBuffer.length();
186    for (Content content : element.getContent()) {
187      if (content instanceof org.jdom2.Text) {
188        textBuffer.append(((org.jdom2.Text) content).getText());
189      } else if (content instanceof Element) {
190        this.addAnnotations(
191            jCas,
192            (Element) content,
193            textBuffer,
194            anchors,
195            anchorTimeIDs,
196            makeInstances,
197            tlinkSourceIDs,
198            tlinkTargetIDs);
199      }
200    }
201    int endOffset = textBuffer.length();
202
203    if (element.getName().equals("TIMEX3")) {
204      String funcInDoc = element.getAttributeValue("functionInDocument");
205      boolean isCreationTime = funcInDoc != null && funcInDoc.equals("CREATION_TIME");
206      Time time = isCreationTime
207          ? new DocumentCreationTime(jCas, startOffset, endOffset)
208          : new Time(jCas, startOffset, endOffset);
209      TimeMLUtil.copyAttributes(element, time, jCas);
210      String anchorTimeID = element.getAttributeValue("anchorTimeID");
211      if (anchorTimeID != null) {
212        anchorTimeIDs.put(time, anchorTimeID);
213      }
214      anchors.put(time.getId(), time);
215      time.addToIndexes();
216    } else if (element.getName().equals("EVENT")) {
217      Event event = new Event(jCas, startOffset, endOffset);
218      TimeMLUtil.copyAttributes(element, event, jCas);
219      anchors.put(event.getId(), event);
220      event.addToIndexes();
221    } else if (element.getName().equals("MAKEINSTANCE")) {
222      makeInstances.add(element);
223    } else if (element.getName().equals("TLINK") && this.loadTlinks) {
224      TemporalLink temporalLink = new TemporalLink(jCas, startOffset, endOffset);
225      TimeMLUtil.copyAttributes(element, temporalLink, jCas);
226      String sourceID = this.getOneOf(element, "eventInstanceID", "eventID", "timeID");
227      String targetID = this.getOneOf(
228          element,
229          "relatedToEventInstance",
230          "relatedToEvent",
231          "relatedToTime");
232      tlinkSourceIDs.put(temporalLink, sourceID);
233      tlinkTargetIDs.put(temporalLink, targetID);
234      temporalLink.addToIndexes();
235    } else if (element.getName().equals("TEXT")) {
236      Text text = new Text(jCas, startOffset, endOffset);
237      text.addToIndexes();
238    } else if (element.getName().toLowerCase().equals("s")) {
239      Sentence sentence = new Sentence(jCas, startOffset, endOffset);
240      sentence.addToIndexes();
241    }
242  }
243
244  private String getOneOf(Element element, String... attributeNames) {
245    for (String name : attributeNames) {
246      String result = element.getAttributeValue(name);
247      if (result != null) {
248        return result;
249      }
250    }
251    throw new RuntimeException(String.format(
252        "unable to find in %s any of the following attributes: %s",
253        element,
254        Arrays.asList(attributeNames)));
255  }
256
257  private Anchor getAnchor(JCas jCas, Map<String, Anchor> anchors, String id)
258      throws AnalysisEngineProcessException {
259    Anchor anchor = anchors.get(id);
260    if (anchor == null) {
261      throw new RuntimeException(String.format(
262          "%s: no anchor for id %s",
263          ViewURIUtil.getURI(jCas),
264          id));
265    }
266    return anchor;
267  }
