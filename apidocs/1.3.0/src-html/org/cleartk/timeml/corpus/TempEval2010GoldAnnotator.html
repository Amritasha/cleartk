159  @Override
160  public void process(JCas jCas) throws AnalysisEngineProcessException {
161
162    // load the sentences and tokens from the view
163    ListMultimap<Integer, String> sentTokens = ArrayListMultimap.create();
164    for (String line : lines(jCas, TimeMLViewName.TEMPEVAL_BASE_SEGMENTATION)) {
165      String[] columns = split(line, "<filename>", "<sent_no>", "<token_no>", "<text>");
166      int sentIndex = new Integer(columns[1]);
167      String text = columns[3];
168      sentTokens.put(sentIndex, text);
169    }
170
171    // create the sentences and tokens
172    Map<String, StringBuilder> textBuilders = new HashMap<String, StringBuilder>();
173    for (String viewName : this.textViews) {
174      StringBuilder textBuilder = new StringBuilder("\n\n"); // leave line for document time
175      JCas view = JCasUtil.getView(jCas, viewName, true);
176      for (int i = 0; i < sentTokens.keySet().size(); ++i) {
177        int sentBegin = textBuilder.length();
178        List<Token> tokens = new ArrayList<Token>();
179        for (String tokenText : sentTokens.get(i)) {
180          int tokenBegin = textBuilder.length();
181          textBuilder.append(tokenText);
182          int tokenEnd = textBuilder.length();
183          textBuilder.append(' ');
184          Token token = new Token(view, tokenBegin, tokenEnd);
185          token.addToIndexes();
186          tokens.add(token);
187        }
188        int sentEnd = textBuilder.length() - 1;
189        textBuilder.setCharAt(sentEnd, '\n');
190        Sentence sentence = new Sentence(view, sentBegin, sentEnd);
191        sentence.addToIndexes();
192      }
193      textBuilders.put(viewName, textBuilder);
194    }
195
196    // add the document creation time
197    for (String line : lines(jCas, TimeMLViewName.TEMPEVAL_DCT)) {
198      String[] dctColumns = split(line, "<filename>", "<dct>");
199      String dctValue = dctColumns[1].replaceAll("(\\d{4})(\\d{2})(\\d{2})", "$1-$2-$3");
200      for (String viewName : this.documentCreationTimeViews) {
201        JCas view = JCasUtil.getView(jCas, viewName, true);
202        DocumentCreationTime docTime = new DocumentCreationTime(view, 1, 1);
203        docTime.setId("t0");
204        docTime.setTimeType("DATE");
205        docTime.setValue(dctValue);
206        docTime.setFunctionInDocument("CREATION_TIME");
207        docTime.addToIndexes();
208      }
209    }
210
211    // add Time annotations
212    addSpans(
213        jCas,
214        TimeMLViewName.TEMPEVAL_TIMEX_EXTENTS,
215        "timex3",
216        this.timeExtentViews,
217        new AnnotationConstructor<Time>() {
218          @Override
219          public Time apply(JCas aJCas, int begin, int end) {
220            return new Time(aJCas, begin, end);
221          }
222        });
223
224    // add Time attributes
225    addAttributes(
226        jCas,
227        TimeMLViewName.TEMPEVAL_TIMEX_ATTRIBUTES,
228        Time.class,
229        this.timeAttributeViews,
230        new AttributeSetter<Time>() {
231          @Override
232          public void apply(Time time, String attrName, String attrValue) {
233            if (attrName.equals("type")) {
234              time.setTimeType(attrValue);
235            } else if (attrName.equals("value")) {
236              time.setValue(attrValue);
237            } else {
238              String message = "Unexpected TIMEX attribute %s=%s";
239              throw new IllegalArgumentException(String.format(message, attrName, attrValue));
240            }
241          }
242        });
243
244    // add Event annotations
245    addSpans(
246        jCas,
247        TimeMLViewName.TEMPEVAL_EVENT_EXTENTS,
248        "event",
249        this.eventExtentViews,
250        new AnnotationConstructor<Event>() {
251          @Override
252          public Event apply(JCas aJCas, int begin, int end) {
253            return new Event(aJCas, begin, end);
254          }
255        });
256
257    // add Event attributes
258    addAttributes(
259        jCas,
260        TimeMLViewName.TEMPEVAL_EVENT_ATTRIBUTES,
261        Event.class,
262        this.eventAttributeViews,
263        new AttributeSetter<Event>() {
264          @Override
265          public void apply(Event event, String attrName, String attrValue) {
266            if (attrName.equals("pos")) {
267              event.setPos(attrValue);
268            } else if (attrName.equals("tense")) {
269              event.setTense(attrValue);
270            } else if (attrName.equals("aspect")) {
271              event.setAspect(attrValue);
272            } else if (attrName.equals("class")) {
273              event.setEventClass(attrValue);
274            } else if (attrName.equals("polarity")) {
275              event.setPolarity(attrValue);
276            } else if (attrName.equals("modality")) {
277              event.setModality(attrValue);
278            } else {
279              String message = "Unexpected EVENT attribute %s=%s";
280              throw new IllegalArgumentException(String.format(message, attrName, attrValue));
281            }
282          }
283        });
284
285    // add TemporalLink annotations
286    addTemporalLinks(
287        jCas,
288        TimeMLViewName.TEMPEVAL_TLINK_DCT_EVENT,
289        textBuilders,
290        this.temporalLinkEventToDocumentCreationTimeViews);
291    addTemporalLinks(
292        jCas,
293        TimeMLViewName.TEMPEVAL_TLINK_TIMEX_EVENT,
294        textBuilders,
295        this.temporalLinkEventToSameSentenceTimeViews);
296    addTemporalLinks(
297        jCas,
298        TimeMLViewName.TEMPEVAL_TLINK_SUBORDINATED_EVENTS,
299        textBuilders,
300        this.temporalLinkEventToSubordinatedEventViews);
301    addTemporalLinks(
302        jCas,
303        TimeMLViewName.TEMPEVAL_TLINK_MAIN_EVENTS,
304        textBuilders,
305        this.temporalLinkMainEventToNextSentenceMainEventViews);
306
307    // set the document text
308    for (String viewName : this.textViews) {
309      JCas view = JCasUtil.getView(jCas, viewName, true);
310      view.setDocumentText(textBuilders.get(viewName).toString());
311    }
312  }

314  private static String[] split(String line, String... expected) {
315    String[] columns = line.split("\t");
316    if (columns.length != expected.length) {
317      throw new IllegalArgumentException(String.format(
318          "Expected % d items, %s, found %d items, %s",
319          expected.length,
320          Joiner.on('\t').join(expected),
321          columns.length,
322          line));
323    }
324    return columns;
325  }
326
327  private static String[] lines(JCas jCas, String viewName) throws AnalysisEngineProcessException {
328    JCas view;
329    try {
330      view = jCas.getView(viewName);
331    } catch (CASException e) {
332      throw new AnalysisEngineProcessException(e);
333    }
334    String text = view.getDocumentText();
335    if (text == null) {
336      throw new IllegalArgumentException("no text in view " + viewName);
337    }
338    return text.length() > 0 ? text.split("\n") : new String[0];
339  }

341  private static interface AnnotationConstructor<T extends Annotation> {
342    public T apply(JCas jCas, int begin, int end);
343  }
344
345  private static <T extends Anchor> void addSpans(
346      JCas jCas,
347      String tabViewName,
348      String elementName,
349      String[] annotationViewNames,
350      AnnotationConstructor<T> constructor) throws AnalysisEngineProcessException {
351    String[] lines = lines(jCas, tabViewName);
352    for (String annotationViewName : annotationViewNames) {
353      JCas view = JCasUtil.getView(jCas, annotationViewName, true);
354      Map<String, T> idMap = new HashMap<String, T>();
355      List<List<Token>> sentenceTokens = new ArrayList<List<Token>>();
356      for (Sentence sentence : JCasUtil.select(view, Sentence.class)) {
357        sentenceTokens.add(JCasUtil.selectCovered(view, Token.class, sentence));
358      }
359      for (String line : lines) {
360        String[] columns = split(
361            line,
362            "<filename>",
363            "<sent_no>",
364            "<token_no>",
365            elementName,
366            "<id>",
367            "1");
368        int sentIndex = Integer.parseInt(columns[1]);
369        int tokenIndex = Integer.parseInt(columns[2]);
370        String id = columns[4];
371        Token token = sentenceTokens.get(sentIndex).get(tokenIndex);
372        if (!idMap.containsKey(id)) {
373          T ann = constructor.apply(view, token.getBegin(), token.getEnd());
374          ann.setId(id);
375          ann.addToIndexes();
376          idMap.put(id, ann);
377        } else {
378          T ann = idMap.get(id);
379          if (token.getBegin() < ann.getBegin()) {
380            ann.setBegin(token.getBegin());
381          }
382          if (token.getEnd() > ann.getEnd()) {
383            ann.setEnd(token.getEnd());
384          }
385        }
386      }
387    }
388  }

390  private static interface AttributeSetter<T extends Annotation> {
391    public void apply(T ann, String attrName, String attrValue);
392  }
393
394  private static <T extends Anchor> void addAttributes(
395      JCas jCas,
396      String tabViewName,
397      Class<T> cls,
398      String[] annotationViewNames,
399      AttributeSetter<T> setter) throws AnalysisEngineProcessException {
400    String[] lines = lines(jCas, tabViewName);
401    for (String annotationViewName : annotationViewNames) {
402      JCas view = JCasUtil.getView(jCas, annotationViewName, false);
403      Map<String, T> idMap = new HashMap<String, T>();
404      for (T anchor : JCasUtil.select(view, cls)) {
405        idMap.put(anchor.getId(), anchor);
406      }
407      for (String line : lines) {
408        String[] columns = split(
409            line,
410            "<filename>",
411            "<sent_no>",
412            "<token_no>",
413            "timex3",
414            "<id>",
415            "1",
416            "<attribute>",
417            "<value>");
418        String id = columns[4];
419        String attrName = columns[6];
420        String attrValue = columns[7];
421        setter.apply(idMap.get(id), attrName, attrValue);
422      }
423    }
424  }
425
426  private static void addTemporalLinks(
427      JCas jCas,
428      String tabViewName,
429      Map<String, StringBuilder> textBuilders,
430      String[] annotationViewNames) throws AnalysisEngineProcessException {
431    String[] lines = lines(jCas, tabViewName);
432    for (String annotationViewName : annotationViewNames) {
433      JCas view = JCasUtil.getView(jCas, annotationViewName, true);
434      Map<String, Anchor> idAnchors = new HashMap<String, Anchor>();
435      for (Anchor anchor : JCasUtil.select(view, Anchor.class)) {
436        idAnchors.put(anchor.getId(), anchor);
437      }
438      StringBuilder textBuilder = textBuilders.get(annotationViewName);
439      for (String line : lines) {
440        String[] columns = split(line, "<filename>", "<eid>", "<tid>", "<relation>");
441        String sourceID = columns[1];
442        String targetID = columns[2];
443        String relation = columns[3];
444        int offset = textBuilder.length();
445        TemporalLink tlink = new TemporalLink(view, offset, offset);
446        tlink.setSource(idAnchors.get(sourceID));
447        tlink.setTarget(idAnchors.get(targetID));
448        tlink.setRelationType(relation);
449        tlink.addToIndexes();
450        textBuilder.append('\n');
451      }
452    }
453  }
