068  public static CollectionReader getCollectionReader(String... dataPaths)
069      throws ResourceInitializationException {
070    List<File> dirs = new ArrayList<File>();
071    for (String path : dataPaths) {
072      dirs.add(new File(path));
073    }
074    return getCollectionReader(dirs);
075  }
076
077  public static CollectionReader getCollectionReader(List<File> dataDirectories)
078      throws ResourceInitializationException {
079    return getCollectionReader(dataDirectories, null);
080  }

082  public static CollectionReader getCollectionReader(
083      List<File> dataDirectories,
084      Set<String> selectedFileNames) throws ResourceInitializationException {
085    // workaround UimaFIT limitation
086    List<String> dirsList = new ArrayList<String>();
087    for (File dir : dataDirectories) {
088      dirsList.add(dir.getPath());
089    }
090    String[] dirs = dirsList.toArray(new String[dirsList.size()]);
091    String[] names = selectedFileNames == null
092        ? null
093        : selectedFileNames.toArray(new String[selectedFileNames.size()]);
094    return CollectionReaderFactory.createCollectionReader(
095        TempEval2010CollectionReader.class,
096        null,
097        PARAM_DATA_DIRECTORIES,
098        dirs,
099        PARAM_SELECTED_FILE_NAMES,
100        names);
101  }

119  protected List<URI> uris;

121  protected int uriIndex;

123  private Map<String, Map<String, String>> viewFileTexts;
124
125  @Override
126  public void initialize(UimaContext context) throws ResourceInitializationException {
127    super.initialize(context);
128
129    try {
130      // assemble URIs for each file in the data
131      this.uriIndex = 0;
132      this.uris = new ArrayList<URI>();
133      for (File dataDirectory : this.dataDirectories) {
134        URI dataURI = dataDirectory.toURI();
135        for (String fileName : getAnnotatedFileNames(dataDirectory)) {
136          if (this.selectedFileNames == null || this.selectedFileNames.contains(fileName)) {
137            URI uri = new URI(dataURI.getScheme(), dataURI.getHost(), dataURI.getPath(), fileName);
138            this.uris.add(uri);
139          }
140        }
141      }
142
143      // group lines by filename
144      this.viewFileTexts = new HashMap<String, Map<String, String>>();
145      for (String viewName : Arrays.asList(
146          TimeMLViewName.TEMPEVAL_BASE_SEGMENTATION,
147          TimeMLViewName.TEMPEVAL_DCT,
148          TimeMLViewName.TEMPEVAL_EVENT_EXTENTS,
149          TimeMLViewName.TEMPEVAL_EVENT_ATTRIBUTES,
150          TimeMLViewName.TEMPEVAL_TIMEX_EXTENTS,
151          TimeMLViewName.TEMPEVAL_TIMEX_ATTRIBUTES,
152          TimeMLViewName.TEMPEVAL_TLINK_DCT_EVENT,
153          TimeMLViewName.TEMPEVAL_TLINK_MAIN_EVENTS,
154          TimeMLViewName.TEMPEVAL_TLINK_SUBORDINATED_EVENTS,
155          TimeMLViewName.TEMPEVAL_TLINK_TIMEX_EVENT)) {
156        // assumes view names are the same as the .tab file names
157        this.viewFileTexts.put(viewName, this.textByFileName(viewName));
158      }
159    } catch (IOException e) {
160      throw new ResourceInitializationException(e);
161    } catch (URISyntaxException e) {
162      throw new ResourceInitializationException(e);
163    }
164  }
165
166  @Override
167  public boolean hasNext() throws IOException, CollectionException {
168    return this.uriIndex < this.uris.size();
169  }
170
171  @Override
172  public void getNext(JCas jCas) throws IOException, CollectionException {
173    URI uri = this.uris.get(this.uriIndex);
174    this.uriIndex += 1;
175    ViewURIUtil.setURI(jCas, uri);
176
177    String fileName = uri.getFragment();
178    for (String viewName : this.viewFileTexts.keySet()) {
179      JCas view;
180      try {
181        view = jCas.createView(viewName);
182      } catch (CASException e) {
183        throw new CollectionException(e);
184      }
185      String text = this.viewFileTexts.get(viewName).get(fileName);
186      view.setDocumentText(text == null ? "" : text);
187    }
188  }
189
190  @Override
191  public Progress[] getProgress() {
192    return new Progress[] { new ProgressImpl(this.uriIndex, this.uris.size(), Progress.ENTITIES) };
193  }

195  private Map<String, String> textByFileName(String tabFileName) throws IOException {
196    // get all variants of the file under all subdirectories
197    List<File> files = new ArrayList<File>();
198    for (File dir : this.dataDirectories) {
199      files.addAll(getTempEvalFiles(dir, tabFileName));
200    }
201
202    // map each file name to its lines
203    ListMultimap<String, String> fileLines = ArrayListMultimap.create();
204    for (File file : files) {
205      for (String line : Files.readLines(file, Charsets.US_ASCII)) {
206        String fileName = getAnnotatedFileName(line);
207        fileLines.put(fileName, line);
208      }
209    }
210
211    // convert lists of lines back into text
212    Map<String, String> fileTexts = new HashMap<String, String>();
213    for (String fileName : fileLines.keySet()) {
214      StringBuilder builder = new StringBuilder();
215      for (String line : fileLines.get(fileName)) {
216        builder.append(line).append('\n');
217      }
218      fileTexts.put(fileName, builder.toString());
219    }
220    return fileTexts;
221  }
222
223  private static List<File> getTempEvalFiles(File dataDirectory, String tabFileName)
224      throws FileNotFoundException {
225
226    // subdirectory is "data" for training, and both "entities" and "relations" for testing
227    List<File> files = new ArrayList<File>();
228    for (String subDir : Arrays.asList("data", "key")) {
229      files.add(new File(new File(dataDirectory, subDir), tabFileName));
230    }
231    // weird special case: dct.txt is dct-en.txt in testing base directory
232    files.add(new File(dataDirectory, tabFileName.replaceAll("\\.txt", "-en.txt")));
233
234    // filter existing files
235    List<File> existingFiles = new ArrayList<File>();
236    for (File file : files) {
237      if (file.exists()) {
238        existingFiles.add(file);
239      }
240    }
241
242    // error if we didn't find at least one
243    if (existingFiles.size() == 0) {
244      throw new FileNotFoundException("Could not find any of " + files);
245    }
246    return existingFiles;
247  }
248
249  protected static String getAnnotatedFileName(String line) {
250    // the filename is the first column
251    String[] parts = line.split("\t", 2);
252    if (parts.length != 2) {
253      throw new IllegalArgumentException("Expected <filename>\t..., found " + line);
254    }
255    return parts[0];
256  }
257
258  public static List<String> getAnnotatedFileNames(File dataDirectory) throws IOException {
259    // look for file names in all the base segmentation files
260    List<String> fileNames = new ArrayList<String>();
261    Set<String> seenFileNames = new HashSet<String>();
262    for (File tabFile : getTempEvalFiles(dataDirectory, "base-segmentation.tab")) {
263      for (String line : Files.readLines(tabFile, Charsets.US_ASCII)) {
264
265        // add the filename to the list if we haven't already seen it
266        String fileName = getAnnotatedFileName(line);
267        if (!seenFileNames.contains(fileName)) {
268          seenFileNames.add(fileName);
269          fileNames.add(fileName);
270        }
271      }
272    }
273    return fileNames;
274  }
