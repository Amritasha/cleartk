024package org.cleartk.timeml.eval;
025
026import java.io.File;
027import java.util.ArrayList;
028import java.util.Arrays;
029import java.util.List;
030
031import org.cleartk.syntax.opennlp.PosTaggerAnnotator;
032import org.cleartk.timeml.corpus.TempEval2010GoldAnnotator;
033import org.cleartk.timeml.corpus.TempEval2010Writer;
034import org.cleartk.timeml.event.EventAspectAnnotator;
035import org.cleartk.timeml.event.EventClassAnnotator;
036import org.cleartk.timeml.event.EventModalityAnnotator;
037import org.cleartk.timeml.event.EventPolarityAnnotator;
038import org.cleartk.timeml.event.EventTenseAnnotator;
039import org.cleartk.timeml.type.Event;
040import org.cleartk.token.stem.snowball.DefaultSnowballStemmer;
041
042/**
043 * TempEval 2010 task B: event attributes
044 * 
045 * Best reported accuracies in TempEval 2010:
046 * <ul>
047 * <li>polarity: 0.99</li>
048 * <li>modality: 0.99</li>
049 * <li>tense: 0.92</li>
050 * <li>aspect: 0.98</li>
051 * <li>class: 0.79</li>
052 * </ul>
053 * 
054 * <br>
055 * Copyright (c) 2011, Regents of the University of Colorado <br>
056 * All rights reserved.
057 * 
058 * @author Steven Bethard
059 */
060public class TempEval2010TaskBAttributes extends TempEval2010Main {
061
062  public static void main(String[] args) throws Exception {
063    new TempEval2010TaskBAttributes().runMain(args);
064  }
065
066  @Override
067  protected TempEval2010Evaluation getEvaluation(File trainDir, File testDir, File outputDir)
068      throws Exception {
069
070    List<ModelInfo<Event>> infos = new ArrayList<ModelInfo<Event>>();
071    infos.add(new EventModelInfo("aspect", EventAspectAnnotator.FACTORY));
072    infos.add(new EventModelInfo("eventClass", EventClassAnnotator.FACTORY));
073    infos.add(new EventModelInfo("modality", EventModalityAnnotator.FACTORY));
074    infos.add(new EventModelInfo("polarity", EventPolarityAnnotator.FACTORY));
075    infos.add(new EventModelInfo("tense", EventTenseAnnotator.FACTORY));
076
077    return new TempEval2010Evaluation(
078        trainDir,
079        testDir,
080        outputDir,
081        Arrays.asList(
082            TempEval2010GoldAnnotator.PARAM_TEXT_VIEWS,
083            TempEval2010GoldAnnotator.PARAM_EVENT_EXTENT_VIEWS),
084        TempEval2010GoldAnnotator.PARAM_EVENT_ATTRIBUTE_VIEWS,
085        TempEval2010Writer.PARAM_EVENT_ATTRIBUTE_VIEW,
086        Arrays.asList(
087            DefaultSnowballStemmer.getDescription("English"),
088            PosTaggerAnnotator.getDescription()),
089        infos);
090  }
091}
