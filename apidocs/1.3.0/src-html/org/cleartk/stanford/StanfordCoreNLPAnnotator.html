091  @Override
092  public void initialize(UimaContext context) throws ResourceInitializationException {
093    super.initialize(context);
094
095    Properties properties = new Properties();
096    properties.put("annotators", "tokenize, ssplit, pos, lemma, ner, parse, dcoref");
097
098    this.processor = new StanfordCoreNLP(properties);
099  }

101  @Override
102  public void process(JCas jCas) throws AnalysisEngineProcessException {
103    Annotation document = this.processor.process(jCas.getDocumentText());
104
105    String lastNETag = "O";
106    int lastNEBegin = -1;
107    int lastNEEnd = -1;
108    for (CoreMap tokenAnn : document.get(TokensAnnotation.class)) {
109
110      // create the token annotation
111      int begin = tokenAnn.get(CharacterOffsetBeginAnnotation.class);
112      int end = tokenAnn.get(CharacterOffsetEndAnnotation.class);
113      String pos = tokenAnn.get(PartOfSpeechAnnotation.class);
114      String lemma = tokenAnn.get(LemmaAnnotation.class);
115      Token token = new Token(jCas, begin, end);
116      token.setPos(pos);
117      token.setLemma(lemma);
118      token.addToIndexes();
119
120      // hackery to convert token-level named entity tag into phrase-level tag
121      String neTag = tokenAnn.get(NamedEntityTagAnnotation.class);
122      if (neTag.equals("O") && !lastNETag.equals("O")) {
123        NamedEntityMention ne = new NamedEntityMention(jCas, lastNEBegin, lastNEEnd);
124        ne.setMentionType(lastNETag);
125        ne.addToIndexes();
126      } else {
127        if (lastNETag.equals("O")) {
128          lastNEBegin = begin;
129        } else if (lastNETag.equals(neTag)) {
130          // do nothing - begin was already set
131        } else {
132          NamedEntityMention ne = new NamedEntityMention(jCas, lastNEBegin, lastNEEnd);
133          ne.setMentionType(lastNETag);
134          ne.addToIndexes();
135          lastNEBegin = begin;
136        }
137        lastNEEnd = end;
138      }
139      lastNETag = neTag;
140    }
141    if (!lastNETag.equals("O")) {
142      NamedEntityMention ne = new NamedEntityMention(jCas, lastNEBegin, lastNEEnd);
143      ne.setMentionType(lastNETag);
144      ne.addToIndexes();
145    }
146
147    // add sentences and trees
148    for (CoreMap sentenceAnn : document.get(SentencesAnnotation.class)) {
149
150      // add the sentence annotation
151      int sentBegin = sentenceAnn.get(CharacterOffsetBeginAnnotation.class);
152      int sentEnd = sentenceAnn.get(CharacterOffsetEndAnnotation.class);
153      Sentence sentence = new Sentence(jCas, sentBegin, sentEnd);
154      sentence.addToIndexes();
155
156      // add the syntactic tree annotation
157      List<CoreLabel> tokenAnns = sentenceAnn.get(TokensAnnotation.class);
158      Tree tree = sentenceAnn.get(TreeAnnotation.class);
159      if (tree.children().length != 1) {
160        throw new RuntimeException("Expected single root node, found " + tree);
161      }
162      tree = tree.firstChild();
163      tree.indexSpans(0);
164      TopTreebankNode root = new TopTreebankNode(jCas);
165      root.setTreebankParse(tree.toString());
166      // TODO: root.setTerminals(v)
167      this.addTreebankNodeToIndexes(root, jCas, tree, tokenAnns);
168
169      // get the dependencies
170      SemanticGraph dependencies = sentenceAnn.get(CollapsedCCProcessedDependenciesAnnotation.class);
171
172      // convert Stanford nodes to UIMA annotations
173      List<Token> tokens = JCasUtil.selectCovered(jCas, Token.class, sentence);
174      Map<IndexedWord, DependencyNode> stanfordToUima = new HashMap<IndexedWord, DependencyNode>();
175      for (IndexedWord stanfordNode : dependencies.vertexSet()) {
176        int indexBegin = stanfordNode.get(BeginIndexAnnotation.class);
177        int indexEnd = stanfordNode.get(EndIndexAnnotation.class);
178        int tokenBegin = tokens.get(indexBegin).getBegin();
179        int tokenEnd = tokens.get(indexEnd - 1).getEnd();
180        DependencyNode node;
181        if (dependencies.getRoots().contains(stanfordNode)) {
182          node = new TopDependencyNode(jCas, tokenBegin, tokenEnd);
183        } else {
184          node = new DependencyNode(jCas, tokenBegin, tokenEnd);
185        }
186        stanfordToUima.put(stanfordNode, node);
187      }
188
189      // create relation annotations for each Stanford dependency
190      ArrayListMultimap<DependencyNode, DependencyRelation> headRelations = ArrayListMultimap.create();
191      ArrayListMultimap<DependencyNode, DependencyRelation> childRelations = ArrayListMultimap.create();
192      for (SemanticGraphEdge stanfordEdge : dependencies.edgeIterable()) {
193        DependencyRelation relation = new DependencyRelation(jCas);
194        DependencyNode head = stanfordToUima.get(stanfordEdge.getGovernor());
195        DependencyNode child = stanfordToUima.get(stanfordEdge.getDependent());
196        String relationType = stanfordEdge.getRelation().toString();
197        if (head == null || child == null || relationType == null) {
198          throw new RuntimeException(String.format(
199              "null elements not allowed in relation:\nrelation=%s\nchild=%s\nhead=%s\n",
200              relation,
201              child,
202              head));
203        }
204        relation.setHead(head);
205        relation.setChild(child);
206        relation.setRelation(relationType);
207        relation.addToIndexes();
208        headRelations.put(child, relation);
209        childRelations.put(head, relation);
210      }
211
212      // set the relations for each node annotation
213      for (DependencyNode node : stanfordToUima.values()) {
214        node.setHeadRelations(UIMAUtil.toFSArray(jCas, headRelations.get(node)));
215        node.setChildRelations(UIMAUtil.toFSArray(jCas, childRelations.get(node)));
216        node.addToIndexes();
217      }
218    }
219
220    // map from spans to named entity mentions
221    Map<Span, NamedEntityMention> spanMentionMap = new HashMap<Span, NamedEntityMention>();
222    for (NamedEntityMention mention : JCasUtil.select(jCas, NamedEntityMention.class)) {
223      spanMentionMap.put(new Span(mention.getBegin(), mention.getEnd()), mention);
224    }
225
226    // add mentions for all entities identified by the coreference system
227    List<NamedEntity> entities = new ArrayList<NamedEntity>();
228    List<List<Token>> sentenceTokens = new ArrayList<List<Token>>();
229    for (Sentence sentence : JCasUtil.select(jCas, Sentence.class)) {
230      sentenceTokens.add(JCasUtil.selectCovered(jCas, Token.class, sentence));
231    }
232    Map<Integer, CorefChain> corefChains = document.get(CorefChainAnnotation.class);
233    for (CorefChain chain : corefChains.values()) {
234      List<NamedEntityMention> mentions = new ArrayList<NamedEntityMention>();
235      for (CorefMention corefMention : chain.getCorefMentions()) {
236
237        // figure out the character span of the token
238        List<Token> tokens = sentenceTokens.get(corefMention.sentNum - 1);
239        int begin = tokens.get(corefMention.startIndex - 1).getBegin();
240        int end = tokens.get(corefMention.endIndex - 2).getEnd();
241
242        // use an existing named entity mention when possible; otherwise create a new one
243        NamedEntityMention mention = spanMentionMap.get(new Span(begin, end));
244        if (mention == null) {
245          mention = new NamedEntityMention(jCas, begin, end);
246          mention.addToIndexes();
247        }
248        mentions.add(mention);
249      }
250
251      // create an entity for the mentions
252      Collections.sort(mentions, new Comparator<NamedEntityMention>() {
253        @Override
254        public int compare(NamedEntityMention m1, NamedEntityMention m2) {
255          return m1.getBegin() - m2.getBegin();
256        }
257      });
258
259      // create mentions and add them to entity
260      NamedEntity entity = new NamedEntity(jCas);
261      entity.setMentions(new FSArray(jCas, mentions.size()));
262      int index = 0;
263      for (NamedEntityMention mention : mentions) {
264        mention.setMentionedEntity(entity);
265        entity.setMentions(index, mention);
266        index += 1;
267      }
268      entities.add(entity);
269    }
270
271    // add singleton entities for any named entities not picked up by coreference system
272    for (NamedEntityMention mention : JCasUtil.select(jCas, NamedEntityMention.class)) {
273      if (mention.getMentionedEntity() == null) {
274        NamedEntity entity = new NamedEntity(jCas);
275        entity.setMentions(new FSArray(jCas, 1));
276        entity.setMentions(0, mention);
277        mention.setMentionedEntity(entity);
278        entity.getMentions();
279        entities.add(entity);
280      }
281    }
282
283    // sort entities by document order
284    Collections.sort(entities, new Comparator<NamedEntity>() {
285      @Override
286      public int compare(NamedEntity o1, NamedEntity o2) {
287        return getFirstBegin(o1) - getFirstBegin(o2);
288      }
289
290      private int getFirstBegin(NamedEntity entity) {
291        int min = Integer.MAX_VALUE;
292        for (NamedEntityMention mention : JCasUtil.select(
293            entity.getMentions(),
294            NamedEntityMention.class)) {
295          if (mention.getBegin() < min) {
296            min = mention.getBegin();
297          }
298        }
299        return min;
300      }
301    });
302
303    // add entities to document
304    for (NamedEntity entity : entities) {
305      entity.addToIndexes();
306    }
307
308  }

310  private FSArray addTreebankNodeChildrenToIndexes(
311      TreebankNode parent,
312      JCas jCas,
313      List<CoreLabel> tokenAnns,
314      Tree tree) {
315    Tree[] childTrees = tree.children();
316
317    // collect all children (except leaves, which are just the words - POS tags are pre-terminals in
318    // a Stanford tree)
319    List<TreebankNode> childNodes = new ArrayList<TreebankNode>();
320    for (Tree child : childTrees) {
321      if (!child.isLeaf()) {
322
323        // set node attributes and add children (mutual recursion)
324        TreebankNode node = new TreebankNode(jCas);
325        node.setParent(parent);
326        this.addTreebankNodeToIndexes(node, jCas, child, tokenAnns);
327        childNodes.add(node);
328      }
329    }
330
331    // convert the child list into an FSArray
332    FSArray childNodeArray = new FSArray(jCas, childNodes.size());
333    for (int i = 0; i < childNodes.size(); ++i) {
334      childNodeArray.set(i, childNodes.get(i));
335    }
336    return childNodeArray;
337  }
338
339  private void addTreebankNodeToIndexes(
340      TreebankNode node,
341      JCas jCas,
342      Tree tree,
343      List<CoreLabel> tokenAnns) {
344    // figure out begin and end character offsets
345    CoreMap label = (CoreMap) tree.label();
346    CoreMap beginToken = tokenAnns.get(label.get(BeginIndexAnnotation.class));
347    CoreMap endToken = tokenAnns.get(label.get(EndIndexAnnotation.class) - 1);
348    int nodeBegin = beginToken.get(CharacterOffsetBeginAnnotation.class);
349    int nodeEnd = endToken.get(CharacterOffsetEndAnnotation.class);
350
351    // set span, node type, children (mutual recursion), and add it to the JCas
352    node.setBegin(nodeBegin);
353    node.setEnd(nodeEnd);
354    node.setNodeType(tree.value());
355    node.setChildren(this.addTreebankNodeChildrenToIndexes(node, jCas, tokenAnns, tree));
356    node.setLeaf(node.getChildren().size() == 0);
357    node.addToIndexes();
358  }
359
360  private static class Span {
361    public int begin;
362
363    public int end;
364
365    public Span(int begin, int end) {
366      this.begin = begin;
367      this.end = end;
368    }
369
370    public boolean equals(Object object) {
371      if (object instanceof Span) {
372        Span that = (Span) object;
373        return this.begin == that.begin && this.end == that.end;
374      } else {
375        return false;
376      }
377    }
378
379    public int hashCode() {
380      return Arrays.hashCode(new int[] { this.begin, this.end });
381    }
382  }
