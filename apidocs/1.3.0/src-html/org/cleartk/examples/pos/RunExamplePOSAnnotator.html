024package org.cleartk.examples.pos;
025
026import java.io.File;
027import java.util.Arrays;
028import java.util.List;
029
030import org.apache.uima.collection.CollectionReader;
031import org.cleartk.syntax.opennlp.SentenceAnnotator;
032import org.cleartk.token.tokenizer.TokenAnnotator;
033import org.cleartk.util.ae.UriToDocumentTextAnnotator;
034import org.cleartk.util.cr.UriCollectionReader;
035import org.uimafit.factory.AggregateBuilder;
036import org.uimafit.factory.AnalysisEngineFactory;
037import org.uimafit.pipeline.SimplePipeline;
038
039/**
040 * <br>
041 * Copyright (c) 2009, Regents of the University of Colorado <br>
042 * All rights reserved.
043 * 
044 * @author Philip Ogren
045 * 
046 *         For examples of using the ExamplePOSAnnotationHandler using different classifiers, please
047 *         see org.cleartk.example.pos.ExamplePosClassifierTest
048 */
049
050public class RunExamplePOSAnnotator {
051
052  public static void main(String[] args) throws Exception {
053    String samplePath = "src/main/resources/data/sample/2008_Sichuan_earthquake.txt";
054    List<File> files = Arrays.asList(new File(samplePath));
055
056    // A collection reader that creates one CAS per file, containing the file's URI
057    CollectionReader reader = UriCollectionReader.getCollectionReaderFromFiles(files);
058
059    // The pipeline of annotators
060    AggregateBuilder builder = new AggregateBuilder();
061
062    // An annotator that reads in the file text
063    builder.add(UriToDocumentTextAnnotator.getDescription());
064
065    // An annotator that adds Sentence annotations
066    builder.add(SentenceAnnotator.getDescription());
067
068    // An annotator that adds Token annotations
069    builder.add(TokenAnnotator.getDescription());
070
071    // The POS annotator, configured to make predictions
072    builder.add(ExamplePOSAnnotator.getClassifierDescription(ExamplePOSAnnotator.DEFAULT_MODEL));
073
074    // An annotator that write out the tokens and their part of speech tags
075    builder.add(AnalysisEngineFactory.createPrimitiveDescription(
076        ExamplePOSPlainTextWriter.class,
077        ExamplePOSPlainTextWriter.PARAM_OUTPUT_DIRECTORY_NAME,
078        ExamplePOSPlainTextWriter.DEFAULT_OUTPUT_DIRECTORY));
079
080    // Run the pipeline of annotators on each of the CASes produced by the reader
081    SimplePipeline.runPipeline(reader, builder.createAggregateDescription());
082
083    System.out.println("Please look at the file generated by this program: "
084        + ExamplePOSPlainTextWriter.DEFAULT_OUTPUT_DIRECTORY + "/2008_Sichuan_earthquake.txt.pos");
085  }
086}
