052/**
053 * <br>
054 * Copyright (c) 2010, Regents of the University of Colorado <br>
055 * All rights reserved.
056 * 
057 * This class takes in treebank files and parses the the PTB-style format and populates a the CAS
058 * with constituent trees and then writes out XMI files. The CAS will contain the raw treebank data
059 * in a view You may find it useful to view the generated XMI files in the CAS Visual Debugger.
060 * 
061 * If you do not have access to PennTreebank data or any other treebank data, then you can run this
062 * script with the following arguments:
063 * <ul>
064 * <li>-td src/main/resources/data/pos/treebank</li>
065 * <li>-o src/main/resources/data/pos/treebank</li>
066 * <li>-suf tree</li>
067 * </ul>
068 * 
069 * To view the resulting xmi file in the CAS Visual Debugger you can run the eclipse launch
070 * configuration labeled "CVD (cleartk-examples)". When the CVD opens select Menu -> File -> Read
071 * XMI CAS File. Now navigate to the single xmi file located in
072 * src/main/resources/data/pos/treebank. Once the file is open in the CVD be sure to examine the
073 * different views by selecting different values from the drop down box labeled "Select View" near
074 * the top of the window.
075 * 
076 * @author Philip Ogren
077 * 
078 */
079public class TreebankParsingExample {
080
081  public static class Options extends Options_ImplBase {
082    @Option(
083        name = "-td",
084        aliases = "--treebankDirectory",
085        usage = "specify the directory containing treebank files",
086        required = true)
087    public String treebankDirectory;
088
089    @Option(
090        name = "-o",
091        aliases = "--outputDirectory",
092        usage = "specify the directory to write the XMI files to",
093        required = true)
094    public String outputDirectory;
095
096    @Option(
097        name = "-suf",
098        aliases = "--treebankFileSuffixes",
099        usage = "specify file suffixes of the treebank files in the treebank directory",
100        multiValued = true)
101    public List<String> treebankFileSuffixes = new ArrayList<String>();
102  }
103
104  public static void main(String[] args) throws UIMAException, IOException {
105    Options options = new Options();
106    options.parseOptions(args);
107
108    // Loads URIS specified by files into URI view
109    String[] suffixes = options.treebankFileSuffixes.toArray(new String[options.treebankFileSuffixes.size()]);
110    File treebankDir = new File(options.treebankDirectory);
111    Collection<File> files = FileUtils.listFiles(treebankDir, suffixes, false);
112    CollectionReader reader = UriCollectionReader.getCollectionReaderFromFiles(files);
113
114    AggregateBuilder builder = new AggregateBuilder();
115
116    // Reads text into TREEBANK_VIEW
117    builder.add(UriToDocumentTextAnnotator.getDescriptionForView(TreebankConstants.TREEBANK_VIEW));
118
119    // Ensures GOLD_VIEW is present
120    builder.add(AnalysisEngineFactory.createPrimitiveDescription(
121        ViewCreatorAnnotator.class,
122        ViewCreatorAnnotator.PARAM_VIEW_NAME,
123        EvaluationConstants.GOLD_VIEW));
124
125    // Parses treebank text into gold view (instead of default sofa view)
126    AnalysisEngineDescription treebankParserDescription = AnalysisEngineFactory.createPrimitiveDescription(TreebankGoldAnnotator.class);
127    builder.add(treebankParserDescription, CAS.NAME_DEFAULT_SOFA, EvaluationConstants.GOLD_VIEW);
128
129    // XMI Writer
130    builder.add(AnalysisEngineFactory.createPrimitiveDescription(
131        XWriter.class,
132        XWriter.PARAM_OUTPUT_DIRECTORY_NAME,
133        options.outputDirectory,
134        XWriter.PARAM_FILE_NAMER_CLASS_NAME,
135        ViewURIFileNamer.class.getName()));
136
137    SimplePipeline.runPipeline(reader, builder.createAggregateDescription());
138
139  }
140}
