019package org.cleartk.classifier.weka;

021import java.io.File;
022import java.io.IOException;
023import java.util.ArrayList;
024import java.util.HashSet;
025import java.util.List;
026import java.util.Map;
027import java.util.Set;
028
029import org.cleartk.classifier.CleartkProcessingException;
030import org.cleartk.classifier.Feature;
031import org.cleartk.classifier.encoder.outcome.StringToStringOutcomeEncoder;
032import org.cleartk.classifier.jar.DataWriter_ImplBase;
033
034import weka.core.Attribute;
035import weka.core.Instances;
036import weka.core.SparseInstance;
037
038/**

047
048public class WekaStringOutcomeDataWriter extends
049    DataWriter_ImplBase<WekaStringOutcomeClassifierBuilder, Iterable<Feature>, String, String> {
050
051  private final String relationTag;
052
053  List<Iterable<Feature>> instanceFeatures;
054
055  List<String> instanceOutcomes;
056
057  Set<String> outcomeValues;
058
059  public WekaStringOutcomeDataWriter(File outputDirectory, String relationTag) throws IOException {
060    super(outputDirectory);
061    this.setFeaturesEncoder(new WekaFeaturesEncoder());
062    this.setOutcomeEncoder(new StringToStringOutcomeEncoder());
063    this.relationTag = relationTag;
064    instanceFeatures = new ArrayList<Iterable<Feature>>();
065    instanceOutcomes = new ArrayList<String>();
066    outcomeValues = new HashSet<String>();
067  }
068
069  public WekaStringOutcomeDataWriter(File outputDirectory) throws IOException {
070    this(outputDirectory, "cleartk-generated");
071  }
072
073  @Override
074  public void writeEncoded(Iterable<Feature> features, String outcome) {
075    this.instanceFeatures.add(features);
076    instanceOutcomes.add(outcome);
077    outcomeValues.add(outcome);
078  }
079
080  @Override
081  public void finish() throws CleartkProcessingException {
082    ArrayList<Attribute> attributes = ((WekaFeaturesEncoder) this.classifierBuilder.getFeaturesEncoder()).getWekaAttributes();
083    Map<String, Attribute> attributeMap = ((WekaFeaturesEncoder) this.classifierBuilder.getFeaturesEncoder()).getWekaAttributeMap();
084
085    Attribute outcomeAttribute = createOutcomeAttribute(attributes.size());
086    attributes.add(outcomeAttribute);
087
088    Instances instances = new Instances(relationTag, attributes, instanceFeatures.size());
089    instances.setClass(outcomeAttribute);
090
091    for (int i = 0; i < instanceFeatures.size(); i++) {
092      SparseInstance instance = new SparseInstance(instances.numAttributes());
093
094      Iterable<Feature> features = instanceFeatures.get(i);
095      for (Feature feature : features) {
096        Attribute attribute = attributeMap.get(feature.getName());
097        Object featureValue = feature.getValue();
098
099        if (featureValue instanceof Number) {
100          double attributeValue = ((Number) feature.getValue()).doubleValue();
101          instance.setValue(attribute, attributeValue);
102        } else if (featureValue instanceof Boolean) {
103          double attributeValue = (Boolean) featureValue ? 1.0d : -1.0d;
104          instance.setValue(attribute, attributeValue);
105        } else {
106          instance.setValue(attribute, featureValue.toString());
107        }
108      }
109
110      instance.setValue(outcomeAttribute, instanceOutcomes.get(i));
111      instances.add(instance);
112    }
113
114    trainingDataWriter.write(instances.toString());
115    super.finish();
116  }
117
118  private Attribute createOutcomeAttribute(int attributeIndex) {
119    // TODO make sure that "outcome" is not the name of an existing feature.
120    return new Attribute("outcome", new ArrayList<String>(this.outcomeValues));
121  }
122
123  @Override
124  protected WekaStringOutcomeClassifierBuilder newClassifierBuilder() {
125    return new WekaStringOutcomeClassifierBuilder();
126  }
127
128}
