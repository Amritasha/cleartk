019package org.cleartk.classifier.weka;
020
021import java.util.ArrayList;
022import java.util.Arrays;
023import java.util.List;
024
025import weka.core.Attribute;
026import weka.core.DenseInstance;
027import weka.core.Instance;
028import weka.core.Instances;
029
030/**
031 * Copyright (c) 2012, Regents of the University of Colorado <br>
032 * All rights reserved.
033 * 
034 * @author Philip Ogren
035 * 
036 */
037
038public class WekaTest {
039
040  public static void main(String[] args) {
041    // Declare the class attribute along with its values
042    Attribute classAttribute = new Attribute("theClass", Arrays.asList("positive", "negative"), 0);
043
044    Attribute attribute1 = new Attribute("firstNumeric", 1);
045    Attribute attribute2 = new Attribute("secondNumeric", 2);
046
047    // Declare a nominal attribute along with its values
048    Attribute attribute3 = new Attribute("aNominal", Arrays.asList("blue", "gray", "black"), 3);
049
050    // Declare the feature vector
051    Instance instance = new DenseInstance(4);
052    instance.setValue(classAttribute, "positive");
053    instance.setValue(attribute1, 1.0);
054
055    Instance instance1 = new DenseInstance(4);
056    instance1.setValue(classAttribute, "positive");
057    instance1.setValue(attribute1, 1.0);
058    instance1.setValue(attribute2, 0.5);
059    instance1.setValue(attribute3, "gray");
060
061    List<Attribute> attributes = Arrays.asList(classAttribute, attribute1, attribute2, attribute3);
062    Instances instances = new Instances("Rel", new ArrayList<Attribute>(attributes), 10);
063
064    // Set class index
065    instances.setClassIndex(0);
066    // add the instance
067    instances.add(instance);
068    instances.add(instance1);
069
070    System.out.println(instances);
071  }
072}
