024package org.cleartk.classifier.svmlight;
025
026import java.util.ArrayList;
027import java.util.Collections;
028import java.util.List;
029import java.util.Map;
030
031import org.cleartk.classifier.CleartkProcessingException;
032import org.cleartk.classifier.Feature;
033import org.cleartk.classifier.ScoredOutcome;
034import org.cleartk.classifier.encoder.features.FeaturesEncoder;
035import org.cleartk.classifier.encoder.outcome.OutcomeEncoder;
036import org.cleartk.classifier.jar.Classifier_ImplBase;
037import org.cleartk.classifier.sigmoid.Sigmoid;
038import org.cleartk.classifier.svmlight.model.SVMlightModel;
039import org.cleartk.classifier.util.featurevector.FeatureVector;
040
041/**
042 * <br>
043 * Copyright (c) 2007-2008, Regents of the University of Colorado <br>
044 * All rights reserved.
045 */
046public class SVMlightStringOutcomeClassifier extends Classifier_ImplBase<FeatureVector, String, Integer> {
047
048  Map<Integer, SVMlightModel> models;
049
050  Map<Integer, Sigmoid> sigmoids;
051
052  public SVMlightStringOutcomeClassifier(
053      FeaturesEncoder<FeatureVector> featuresEncoder,
054      OutcomeEncoder<String, Integer> outcomeEncoder,
055      Map<Integer, SVMlightModel> models,
056      Map<Integer, Sigmoid> sigmoids) {
057    super(featuresEncoder, outcomeEncoder);
058    this.models = models;
059    this.sigmoids = sigmoids;
060  }
061
062  public String classify(List<Feature> features) throws CleartkProcessingException {
063    FeatureVector featureVector = this.featuresEncoder.encodeAll(features);
064
065    int maxScoredIndex = 0;
066    double maxScore = 0;
067    boolean first = true;
068    for (int i : models.keySet()) {
069      double score = score(featureVector, i);
070      if (first || score > maxScore) {
071        first = false;
072        maxScore = score;
073        maxScoredIndex = i;
074      }
075    }
076
077    return outcomeEncoder.decode(maxScoredIndex);
078  }
079
080  @Override
081  public List<ScoredOutcome<String>> score(List<Feature> features, int maxResults)
082      throws CleartkProcessingException {
083    FeatureVector featureVector = this.featuresEncoder.encodeAll(features);
084
085    List<ScoredOutcome<String>> results = new ArrayList<ScoredOutcome<String>>();
086    for (int i : models.keySet()) {
087      double score = score(featureVector, i);
088      String name = outcomeEncoder.decode(i);
089
090      results.add(new ScoredOutcome<String>(name, score));
091    }
092    Collections.sort(results);
093
094    return results.subList(0, Math.min(maxResults, results.size()));
095  }
096
097  private double score(FeatureVector fv, int i) {
098    return sigmoids.get(i).evaluate(models.get(i).evaluate(fv));
099  }
100}
