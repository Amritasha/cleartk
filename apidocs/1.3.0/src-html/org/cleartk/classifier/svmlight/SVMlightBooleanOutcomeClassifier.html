024package org.cleartk.classifier.svmlight;
025
026import java.util.ArrayList;
027import java.util.List;
028
029import org.cleartk.classifier.CleartkProcessingException;
030import org.cleartk.classifier.Feature;
031import org.cleartk.classifier.ScoredOutcome;
032import org.cleartk.classifier.encoder.features.FeaturesEncoder;
033import org.cleartk.classifier.encoder.outcome.OutcomeEncoder;
034import org.cleartk.classifier.sigmoid.Sigmoid;
035import org.cleartk.classifier.svmlight.model.SVMlightModel;
036import org.cleartk.classifier.util.featurevector.FeatureVector;
037
038/**
039 * <br>
040 * Copyright (c) 2007-2011, Regents of the University of Colorado <br>
041 * All rights reserved.
042 * 
043 * @author Steven Bethard
044 */
045public class SVMlightBooleanOutcomeClassifier extends SVMlightClassifier_ImplBase<Boolean, Boolean> {
046
047  protected Sigmoid sigmoid;
048
049  public SVMlightBooleanOutcomeClassifier(
050      FeaturesEncoder<FeatureVector> featuresEncoder,
051      OutcomeEncoder<Boolean, Boolean> outcomeEncoder,
052      SVMlightModel model,
053      Sigmoid sigmoid) {
054    super(featuresEncoder, outcomeEncoder, model);
055    this.sigmoid = sigmoid;
056  }
057
058  @Override
059  protected Boolean predictionToOutcome(double prediction) {
060    return this.sigmoid.evaluate(prediction) > 0.5;
061  }
062
063  @Override
064  public List<ScoredOutcome<Boolean>> score(List<Feature> features, int maxResults)
065      throws CleartkProcessingException {
066
067    List<ScoredOutcome<Boolean>> resultList = new ArrayList<ScoredOutcome<Boolean>>();
068    if (maxResults > 0)
069      resultList.add(this.score(features));
070    if (maxResults > 1) {
071      ScoredOutcome<Boolean> v1 = resultList.get(0);
072      ScoredOutcome<Boolean> v2 = new ScoredOutcome<Boolean>(!v1.getOutcome(), 1 - v1.getScore());
073      resultList.add(v2);
074    }
075    return resultList;
076  }
077
078  private ScoredOutcome<Boolean> score(List<Feature> features) throws CleartkProcessingException {
079    FeatureVector featureVector = featuresEncoder.encodeAll(features);
080
081    double prediction = sigmoid.evaluate(model.evaluate(featureVector));
082    boolean encodedResult = (prediction > 0.5);
083
084    if (encodedResult) {
085      return new ScoredOutcome<Boolean>(true, prediction);
086    } else {
087      return new ScoredOutcome<Boolean>(false, 1 - prediction);
088    }
089  }
090}
