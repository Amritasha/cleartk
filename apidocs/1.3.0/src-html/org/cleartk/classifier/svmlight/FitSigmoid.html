024package org.cleartk.classifier.svmlight;
025
026import java.io.BufferedReader;
027import java.io.File;
028import java.io.FileReader;
029import java.io.IOException;
030
031import org.cleartk.classifier.sigmoid.LinWengPlatt;
032import org.cleartk.classifier.sigmoid.LinWengPlatt.ConvergenceFailure;
033import org.cleartk.classifier.sigmoid.Sigmoid;
034import org.cleartk.classifier.svmlight.model.SVMlightModel;
035import org.cleartk.classifier.util.featurevector.FeatureVector;
036import org.cleartk.classifier.util.featurevector.InvalidFeatureVectorValueException;
037import org.cleartk.classifier.util.featurevector.SparseFeatureVector;
038
039/**
040 * <br>
041 * Copyright (c) 2009, Regents of the University of Colorado <br>
042 * All rights reserved.
043 * 
044 * This class implements an algorithm to fit a sigmoid function to the output of an SVM classifier.
045 * The algorithm is the one introduced by Hsuan-Tien Lin, Chih-Jen Lin, and Ruby C. Weng (who were
046 * in turn extending work by J. Platt), and this implementation is a direct translation of their
047 * pseudo-code as presented in
048 * 
049 * Lin, Lin, Weng. A note on Platt's probabilistic outputs for support vector machines. In Machine
050 * Learning, vol. 68, pp. 267-276, 2007.
051 * 
052 * @author Philipp G. Wetzler
053 */
054public class FitSigmoid {
055
056  public static Sigmoid fit(File svmlightModelFile, File trainingDataFile)
057      throws ConvergenceFailure, IOException, InvalidFeatureVectorValueException {
058    SVMlightModel model = SVMlightModel.fromFile(svmlightModelFile);
059
060    BufferedReader r = new BufferedReader(new FileReader(trainingDataFile));
061    int lines = 0;
062    while (r.readLine() != null)
063      lines += 1;
064    r.close();
065
066    boolean[] labels = new boolean[lines];
067    double[] decisionValues = new double[lines];
068    int i = 0;
069    r = new BufferedReader(new FileReader(trainingDataFile));
070    String line = r.readLine();
071    while (line != null) {
072      TrainingInstance ti = parseTI(line.trim());
073      labels[i] = ti.getLabel();
074      decisionValues[i] = model.evaluate(ti.getFeatureVector());
075      i += 1;
076      line = r.readLine();
077    }
078    r.close();
079
080    return LinWengPlatt.fit(decisionValues, labels);
081
082  }
083
084  private static TrainingInstance parseTI(String line) throws InvalidFeatureVectorValueException {
085    String[] fields = line.split(" ");
086
087    boolean label = fields[0].trim().equals("+1");
088
089    FeatureVector fv = new SparseFeatureVector();
090
091    for (int i = 1; i < fields.length; i++) {
092      String[] parts = fields[i].split(":");
093      int featureIndex = Integer.valueOf(parts[0]);
094      double featureValue = Double.valueOf(parts[1]);
095      fv.set(featureIndex, featureValue);
096    }
097
098    return new TrainingInstance(label, fv);
099  }
100
101  private static class TrainingInstance {
102
103    public TrainingInstance(boolean label, FeatureVector featureVector) {
104      this.label = label;
105      this.featureVector = featureVector;
106    }
107
108    public boolean getLabel() {
109      return label;
110    }
111
112    public FeatureVector getFeatureVector() {
113      return featureVector;
114    }
115
116    private boolean label;
117
118    private FeatureVector featureVector;
119  }
120}
