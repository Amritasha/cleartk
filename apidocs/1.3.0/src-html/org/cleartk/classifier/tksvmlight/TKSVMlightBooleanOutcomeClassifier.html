049/**
050 * A Tree Kernel SVM light classifier implementation. All features named with the prefix "TK_"
051 * treated as Tree Kernels.
052 * 
053 * <br>
054 * Copyright (c) 2007-2008, Regents of the University of Colorado <br>
055 * All rights reserved.
056 * 
057 * @author Daryl Lonnon
058 * @see TKSVMlightStringOutcomeClassifier
059 */
060public class TKSVMlightBooleanOutcomeClassifier extends
061    Classifier_ImplBase<TreeFeatureVector, Boolean, Boolean> {
062
063  static Logger logger = UIMAFramework.getLogger(TKSVMlightBooleanOutcomeClassifier.class);
064
065  File mFile;
066
067  /**
068   * Constructor
069   * 
070   * @param featuresEncoder
071   *          The features encoder used by this classifier.
072   * @param outcomeEncoder
073   *          The outcome encoder used by this classifier.
074   * @param modelFile
075   *          The file for the model used by this classifier.
076   */
077  public TKSVMlightBooleanOutcomeClassifier(
078      FeaturesEncoder<TreeFeatureVector> featuresEncoder,
079      OutcomeEncoder<Boolean, Boolean> outcomeEncoder,
080      File modelFile) {
081    super(featuresEncoder, outcomeEncoder);
082    mFile = modelFile;
083  }
084
085  /**
086   * Classify a features list.
087   * 
088   * @param features
089   *          The feature list to classify.
090   * @return A Boolean of whether the features match this classification.
091   */
092  public Boolean classify(List<Feature> features) throws CleartkProcessingException {
093
094    ScoredOutcome<Boolean> s = score(features);
095    return s.getOutcome();
096  }
097
098  /**
099   * Score a list of features against the model.
100   * 
101   * @param features
102   *          The features to classify
103   * @param maxResults
104   *          The maximum number of results to return in the list (at most 2).
105   * @return A list of scored outcomes ordered by likelihood.
106   */
107  @Override
108  public List<ScoredOutcome<Boolean>> score(List<Feature> features, int maxResults)
109      throws CleartkProcessingException {
110
111    List<ScoredOutcome<Boolean>> resultList = new ArrayList<ScoredOutcome<Boolean>>();
112    if (maxResults > 0)
113      resultList.add(this.score(features));
114    if (maxResults > 1) {
115      ScoredOutcome<Boolean> v1 = resultList.get(0);
116      ScoredOutcome<Boolean> v2 = new ScoredOutcome<Boolean>(!v1.getOutcome(), 1 - v1.getScore());
117      resultList.add(v2);
118    }
119    return resultList;
120  }
121
122  private ScoredOutcome<Boolean> score(List<Feature> features) throws CleartkProcessingException {
123    TreeFeatureVector featureVector = featuresEncoder.encodeAll(features);
124    double prediction = tkSvmLightPredict(mFile, featureVector);
125
126    // I got this from the svmlight classifier, shouldn't this be over 0.0 (doesn't it range from
127    // -1.0 to 1.0
128    boolean encodedResult = (prediction > 0.5);
129
130    if (encodedResult) {
131      return new ScoredOutcome<Boolean>(true, prediction);
132    } else {
133      return new ScoredOutcome<Boolean>(false, 1 - prediction);
134    }
135  }
136
137  /**
138   * Predict which side of the line a feature vector resides upon for a particular svm model.
139   * 
140   * @param mFile
141   *          The model file to predict against.
142   * @param featureVector
143   *          The feature vector to predict for.
144   * @return A double that represents which side of the line the feature vector resides (negative
145   *         below, positive above).
146   */
147  // Moving this into public space so it can be used by one versus all thing.
148  public static double tkSvmLightPredict(File mFile, TreeFeatureVector featureVector)
149      throws CleartkProcessingException {
150    double prediction;
151
152    try {
153      File cFile = File.createTempFile("tkclassify", ".txt");
154      File oFile = File.createTempFile("tkoutput", ".out");
155      BufferedWriter out = new BufferedWriter(new FileWriter(cFile));
156      out.write("0");
157      out.write(TKSVMlightBooleanOutcomeDataWriter.createString(featureVector));
158      out.write("\n");
159      out.close();
160
161      String executable = "tk_svm_classify";
162      String[] command = new String[4];
163      command[0] = executable;
164      command[1] = cFile.getPath();
165      command[2] = mFile.getPath();
166      command[3] = oFile.getPath();
167
168      logger.log(Level.FINE, "classifying with tree kernel svmlight using the following command: "
169          + Joiner.on(" ").join(command));
170      Process process = Runtime.getRuntime().exec(command);
171      process.getOutputStream().close();
172      output(process.getInputStream(), System.out);
173      output(process.getErrorStream(), System.err);
174      process.waitFor();
175
176      BufferedReader in = new BufferedReader(new FileReader(oFile));
177      prediction = Double.parseDouble(in.readLine());
178      in.close();
179    } catch (IOException e) {
180      throw new CleartkProcessingException(e);
181    } catch (InterruptedException e) {
182      throw new CleartkProcessingException(e);
183    }
184    return prediction;
185  }
186
187  private static void output(InputStream input, PrintStream output) throws IOException {
188    byte[] buffer = new byte[128];
189    int count = input.read(buffer);
190    while (count != -1) {
191      output.write(buffer, 0, count);
192      count = input.read(buffer);
193    }
194  }
