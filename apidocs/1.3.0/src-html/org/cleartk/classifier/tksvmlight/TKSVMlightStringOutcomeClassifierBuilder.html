024package org.cleartk.classifier.tksvmlight;
025
026import java.io.BufferedReader;
027import java.io.BufferedWriter;
028import java.io.File;
029import java.io.FileWriter;
030import java.io.IOException;
031import java.io.InputStreamReader;
032import java.util.TreeMap;
033import java.util.jar.JarEntry;
034import java.util.jar.JarInputStream;
035import java.util.jar.JarOutputStream;
036
037import org.cleartk.classifier.jar.ClassifierBuilder_ImplBase;
038import org.cleartk.classifier.jar.JarStreams;
039
040/**
041 * A class that provided interfaces to package and unpackage a
042 * {@link TKSVMlightStringOutcomeClassifier} into a jar file.
043 * 
044 * <br>
045 * Copyright (c) 2007-2008, Regents of the University of Colorado <br>
046 * All rights reserved.
047 * 
048 * @author Daryl Lonnon
049 */
050public class TKSVMlightStringOutcomeClassifierBuilder
051    extends
052    ClassifierBuilder_ImplBase<TKSVMlightStringOutcomeClassifier, TreeFeatureVector, String, Integer> {
053
054  /**
055   * @param dir
056   *          The directory that contains the all false training data.
057   */
058  public File getTrainingDataFile(File dir) {
059    return new File(dir, "training-data-allfalse.svmlight");
060  }
061
062  /**
063   * Get the training data file for a specific label.
064   * 
065   * @param dir
066   *          The directory to keep the training data.
067   * @param label
068   *          The integer number that represents an index for a particular label.
069   * @return The file of the training data to be used to train the model.
070   */
071  public File getTrainingDataFile(File dir, int label) {
072    return new File(dir, String.format("training-data-%d.svmlight", label));
073  }
074
075  /**
076   * Train the OVASVMTK classifier.
077   * 
078   * @param dir
079   *          The directory where the training data has been written.
080   * @param args
081   *          The arguments to be used by the tk_svm_classify command. Note: -t 5 is used to specify
082   *          the use of Tree Kernels.
083   */
084  public void trainClassifier(File dir, String... args) throws Exception {
085    for (File file : dir.listFiles()) {
086      if (file.getName().matches("training-data-\\d+.svmlight")) {
087        TKSVMlightBooleanOutcomeClassifierBuilder.train(file.getPath(), args);
088      }
089    }
090  }
091
092  /**
093   * Package the classifier into a jar file.
094   */
095  @Override
096  protected void packageClassifier(File dir, JarOutputStream modelStream) throws IOException {
097    super.packageClassifier(dir, modelStream);
098
099    int label = 1;
100    while (true) {
101      File modelFile = new File(dir, String.format("training-data-%d.svmlight.model", label));
102
103      if (!modelFile.exists()) {
104        break;
105      }
106
107      String modelName = String.format("model-%d.svmlight", label);
108      JarStreams.putNextJarEntry(modelStream, modelName, modelFile);
109
110      label += 1;
111    }
112  }
113
114  private TreeMap<Integer, File> models;
115
116  /**
117   * Unpackage the classifier out of a JarInputStream.
118   */
119  @Override
120  protected void unpackageClassifier(JarInputStream modelStream) throws IOException {
121    super.unpackageClassifier(modelStream);
122    this.models = new TreeMap<Integer, File>();
123    File model;
124
125    int label = 1;
126    while ((model = getNextModel(modelStream, label)) != null) {
127      this.models.put(label, model);
128      label += 1;
129    }
130
131    if (this.models.isEmpty()) {
132      throw new IOException(String.format("no models found in %s", modelStream));
133    }
134  }
135
136  /**
137   * Create a OVATKSVMlightClassifier.
138   */
139  @Override
140  protected TKSVMlightStringOutcomeClassifier newClassifier() {
141    return new TKSVMlightStringOutcomeClassifier(
142        this.featuresEncoder,
143        this.outcomeEncoder,
144        this.models);
145  }
146
147  private static File getNextModel(JarInputStream modelStream, int label) throws IOException {
148    File mFile = File.createTempFile(String.format("model-%d", label), ".svmlight");
149    // look for a next entry or return null if there isn't one
150    JarEntry entry = modelStream.getNextJarEntry();
151    if (entry == null) {
152      return null;
153    }
154
155    // make sure the name was the model we expected
156    String expectedName = String.format("model-%d.svmlight", label);
157    if (!entry.getName().equals(expectedName)) {
158      throw new IOException(String.format(
159          "expected next jar entry to be %s, found %s",
160          expectedName,
161          entry.getName()));
162    }
163
164    BufferedWriter out = new BufferedWriter(new FileWriter(mFile));
165    BufferedReader in = new BufferedReader(new InputStreamReader(modelStream));
166    String line;
167    while ((line = in.readLine()) != null) {
168      out.append(line);
169      out.append("\n");
170    }
171    out.close();
172    // read the model from the jar stream
173    return mFile;
174  }
175}
