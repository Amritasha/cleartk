024package org.cleartk.classifier.grmm;
025
026import java.io.PrintWriter;
027import java.io.StringWriter;
028import java.util.ArrayList;
029import java.util.Arrays;
030import java.util.List;
031
032import org.cleartk.classifier.CleartkProcessingException;
033import org.cleartk.classifier.Feature;
034import org.cleartk.classifier.encoder.features.FeaturesEncoder;
035import org.cleartk.classifier.encoder.features.NameNumber;
036import org.cleartk.classifier.encoder.outcome.OutcomeEncoder;
037import org.cleartk.classifier.jar.SequenceClassifier_ImplBase;
038
039import cc.mallet.grmm.learning.ACRF;
040import cc.mallet.pipe.Pipe;
041import cc.mallet.types.Instance;
042import cc.mallet.types.Labels;
043import cc.mallet.types.LabelsSequence;
044
045/**
046 * <br>
047 * Copyright (c) 2010, University of Würzburg<br>
048 * All rights reserved.
049 * <p>
050 * 
051 * @author Martin Toepfer
052 */
053public class GrmmClassifier extends
054    SequenceClassifier_ImplBase<List<NameNumber>, String[], String[]> {
055
056  protected ACRF acrf;
057
058  protected String outcomeExample;
059
060  public GrmmClassifier(
061      FeaturesEncoder<List<NameNumber>> featuresEncoder,
062      OutcomeEncoder<String[], String[]> outcomeEncoder,
063      ACRF acrf,
064      String outcomeExample) {
065    super(featuresEncoder, outcomeEncoder);
066    this.acrf = acrf;
067    this.outcomeExample = outcomeExample;
068  }
069
070  /**
071   * This method classifies several instances at once
072   * 
073   * @param features
074   *          a list of lists of features - each list in the list represents one instance to be
075   *          classified. The list should correspond to some logical sequence of instances to be
076   *          classified (e.g. tokens in a sentence or lines in a document) that corresponds to the
077   *          model that has been built for this classifier.
078   */
079  public List<String[]> classify(final List<List<Feature>> features)
080      throws CleartkProcessingException {
081    // generate format that is appropriate for the acrf input pipe:
082    String data = "";
083    {
084      StringWriter out = new StringWriter();
085      PrintWriter printWriter = new PrintWriter(out);
086      for (List<Feature> f : features) {
087        List<NameNumber> nameNumbers = this.featuresEncoder.encodeAll(f);
088        GrmmDataWriter.writeEncoded(nameNumbers, this.outcomeExample.split(" "), printWriter);
089      }
090      data = out.toString();
091    }
092    // classify:
093    Pipe pipe = acrf.getInputPipe();
094    Instance unprocessedInstance = new Instance(data, null, "", null);
095    Instance instance = pipe.newIteratorFrom(Arrays.asList(unprocessedInstance).iterator()).next();
096    LabelsSequence bestLabels = acrf.getBestLabels(instance);
097    List<String[]> returnValues = new ArrayList<String[]>(features.size());
098    for (int i = 0; i < bestLabels.size(); i++) {
099      Labels labels = bestLabels.getLabels(i);
100      String[] outcomes = new String[labels.size()];
101      for (int j = 0; j < labels.size(); j++) {
102        outcomes[j] = labels.get(j).getBestLabel().toString();
103      }
104      returnValues.add(outcomes);
105    }
106    return returnValues;
107  }
108}
