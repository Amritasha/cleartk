025package org.cleartk.classifier.viterbi;
026
027import java.util.ArrayList;
028import java.util.Collections;
029import java.util.List;
030
031import org.apache.uima.UimaContext;
032import org.apache.uima.resource.ResourceInitializationException;
033import org.cleartk.classifier.Feature;
034import org.cleartk.classifier.feature.WindowFeature;
035import org.cleartk.classifier.feature.WindowNGramFeature;
036import org.cleartk.util.CleartkInitializationException;
037import org.uimafit.component.initialize.ConfigurationParameterInitializer;
038import org.uimafit.descriptor.ConfigurationParameter;
039import org.uimafit.factory.ConfigurationParameterFactory;
040

049
050public class DefaultOutcomeFeatureExtractor implements OutcomeFeatureExtractor {
051
052  private static final long serialVersionUID = 7476684786572310025L;
053
054  public static final String PARAM_MOST_RECENT_OUTCOME = ConfigurationParameterFactory
055      .createConfigurationParameterName(DefaultOutcomeFeatureExtractor.class, "mostRecentOutcome");
056
057  @ConfigurationParameter(description = "indicates the position of the first (most recent) outcome to include. For example, the default value of 1 means that if the outcomes produced so far by the classifier were [A, B, C, D], then the first outcome to be used as a feature would be D since it is the most recent.", defaultValue = "1")
058  private int mostRecentOutcome = 1;
059
060  public static final String PARAM_LEAST_RECENT_OUTCOME = ConfigurationParameterFactory
061      .createConfigurationParameterName(DefaultOutcomeFeatureExtractor.class, "leastRecentOutcome");
062
063  @ConfigurationParameter(description = "indicates the position of the last (least recent) outcome to include. For example, the default value of 3 means that if the outcomes produced so far by the classifier were [A, B, C, D], then the last outcome to be used as a feature would be B since and is considered the least recent.", defaultValue = "3")
064  private int leastRecentOutcome = 3;
065
066  public static final String PARAM_USE_BIGRAM = ConfigurationParameterFactory
067      .createConfigurationParameterName(DefaultOutcomeFeatureExtractor.class, "useBigram");
068
069  @ConfigurationParameter(description = "when true indicates that bigrams of outcomes should be included as features", defaultValue = "true")
070  private boolean useBigram = true;
071
072  public static final String PARAM_USE_TRIGRAM = ConfigurationParameterFactory
073      .createConfigurationParameterName(DefaultOutcomeFeatureExtractor.class, "useTrigram");
074
075  @ConfigurationParameter(defaultValue = "true", description = "indicates that trigrams of outcomes should be included as features")
076  private boolean useTrigram = true;
077
078  public static final String PARAM_USE4GRAM = ConfigurationParameterFactory
079      .createConfigurationParameterName(DefaultOutcomeFeatureExtractor.class, "use4gram");
080
081  @ConfigurationParameter(defaultValue = "false", description = "indicates that 4-grams of outcomes should be included as features")
082  private boolean use4gram = false;
083
084  public void initialize(UimaContext context) throws ResourceInitializationException {
085    ConfigurationParameterInitializer.initialize(this, context);
086
087    if (mostRecentOutcome < 1) {
088      throw CleartkInitializationException.parameterLessThan(
089          PARAM_MOST_RECENT_OUTCOME,
090          1,
091          mostRecentOutcome);
092    }
093
094    if (leastRecentOutcome < mostRecentOutcome) {
095      throw CleartkInitializationException.parameterLessThan(
096          PARAM_LEAST_RECENT_OUTCOME,
097          mostRecentOutcome,
098          leastRecentOutcome);
099    }
100
101  }
102
103  public List<Feature> extractFeatures(List<Object> previousOutcomes) {
104    if (previousOutcomes == null || previousOutcomes.size() == 0) {
105      return Collections.emptyList();
106    }
107
108    List<Feature> features = new ArrayList<Feature>();
109
110    for (int i = mostRecentOutcome; i <= leastRecentOutcome; i++) {
111      int index = previousOutcomes.size() - i;
112      if (index >= 0) {
113        Feature feature = new WindowFeature(
114            "PreviousOutcome",
115            previousOutcomes.get(index),
116            WindowFeature.ORIENTATION_LEFT,
117            i,
118            (Feature) null);
119        features.add(feature);
120      }
121    }
122
123    if (useBigram && previousOutcomes.size() >= 2) {
124      int size = previousOutcomes.size();
125      String featureValue = previousOutcomes.get(size - 1).toString() + "_"
126          + previousOutcomes.get(size - 2);
127      Feature feature = new WindowNGramFeature(
128          "PreviousOutcomes",
129          featureValue,
130          WindowNGramFeature.ORIENTATION_LEFT,
131          WindowNGramFeature.DIRECTION_LEFT_TO_RIGHT,
132          "_",
133          2,
134          1,
135          (List<Feature>) null);
136      features.add(feature);
137
138    }
139
140    if (useTrigram && previousOutcomes.size() >= 3) {
141      int size = previousOutcomes.size();
142      String featureValue = previousOutcomes.get(size - 1).toString() + "_"
143          + previousOutcomes.get(size - 2) + "_" + previousOutcomes.get(size - 3);
144      Feature feature = new WindowNGramFeature(
145          "PreviousOutcomes",
146          featureValue,
147          WindowNGramFeature.ORIENTATION_LEFT,
148          WindowNGramFeature.DIRECTION_LEFT_TO_RIGHT,
149          "_",
150          3,
151          1,
152          (List<Feature>) null);
153      features.add(feature);
154
155    }
156
157    if (use4gram && previousOutcomes.size() >= 4) {
158      int size = previousOutcomes.size();
159      String featureValue = previousOutcomes.get(size - 1).toString() + "_"
160          + previousOutcomes.get(size - 2) + "_" + previousOutcomes.get(size - 3) + "_"
161          + previousOutcomes.get(size - 4);
162      Feature feature = new WindowNGramFeature(
163          "PreviousOutcomes",
164          featureValue,
165          WindowNGramFeature.ORIENTATION_LEFT,
166          WindowNGramFeature.DIRECTION_LEFT_TO_RIGHT,
167          "_",
168          4,
169          1,
170          (List<Feature>) null);
171      features.add(feature);
172    }
173
174    return features;
175  }
