026import java.util.ArrayList;
027import java.util.Collections;
028import java.util.Iterator;
029import java.util.List;

031import org.cleartk.classifier.CleartkProcessingException;
032import org.cleartk.classifier.Feature;
033import org.cleartk.classifier.ScoredOutcome;
034import org.cleartk.classifier.encoder.CleartkEncoderException;
035import org.cleartk.classifier.encoder.features.FeaturesEncoder;
036import org.cleartk.classifier.encoder.features.NameNumber;
037import org.cleartk.classifier.encoder.outcome.OutcomeEncoder;
038import org.cleartk.classifier.jar.Classifier_ImplBase;
039
040import cc.mallet.classify.Classification;
041import cc.mallet.classify.Classifier;
042import cc.mallet.types.Alphabet;
043import cc.mallet.types.FeatureVector;
044import cc.mallet.types.Instance;
045import cc.mallet.types.Labeling;

056 */
057public abstract class MalletClassifier_ImplBase<OUTCOME_TYPE> extends
058    Classifier_ImplBase<List<NameNumber>, OUTCOME_TYPE, String> {
059
060  protected Classifier classifier;
061
062  Alphabet alphabet;
063
064  public MalletClassifier_ImplBase(
065      FeaturesEncoder<List<NameNumber>> featuresEncoder,
066      OutcomeEncoder<OUTCOME_TYPE, String> outcomeEncoder,
067      Classifier classifier) {
068    super(featuresEncoder, outcomeEncoder);
069    this.classifier = classifier;
070    this.alphabet = classifier.getAlphabet();
071  }
072
073  public OUTCOME_TYPE classify(List<Feature> features) throws CleartkProcessingException {
074    Classification classification = classifier.classify(toInstance(features));
075    String returnValue = classification.getLabeling().getBestLabel().toString();
076    return outcomeEncoder.decode(returnValue);
077  }
078
079  @Override
080  public List<ScoredOutcome<OUTCOME_TYPE>> score(List<Feature> features, int maxResults)
081      throws CleartkProcessingException {
082    Classification classification = classifier.classify(toInstance(features));
083    List<ScoredOutcome<OUTCOME_TYPE>> returnValues = new ArrayList<ScoredOutcome<OUTCOME_TYPE>>(
084        maxResults);
085    Labeling labeling = classification.getLabeling();
086
087    if (maxResults == 1) {
088      String bestOutcome = labeling.getBestLabel().toString();
089      OUTCOME_TYPE outcome = outcomeEncoder.decode(bestOutcome);
090      double score = labeling.getBestValue();
091      returnValues.add(new ScoredOutcome<OUTCOME_TYPE>(outcome, score));
092      return returnValues;
093    }
094
095    for (int i = 0; i < labeling.numLocations(); i++) {
096      String label = labeling.getLabelAtRank(i).toString();
097      OUTCOME_TYPE outcome = outcomeEncoder.decode(label);
098      double score = labeling.getValueAtRank(i);
099      returnValues.add(new ScoredOutcome<OUTCOME_TYPE>(outcome, score));
100    }
101
102    Collections.sort(returnValues);
103    if (returnValues.size() > maxResults) {
104      return returnValues.subList(0, maxResults);
105    } else {
106      return returnValues;
107    }
108
109  }
110
111  public Instance[] toInstances(List<List<Feature>> features) throws CleartkEncoderException {
112
113    Instance[] instances = new Instance[features.size()];
114    for (int i = 0; i < features.size(); i++) {
115      instances[i] = toInstance(features.get(i));
116    }
117    return instances;
118  }
119
120  public Instance toInstance(List<Feature> features) throws CleartkEncoderException {
121    List<NameNumber> nameNumbers = featuresEncoder.encodeAll(features);
122
123    Iterator<NameNumber> nameNumberIterator = nameNumbers.iterator();
124    while (nameNumberIterator.hasNext()) {
125      NameNumber nameNumber = nameNumberIterator.next();
126      if (!alphabet.contains(nameNumber.name))
127        nameNumberIterator.remove();
128    }
129
130    String[] keys = new String[nameNumbers.size()];
131    double[] values = new double[nameNumbers.size()];
132
133    for (int i = 0; i < nameNumbers.size(); i++) {
134      NameNumber nameNumber = nameNumbers.get(i);
135      keys[i] = nameNumber.name;
136      values[i] = nameNumber.number.doubleValue();
137    }
138
139    int[] keyIndices = FeatureVector.getObjectIndices(keys, alphabet, true);
140    FeatureVector fv = new FeatureVector(alphabet, keyIndices, values);
141
142    Instance instance = new Instance(fv, null, null, null);
143    return instance;
144  }
145}
