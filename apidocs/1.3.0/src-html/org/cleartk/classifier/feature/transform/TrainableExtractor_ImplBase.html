024package org.cleartk.classifier.feature.transform;
025
026import java.util.ArrayList;
027import java.util.List;
028import java.util.Stack;
029
030import org.cleartk.classifier.Feature;
031import org.cleartk.classifier.Instance;
032
033/**
034 * Copyright (c) 2012, Regents of the University of Colorado <br>
035 * All rights reserved.
036 */
037
038public abstract class TrainableExtractor_ImplBase<OUTCOME_T> implements
039    TrainableExtractor<OUTCOME_T> {
040
041  protected String name;
042
043  public TrainableExtractor_ImplBase(String name) {
044    this.name = name;
045  }
046
047  protected boolean isTransformable(Feature feature) {
048    return feature instanceof TransformableFeature && this.name.equals(feature.getName());
049  }
050
051  /**
052   * Recursively look through all features in an instance to find the features this extractor is
053   * responsible for extracting
054   */
055  protected List<TransformableFeature> selectTransformableFeatures(Instance<OUTCOME_T> instance) {
056
057    List<TransformableFeature> features = new ArrayList<TransformableFeature>();
058    Stack<Feature> featuresToProcess = new Stack<Feature>();
059    featuresToProcess.addAll(instance.getFeatures());
060
061    while (!featuresToProcess.isEmpty()) {
062      Feature currFeature = featuresToProcess.pop();
063      if (currFeature instanceof TransformableFeature) {
064        TransformableFeature transformable = (TransformableFeature) currFeature;
065        if (transformable.getName() == this.name) {
066          features.add(transformable);
067        } else {
068          featuresToProcess.addAll(transformable.getFeatures());
069        }
070      }
071    }
072    return features;
073  }
074
075}
