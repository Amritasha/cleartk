044public class WindowFeature extends Feature {
045  private static final long serialVersionUID = 1L;
046
047  public static final String ORIENTATION_LEFT = "L";
048
049  public static final String ORIENTATION_RIGHT = "R";
050
051  public static final String ORIENTATION_MIDDLE = "M";
052
053  public static final String ORIENTATION_MIDDLE_REVERSE = "MR";
054
055  /**
056   * This method returns the annotation at index 0 w.r.t. the orientation of the extractor, the
057   * annotation we are extracting features for and the window the provides the boundaries for
058   * feature extraction.
059   * 
060   * @param jCas
061   *          The JCas containing the focus annotation
062   * @param itemClass
063   *          The type of Annotation around the focus annotation to select
064   * @param focusAnnotation
065   *          The annotation from which the orientation is defined.
066   * @param orientation
067   *          The orientation defining which annotation is the start.
068   * @return the annotation at index 0:
069   *         <ul>
070   *         <li>for ORIENTATION_LEFT return the annotation adjacent and to the left of annotation
071   *         <li>for ORIENTATION_RIGHT return the annotation adjacent and to the right of annotation
072   *         <li>for ORIENTATION_MIDDLE return the left most annotation completely contained in
073   *         annotation, null if does not exist
074   *         <li>for ORIENTATION_MIDDLE_REVERSE return the right most annotation completely
075   *         contained in annotation, null if does not exist
076   *         </ul>
077   */
078  public static <T extends Annotation> T getStartAnnotation(
079      JCas jCas,
080      Class<T> itemClass,
081      Annotation focusAnnotation,
082      String orientation) {
083    if (orientation.equals(WindowFeature.ORIENTATION_LEFT)) {
084      List<T> anns = JCasUtil.selectPreceding(jCas, itemClass, focusAnnotation, 1);
085      if (anns.size() > 0 && anns.get(0) == focusAnnotation) {
086        anns = JCasUtil.selectPreceding(jCas, itemClass, focusAnnotation, 2).subList(0, 1);
087      }
088      return anns.size() == 0 ? null : anns.get(0);
089    } else if (orientation.equals(WindowFeature.ORIENTATION_RIGHT)) {
090      List<T> anns = JCasUtil.selectFollowing(jCas, itemClass, focusAnnotation, 1);
091      if (anns.size() > 0 && anns.get(0) == focusAnnotation) {
092        anns = JCasUtil.selectFollowing(jCas, itemClass, focusAnnotation, 2).subList(1, 2);
093      }
094      return anns.size() == 0 ? null : anns.get(0);
095    } else if (orientation.equals(WindowFeature.ORIENTATION_MIDDLE)) {
096      List<T> anns = JCasUtil.selectCovered(jCas, itemClass, focusAnnotation);
097      if (anns.size() == 0) {
098        return itemClass.isInstance(focusAnnotation) ? itemClass.cast(focusAnnotation) : null;
099      } else {
100        return anns.get(0);
101      }
102    } else if (orientation.equals(WindowFeature.ORIENTATION_MIDDLE_REVERSE)) {
103      List<T> anns = JCasUtil.selectCovered(jCas, itemClass, focusAnnotation);
104      if (anns.size() == 0) {
105        return itemClass.isInstance(focusAnnotation) ? itemClass.cast(focusAnnotation) : null;
106      } else {
107        return anns.get(anns.size() - 1);
108      }
109    }
110    throw new IllegalArgumentException("Unknown orientation: " + orientation);
111  }
112
113  private String orientation = null;
114
115  private Integer position = null;
116
117  private Feature windowedFeature = null;
118
119  private Integer outOfBoundsDistance = 0;
120
121  public WindowFeature(
122      String name,
123      Object value,
124      String orientation,
125      Integer position,
126      Feature windowedFeature,
127      Integer outOfBoundsDistance) {
128    super(value);
129    this.orientation = orientation;
130    this.position = position;
131    this.windowedFeature = windowedFeature;
132    this.outOfBoundsDistance = outOfBoundsDistance;
133    this.name = createName(name);
134  }
135
136  public WindowFeature(
137      String name,
138      Object value,
139      String orientation,
140      Integer position,
141      Feature windowedFeature) {
142    this(name, value, orientation, position, windowedFeature, null);
143  }
144
145  public WindowFeature(
146      String name,
147      Object value,
148      String orientation,
149      Integer position,
150      Integer outOfBoundsDistance) {
151    this(name, value, orientation, position, null, outOfBoundsDistance);
152  }
153
154  private String createName(String namePrefix) {
155    if (namePrefix == null)
156      namePrefix = "Window";
157
158    StringBuffer sb = new StringBuffer();
159    if (orientation != null)
160      sb.append(orientation);
161    if (position != null)
162      sb.append(position);
163
164    if (outOfBoundsDistance != null && outOfBoundsDistance > 0)
165      sb.append("OOB" + outOfBoundsDistance);
166
167    String windowedFeatureName = null;
168    if (windowedFeature != null)
169      windowedFeatureName = windowedFeature.getName();
170
171    return Feature.createName(namePrefix, sb.toString(), windowedFeatureName);
172
173  }
174
175  public String getOrientation() {
176    return orientation;
177  }
178
179  public int getOutOfBoundsDistance() {
180    return outOfBoundsDistance;
181  }
182
183  public int getPosition() {
184    return position;
185  }
186
187  public Feature getWindowedFeature() {
188    return windowedFeature;
189  }
