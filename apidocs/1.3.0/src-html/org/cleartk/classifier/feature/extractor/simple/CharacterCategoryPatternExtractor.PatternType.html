024package org.cleartk.classifier.feature.extractor.simple;
025
026import java.util.Arrays;
027import java.util.List;
028
029import org.apache.uima.jcas.JCas;
030import org.apache.uima.jcas.tcas.Annotation;
031import org.cleartk.classifier.Feature;
032import org.cleartk.classifier.feature.extractor.CleartkExtractorException;
033
034/**
035 * A feature extractor that generates a pattern based on the <a
036 * href="http://unicode.org/reports/tr49/">Unicode categories</a> of each of the characters in the
037 * annotation text. For example, "A-z0" is an uppercase letter, followed by a dash, followed by a
038 * lowercase letter, followed by a digit, and so would get the pattern "LuPdLlNd".
039 * 
040 * <br>
041 * Copyright (c) 2011, Regents of the University of Colorado <br>
042 * All rights reserved.
043 * 
044 * @author Steven Bethard
045 */
046public class CharacterCategoryPatternExtractor implements SimpleFeatureExtractor {
047
048  /**
049   * The type of pattern to generate in feature values.
050   */
051  public static enum PatternType {
052    /**
053     * The standard pattern, where one category abbreviation is added to the feature value for each
054     * character in the text.
055     */
056    ONE_PER_CHAR,
057    /**
058     * A simplified pattern, where if the same category appears many times in a row, the category is
059     * added to the feature value only once. For example "XX00" would get the pattern "LuNd" since
060     * there are two uppercase letters followed by two digits.
061     */
062    REPEATS_MERGED
063  }
064
065  private PatternType patternType;
066
067  private String name;
068
069  /**
070   * Create the standard feature extractor, where one category is added to the feature value for
071   * each character in the text. See {@link PatternType#ONE_PER_CHAR}.
072   */
073  public CharacterCategoryPatternExtractor() {
074    this(PatternType.ONE_PER_CHAR);
075  }
076
077  /**
078   * Create a feature extractor with the specified pattern type. See {@link PatternType} for the
079   * acceptable pattern types.
080   * 
081   * @param patternType
082   *          The type of pattern to generate in feature values.
083   */
084  public CharacterCategoryPatternExtractor(PatternType patternType) {
085    this.patternType = patternType;
086    switch (this.patternType) {
087      case ONE_PER_CHAR:
088        this.name = "CharPattern";
089        break;
090      case REPEATS_MERGED:
091        this.name = "CharPatternRepeatsMerged";
092        break;
093    }
094  }
095
096  @Override
097  public List<Feature> extract(JCas view, Annotation focusAnnotation)
098      throws CleartkExtractorException {
099    StringBuilder builder = new StringBuilder();
100    String text = focusAnnotation.getCoveredText();
101    String lastType = null;
102    for (int i = 0; i < text.length(); i += 1) {
103      String type;
104      int typeInt = Character.getType(text.charAt(i));
105      switch (typeInt) {
106        case Character.CONTROL:
107          type = "CC";
108          break;
109        case Character.FORMAT:
110          type = "Cf";
111          break;
112        case Character.UNASSIGNED:
113          type = "Cn";
114          break;
115        case Character.PRIVATE_USE:
116          type = "Co";
117          break;
118        case Character.SURROGATE:
119          type = "Cs";
120          break;
121        case Character.LOWERCASE_LETTER:
122          type = "Ll";
123          break;
124        case Character.MODIFIER_LETTER:
125          type = "Lm";
126          break;
127        case Character.OTHER_LETTER:
128          type = "Lo";
129          break;
130        case Character.TITLECASE_LETTER:
131          type = "Lt";
132          break;
133        case Character.UPPERCASE_LETTER:
134          type = "Lu";
135          break;
136        case Character.COMBINING_SPACING_MARK:
137          type = "Mc";
138          break;
139        case Character.ENCLOSING_MARK:
140          type = "Me";
141          break;
142        case Character.NON_SPACING_MARK:
143          type = "Mn";
144          break;
145        case Character.DECIMAL_DIGIT_NUMBER:
146          type = "Nd";
147          break;
148        case Character.LETTER_NUMBER:
149          type = "Nl";
150          break;
151        case Character.OTHER_NUMBER:
152          type = "No";
153          break;
154        case Character.CONNECTOR_PUNCTUATION:
155          type = "Pc";
156          break;
157        case Character.DASH_PUNCTUATION:
158          type = "Pd";
159          break;
160        case Character.END_PUNCTUATION:
161          type = "Pe";
162          break;
163        case Character.FINAL_QUOTE_PUNCTUATION:
164          type = "Pf";
165          break;
166        case Character.INITIAL_QUOTE_PUNCTUATION:
167          type = "Pi";
168          break;
169        case Character.OTHER_PUNCTUATION:
170          type = "Po";
171          break;
172        case Character.START_PUNCTUATION:
173          type = "Ps";
174          break;
175        case Character.CURRENCY_SYMBOL:
176          type = "Sc";
177          break;
178        case Character.MODIFIER_SYMBOL:
179          type = "Sk";
180          break;
181        case Character.MATH_SYMBOL:
182          type = "Sm";
183          break;
184        case Character.OTHER_SYMBOL:
185          type = "So";
186          break;
187        case Character.LINE_SEPARATOR:
188          type = "Zl";
189          break;
190        case Character.PARAGRAPH_SEPARATOR:
191          type = "Zp";
192          break;
193        case Character.SPACE_SEPARATOR:
194          type = "Zs";
195          break;
196        default:
197          throw new RuntimeException("Unknown character type: " + typeInt);
198      }
199      switch (this.patternType) {
200        case ONE_PER_CHAR:
201          builder.append(type);
202          break;
203        case REPEATS_MERGED:
204          if (!type.equals(lastType)) {
205            builder.append(type);
206          }
207          break;
208      }
209      lastType = type;
210    }
211    return Arrays.asList(new Feature(this.name, builder.toString()));
212  }
213}
