026import java.util.Collections;
027import java.util.List;
028
029import org.cleartk.classifier.Feature;

041
042public class WindowNGramFeature extends Feature {
043  private static final long serialVersionUID = 1L;
044
045  public static final String ORIENTATION_LEFT = "L";
046
047  public static final String ORIENTATION_RIGHT = "R";
048
049  public static final String ORIENTATION_MIDDLE = "M";
050
051  public static final String ORIENTATION_MIDDLE_REVERSE = "MR";
052
053  public static final String DIRECTION_LEFT_TO_RIGHT = "L2R";
054
055  public static final String DIRECTION_RIGHT_TO_LEFT = "R2L";
056
057  private String orientation = null;
058
059  private String direction = null;
060
061  private String separator;
062
063  private Integer size = null;
064
065  private Integer index = null;
066
067  private List<Feature> windowedFeatures = null;
068
069  public WindowNGramFeature(
070      String name,
071      Object value,
072      String orientation,
073      String direction,
074      String separator,
075      Integer size,
076      Integer index,
077      List<Feature> windowedFeatures) {
078    super(value);
079    this.orientation = orientation;
080    this.direction = direction;
081    this.separator = separator;
082    this.size = size;
083    this.index = index;
084    this.windowedFeatures = windowedFeatures;
085    this.name = createName(name);
086  }
087
088  public Integer getIndex() {
089    return index;
090  }
091
092  private String createName(String namePrefix) {
093    if (namePrefix == null)
094      namePrefix = "WindowNGram";
095
096    StringBuffer returnValue = new StringBuffer();
097    returnValue.append(orientation + index + "_" + size + "gram_" + direction);
098
099    String windowedFeatureName = null;
100    if (windowedFeatures != null && windowedFeatures.size() > 0)
101      windowedFeatureName = windowedFeatures.get(0).getName();
102
103    return Feature.createName(namePrefix, returnValue.toString(), windowedFeatureName);
104
105  }
106
107  public String getOrientation() {
108    return orientation;
109  }
110
111  public int getSize() {
112    return size;
113  }
114
115  public String getDirection() {
116    return direction;
117  }
118
119  public String getSeparator() {
120    return separator;
121  }
122
123  public List<Feature> getWindowedFeatures() {
124    return Collections.unmodifiableList(windowedFeatures);
125  }
126
127}
