024package org.cleartk.classifier.jar;
025
026import java.io.BufferedInputStream;
027import java.io.FileInputStream;
028import java.io.IOException;
029import java.io.InputStream;
030import java.net.MalformedURLException;
031import java.net.URL;
032import java.util.jar.JarInputStream;
033
034import org.apache.uima.UimaContext;
035import org.apache.uima.resource.ResourceInitializationException;
036import org.uimafit.component.initialize.ConfigurationParameterInitializer;
037import org.uimafit.descriptor.ConfigurationParameter;
038import org.uimafit.factory.ConfigurationParameterFactory;
039import org.uimafit.factory.initializable.Initializable;
040
041/**
042 * <br>
043 * Copyright (c) 2011, Regents of the University of Colorado <br>
044 * All rights reserved.
045 */
046public abstract class GenericJarClassifierFactory<CLASSIFIER_TYPE> implements Initializable {
047
048  public static final String PARAM_CLASSIFIER_JAR_PATH = ConfigurationParameterFactory
049      .createConfigurationParameterName(GenericJarClassifierFactory.class, "classifierJarPath");
050
051  @ConfigurationParameter(mandatory = true, description = "provides the path to the jar file that should be used to instantiate the classifier.")
052  private String classifierJarPath;
053
054  public void setClassifierJarPath(String classifierJarPath) {
055    this.classifierJarPath = classifierJarPath;
056  }
057
058  public void initialize(UimaContext context) throws ResourceInitializationException {
059    ConfigurationParameterInitializer.initialize(this, context);
060  }
061
062  public CLASSIFIER_TYPE createClassifier() throws IOException {
063    InputStream stream;
064    try {
065      stream = new URL(this.classifierJarPath).openStream();
066    } catch (MalformedURLException e) {
067      stream = new FileInputStream(this.classifierJarPath);
068    }
069    stream = new BufferedInputStream(stream);
070    JarInputStream modelStream = new JarInputStream(stream);
071    JarClassifierBuilder<?> builder = JarClassifierBuilder.fromManifest(modelStream.getManifest());
072    try {
073      return this.getClassifierClass().cast(builder.loadClassifier(modelStream));
074    } finally {
075      stream.close();
076    }
077  }
078
079  protected abstract Class<CLASSIFIER_TYPE> getClassifierClass();
080}
