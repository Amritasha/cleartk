081  private static void exit_with_help() {
082    System.out.print("Usage: svm_train [options] training_set_file [model_file]\n"
083        + "options:\n"
084        + "-s svm_type : set type of SVM (default 0)\n"
085        + "  0 -- C-SVC\n"
086        + "  1 -- nu-SVC\n"
087        + "  2 -- one-class SVM\n"
088        + "  3 -- epsilon-SVR\n"
089        + "  4 -- nu-SVR\n"
090        + "-t kernel_type : set type of kernel function (default 2)\n"
091        + "  0 -- linear: u'*v\n"
092        + "  1 -- polynomial: (gamma*u'*v + coef0)^degree\n"
093        + "  2 -- radial basis function: exp(-gamma*|u-v|^2)\n"
094        + "  3 -- sigmoid: tanh(gamma*u'*v + coef0)\n"
095        + "  4 -- precomputed kernel (kernel values in training_set_file)\n"
096        + "-d degree : set degree in kernel function (default 3)\n"
097        + "-g gamma : set gamma in kernel function (default 1/num_features)\n"
098        + "-r coef0 : set coef0 in kernel function (default 0)\n"
099        + "-c cost : set the parameter C of C-SVC, epsilon-SVR, and nu-SVR (default 1)\n"
100        + "-n nu : set the parameter nu of nu-SVC, one-class SVM, and nu-SVR (default 0.5)\n"
101        + "-p epsilon : set the epsilon in loss function of epsilon-SVR (default 0.1)\n"
102        + "-m cachesize : set cache memory size in MB (default 100)\n"
103        + "-e epsilon : set tolerance of termination criterion (default 0.001)\n"
104        + "-h shrinking : whether to use the shrinking heuristics, 0 or 1 (default 1)\n"
105        + "-b probability_estimates : whether to train a SVC or SVR model for probability estimates, 0 or 1 (default 0)\n"
106        + "-wi weight : set the parameter C of class i to weight*C, for C-SVC (default 1)\n"
107        + "-v n : n-fold cross validation mode\n" + "-q : quiet mode (no outputs)\n");
108    System.exit(1);
109  }

111  private void do_cross_validation() {
112    int i;
113    int total_correct = 0;
114    double total_error = 0;
115    double sumv = 0, sumy = 0, sumvv = 0, sumyy = 0, sumvy = 0;
116    double[] target = new double[prob.l];
117
118    svm.svm_cross_validation(prob, param, nr_fold, target);
119    if (param.svm_type == svm_parameter.EPSILON_SVR || param.svm_type == svm_parameter.NU_SVR) {
120      for (i = 0; i < prob.l; i++) {
121        double y = prob.y[i];
122        double v = target[i];
123        total_error += (v - y) * (v - y);
124        sumv += v;
125        sumy += y;
126        sumvv += v * v;
127        sumyy += y * y;
128        sumvy += v * y;
129      }
130      System.out.print("Cross Validation Mean squared error = " + total_error / prob.l + "\n");
131      System.out.print("Cross Validation Squared correlation coefficient = "
132          + ((prob.l * sumvy - sumv * sumy) * (prob.l * sumvy - sumv * sumy))
133          / ((prob.l * sumvv - sumv * sumv) * (prob.l * sumyy - sumy * sumy)) + "\n");
134    } else {
135      for (i = 0; i < prob.l; i++)
136        if (target[i] == prob.y[i])
137          ++total_correct;
138      System.out.print("Cross Validation Accuracy = " + 100.0 * total_correct / prob.l + "%\n");
139    }
140  }
141
142  private void run(String argv[]) throws IOException {
143    parse_command_line(argv);
144    read_problem();
145    error_msg = svm.svm_check_parameter(prob, param);
146
147    if (error_msg != null) {
148      System.err.print("ERROR: " + error_msg + "\n");
149      System.exit(1);
150    }
151
152    if (cross_validation != 0) {
153      do_cross_validation();
154    } else {
155      model = svm.svm_train(prob, param);
156      svm.svm_save_model(model_file_name, model);
157    }
158  }

161    LIBSVMTrain t = new LIBSVMTrain();
162    t.run(argv);
163  }

165  private static double atof(String s) {
166    double d = Double.valueOf(s).doubleValue();
167    if (Double.isNaN(d) || Double.isInfinite(d)) {
168      System.err.print("NaN or Infinity in input\n");
169      System.exit(1);
170    }
171    return (d);
172  }
173
174  private static int atoi(String s) {
175    return Integer.parseInt(s);
176  }
177
178  private void parse_command_line(String argv[]) {
179    int i;
180    svm_print_interface print_func = null; // default printing to stdout
181
182    param = new svm_parameter();
183    // default values
184    param.svm_type = svm_parameter.C_SVC;
185    param.kernel_type = svm_parameter.RBF;
186    param.degree = 3;
187    param.gamma = 0; // 1/num_features
188    param.coef0 = 0;
189    param.nu = 0.5;
190    param.cache_size = 100;
191    param.C = 1;
192    param.eps = 1e-3;
193    param.p = 0.1;
194    param.shrinking = 1;
195    param.probability = 0;
196    param.nr_weight = 0;
197    param.weight_label = new int[0];
198    param.weight = new double[0];
199    cross_validation = 0;
200
201    // parse options
202    for (i = 0; i < argv.length; i++) {
203      if (argv[i].charAt(0) != '-')
204        break;
205      if (++i >= argv.length)
206        exit_with_help();
207      switch (argv[i - 1].charAt(1)) {
208        case 's':
209          param.svm_type = atoi(argv[i]);
210          break;
211        case 't':
212          param.kernel_type = atoi(argv[i]);
213          break;
214        case 'd':
215          param.degree = atoi(argv[i]);
216          break;
217        case 'g':
218          param.gamma = atof(argv[i]);
219          break;
220        case 'r':
221          param.coef0 = atof(argv[i]);
222          break;
223        case 'n':
224          param.nu = atof(argv[i]);
225          break;
226        case 'm':
227          param.cache_size = atof(argv[i]);
228          break;
229        case 'c':
230          param.C = atof(argv[i]);
231          break;
232        case 'e':
233          param.eps = atof(argv[i]);
234          break;
235        case 'p':
236          param.p = atof(argv[i]);
237          break;
238        case 'h':
239          param.shrinking = atoi(argv[i]);
240          break;
241        case 'b':
242          param.probability = atoi(argv[i]);
243          break;
244        case 'q':
245          print_func = svm_print_null;
246          i--;
247          break;
248        case 'v':
249          cross_validation = 1;
250          nr_fold = atoi(argv[i]);
251          if (nr_fold < 2) {
252            System.err.print("n-fold cross validation: n must >= 2\n");
253            exit_with_help();
254          }
255          break;
256        case 'w':
257          ++param.nr_weight;
258          {
259            int[] old = param.weight_label;
260            param.weight_label = new int[param.nr_weight];
261            System.arraycopy(old, 0, param.weight_label, 0, param.nr_weight - 1);
262          }
263
264          {
265            double[] old = param.weight;
266            param.weight = new double[param.nr_weight];
267            System.arraycopy(old, 0, param.weight, 0, param.nr_weight - 1);
268          }
269
270          param.weight_label[param.nr_weight - 1] = atoi(argv[i - 1].substring(2));
271          param.weight[param.nr_weight - 1] = atof(argv[i]);
272          break;
273        default:
274          System.err.print("Unknown option: " + argv[i - 1] + "\n");
275          exit_with_help();
276      }
277    }
278
279    svm.svm_set_print_string_function(print_func);
280
281    // determine filenames
282
283    if (i >= argv.length)
284      exit_with_help();
285
286    input_file_name = argv[i];
287
288    if (i < argv.length - 1)
289      model_file_name = argv[i + 1];
290    else {
291      int p = argv[i].lastIndexOf('/');
292      ++p; // whew...
293      model_file_name = argv[i].substring(p) + ".model";
294    }
295  }

297  // read in a problem (in svmlight format)
298
299  private void read_problem() throws IOException {
300    BufferedReader fp = new BufferedReader(new FileReader(input_file_name));
301    Vector<Double> vy = new Vector<Double>();
302    Vector<svm_node[]> vx = new Vector<svm_node[]>();
303    int max_index = 0;
304
305    while (true) {
306      String line = fp.readLine();
307      if (line == null)
308        break;
309
310      StringTokenizer st = new StringTokenizer(line, " \t\n\r\f:");
311
312      vy.addElement(atof(st.nextToken()));
313      int m = st.countTokens() / 2;
314      svm_node[] x = new svm_node[m];
315      for (int j = 0; j < m; j++) {
316        x[j] = new svm_node();
317        x[j].index = atoi(st.nextToken());
318        x[j].value = atof(st.nextToken());
319      }
320      if (m > 0)
321        max_index = Math.max(max_index, x[m - 1].index);
322      vx.addElement(x);
323    }
324
325    prob = new svm_problem();
326    prob.l = vy.size();
327    prob.x = new svm_node[prob.l][];
328    for (int i = 0; i < prob.l; i++)
329      prob.x[i] = vx.elementAt(i);
330    prob.y = new double[prob.l];
331    for (int i = 0; i < prob.l; i++)
332      prob.y[i] = vy.elementAt(i);
333
334    if (param.gamma == 0 && max_index > 0)
335      param.gamma = 1.0 / max_index;
336
337    if (param.kernel_type == svm_parameter.PRECOMPUTED)
338      for (int i = 0; i < prob.l; i++) {
339        if (prob.x[i][0].index != 0) {
340          System.err.print("Wrong kernel matrix: first column must be 0:sample_serial_number\n");
341          System.exit(1);
342        }
343        if ((int) prob.x[i][0].value <= 0 || (int) prob.x[i][0].value > max_index) {
344          System.err.print("Wrong input format: sample_serial_number out of range\n");
345          System.exit(1);
346        }
347      }
348
349    fp.close();
350  }
