098  @Override
099  public void initialize(UimaContext context) throws ResourceInitializationException {
100    super.initialize(context);
101
102    SimpleFeatureExtractor[] tokenExtractors = {
103        new CoveredTextExtractor(),
104        new TypePathExtractor(Token.class, "stem"),
105        new TypePathExtractor(Token.class, "pos") };
106
107    SimpleFeatureExtractor[] constituentExtractors = {
108        new TypePathExtractor(TreebankNode.class, "nodeType"),
109        // new TypePathExtractor(TreebankNode.class, "nodeTags"),
110        new HeadWordExtractor(new CombinedExtractor(tokenExtractors), true) };
111
112    predicateTokenExtractor = new CombinedExtractor(tokenExtractors);
113    predicateNodeExtractor = new SubCategorizationExtractor("PredicateNode");
114
115    pathExtractor = new SyntacticPathExtractor(
116        new TypePathExtractor(TreebankNode.class, "nodeType"));
117    partialPathExtractor = new SyntacticPathExtractor(new TypePathExtractor(
118        TreebankNode.class,
119        "nodeType"), true);
120    relPosExtractor = new RelativePositionExtractor();
121    distanceExtractor = new DistanceExtractor("ConstituentPredicate", Token.class);
122
123    constituentExtractor = new CombinedExtractor(constituentExtractors);
124    leftSiblingExtractor = new CombinedExtractor(constituentExtractors);
125    rightSiblingExtractor = new CombinedExtractor(constituentExtractors);
126    parentExtractor = new CombinedExtractor(constituentExtractors);
127    firstAndLastWordExtractor = new CleartkExtractor(Token.class, new NamingExtractor(
128        "Constituent",
129        new CombinedExtractor(tokenExtractors)), new FirstCovered(1), new LastCovered(1));
130  }

132  @Override
133  public void process(JCas jCas) throws AnalysisEngineProcessException {
134    for (Sentence sentence : JCasUtil.select(jCas, Sentence.class)) {
135      processSentence(jCas, sentence);
136    }
137  }
138
139  private void processSentence(JCas jCas, Sentence sentence) throws AnalysisEngineProcessException {
140
141    List<TreebankNode> sentenceConstituents = new ArrayList<TreebankNode>(80);
142    TopTreebankNode top = AnnotationUtil.selectFirstMatching(jCas, TopTreebankNode.class, sentence);
143    if (top == null) {
144      throw CleartkExtractorException.noAnnotationInWindow(TopTreebankNode.class, sentence);
145    }
146    collectConstituents(top, sentenceConstituents);
147
148    List<List<Feature>> sentenceConstituentFeatures = new ArrayList<List<Feature>>(
149        sentenceConstituents.size());
150    for (TreebankNode constituent : sentenceConstituents)
151      sentenceConstituentFeatures.add(extractConstituentFeatures(jCas, constituent, sentence));
152
153    List<Predicate> predicates = JCasUtil.selectCovered(jCas, Predicate.class, sentence);
154    for (Predicate predicate : predicates) {
155      processPredicate(jCas, sentence, predicate, sentenceConstituents, sentenceConstituentFeatures);
156    }
157
158  }
159
160  private void processPredicate(
161      JCas jCas,
162      Sentence sentence,
163      Predicate predicate,
164      List<TreebankNode> constituents,
165      List<List<Feature>> constituentFeatures) throws AnalysisEngineProcessException {
166    TreebankNode predicateNode = TreebankNodeUtil.selectMatchingLeaf(
167        jCas,
168        predicate.getAnnotation());
169    List<Token> predicateTokens = JCasUtil.selectCovered(
170        jCas,
171        Token.class,
172        predicate.getAnnotation());
173    Token predicateToken = predicateTokens.get(0);
174
175    List<Feature> predicateFeatures = new ArrayList<Feature>(20);
176    predicateFeatures.addAll(predicateTokenExtractor.extract(jCas, predicateToken));
177    predicateFeatures.addAll(predicateNodeExtractor.extract(jCas, predicateNode));
178
179    for (int i = 0; i < constituents.size(); i++) {
180      TreebankNode constituent = constituents.get(i);
181
182      Instance<String> instance = new Instance<String>();
183
184      instance.addAll(predicateFeatures);
185      instance.addAll(constituentFeatures.get(i));
186      instance.addAll(relPosExtractor.extract(jCas, constituent, predicate.getAnnotation()));
187      instance.addAll(pathExtractor.extract(jCas, constituent, predicateNode));
188      instance.addAll(partialPathExtractor.extract(jCas, constituent, predicateNode));
189      instance.addAll(distanceExtractor.extract(jCas, constituent, predicateNode));
190
191      instance.setOutcome("NULL");
192
193      for (Argument arg : UIMAUtil.toList(predicate.getArguments(), Argument.class)) {
194        if (!(arg instanceof SemanticArgument))
195          continue;
196
197        SemanticArgument sarg = (SemanticArgument) arg;
198
199        if (sarg.getAnnotation() == constituent && !sarg.getLabel().equals("rel")) {
200          if (sarg.getFeature() != null) {
201            instance.setOutcome(sarg.getLabel() + "-" + sarg.getFeature());
202          } else {
203            instance.setOutcome(sarg.getLabel());
204          }
205          break;
206        }
207      }
208
209      if (this.isTraining()) {
210        this.dataWriter.write(instance);
211      } else {
212        String outcome = this.classifier.classify(instance.getFeatures());
213        if (!outcome.equals("NULL")) {
214          SemanticArgument arg = new SemanticArgument(jCas);
215          arg.setAnnotation(constituent);
216          arg.setBegin(constituent.getBegin());
217          arg.setEnd(constituent.getEnd());
218
219          String[] parts = outcome.split("-", 1);
220          arg.setLabel(parts[0]);
221          if (parts.length > 1)
222            arg.setFeature(parts[1]);
223
224          arg.addToIndexes();
225
226          List<Argument> args;
227          if (predicate.getArguments() != null)
228            args = UIMAUtil.toList(predicate.getArguments(), Argument.class);
229          else
230            args = new ArrayList<Argument>(1);
231          args.add(arg);
232          predicate.setArguments(UIMAUtil.toFSArray(jCas, args));
233        }
234      }
235    }
236
237  }
238
239  private List<Feature> extractConstituentFeatures(
240      JCas jCas,
241      TreebankNode constituent,
242      Sentence sentence) throws CleartkExtractorException {
243    List<Feature> features = new ArrayList<Feature>(20);
244    features.addAll(constituentExtractor.extract(jCas, constituent));
245    features.addAll(firstAndLastWordExtractor.extractWithin(jCas, constituent, sentence));
246
247    TreebankNode parent = constituent.getParent();
248    if (parent != null) {
249      features.addAll(parentExtractor.extract(jCas, parent));
250
251      int constituentPosition = 0;
252      while (parent.getChildren(constituentPosition) != constituent)
253        constituentPosition += 1;
254
255      if (constituentPosition > 0)
256        features.addAll(leftSiblingExtractor.extract(
257            jCas,
258            parent.getChildren(constituentPosition - 1)));
259
260      if (constituentPosition < parent.getChildren().size() - 1)
261        features.addAll(rightSiblingExtractor.extract(
262            jCas,
263            parent.getChildren(constituentPosition + 1)));
264    }
265
266    return features;
267  }
268
269  private void collectConstituents(TreebankNode top, List<TreebankNode> constituents) {
270    if (!(top instanceof TopTreebankNode))
271      constituents.add(top);
272
273    if (top.getChildren() == null)
274      return;
275
276    int numberOfChildren = top.getChildren().size();
277    for (int i = 0; i < numberOfChildren; i++) {
278      collectConstituents(top.getChildren(i), constituents);
279    }
280  }
