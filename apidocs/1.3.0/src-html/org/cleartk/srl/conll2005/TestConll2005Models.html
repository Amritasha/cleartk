024package org.cleartk.srl.conll2005;
025
026import java.io.File;
027
028import org.cleartk.classifier.jar.GenericJarClassifierFactory;
029import org.cleartk.srl.ArgumentClassifier;
030import org.cleartk.srl.ArgumentIdentifier;
031import org.cleartk.token.stem.snowball.DefaultSnowballStemmer;
032import org.uimafit.factory.AnalysisEngineFactory;
033import org.uimafit.pipeline.SimplePipeline;
034
035/**
036 * <br>
037 * Copyright (c) 2010, Regents of the University of Colorado <br>
038 * All rights reserved.
039 * 
040 * @author Philipp Wetzler
041 */
042public class TestConll2005Models {
043
044  private static final File conll2005File = new File("../../ClearTK-data/data/conll2005/devset");
045
046  // private static final File predicateIdentificationModel = new
047  // File("scratch/CoNLL2005/predicateIdentification/model.jar");
048  private static final File argumentIdentificationModel = new File(
049      "scratch/CoNLL2005/argumentIdentification/model.jar");
050
051  private static final File argumentClassificationModel = new File(
052      "scratch/CoNLL2005/argumentClassification/model.jar");
053
054  private static final File outputFile = new File("scratch/CoNLL2005/results/props-devset");
055
056  public static void main(String[] args) throws Exception {
057    outputFile.getParentFile().mkdirs();
058
059    SimplePipeline.runPipeline(
060        Conll2005GoldReader.getCollectionReader(conll2005File.toString()),
061        AnalysisEngineFactory.createPrimitiveDescription(
062            Conll2005GoldAnnotator.class,
063            Conll2005GoldAnnotator.PARAM_HAS_VERB_SENSES,
064            false),
065        DefaultSnowballStemmer.getDescription("English"),
066        // CleartkComponents.createCleartkAnnotator(
067        // PredicateAnnotator.class,
068        // predicateIdentificationModel.toString()),
069        AnalysisEngineFactory.createPrimitiveDescription(
070            ArgumentIdentifier.class,
071            GenericJarClassifierFactory.PARAM_CLASSIFIER_JAR_PATH,
072            argumentIdentificationModel.toString()),
073        AnalysisEngineFactory.createPrimitiveDescription(
074            ArgumentClassifier.class,
075            GenericJarClassifierFactory.PARAM_CLASSIFIER_JAR_PATH,
076            argumentClassificationModel.toString()),
077        AnalysisEngineFactory.createPrimitiveDescription(
078            Conll2005Writer.class,
079            Conll2005Writer.PARAM_OUTPUT_FILE,
080            outputFile.toString()));
081  }
082
083}
