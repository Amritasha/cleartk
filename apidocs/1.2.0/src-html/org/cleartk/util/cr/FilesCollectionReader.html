245      @Override
246      public void initialize(UimaContext context) throws ResourceInitializationException {
247        // raise an exception if the root file does not exist
248        if (!this.rootFile.exists()) {
249          String format = "file or directory %s does not exist";
250          String message = String.format(format, rootFile.getPath());
251          throw new ResourceInitializationException(new IOException(message));
252        }
253    
254        if (rootFile.isFile()) {
255          files = Arrays.asList(rootFile).iterator();
256          filesCount = 1;
257        } else {
258    
259          files = createFileIterator();
260          filesCount = countFiles(createFileIterator());
261        }
262      }
263    
264      protected Iterator<File> createFileIterator() throws ResourceInitializationException {
265        IOFileFilter fileFilter = TrueFileFilter.INSTANCE;
266    
267        if (suffixes != null) {
268          fileFilter = new AndFileFilter(fileFilter, new SuffixFileFilter(suffixes));
269        }
270    
271        if (patterns != null && patterns.length > 0) {
272    
273          IOFileFilter patternFilter = new RegexFileFilter(Pattern.compile(patterns[0]));
274          if (patterns.length > 1) {
275            for (int i = 1; i < patterns.length; i++) {
276              patternFilter = new OrFileFilter(patternFilter, new RegexFileFilter(patterns[i]));
277            }
278          }
279          fileFilter = new AndFileFilter(fileFilter, patternFilter);
280    
281        }
282    
283        if (nameFilesFileNames != null) {
284          List<String> fileNamesFromLists = new ArrayList<String>();
285          try {
286            for (String fileNamesList : nameFilesFileNames) {
287              fileNamesFromLists.addAll(Arrays.asList(FileUtil.loadListOfStrings(new File(fileNamesList))));
288            }
289            fileFilter = new AndFileFilter(fileFilter, new NameFileFilter(fileNamesFromLists));
290          } catch (IOException ioe) {
291            throw new ResourceInitializationException(ioe);
292          }
293        }
294    
295        if (fileNames != null) {
296          fileFilter = new AndFileFilter(fileFilter, new NameFileFilter(fileNames));
297        }
298    
299        IOFileFilter directoryFilter = TrueFileFilter.INSTANCE;
300    
301        if (ignoreSystemFiles) {
302          directoryFilter = new RegexFileFilter("^[^\\.].*$");
303          fileFilter = new AndFileFilter(fileFilter, new RegexFileFilter("^[^\\.].*$"));
304        }
305    
306        return org.apache.commons.io.FileUtils.iterateFiles(rootFile, fileFilter, directoryFilter);
307    
308      }
309    
310      public void getNext(JCas jCas) throws IOException, CollectionException {
311        if (!hasNext()) {
312          throw new RuntimeException("getNext(jCas) was called but hasNext() returns false");
313        }
314        // get a JCas object
315        JCas view;
316        try {
317          view = ViewCreatorAnnotator.createViewSafely(jCas, this.viewName);
318        } catch (AnalysisEngineProcessException e) {
319          throw new CollectionException(e);
320        }
321    
322        // set the document's text
323        String text = FileUtils.file2String(currentFile, this.encoding);
324        view.setSofaDataString(text, "text/plain");
325    
326        // set language if it was specified
327        if (this.language != null) {
328          view.setDocumentLanguage(this.language);
329        }
330    
331        // set the document URI
332        ViewURIUtil.setURI(jCas, currentFile.toURI());
333    
334        completed++;
335        currentFile = null;
336      }
337    
338      protected int countFiles(Iterator<File> tempFiles) {
339        int count = 0;
340        while (tempFiles.hasNext()) {
341          File file = tempFiles.next();
342          if (file.isFile())
343            count++;
344        }
345        return count;
346      }
347    
348      public Progress[] getProgress() {
349        Progress progress = new ProgressImpl(completed, filesCount, Progress.ENTITIES);
350        return new Progress[] { progress };
351      }
352    
353      public boolean hasNext() throws IOException, CollectionException {
354        if (currentFile != null) {
355          return true;
356        }
357        while (this.files.hasNext()) {
358          currentFile = files.next();
359          if (currentFile.isFile()) {
360            return true;
361          }
362        }
363        return false;
364      }
