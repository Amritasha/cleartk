024    package org.cleartk.util.cr;

026    import java.io.FileInputStream;
027    import java.io.IOException;
028    import java.util.Arrays;
029    
030    import org.apache.uima.UimaContext;
031    import org.apache.uima.cas.impl.XCASDeserializer;
032    import org.apache.uima.cas.impl.XCASSerializer;

034    import org.apache.uima.cas.impl.XmiCasSerializer;
035    import org.apache.uima.collection.CollectionException;
036    import org.apache.uima.jcas.JCas;
037    import org.apache.uima.resource.ResourceInitializationException;
038    import org.cleartk.util.CleartkInitializationException;
039    import org.uimafit.descriptor.ConfigurationParameter;
040    import org.uimafit.factory.ConfigurationParameterFactory;
041    import org.xml.sax.SAXException;
042

049     * @see XCASSerializer

051    
052    public class XReader extends FilesCollectionReader {
053    
054      public static final String XMI = "XMI";
055    
056      public static final String XCAS = "XCAS";
057    
058      public static final String PARAM_XML_SCHEME = ConfigurationParameterFactory
059          .createConfigurationParameterName(XReader.class, "xmlScheme");
060    
061      @ConfigurationParameter(defaultValue = "XMI", description = "specifies the UIMA XML serialization scheme that should be used. Valid values for this parameter are 'XMI' and 'XCAS'")
062      private String xmlScheme;
063    
064      @Override
065      public void initialize(UimaContext context) throws ResourceInitializationException {
066        super.initialize(context);
067    
068        if (!xmlScheme.equals(XMI) && !xmlScheme.equals(XCAS))
069          throw CleartkInitializationException.invalidParameterValueSelectFrom(
070              PARAM_XML_SCHEME,
071              Arrays.asList(XMI, XCAS),
072              xmlScheme);
073      }
074    
075      public void getNext(JCas jCas) throws IOException, CollectionException {
076        if (!hasNext()) {
077          throw new RuntimeException("getNext(jCas) was called but hasNext() returns false");
078        }
079    
080        FileInputStream inputStream = new FileInputStream(currentFile);
081    
082        try {
083          if (xmlScheme.equals(XMI))
084            XmiCasDeserializer.deserialize(inputStream, jCas.getCas());
085          else
086            XCASDeserializer.deserialize(inputStream, jCas.getCas());
087        } catch (SAXException e) {
088          throw new CollectionException(e);
089        } finally {
090          inputStream.close();
091        }
092    
093        completed++;
094        currentFile = null;
095      }
096    
097    }
