024    package org.cleartk.util.ae;
025    
026    import java.io.IOException;
027    import java.io.InputStreamReader;
028    import java.net.MalformedURLException;
029    import java.net.URI;
030    
031    import org.apache.uima.analysis_engine.AnalysisEngineDescription;
032    import org.apache.uima.analysis_engine.AnalysisEngineProcessException;
033    import org.apache.uima.cas.CAS;
034    import org.apache.uima.jcas.JCas;
035    import org.apache.uima.resource.ResourceInitializationException;
036    import org.cleartk.util.ViewURIUtil;
037    import org.uimafit.component.JCasAnnotator_ImplBase;
038    import org.uimafit.component.ViewCreatorAnnotator;
039    import org.uimafit.factory.AggregateBuilder;
040    import org.uimafit.factory.AnalysisEngineFactory;
041    
042    import com.google.common.io.CharStreams;
043    
044    /**
045     * <br>
046     * Copyright (c) 2012, Regents of the University of Colorado <br>
047     * All rights reserved.
048     * <p>
049     * 
050     * This Analysis Engine reads in the contents of the URI in the default sofa, and puts it in the CAS
051     * 
052     * @author Lee Becker
053     * 
054     */
055    public class UriToDocumentTextAnnotator extends JCasAnnotator_ImplBase {
056    
057      public static AnalysisEngineDescription getDescription() throws ResourceInitializationException {
058        return AnalysisEngineFactory.createPrimitiveDescription(UriToDocumentTextAnnotator.class);
059      }
060    
061      /**
062       * This description will read the contents into the specify view. If the view does not exist, it
063       * will make it as needed.
064       */
065      public static AnalysisEngineDescription getDescriptionForView(String targetViewName)
066          throws ResourceInitializationException {
067        AggregateBuilder builder = new AggregateBuilder();
068        builder.add(AnalysisEngineFactory.createPrimitiveDescription(
069            ViewCreatorAnnotator.class,
070            ViewCreatorAnnotator.PARAM_VIEW_NAME,
071            targetViewName));
072        builder.add(UriToDocumentTextAnnotator.getDescription(), CAS.NAME_DEFAULT_SOFA, targetViewName);
073        return builder.createAggregateDescription();
074      }
075    
076      @Override
077      public void process(JCas jCas) throws AnalysisEngineProcessException {
078    
079        URI uri = ViewURIUtil.getURI(jCas);
080        String content;
081    
082        try {
083          content = CharStreams.toString(new InputStreamReader(uri.toURL().openStream()));
084          jCas.setSofaDataString(content, "text/plain");
085        } catch (MalformedURLException e) {
086          throw new AnalysisEngineProcessException(e);
087        } catch (IOException e) {
088          throw new AnalysisEngineProcessException(e);
089        }
090      }
091    }
