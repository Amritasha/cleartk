205      @Override
206      public void initialize(UimaContext context) throws ResourceInitializationException {
207        try {
208          super.initialize(context);
209    
210          if ((outputDirectoryName == null && outputFileName == null)
211              || (outputDirectoryName != null && outputFileName != null)) {
212            throw CleartkInitializationException.notExactlyOneParameterSet(
213                PARAM_OUTPUT_DIRECTORY_NAME,
214                outputDirectoryName,
215                PARAM_OUTPUT_FILE_NAME,
216                outputFileName);
217          }
218    
219          if (outputDirectoryName != null) {
220            outputDirectory = new File(outputDirectoryName);
221            if (!this.outputDirectory.exists()) {
222              this.outputDirectory.mkdirs();
223            }
224          }
225    
226          if (outputFileName != null) {
227            outputFile = new File(outputFileName);
228            if (!outputFile.getParentFile().exists()) {
229              outputFile.getParentFile().mkdirs();
230            }
231            out = new PrintStream(outputFile);
232          }
233    
234          outputAnnotationClass = InitializableFactory.getClass(
235              outputAnnotationClassName,
236              Annotation.class);
237    
238          Class<? extends AnnotationWriter<ANNOTATION_TYPE>> annotationWriterClass = ReflectionUtil
239              .uncheckedCast(Class
240                  .forName(annotationWriterClassName)
241                  .asSubclass(AnnotationWriter.class));
242          annotationWriter = InitializableFactory.create(
243              context,
244              annotationWriterClassName,
245              annotationWriterClass);
246    
247          java.lang.reflect.Type annotationType = ReflectionUtil.getTypeArgument(
248              AnnotationWriter.class,
249              "ANNOTATION_TYPE",
250              this.annotationWriter);
251    
252          if (!ReflectionUtil.isAssignableFrom(annotationType, outputAnnotationClass)) {
253            throw CleartkInitializationException.incompatibleTypeParameterAndType(
254                this.annotationWriter,
255                "ANNOTATION_TYPE",
256                annotationType,
257                outputAnnotationClass);
258          }
259    
260          if (blockAnnotationClassName != null) {
261    
262            Class<? extends BlockWriter<BLOCK_TYPE>> blockWriterClass = ReflectionUtil
263                .uncheckedCast(Class.forName(blockWriterClassName).asSubclass(BlockWriter.class));
264            this.blockWriter = InitializableFactory.create(
265                context,
266                blockWriterClassName,
267                blockWriterClass);
268    
269            if (blockAnnotationClassName.equals("org.apache.uima.jcas.tcas.DocumentAnnotation")) {
270              blockOnDocument = true;
271            } else {
272              blockAnnotationClass = Class.forName(blockAnnotationClassName).asSubclass(
273                  Annotation.class);
274    
275              java.lang.reflect.Type blockType = ReflectionUtil.getTypeArgument(
276                  BlockWriter.class,
277                  "BLOCK_TYPE",
278                  this.blockWriter);
279    
280              if (!ReflectionUtil.isAssignableFrom(blockType, blockAnnotationClass)) {
281                throw CleartkInitializationException.incompatibleTypeParameterAndType(
282                    this.blockWriter,
283                    "BLOCK_TYPE",
284                    blockType,
285                    blockAnnotationClass);
286              }
287            }
288          }
289    
290          if (fileSuffix == null) {
291            fileSuffix = "";
292          } else if (!fileSuffix.startsWith(".")) {
293            fileSuffix = "." + fileSuffix;
294          }
295        } catch (Exception e) {
296          throw new ResourceInitializationException(e);
297        }
298    
299      }
300    
301      private void initializeTypes(JCas jCas) throws AnalysisEngineProcessException {
302        try {
303          outputAnnotationType = UIMAUtil.getCasType(jCas, outputAnnotationClass);
304          if (blockAnnotationClass != null) {
305            blockAnnotationType = UIMAUtil.getCasType(jCas, blockAnnotationClass);
306          }
307        } catch (Exception e) {
308          throw new AnalysisEngineProcessException(e);
309        }
310        typesInitialized = true;
311      }
312    
313      @SuppressWarnings("unchecked")
314      @Override
315      public void process(JCas jCas) throws AnalysisEngineProcessException {
316        if (!typesInitialized)
317          initializeTypes(jCas);
318    
319        try {
320          if (outputDirectory != null) {
321            String id = (new File(ViewURIUtil.getURI(jCas))).getName();
322            while (id.endsWith(".")) {
323              id = id.substring(0, id.length() - 1);
324            }
325            out = new PrintStream(new File(outputDirectory, id + fileSuffix));
326          }
327    
328          if (blockOnDocument) {
329            BLOCK_TYPE documentAnnotation = (BLOCK_TYPE) jCas.getDocumentAnnotationFs();
330            out.print(blockWriter.writeBlock(jCas, documentAnnotation));
331            FSIterator<Annotation> outputAnnotations = jCas
332                .getAnnotationIndex(outputAnnotationType)
333                .iterator();
334            while (outputAnnotations.hasNext()) {
335              ANNOTATION_TYPE outputAnnotation = (ANNOTATION_TYPE) outputAnnotations.next();
336              out.println(annotationWriter.writeAnnotation(jCas, outputAnnotation));
337            }
338          } else if (blockAnnotationType != null) {
339            FSIterator<Annotation> blocks = jCas.getAnnotationIndex(blockAnnotationType).iterator();
340            while (blocks.hasNext()) {
341              BLOCK_TYPE blockAnnotation = (BLOCK_TYPE) blocks.next();
342              out.print(blockWriter.writeBlock(jCas, blockAnnotation));
343              FSIterator<Annotation> outputAnnotations = jCas
344                  .getAnnotationIndex(outputAnnotationType)
345                  .subiterator(blockAnnotation);
346              while (outputAnnotations.hasNext()) {
347                ANNOTATION_TYPE outputAnnotation = (ANNOTATION_TYPE) outputAnnotations.next();
348                out.println(annotationWriter.writeAnnotation(jCas, outputAnnotation));
349              }
350            }
351          }
352    
353          else {
354            FSIterator<Annotation> outputAnnotations = jCas
355                .getAnnotationIndex(outputAnnotationType)
356                .iterator();
357            while (outputAnnotations.hasNext()) {
358              ANNOTATION_TYPE outputAnnotation = (ANNOTATION_TYPE) outputAnnotations.next();
359              out.println(annotationWriter.writeAnnotation(jCas, outputAnnotation));
360            }
361          }
362    
363          if (outputDirectory != null) {
364            out.flush();
365            out.close();
366          }
367        } catch (FileNotFoundException fnfe) {
368          throw new AnalysisEngineProcessException(fnfe);
369        }
370      }
371    
372      @Override
373      public void collectionProcessComplete() throws AnalysisEngineProcessException {
374        if (outputFile != null) {
375          out.flush();
376          out.close();
377        }
378        // TODO Auto-generated method stub
379        super.collectionProcessComplete();
380      }
