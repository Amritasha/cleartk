122      @Override
123      public void initialize(UimaContext context) throws ResourceInitializationException {
124        super.initialize(context);
125        try {
126          URL parserModelURL = this.parserModelFileName == null
127              ? ClearParserSRL.class.getResource(DEFAULT_MODEL_FILE_NAME)
128              : new File(this.parserModelFileName).toURI().toURL();
129          parser = ClearParserUtil.createSRLParser(parserModelURL.openStream());
130    
131        } catch (MalformedURLException e) {
132          throw new ResourceInitializationException(e);
133        } catch (IOException e) {
134          throw new ResourceInitializationException(e);
135        }
136      }

138      @Override
139      public void process(JCas jCas) throws AnalysisEngineProcessException {
140        for (Sentence sentence : JCasUtil.select(jCas, Sentence.class)) {
141          List<Token> tokens = JCasUtil.selectCovered(jCas, Token.class, sentence);
142          DepTree tree = new DepTree();
143    
144          // Build map between CAS dependency node and id for later creation of
145          // ClearParser dependency node/tree
146          Map<DependencyNode, Integer> depNodeToID = new HashMap<DependencyNode, Integer>();
147          int nodeId = 1;
148          for (DependencyNode depNode : JCasUtil.selectCovered(jCas, DependencyNode.class, sentence)) {
149            if (depNode instanceof TopDependencyNode) {
150              depNodeToID.put(depNode, 0);
151            } else {
152              depNodeToID.put(depNode, nodeId);
153              nodeId++;
154            }
155          }
156    
157          // Initialize Token / Sentence info for the ClearParser Semantic Role Labeler
158          for (int i = 0; i < tokens.size(); i++) {
159            Token token = tokens.get(i);
160    
161            // Determine HeadId
162            DepNode node = new DepNode();
163            DependencyNode casDepNode = JCasUtil.selectCovered(jCas, DependencyNode.class, token).get(0);
164            DependencyRelation headRelation = (DependencyRelation) casDepNode.getHeadRelations().get(0);
165            DependencyNode head = headRelation.getHead();
166            int headId = depNodeToID.get(head);
167    
168            // Populate Dependency Node / Tree information
169            node.id = i + 1;
170            node.form = token.getCoveredText();
171            node.pos = token.getPos();
172            node.lemma = token.getLemma();
173            node.setHead(headId, headRelation.getRelation(), 0);
174            tree.add(node);
175          }
176          tree.setPredicates(AbstractReader.LANG_EN);
177    
178          // Run the SRL
179          parser.parse(tree);
180    
181          // Convert ClearParser SRL output to CAS types
182          extractSRLInfo(jCas, tokens, tree);
183        }
184      }
185    
186      /**
187       * Converts the output from the ClearParser Semantic Role Labeler to the ClearTK Predicate and
188       * SemanticArgument Types.
189       * 
190       * @param jCas
191       * @param tokens
192       *          - In order list of tokens
193       * @param tree
194       *          - DepdendencyTree output by ClearParser SRLPredict
195       */
196      private void extractSRLInfo(JCas jCas, List<Token> tokens, DepTree tree) {
197        Map<Integer, Predicate> headIdToPredicate = new HashMap<Integer, Predicate>();
198        Map<Predicate, List<SemanticArgument>> predicateArguments = new HashMap<Predicate, List<SemanticArgument>>();
199    
200        // Start at node 1, since node 0 is considered the head of the sentence
201        for (int i = 1; i < tree.size(); i++) {
202          // Every ClearParser parserNode will contain an srlInfo field.
203          DepNode parserNode = tree.get(i);
204          Token token = tokens.get(i - 1);
205          if (parserNode.srlInfo == null) {
206            continue;
207          }
208    
209          if (parserNode.srlInfo.isPredicate()) {
210            int headId = i;
211            if (!headIdToPredicate.containsKey(headId)) {
212              // We have not encountered this predicate yet, so create it
213              Predicate pred = this.createPredicate(jCas, parserNode.srlInfo.rolesetId, token);
214              headIdToPredicate.put(headId, pred);
215            }
216          } else {
217            for (SRLHead head : parserNode.srlInfo.heads) {
218              Predicate predicate;
219    
220              // Determine which predicate this argument belongs to
221              if (!headIdToPredicate.containsKey(head.headId)) {
222                // The predicate hasn't been encountered, so create it
223                Token headToken = tokens.get(head.headId - 1);
224                predicate = this.createPredicate(jCas, parserNode.srlInfo.rolesetId, headToken);
225                headIdToPredicate.put(head.headId, predicate);
226              } else {
227                predicate = headIdToPredicate.get(head.headId);
228              }
229    
230              // Append this argument to the predicate's list of arguments
231              if (!predicateArguments.containsKey(predicate)) {
232                predicateArguments.put(predicate, new ArrayList<SemanticArgument>());
233              }
234              List<SemanticArgument> argumentList = predicateArguments.get(predicate);
235    
236              // Create the semantic argument and store for later link creation
237              SemanticArgument argument = createArgument(jCas, head, token);
238              argumentList.add(argument);
239            }
240          }
241        }
242    
243        // Store Arguments in Predicate
244        for (Map.Entry<Predicate, List<SemanticArgument>> entry : predicateArguments.entrySet()) {
245          Predicate predicate = entry.getKey();
246          List<SemanticArgument> arguments = entry.getValue();
247          predicate.setArguments(UIMAUtil.toFSArray(jCas, arguments));
248        }
249      }
250    
251      private Predicate createPredicate(JCas jCas, String rolesetId, Token token) {
252        Predicate pred = new Predicate(jCas, token.getBegin(), token.getEnd());
253        pred.setFrameSet(rolesetId);
254        pred.addToIndexes();
255        return pred;
256      }
257    
258      private SemanticArgument createArgument(JCas jCas, SRLHead head, Token token) {
259        SemanticArgument argument = new SemanticArgument(jCas, token.getBegin(), token.getEnd());
260        argument.setLabel(head.label);
261        argument.addToIndexes();
262        return argument;
263      }
