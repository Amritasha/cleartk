024    package org.cleartk.srl.propbank.util;
025    
026    import org.apache.uima.jcas.JCas;
027    import org.apache.uima.jcas.tcas.Annotation;
028    import org.cleartk.syntax.constituent.type.TopTreebankNode;
029    
030    /**
031     * <br>
032     * Copyright (c) 2007-2008, Regents of the University of Colorado <br>
033     * All rights reserved.
034     * 
035     * 
036     * 
037     * <p>
038     * A <em>PropbankRelation object</em> represents the relation (or a sub-relation) of one label of an
039     * entry in Propbank.
040     * </p>
041     * 
042     * @author Philipp Wetzler
043     */
044    public abstract class PropbankRelation {
045      /**
046       * Parses a relation taken form a Propbank entry and returns its representation as the appropriate
047       * <em>PropbankRelation</em> object.
048       * 
049       * @param s
050       *          the textual representation of a relation taken from <tt>prop.txt</tt>
051       * 
052       * @return a <em>PropbankRelation</em> object representing <b>s</b>
053       */
054      public static PropbankRelation fromString(String s) {
055        if (s.contains("*")) {
056          return PropbankCorefRelation.fromString(s);
057        } else if (s.contains(",")) {
058          return PropbankSplitRelation.fromString(s);
059        } else if (s.contains(":")) {
060          return PropbankNodeRelation.fromString(s);
061        } else {
062          return PropbankTerminalRelation.fromString(s);
063        }
064      }
065    
066      /**
067       * Convert to an appropriate ClearTK annotation and add it to <b>jCas</b> if necessary.
068       * 
069       * @param jCas
070       *          the view where the annotation will be added
071       * @param topNode
072       *          the top node annotation of the corresponding Treebank parse
073       * @return the corresponding annotation
074       */
075      public abstract Annotation convert(JCas jCas, TopTreebankNode topNode);
076    
077      /**
078       * Re-generate the text that this object was parsed from.
079       */
080      @Override
081      public abstract String toString();
082    }
