032    import java.io.IOException;

034    import java.net.URI;

036    import java.util.HashMap;
037    import java.util.List;
038    import java.util.Map;

040    import org.apache.uima.jcas.JCas;
041    import org.apache.uima.jcas.tcas.Annotation;
042    import org.cleartk.classifier.Feature;
043    import org.cleartk.classifier.Instance;
044    import org.cleartk.classifier.feature.extractor.CleartkExtractorException;
045    import org.cleartk.classifier.feature.extractor.simple.SimpleFeatureExtractor;
046    import org.cleartk.classifier.feature.transform.OneToOneTrainableExtractor_ImplBase;
047    import org.cleartk.classifier.feature.transform.TransformableFeature;

049    /**
050     * Scales features extracted by its subextractor to range 0-1, by scaling by the minimum and maximum
051     * values
052     * <p>
053     * 
054     * Copyright (c) 2012, Regents of the University of Colorado <br>
055     * All rights reserved.
056     * 
057     * @author Lee Becker
058     * 
059     */
060    public class MinMaxNormalizationExtractor<OUTCOME_T> extends
061        OneToOneTrainableExtractor_ImplBase<OUTCOME_T> implements SimpleFeatureExtractor {
062    
063      private SimpleFeatureExtractor subExtractor;
064    
065      private boolean isTrained;
066    
067      // This is read in after training for use in transformation
068      private Map<String, MinMaxPair> minMaxMap;
069    
070      public MinMaxNormalizationExtractor(String name) {
071        this(name, null);
072      }
073    
074      public MinMaxNormalizationExtractor(String name, SimpleFeatureExtractor subExtractor) {
075        super(name);
076        this.subExtractor = subExtractor;
077        this.isTrained = false;
078      }
079    
080      @Override
081      protected Feature transform(Feature feature) {
082        String featureName = feature.getName();
083        MinMaxPair stats = this.minMaxMap.get(featureName);
084        double value = ((Number) feature.getValue()).doubleValue();
085        return new Feature("MINMAX_NORMED_" + featureName, (value - stats.min)
086            / (stats.max - stats.min));
087      }
088    
089      @Override
090      public List<Feature> extract(JCas view, Annotation focusAnnotation)
091          throws CleartkExtractorException {
092    
093        List<Feature> extracted = this.subExtractor.extract(view, focusAnnotation);
094        List<Feature> result = new ArrayList<Feature>();
095        if (this.isTrained) {
096          // We have trained / loaded a MinMax model, so now fix up the values
097          for (Feature feature : extracted) {
098            result.add(this.transform(feature));
099          }
100        } else {
101          // We haven't trained this extractor yet, so just mark the existing features
102          // for future modification, by creating one mega container feature
103          result.add(new TransformableFeature(this.name, extracted));
104        }
105    
106        return result;
107      }
108    
109      @Override
110      public void train(Iterable<Instance<OUTCOME_T>> instances) {
111        Map<String, MinMaxRunningStat> featureStatsMap = new HashMap<String, MinMaxRunningStat>();
112    
113        // keep a running mean and standard deviation for all applicable features
114        for (Instance<OUTCOME_T> instance : instances) {
115          // Grab the matching zmus (zero mean, unit stddev) features from the set of all features in an
116          // instance
117          for (Feature feature : instance.getFeatures()) {
118            if (this.isTransformable(feature)) {
119              // ZMUS features contain a list of features, these are actually what get added
120              // to our document frequency map
121              for (Feature untransformedFeature : ((TransformableFeature) feature).getFeatures()) {
122                String featureName = untransformedFeature.getName();
123                Object featureValue = untransformedFeature.getValue();
124                if (featureValue instanceof Number) {
125                  MinMaxRunningStat stats;
126                  if (featureStatsMap.containsKey(featureName)) {
127                    stats = featureStatsMap.get(featureName);
128                  } else {
129                    stats = new MinMaxRunningStat();
130                    featureStatsMap.put(featureName, stats);
131                  }
132                  stats.add(((Number) featureValue).doubleValue());
133                } else {
134                  throw new IllegalArgumentException("Cannot normalize non-numeric feature values");
135                }
136              }
137            }
138          }
139        }
140    
141        this.minMaxMap = new HashMap<String, MinMaxPair>();
142        for (Map.Entry<String, MinMaxRunningStat> entry : featureStatsMap.entrySet()) {
143          MinMaxRunningStat stats = entry.getValue();
144          this.minMaxMap.put(entry.getKey(), new MinMaxPair(stats.min(), stats.max()));
145        }
146    
147        this.isTrained = true;
148      }
149    
150      @Override
151      public void save(URI zmusDataUri) throws IOException {
152        // Write out tab separated values: feature_name, mean, stddev
153        File out = new File(zmusDataUri);
154        BufferedWriter writer = null;
155        writer = new BufferedWriter(new FileWriter(out));
156    
157        for (Map.Entry<String, MinMaxPair> entry : this.minMaxMap.entrySet()) {
158          MinMaxPair pair = entry.getValue();
159          writer.append(String.format("%s\t%f\t%f\n", entry.getKey(), pair.min, pair.max));
160        }
161        writer.close();
162      }
163    
164      @Override
165      public void load(URI zmusDataUri) throws IOException {
166        // Reads in tab separated values (feature name, min, max)
167        File in = new File(zmusDataUri);
168        BufferedReader reader = null;
169        this.minMaxMap = new HashMap<String, MinMaxPair>();
170        reader = new BufferedReader(new FileReader(in));
171        String line = null;
172        while ((line = reader.readLine()) != null) {
173          String[] featureMeanStddev = line.split("\\t");
174          this.minMaxMap.put(
175              featureMeanStddev[0],
176              new MinMaxPair(
177                  Double.parseDouble(featureMeanStddev[1]),
178                  Double.parseDouble(featureMeanStddev[2])));
179        }
180        reader.close();
181    
182        this.isTrained = true;
183      }
184    
185      private static class MinMaxPair {
186    
187        public MinMaxPair(double min, double max) {
188          this.min = min;
189          this.max = max;
190        }
191    
192        public double min;
193    
194        public double max;
195      }
196    
197      public static class MinMaxRunningStat implements Serializable {
198    
199        /**
200         * 
201         */
202        private static final long serialVersionUID = 1L;
203    
204        public MinMaxRunningStat() {
205          this.clear();
206        }
207    
208        public void add(double x) {
209          this.n++;
210    
211          if (x < min) {
212            this.min = x;
213          }
214    
215          if (x > max) {
216            this.max = x;
217          }
218        }
219    
220        public void clear() {
221          this.n = 0;
222          this.min = Double.MAX_VALUE;
223          this.max = Double.MIN_VALUE;
224        }
225    
226        public int getNumSamples() {
227          return this.n;
228        }
229    
230        public double min() {
231          return this.min;
232        }
233    
234        public double max() {
235          return this.max;
236        }
237    
238        private double min;
239    
240        private double max;
241    
242        private int n;
243    
244      }
