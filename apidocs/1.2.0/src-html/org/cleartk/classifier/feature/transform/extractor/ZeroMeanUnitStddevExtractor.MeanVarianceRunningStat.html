051    /**
052     * Scales features produced by its subextractor to have mean=0, stddev=1 for a given feature
053     * <p>
054     * 
055     * Copyright (c) 2012, Regents of the University of Colorado <br>
056     * All rights reserved.
057     * 
058     * @author Lee Becker
059     * 
060     */
061    public class ZeroMeanUnitStddevExtractor<OUTCOME_T> extends
062        OneToOneTrainableExtractor_ImplBase<OUTCOME_T> implements SimpleFeatureExtractor {
063    
064      private SimpleFeatureExtractor subExtractor;
065    
066      private boolean isTrained;
067    
068      // This is read in after training for use in transformation
069      private Map<String, MeanStddevTuple> meanStddevMap;
070    
071      public ZeroMeanUnitStddevExtractor(String name) {
072        this(name, null);
073      }
074    
075      public ZeroMeanUnitStddevExtractor(String name, SimpleFeatureExtractor subExtractor) {
076        super(name);
077        this.subExtractor = subExtractor;
078        this.isTrained = false;
079      }
080    
081      @Override
082      protected Feature transform(Feature feature) {
083        String featureName = feature.getName();
084        MeanStddevTuple stats = this.meanStddevMap.get(featureName);
085        double value = ((Number) feature.getValue()).doubleValue();
086        return new Feature("ZMUS_" + featureName, (value - stats.mean) / stats.stddev);
087      }
088    
089      @Override
090      public List<Feature> extract(JCas view, Annotation focusAnnotation)
091          throws CleartkExtractorException {
092    
093        List<Feature> extracted = this.subExtractor.extract(view, focusAnnotation);
094        List<Feature> result = new ArrayList<Feature>();
095        if (this.isTrained) {
096          // We have trained / loaded a ZMUS model, so now fix up the values
097          for (Feature feature : extracted) {
098            result.add(this.transform(feature));
099          }
100        } else {
101          // We haven't trained this extractor yet, so just mark the existing features
102          // for future modification, by creating one mega container feature
103          result.add(new TransformableFeature(this.name, extracted));
104        }
105    
106        return result;
107      }
108    
109      @Override
110      public void train(Iterable<Instance<OUTCOME_T>> instances) {
111        Map<String, MeanVarianceRunningStat> featureStatsMap = new HashMap<String, MeanVarianceRunningStat>();
112    
113        // keep a running mean and standard deviation for all applicable features
114        for (Instance<OUTCOME_T> instance : instances) {
115          // Grab the matching zmus (zero mean, unit stddev) features from the set of all features in an
116          // instance
117          for (Feature feature : instance.getFeatures()) {
118            if (this.isTransformable(feature)) {
119              // ZMUS features contain a list of features, these are actually what get added
120              // to our document frequency map
121              for (Feature untransformedFeature : ((TransformableFeature) feature).getFeatures()) {
122                String featureName = untransformedFeature.getName();
123                Object featureValue = untransformedFeature.getValue();
124                if (featureValue instanceof Number) {
125                  MeanVarianceRunningStat stats;
126                  if (featureStatsMap.containsKey(featureName)) {
127                    stats = featureStatsMap.get(featureName);
128                  } else {
129                    stats = new MeanVarianceRunningStat();
130                    featureStatsMap.put(featureName, stats);
131                  }
132                  stats.add(((Number) featureValue).doubleValue());
133                } else {
134                  throw new IllegalArgumentException("Cannot normalize non-numeric feature values");
135                }
136              }
137            }
138          }
139        }
140    
141        this.meanStddevMap = new HashMap<String, MeanStddevTuple>();
142        for (Map.Entry<String, MeanVarianceRunningStat> entry : featureStatsMap.entrySet()) {
143          MeanVarianceRunningStat stats = entry.getValue();
144          this.meanStddevMap.put(entry.getKey(), new MeanStddevTuple(stats.mean(), stats.stddev()));
145        }
146    
147        this.isTrained = true;
148      }
149    
150      @Override
151      public void save(URI zmusDataUri) throws IOException {
152        // Write out tab separated values: feature_name, mean, stddev
153        File out = new File(zmusDataUri);
154        BufferedWriter writer = null;
155        writer = new BufferedWriter(new FileWriter(out));
156    
157        for (Map.Entry<String, MeanStddevTuple> entry : this.meanStddevMap.entrySet()) {
158          MeanStddevTuple tuple = entry.getValue();
159          writer.append(String.format("%s\t%f\t%f\n", entry.getKey(), tuple.mean, tuple.stddev));
160        }
161        writer.close();
162      }
163    
164      @Override
165      public void load(URI zmusDataUri) throws IOException {
166        // Reads in tab separated values (feature name, mean, stddev)
167        File in = new File(zmusDataUri);
168        BufferedReader reader = null;
169        this.meanStddevMap = new HashMap<String, MeanStddevTuple>();
170        reader = new BufferedReader(new FileReader(in));
171        String line = null;
172        while ((line = reader.readLine()) != null) {
173          String[] featureMeanStddev = line.split("\\t");
174          this.meanStddevMap.put(
175              featureMeanStddev[0],
176              new MeanStddevTuple(
177                  Double.parseDouble(featureMeanStddev[1]),
178                  Double.parseDouble(featureMeanStddev[2])));
179        }
180    
181        reader.close();
182    
183        this.isTrained = true;
184      }
185    
186      public static class MeanStddevTuple {
187    
188        public MeanStddevTuple(double mean, double stddev) {
189          this.mean = mean;
190          this.stddev = stddev;
191        }
192    
193        public double mean;
194    
195        public double stddev;
196      }
197    
198      public static class MeanVarianceRunningStat implements Serializable {
199    
200        private static final long serialVersionUID = 1L;
201    
202        public MeanVarianceRunningStat() {
203          this.clear();
204        }
205    
206        public void init(int n, double mean, double variance) {
207          this.numSamples = n;
208          this.meanNew = mean;
209          this.varNew = variance;
210        }
211    
212        public void add(double x) {
213          numSamples++;
214    
215          if (this.numSamples == 1) {
216            meanOld = meanNew = x;
217            varOld = 0.0;
218          } else {
219            meanNew = meanOld + (x - meanOld) / numSamples;
220            varNew = varOld + (x - meanOld) * (x - meanNew);
221    
222            // set up for next iteration
223            meanOld = meanNew;
224            varOld = varNew;
225          }
226        }
227    
228        public void clear() {
229          this.numSamples = 0;
230        }
231    
232        public int getNumSamples() {
233          return this.numSamples;
234        }
235    
236        public double mean() {
237          return (this.numSamples > 0) ? meanNew : 0.0;
238        }
239    
240        public double variance() {
241          return (this.numSamples > 1) ? varNew / (this.numSamples) : 0.0;
242        }
243    
244        public double stddev() {
245          return Math.sqrt(this.variance());
246        }
247    
248        public double variancePop() {
249          return (this.numSamples > 1) ? varNew / (this.numSamples - 1) : 0.0;
250        }
251    
252        public double stddevPop() {
253          return Math.sqrt(this.variancePop());
254        }
255    
256        private void writeObject(ObjectOutputStream out) throws IOException {
257          out.defaultWriteObject();
258          out.writeInt(numSamples);
259          out.writeDouble(meanNew);
260          out.writeDouble(varNew);
261    
262        }
263    
264        private void readObject(ObjectInputStream in) throws IOException, ClassNotFoundException {
265          in.defaultReadObject();
266          numSamples = in.readInt();
267          meanOld = meanNew = in.readDouble();
268          varOld = varNew = in.readDouble();
269        }
270    
271        private int numSamples;
272    
273        private double meanOld;
274    
275        private double meanNew;
276    
277        private double varOld;
278    
279        private double varNew;
280      }
