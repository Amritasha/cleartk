025    package org.cleartk.classifier.feature.transform;
026    
027    import java.io.BufferedOutputStream;
028    import java.io.File;
029    import java.io.FileNotFoundException;
030    import java.io.FileOutputStream;
031    import java.io.IOException;
032    import java.io.ObjectOutputStream;
033    
034    import org.cleartk.classifier.CleartkProcessingException;
035    import org.cleartk.classifier.DataWriter;
036    import org.cleartk.classifier.Instance;
037    
038    /**
039     * 
040     * <br>
041     * Copyright (c) 2011-2012, Regents of the University of Colorado <br>
042     * All rights reserved.
043     * 
044     * This training data (instance) consumer produces intermediate storage of instances for later use
045     * in feature transformation (i.e. normalization, clustering, etc...)
046     * 
047     * @author Lee Becker
048     * 
049     * @param <OUTCOME_T>
050     *          - The outcome type of the instances it writes out
051     */
052    public class InstanceDataWriter<OUTCOME_T> implements DataWriter<OUTCOME_T> {
053    
054      ObjectOutputStream objout;
055    
056      public static String INSTANCES_OUTPUT_FILENAME = "training-data.instances";
057    
058      public InstanceDataWriter(File outputDirectory) {
059        if (!outputDirectory.exists()) {
060          outputDirectory.mkdirs();
061        }
062    
063        // Initialize Object Serializer
064        File outputFile = new File(outputDirectory, INSTANCES_OUTPUT_FILENAME);
065        try {
066          this.objout = new ObjectOutputStream(new BufferedOutputStream(
067              new FileOutputStream(outputFile)));
068        } catch (FileNotFoundException e) {
069          e.printStackTrace();
070        } catch (IOException e) {
071          e.printStackTrace();
072        }
073      }
074    
075      @Override
076      public void write(Instance<OUTCOME_T> instance) throws CleartkProcessingException {
077        try {
078          this.objout.writeObject(instance);
079        } catch (IOException e) {
080          throw new CleartkProcessingException("", "Unable to write Instance", e);
081        }
082      }
083    
084      @Override
085      public void finish() throws CleartkProcessingException {
086        try {
087          // We need to add a "null" instance terminator to gracefully handle the iteration while
088          // reading in serialized objects from file
089          InstanceStream.Terminator<OUTCOME_T> terminator = new InstanceStream.Terminator<OUTCOME_T>();
090          this.objout.writeObject(terminator);
091    
092          this.objout.close();
093        } catch (IOException e) {
094          throw new CleartkProcessingException("", "Unable to write terminal instance", e);
095        }
096    
097      }
098    
099    }
