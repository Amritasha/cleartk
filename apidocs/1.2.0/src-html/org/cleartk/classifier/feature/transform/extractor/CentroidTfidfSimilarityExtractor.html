057    public class CentroidTfidfSimilarityExtractor<OUTCOME_T> extends TfidfExtractor<OUTCOME_T> {
058    
059      private Map<String, Double> centroidMap;
060    
061      private SimilarityFunction simFunction;
062    
063      private static String docFreqFileSuffix = "_tfidf-centroid-extractor_idfmap.dat";
064    
065      private static String centroidMapFileSuffix = "_tfidf-centroid-extractor_centroidmap.dat";
066    
067      public static URI getDocumentFrequencyDataURI(String name, URI baseURI)
068          throws MalformedURLException, URISyntaxException {
069        return new URL(baseURI.toURL(), name + docFreqFileSuffix).toURI();
070      }
071    
072      public static URI getCentroidDataURI(String name, URI baseURI) throws MalformedURLException,
073          URISyntaxException {
074        return new URL(baseURI.toURL(), name + centroidMapFileSuffix).toURI();
075      }
076    
077      public CentroidTfidfSimilarityExtractor(String name) {
078        super(name);
079      }
080    
081      public CentroidTfidfSimilarityExtractor(String name, SimpleFeatureExtractor extractor) {
082        super(name);
083        this.subExtractor = extractor;
084        this.isTrained = false;
085        this.idfMap = new IDFMap();
086      }
087    
088      @Override
089      public Instance<OUTCOME_T> transform(Instance<OUTCOME_T> instance) {
090        List<Feature> features = new ArrayList<Feature>();
091        List<Feature> featuresToTransform = new ArrayList<Feature>();
092        for (Feature feature : instance.getFeatures()) {
093          if (this.isTransformable(feature)) {
094            // Store off features for later similarity computation
095            featuresToTransform.addAll(((TransformableFeature) feature).getFeatures());
096          } else {
097            // pass through non-transformable features
098            features.add(feature);
099          }
100        }
101    
102        // Create centroid similarity feature
103        Map<String, Double> featureMap = this.featuresToFeatureMap(featuresToTransform);
104        features.add(new Feature(this.name, new Double(this.simFunction.distance(
105            featureMap,
106            centroidMap))));
107    
108        return new Instance<OUTCOME_T>(instance.getOutcome(), features);
109      }
110    
111      public Map<String, Double> featuresToFeatureMap(List<Feature> features) {
112        Map<String, Double> featureMap = new HashMap<String, Double>();
113        for (Feature feature : features) {
114          String termName = feature.getName();
115          int tf = (Integer) feature.getValue();
116          featureMap.put(termName, tf * this.idfMap.getIDF(termName));
117        }
118        return featureMap;
119      }
120    
121      @Override
122      public List<Feature> extract(JCas view, Annotation focusAnnotation)
123          throws CleartkExtractorException {
124    
125        List<Feature> extracted = this.subExtractor.extract(view, focusAnnotation);
126        List<Feature> result = new ArrayList<Feature>();
127        if (this.isTrained) {
128          // We have trained / loaded a centroid tf*idf model, so now compute
129          // a cosine similarity for the extracted values
130          Map<String, Double> extractedFeatureMap = this.featuresToFeatureMap(extracted);
131          result.add(new Feature(name, this.simFunction.distance(extractedFeatureMap, centroidMap)));
132    
133        } else {
134          // We haven't trained this extractor yet, so just mark the existing features
135          // for future modification, by creating one mega container feature
136          result.add(new TransformableFeature(this.name, extracted));
137        }
138    
139        return result;
140      }
141    
142      protected Map<String, Double> computeCentroid(Iterable<Instance<OUTCOME_T>> instances, IDFMap idfs) {
143    
144        // Now compute centroid of all applicable terms (features) in all instances
145        int numDocuments = idfs.getTotalDocumentCount();
146        Map<String, Double> newCentroidMap = new HashMap<String, Double>();
147        for (Instance<OUTCOME_T> instance : instances) {
148    
149          // Grab the matching tf*idf features from the set of all features in an instance
150          for (Feature feature : instance.getFeatures()) {
151            if (this.isTransformable(feature)) {
152              // tf*idf features contain a list of features, these are actually what get added
153              // to our document frequency map
154              for (Feature untransformedFeature : ((TransformableFeature) feature).getFeatures()) {
155                String termName = untransformedFeature.getName();
156                int tf = (Integer) untransformedFeature.getValue();
157                double tfidf = tf * idfs.getIDF(termName);
158                double sumTfidf = (newCentroidMap.containsKey(termName))
159                    ? sumTfidf = newCentroidMap.get(termName)
160                    : 0.0;
161                newCentroidMap.put(termName, sumTfidf + tfidf);
162              }
163            }
164          }
165    
166          for (Map.Entry<String, Double> entry : newCentroidMap.entrySet()) {
167            double mean = entry.getValue() / numDocuments;
168            newCentroidMap.put(entry.getKey(), mean);
169          }
170        }
171        return newCentroidMap;
172      }
173    
174      @Override
175      public void train(Iterable<Instance<OUTCOME_T>> instances) {
176        this.idfMap = this.createIdfMap(instances);
177        this.centroidMap = this.computeCentroid(instances, this.idfMap);
178        this.isTrained = true;
179        this.simFunction = new FixedCosineSimilarity(this.centroidMap);
180      }
181    
182      @Override
183      public void save(URI baseURI) throws IOException {
184        URI documentFreqDataURI;
185        URI centroidDataURI;
186        try {
187          documentFreqDataURI = getDocumentFrequencyDataURI(this.name, baseURI);
188          centroidDataURI = getCentroidDataURI(this.name, baseURI);
189        } catch (URISyntaxException e) {
190          throw new IOException(e);
191        }
192    
193        // Save off idfMap document frequency data
194        this.idfMap.save(documentFreqDataURI);
195    
196        // Save off centroid map data
197        File out = new File(centroidDataURI);
198        BufferedWriter writer = null;
199        writer = new BufferedWriter(new FileWriter(out));
200    
201        for (Map.Entry<String, Double> entry : this.centroidMap.entrySet()) {
202          writer.append(String.format("%s\t%f\n", entry.getKey(), entry.getValue()));
203        }
204        writer.close();
205      }
206    
207      public void load(URI baseURI) throws IOException {
208        URI documentFreqDataURI;
209        URI centroidDataURI;
210        try {
211          documentFreqDataURI = getDocumentFrequencyDataURI(this.name, baseURI);
212          centroidDataURI = getCentroidDataURI(this.name, baseURI);
213        } catch (URISyntaxException e) {
214          throw new IOException(e);
215        }
216    
217        // Load document frequency data
218        this.idfMap.load(documentFreqDataURI);
219    
220        // Reads in centroid map as tab separated values (feature name, mean-tfidf value)
221        File in = new File(centroidDataURI);
222        BufferedReader reader = null;
223        this.centroidMap = new HashMap<String, Double>();
224        reader = new BufferedReader(new FileReader(in));
225        String line = null;
226        while ((line = reader.readLine()) != null) {
227          String[] featureMeanTfidf = line.split("\\t");
228          double tfidf = Double.parseDouble(featureMeanTfidf[1]);
229          this.centroidMap.put(featureMeanTfidf[0], tfidf);
230        }
231        reader.close();
232    
233        this.simFunction = new FixedCosineSimilarity(this.centroidMap);
234        this.isTrained = true;
235      }
