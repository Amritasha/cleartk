039    import org.cleartk.classifier.jar.DefaultDataWriterFactory;

048     * <p>

050     * @author Martin Toepfer
051     * @deprecated Use {@link DefaultDataWriterFactory} with {@link GrmmDataWriter}.
052     */
053    @Deprecated
054    public class DefaultGrmmDataWriterFactory extends
055        SequenceDataWriterFactory_ImplBase<List<NameNumber>, String[], String[]> {
056    
057      public static final String PARAM_COMPRESS = ConfigurationParameterFactory.createConfigurationParameterName(
058          DefaultGrmmDataWriterFactory.class,
059          "compress");
060    
061      @ConfigurationParameter(
062          description = "indicates whether the FeaturesEncoder should compress the feature names",
063          defaultValue = "false")
064      private boolean compress;
065    
066      public static final String PARAM_SORT = ConfigurationParameterFactory.createConfigurationParameterName(
067          DefaultGrmmDataWriterFactory.class,
068          "sort");
069    
070      @ConfigurationParameter(
071          description = "indicates that the FeaturesEncoder should write the feature names in sorted order",
072          defaultValue = "false")
073      private boolean sort;
074    
075      @Override
076      public void initialize(UimaContext uimaContext) throws ResourceInitializationException {
077        super.initialize(uimaContext);
078      }
079    
080      public SequenceDataWriter<String[]> createDataWriter() throws IOException {
081        GrmmDataWriter dataWriter = new GrmmDataWriter(outputDirectory);
082    
083        if (!this.setEncodersFromFileSystem(dataWriter)) {
084          NameNumberFeaturesEncoder fe = new NameNumberFeaturesEncoder(compress, sort);
085          fe.addEncoder(new NumberEncoder());
086          fe.addEncoder(new BooleanEncoder());
087          fe.addEncoder(new StringEncoder());
088          dataWriter.setFeaturesEncoder(fe);
089          dataWriter.setOutcomeEncoder(new StringArrayToStringArrayEncoder());
090        }
091    
092        return dataWriter;
093      }
094    }
