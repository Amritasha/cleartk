024    package org.cleartk.classifier.chunking;
025    
026    import java.util.HashMap;
027    import java.util.List;
028    import java.util.Map;
029    
030    import org.apache.uima.cas.Feature;
031    import org.apache.uima.jcas.JCas;
032    import org.apache.uima.jcas.tcas.Annotation;
033    import org.uimafit.util.JCasUtil;
034    
035    /**
036     * Chunking with Inside-Outside labels
037     * 
038     * <br>
039     * Copyright (c) 2012, Regents of the University of Colorado <br>
040     * All rights reserved.
041     * 
042     * @author Steven Bethard
043     */
044    public class IOChunking<SUB_CHUNK_TYPE extends Annotation, CHUNK_TYPE extends Annotation> extends
045        Chunking_ImplBase<SUB_CHUNK_TYPE, CHUNK_TYPE> {
046    
047      public IOChunking(
048          Class<? extends SUB_CHUNK_TYPE> subChunkClass,
049          Class<? extends CHUNK_TYPE> chunkClass) {
050        super(subChunkClass, chunkClass, null);
051      }
052    
053      public IOChunking(
054          Class<? extends SUB_CHUNK_TYPE> subChunkClass,
055          Class<? extends CHUNK_TYPE> chunkClass,
056          String featureName) {
057        super(subChunkClass, chunkClass, featureName);
058      }
059    
060      @Override
061      protected Map<SUB_CHUNK_TYPE, String> getSubChunkToOutcomeMap(
062          JCas jCas,
063          List<SUB_CHUNK_TYPE> chunkComponents,
064          List<CHUNK_TYPE> chunks) {
065        Feature feature = this.getFeature(jCas);
066        Map<SUB_CHUNK_TYPE, String> subChunkToOutcome = new HashMap<SUB_CHUNK_TYPE, String>();
067        for (CHUNK_TYPE chunk : chunks) {
068          String suffix = this.getOutcomeSuffix(chunk, feature);
069          for (SUB_CHUNK_TYPE chunkComponent : JCasUtil.selectCovered(this.subChunkClass, chunk)) {
070            subChunkToOutcome.put(chunkComponent, "I" + suffix);
071          }
072        }
073        return subChunkToOutcome;
074      }
075    
076      @Override
077      protected boolean isEndOfChunk(
078          char currPrefix,
079          String currLabel,
080          char nextPrefix,
081          String nextLabel) {
082        return nextPrefix == 'O' || !nextLabel.equals(currLabel);
083      }
084    
085    }
