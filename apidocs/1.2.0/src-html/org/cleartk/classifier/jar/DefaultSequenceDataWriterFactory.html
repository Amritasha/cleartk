024    package org.cleartk.classifier.jar;
025    
026    import java.io.IOException;
027    
028    import org.cleartk.classifier.SequenceDataWriter;
029    import org.cleartk.classifier.SequenceDataWriterFactory;
030    import org.uimafit.descriptor.ConfigurationParameter;
031    import org.uimafit.factory.ConfigurationParameterFactory;
032    
033    /**
034     * A {@link SequenceDataWriterFactory} that creates a data writer from the class given by
035     * {@link #PARAM_DATA_WRITER_CLASS_NAME} and the directory given by {@link #PARAM_OUTPUT_DIRECTORY}.
036     * 
037     * <br>
038     * Copyright (c) 2012, Regents of the University of Colorado <br>
039     * All rights reserved.
040     * 
041     * @author Steven Bethard
042     */
043    public class DefaultSequenceDataWriterFactory<OUTCOME_TYPE> extends
044        GenericDataWriterFactory<OUTCOME_TYPE> implements SequenceDataWriterFactory<OUTCOME_TYPE> {
045    
046      public static final String PARAM_DATA_WRITER_CLASS_NAME = ConfigurationParameterFactory.createConfigurationParameterName(
047          DefaultSequenceDataWriterFactory.class,
048          "dataWriterClassName");
049    
050      @ConfigurationParameter(
051          mandatory = true,
052          description = "provides the full name of the data writer class to be used.")
053      private String dataWriterClassName;
054    
055      @Override
056      @SuppressWarnings({ "unchecked" })
057      public SequenceDataWriter<OUTCOME_TYPE> createDataWriter() throws IOException {
058        return this.createDataWriter(this.dataWriterClassName, SequenceDataWriter.class);
059      }
060    
061    }
