024    package org.cleartk.classifier.jar;
025    
026    import java.io.File;
027    import java.io.FileNotFoundException;
028    import java.util.List;
029    
030    import org.cleartk.classifier.CleartkProcessingException;
031    import org.cleartk.classifier.Instance;
032    import org.cleartk.classifier.SequenceClassifier;
033    import org.cleartk.classifier.SequenceDataWriter;
034    import org.cleartk.classifier.encoder.features.FeaturesEncoder;
035    import org.cleartk.classifier.encoder.outcome.OutcomeEncoder;
036    
037    /**
038     * Superclass for {@link SequenceDataWriter} implementations that write training data to a file
039     * using {@link FeaturesEncoder}s and {@link OutcomeEncoder}s.
040     * 
041     * <br>
042     * Copyright (c) 2011, Regents of the University of Colorado <br>
043     * All rights reserved.
044     * 
045     * @author Steven Bethard
046     * @author Philip Ogren
047     */
048    public abstract class SequenceDataWriter_ImplBase<CLASSIFIER_BUILDER_TYPE extends EncodingJarClassifierBuilder<? extends SequenceClassifier<OUTCOME_TYPE>, ENCODED_FEATURES_TYPE, OUTCOME_TYPE, ENCODED_OUTCOME_TYPE>, ENCODED_FEATURES_TYPE, OUTCOME_TYPE, ENCODED_OUTCOME_TYPE>
049        extends
050        EncodingDirectoryDataWriter<CLASSIFIER_BUILDER_TYPE, SequenceClassifier<OUTCOME_TYPE>, ENCODED_FEATURES_TYPE, OUTCOME_TYPE, ENCODED_OUTCOME_TYPE>
051        implements SequenceDataWriter<OUTCOME_TYPE> {
052    
053      public SequenceDataWriter_ImplBase(File outputDirectory) throws FileNotFoundException {
054        super(outputDirectory);
055      }
056    
057      public void write(List<Instance<OUTCOME_TYPE>> instances) throws CleartkProcessingException {
058        for (Instance<OUTCOME_TYPE> instance : instances) {
059          writeEncoded(
060              this.classifierBuilder.getFeaturesEncoder().encodeAll(instance.getFeatures()),
061              this.classifierBuilder.getOutcomeEncoder().encode(instance.getOutcome()));
062        }
063        this.writeEndSequence();
064      }
065    
066      /**
067       * Write the encoded features and encoded outcome to the output directory.
068       * 
069       * @param features
070       *          The encoded features.
071       * @param outcome
072       *          The encoded outcome.
073       */
074      protected abstract void writeEncoded(ENCODED_FEATURES_TYPE features, ENCODED_OUTCOME_TYPE outcome)
075          throws CleartkProcessingException;
076    
077      /**
078       * Write the marker for the end of a sequence. This will be called at the end of each
079       * {@link #write}, after all instances have been written by {@link #writeEncoded}.
080       */
081      protected abstract void writeEndSequence();
082    
083    }
