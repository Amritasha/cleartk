024    package org.cleartk.classifier.libsvm;
025    
026    import java.util.ArrayList;
027    import java.util.Collections;
028    import java.util.List;
029    
030    import org.cleartk.classifier.CleartkProcessingException;
031    import org.cleartk.classifier.Feature;
032    import org.cleartk.classifier.ScoredOutcome;
033    import org.cleartk.classifier.encoder.features.FeaturesEncoder;
034    import org.cleartk.classifier.encoder.outcome.OutcomeEncoder;
035    import org.cleartk.classifier.jar.Classifier_ImplBase;
036    import org.cleartk.classifier.util.featurevector.FeatureVector;
037    
038    import com.google.common.base.Function;
039    import com.google.common.collect.Lists;
040    import com.google.common.collect.Ordering;
041    
042    /**
043     * <br>
044     * Copyright (c) 2007-2008, Regents of the University of Colorado <br>
045     * All rights reserved.
046     * 
047     * 
048     * @author Philipp Wetzler
049     * 
050     */
051    public abstract class LIBSVMClassifier<OUTCOME_TYPE, ENCODED_OUTCOME_TYPE> extends
052        Classifier_ImplBase<FeatureVector, OUTCOME_TYPE, ENCODED_OUTCOME_TYPE> {
053    
054      public LIBSVMClassifier(
055          FeaturesEncoder<FeatureVector> featuresEncoder,
056          OutcomeEncoder<OUTCOME_TYPE, ENCODED_OUTCOME_TYPE> outcomeEncoder,
057          libsvm.svm_model model) {
058        super(featuresEncoder, outcomeEncoder);
059        this.model = model;
060      }
061    
062      public OUTCOME_TYPE classify(List<Feature> features) throws CleartkProcessingException {
063        FeatureVector featureVector = this.featuresEncoder.encodeAll(features);
064    
065        ENCODED_OUTCOME_TYPE encodedOutcome = decodePrediction(libsvm.svm.svm_predict(
066            this.model,
067            convertToLIBSVM(featureVector)));
068    
069        return outcomeEncoder.decode(encodedOutcome);
070      }
071    
072      @Override
073      public List<ScoredOutcome<OUTCOME_TYPE>> score(List<Feature> features, int maxResults)
074          throws CleartkProcessingException {
075        FeatureVector featureVector = this.featuresEncoder.encodeAll(features);
076        double[] decisionValues = new double[this.model.nr_class];
077        libsvm.svm.svm_predict_probability(this.model, convertToLIBSVM(featureVector), decisionValues);
078        List<ScoredOutcome<OUTCOME_TYPE>> results = Lists.newArrayList();
079        for (int i = 0; i < this.model.nr_class; ++i) {
080          int intLabel = this.model.label[i];
081          OUTCOME_TYPE outcome = this.outcomeEncoder.decode(this.decodePrediction(intLabel));
082          results.add(new ScoredOutcome<OUTCOME_TYPE>(outcome, decisionValues[i]));
083        }
084        Collections.sort(results, this.orderByScore);
085        return results.subList(0, Math.min(results.size(), maxResults));
086      }
087    
088      protected static libsvm.svm_node[] convertToLIBSVM(FeatureVector featureVector) {
089        List<libsvm.svm_node> nodes = new ArrayList<libsvm.svm_node>();
090    
091        for (FeatureVector.Entry entry : featureVector) {
092          libsvm.svm_node node = new libsvm.svm_node();
093          node.index = entry.index;
094          node.value = entry.value;
095          nodes.add(node);
096        }
097    
098        return nodes.toArray(new libsvm.svm_node[nodes.size()]);
099      }
100    
101      protected abstract ENCODED_OUTCOME_TYPE decodePrediction(double prediction);
102    
103      protected libsvm.svm_model model;
104    
105      protected Ordering<ScoredOutcome<OUTCOME_TYPE>> orderByScore = Ordering.natural().reverse().onResultOf(
106          new Function<ScoredOutcome<OUTCOME_TYPE>, Double>() {
107            @Override
108            public Double apply(ScoredOutcome<OUTCOME_TYPE> scoredOutcome) {
109              return scoredOutcome.getScore();
110            }
111          });
112    
113    }
