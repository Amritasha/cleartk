050    public class SVMlightStringOutcomeClassifierBuilder extends
051        ClassifierBuilder_ImplBase<SVMlightStringOutcomeClassifier, FeatureVector, String, Integer> {
052    
053      public File getTrainingDataFile(File dir) {
054        return new File(dir, "training-data-allfalse.svmlight");
055      }
056    
057      public File getTrainingDataFile(File dir, int label) {
058        return new File(dir, String.format("training-data-%d.svmlight", label));
059      }
060    
061      public void trainClassifier(File dir, String... args) throws Exception {
062        SVMlightBooleanOutcomeClassifierBuilder builder = new SVMlightBooleanOutcomeClassifierBuilder();
063        for (File file : dir.listFiles()) {
064          if (file.getName().matches("training-data-\\d+.svmlight")) {
065            builder.trainClassifier(dir, file, args);
066    
067            Sigmoid s = FitSigmoid.fit(new File(file.toString() + ".model"), file);
068    
069            ObjectOutput o = new ObjectOutputStream(new FileOutputStream(new File(file.toString()
070                + ".sigmoid")));
071            o.writeObject(s);
072            o.close();
073          }
074        }
075      }
076    
077      @Override
078      protected void packageClassifier(File dir, JarOutputStream modelStream) throws IOException {
079        super.packageClassifier(dir, modelStream);
080    
081        int label = 1;
082        while (true) {
083          File modelFile = new File(dir, String.format("training-data-%d.svmlight.model", label));
084          File sigmoidFile = new File(dir, String.format("training-data-%d.svmlight.sigmoid", label));
085          if (!modelFile.exists()) {
086            break;
087          }
088    
089          String modelName = String.format("model-%d.svmlight", label);
090          String sigmoidName = String.format("model-%d.sigmoid", label);
091          JarStreams.putNextJarEntry(modelStream, modelName, modelFile);
092          JarStreams.putNextJarEntry(modelStream, sigmoidName, sigmoidFile);
093    
094          label += 1;
095        }
096      }
097    
098      private TreeMap<Integer, SVMlightModel> models;
099    
100      private TreeMap<Integer, Sigmoid> sigmoids;
101    
102      @Override
103      protected void unpackageClassifier(JarInputStream modelStream) throws IOException {
104        super.unpackageClassifier(modelStream);
105        this.models = new TreeMap<Integer, SVMlightModel>();
106        this.sigmoids = new TreeMap<Integer, Sigmoid>();
107    
108        int label = 1;
109        SVMlightModel model;
110        while ((model = getNextModel(modelStream, label)) != null) {
111          this.models.put(label, model);
112    
113          JarStreams.getNextJarEntry(modelStream, String.format("model-%d.sigmoid", label));
114          try {
115            this.sigmoids.put(label, (Sigmoid) new ObjectInputStream(modelStream).readObject());
116          } catch (ClassNotFoundException e) {
117            throw new IOException(e);
118          }
119    
120          label += 1;
121        }
122    
123        if (this.models.isEmpty()) {
124          throw new IOException(String.format("no models found in %s", modelStream));
125        }
126      }
127    
128      @Override
129      protected SVMlightStringOutcomeClassifier newClassifier() {
130        return new SVMlightStringOutcomeClassifier(
131            this.featuresEncoder,
132            this.outcomeEncoder,
133            this.models,
134            this.sigmoids);
135      }
136    
137      private static SVMlightModel getNextModel(JarInputStream modelStream, int label)
138          throws IOException {
139        // look for a next entry or return null if there isn't one
140        JarEntry entry = modelStream.getNextJarEntry();
141        if (entry == null) {
142          return null;
143        }
144    
145        // make sure the name was the model we expected
146        String expectedName = String.format("model-%d.svmlight", label);
147        if (!entry.getName().equals(expectedName)) {
148          throw new IOException(String.format(
149              "expected next jar entry to be %s, found %s",
150              expectedName,
151              entry.getName()));
152        }
153    
154        // read the model from the jar stream
155        return SVMlightModel.fromInputStream(modelStream);
156      }
