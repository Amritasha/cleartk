055      public static IDFMap read(File file) throws IOException {
056        Connection connection;
057        PreparedStatement stat;
058        ResultSet rs;
059        IDFMap idfMap = new IDFMap();
060    
061        try {
062          connection = openDB(file);
063    
064          stat = connection.prepareStatement("select value from globals " + "where name = ?");
065          try {
066            stat.setString(1, "totalDocumentCount");
067            rs = stat.executeQuery();
068            try {
069              rs.next();
070              idfMap.totalDocumentCount = rs.getInt(1);
071            } finally {
072              rs.close();
073            }
074          } finally {
075            stat.close();
076          }
077    
078          stat = connection.prepareStatement("select key, count from documentcounts ");
079          try {
080            rs = stat.executeQuery();
081            try {
082              while (rs.next()) {
083                String key = rs.getString("key");
084                int count = rs.getInt("count");
085                idfMap.documentCounts.put(key, count);
086              }
087            } finally {
088              rs.close();
089            }
090          } finally {
091            stat.close();
092          }
093    
094          connection.close();
095    
096          return idfMap;
097        } catch (SQLException e) {
098          throw new IOException(e.toString());
099        }
100      }
101    
102      public IDFMap() {
103        this.totalDocumentCount = 0;
104        this.documentCounts = new HashMap<String, Integer>();
105      }
106    
107      public void consume(Counts counts) {
108        totalDocumentCount += 1;
109    
110        for (Object value : counts.getValues()) {
111          if (counts.getCount(value) == 0)
112            continue;
113    
114          String valueString = value.toString();
115          if (documentCounts.containsKey(valueString))
116            documentCounts.put(valueString, documentCounts.get(valueString) + 1);
117          else
118            documentCounts.put(valueString, 1);
119        }
120      }
121    
122      public Double getIDF(Object key) {
123        String keyString = key.toString();
124        Double documentCount = documentCounts.containsKey(keyString)
125            ? documentCounts.get(keyString)
126            : 0.0;
127    
128        return Math.log((totalDocumentCount + 1) / (documentCount + 1));
129      }
130    
131      public int getTotalDocumentCount() {
132        return totalDocumentCount;
133      }
134    
135      public int getDocumentCount(Object key) {
136        String keyString = key.toString();
137        return documentCounts.containsKey(keyString) ? documentCounts.get(keyString) : 0;
138      }
139    
140      public Iterator<?> getValues() {
141        return documentCounts.keySet().iterator();
142      }
143    
144      public void write(File file) throws IOException {
145        try {
146          File tempFile = new File(file.toString() + "_temp");
147          if (tempFile.exists())
148            tempFile.delete();
149    
150          Connection connection = createDB(tempFile);
151    
152          PreparedStatement stat = connection.prepareStatement("insert into globals "
153              + "(name, value) " + "values " + "(?, ?)");
154          try {
155            stat.setString(1, "totalDocumentCount");
156            stat.setInt(2, totalDocumentCount);
157            stat.execute();
158          } finally {
159            stat.close();
160          }
161    
162          for (String key : documentCounts.keySet()) {
163            stat = connection.prepareStatement("insert into documentcounts " + "(key, count) "
164                + "values " + "(?, ?)");
165            try {
166              stat.setString(1, key);
167              stat.setInt(2, documentCounts.get(key));
168              stat.execute();
169            } finally {
170              stat.close();
171            }
172          }
173    
174          connection.commit();
175          connection.close();
176    
177          if (file.exists())
178            file.delete();
179    
180          tempFile.renameTo(file);
181        } catch (SQLException e) {
182          throw new IOException(e.toString());
183        }
184      }
185    
186      private static Connection createDB(File file) throws SQLException {
187        Connection connection = null;
188        PreparedStatement stat;
189        try {
190          Class.forName("org.sqlite.JDBC");
191          connection = DriverManager.getConnection("jdbc:sqlite:" + file.toString());
192          connection.setAutoCommit(false);
193    
194          stat = connection.prepareStatement("create table documentcounts " + "(key text, "
195              + "count integer)");
196          try {
197            stat.execute();
198          } finally {
199            stat.close();
200          }
201    
202          stat = connection.prepareStatement("create table globals " + "(name text, " + "value)");
203          try {
204            stat.execute();
205          } finally {
206            stat.close();
207          }
208    
209          connection.commit();
210          return connection;
211        } catch (SQLException e) {
212          if (connection != null)
213            try {
214              connection.rollback();
215            } catch (SQLException e1) {
216            }
217          throw e;
218        } catch (ClassNotFoundException e) {
219          throw new TypeNotPresentException("org.sqlite.JDBC", e);
220        }
221      }
222    
223      private static Connection openDB(File file) throws SQLException {
224        Connection connection = null;
225        try {
226          Class.forName("org.sqlite.JDBC");
227          connection = DriverManager.getConnection("jdbc:sqlite:" + file.toString());
228          connection.setAutoCommit(false);
229    
230          return connection;
231        } catch (SQLException e) {
232          if (connection != null)
233            try {
234              connection.rollback();
235            } catch (SQLException e1) {
236            }
237          throw e;
238        } catch (ClassNotFoundException e) {
239          throw new TypeNotPresentException("org.sqlite.JDBC", e);
240        }
241      }
