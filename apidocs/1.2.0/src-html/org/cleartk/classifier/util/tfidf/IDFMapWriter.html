024    package org.cleartk.classifier.util.tfidf;

039    import org.cleartk.classifier.feature.Counts;
040    import org.cleartk.classifier.feature.FeatureCollection;

050    @Deprecated
051    public class IDFMapWriter<OUTCOME_TYPE> implements DataWriter<OUTCOME_TYPE> {
052    
053      public IDFMapWriter(File outputDirectory) {
054        this.outputDirectory = outputDirectory;
055      }
056    
057      private File getIDFMapFile(String identifier) {
058        if (identifier == null)
059          identifier = "default";
060        else
061          identifier = identifier.toLowerCase();
062    
063        if (identifier.equals("default")) {
064          return new File(outputDirectory, "idfmap");
065        } else {
066          return new File(outputDirectory, "idfmap_" + identifier);
067        }
068      }
069    
070      private IDFMap getIDFMap(String identifier) throws IOException {
071        if (identifier == null)
072          identifier = "default";
073        else
074          identifier = identifier.toLowerCase();
075    
076        if (idfMaps.containsKey(identifier)) {
077          return idfMaps.get(identifier);
078        } else {
079          File idfMapFile = getIDFMapFile(identifier);
080    
081          IDFMap idfMap;
082          if (idfMapFile.exists()) {
083            logger.info(String.format(
084                "load existing idf map \"%s\" from %s",
085                identifier,
086                idfMapFile.toString()));
087            idfMap = IDFMap.read(idfMapFile);
088          } else {
089            logger.info(String.format("initialize new idf map \"%s\"", identifier));
090            idfMap = new IDFMap();
091          }
092    
093          idfMaps.put(identifier, idfMap);
094          return idfMap;
095        }
096      }
097    
098      public boolean isTraining() {
099        return false;
100      }
101    
102      public void write(Instance<OUTCOME_TYPE> instance) throws CleartkProcessingException {
103        consumeFeatures(instance.getFeatures());
104      }
105    
106      private void consumeFeatures(Collection<Feature> features) throws CleartkProcessingException {
107        for (Feature feature : features) {
108          if (feature.getValue() instanceof Counts) {
109            Counts counts = (Counts) feature.getValue();
110    
111            IDFMap idfMap;
112            try {
113              idfMap = getIDFMap(counts.getIdentifier());
114            } catch (IOException e) {
115              throw new CleartkProcessingException(e);
116            }
117            idfMap.consume(counts);
118          } else if (feature.getValue() instanceof FeatureCollection) {
119            FeatureCollection fc = (FeatureCollection) feature.getValue();
120            consumeFeatures(fc.getFeatures());
121          }
122        }
123      }
124    
125      public void finish() throws CleartkProcessingException {
126        List<IOException> exceptions = new ArrayList<IOException>();
127    
128        for (String identifier : idfMaps.keySet()) {
129          IDFMap idfMap = idfMaps.get(identifier);
130          File idfMapFile = getIDFMapFile(identifier);
131          try {
132            logger.info(String.format(
133                "write idf map \"%s\" to %s.  size=" + idfMap.getTotalDocumentCount(),
134                identifier,
135                idfMapFile.toString()));
136            idfMap.write(idfMapFile);
137          } catch (IOException e1) {
138            exceptions.add(e1);
139            continue;
140          }
141        }
142    
143        if (exceptions.size() > 0) {
144          if (exceptions.size() == 1) {
145            throw new CleartkProcessingException(exceptions.get(0));
146          } else {
147            throw CleartkProcessingException.multipleExceptions(exceptions);
148          }
149        }
150      }
151    
152      private Map<String, IDFMap> idfMaps = new HashMap<String, IDFMap>();
153    
154      private File outputDirectory;
155    
156      private Logger logger = Logger.getLogger(this.getClass().getName());
157    
158    }
