049    /**
050     * A Tree Kernel SVM light classifier implementation. All features named with the prefix "TK_"
051     * treated as Tree Kernels.
052     * 
053     * <br>
054     * Copyright (c) 2007-2008, Regents of the University of Colorado <br>
055     * All rights reserved.
056     * 
057     * @author Daryl Lonnon
058     * @version 0.2.1
059     * @see OVATKSVMlightClassifier
060     * @deprecated Use {@link TKSVMlightBooleanOutcomeClassifier} instead.
061     */
062    @Deprecated
063    public class TKSVMlightClassifier extends Classifier_ImplBase<TreeFeatureVector, Boolean, Boolean> {
064    
065      static Logger logger = UIMAFramework.getLogger(TKSVMlightClassifier.class);
066    
067      File mFile;
068    
069      /**
070       * Constructor
071       * 
072       * @param featuresEncoder
073       *          The features encoder used by this classifier.
074       * @param outcomeEncoder
075       *          The outcome encoder used by this classifier.
076       * @param modelFile
077       *          The file for the model used by this classifier.
078       */
079      public TKSVMlightClassifier(
080          FeaturesEncoder<TreeFeatureVector> featuresEncoder,
081          OutcomeEncoder<Boolean, Boolean> outcomeEncoder,
082          File modelFile) {
083        super(featuresEncoder, outcomeEncoder);
084        mFile = modelFile;
085      }
086    
087      /**
088       * Classify a features list.
089       * 
090       * @param features
091       *          The feature list to classify.
092       * @return A Boolean of whether the features match this classification.
093       */
094      public Boolean classify(List<Feature> features) throws CleartkProcessingException {
095    
096        ScoredOutcome<Boolean> s = score(features);
097        return s.getOutcome();
098      }
099    
100      /**
101       * Score a list of features against the model.
102       * 
103       * @param features
104       *          The features to classify
105       * @param maxResults
106       *          The maximum number of results to return in the list (at most 2).
107       * @return A list of scored outcomes ordered by likelihood.
108       */
109      @Override
110      public List<ScoredOutcome<Boolean>> score(List<Feature> features, int maxResults)
111          throws CleartkProcessingException {
112    
113        List<ScoredOutcome<Boolean>> resultList = new ArrayList<ScoredOutcome<Boolean>>();
114        if (maxResults > 0)
115          resultList.add(this.score(features));
116        if (maxResults > 1) {
117          ScoredOutcome<Boolean> v1 = resultList.get(0);
118          ScoredOutcome<Boolean> v2 = new ScoredOutcome<Boolean>(!v1.getOutcome(), 1 - v1.getScore());
119          resultList.add(v2);
120        }
121        return resultList;
122      }
123    
124      private ScoredOutcome<Boolean> score(List<Feature> features) throws CleartkProcessingException {
125        TreeFeatureVector featureVector = featuresEncoder.encodeAll(features);
126        double prediction = tkSvmLightPredict(mFile, featureVector);
127    
128        // I got this from the svmlight classifier, shouldn't this be over 0.0 (doesn't it range from
129        // -1.0 to 1.0
130        boolean encodedResult = (prediction > 0.5);
131    
132        if (encodedResult) {
133          return new ScoredOutcome<Boolean>(true, prediction);
134        } else {
135          return new ScoredOutcome<Boolean>(false, 1 - prediction);
136        }
137      }
138    
139      /**
140       * Predict which side of the line a feature vector resides upon for a particular svm model.
141       * 
142       * @param mFile
143       *          The model file to predict against.
144       * @param featureVector
145       *          The feature vector to predict for.
146       * @return A double that represents which side of the line the feature vector resides (negative
147       *         below, positive above).
148       */
149      // Moving this into public space so it can be used by one versus all thing.
150      public static double tkSvmLightPredict(File mFile, TreeFeatureVector featureVector)
151          throws CleartkProcessingException {
152        double prediction;
153    
154        try {
155          File cFile = File.createTempFile("tkclassify", ".txt");
156          File oFile = File.createTempFile("tkoutput", ".out");
157          BufferedWriter out = new BufferedWriter(new FileWriter(cFile));
158          out.write("0");
159          out.write(TKSVMlightDataWriter.createString(featureVector));
160          out.close();
161    
162          String executable = "tk_svm_classify";
163          String[] command = new String[4];
164          command[0] = executable;
165          command[1] = cFile.getPath();
166          command[2] = mFile.getPath();
167          command[3] = oFile.getPath();
168    
169          logger.log(Level.FINE, "classifying with tree kernel svmlight using the following command: "
170              + Joiner.on(" ").join(command));
171          Process process = Runtime.getRuntime().exec(command);
172          process.getOutputStream().close();
173          output(process.getInputStream(), System.out);
174          output(process.getErrorStream(), System.err);
175          process.waitFor();
176    
177          BufferedReader in = new BufferedReader(new FileReader(oFile));
178          prediction = Double.parseDouble(in.readLine());
179          in.close();
180        } catch (IOException e) {
181          throw new CleartkProcessingException(e);
182        } catch (InterruptedException e) {
183          throw new CleartkProcessingException(e);
184        }
185        return prediction;
186      }
187    
188      private static void output(InputStream input, PrintStream output) throws IOException {
189        byte[] buffer = new byte[128];
190        int count = input.read(buffer);
191        while (count != -1) {
192          output.write(buffer, 0, count);
193          count = input.read(buffer);
194        }
195      }
