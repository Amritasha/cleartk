024    package org.cleartk.classifier.tksvmlight;
025    
026    import java.io.File;
027    import java.util.ArrayList;
028    import java.util.Collections;
029    import java.util.List;
030    import java.util.Map;
031    
032    import org.cleartk.classifier.CleartkProcessingException;
033    import org.cleartk.classifier.Feature;
034    import org.cleartk.classifier.ScoredOutcome;
035    import org.cleartk.classifier.encoder.features.FeaturesEncoder;
036    import org.cleartk.classifier.encoder.outcome.OutcomeEncoder;
037    import org.cleartk.classifier.jar.Classifier_ImplBase;
038    
039    /**
040     * A One versus All Tree Kernel SVM light classifier implementation. All features named with the
041     * prefix "TK_" treated as Tree Kernels.
042     * 
043     * <br>
044     * Copyright (c) 2007-2008, Regents of the University of Colorado <br>
045     * All rights reserved.
046     * 
047     * @author Daryl Lonnon
048     * @see TKSVMlightBooleanOutcomeClassifier
049     */
050    public class TKSVMlightStringOutcomeClassifier extends
051        Classifier_ImplBase<TreeFeatureVector, String, Integer> {
052    
053      Map<Integer, File> models;
054    
055      /**
056       * Constructor
057       * 
058       * @param featuresEncoder
059       *          The features encoder used by this classifier.
060       * @param outcomeEncoder
061       *          The outcome encoder used by this classifier.
062       * @param models
063       *          The files for the models used by this classifier.
064       */
065      public TKSVMlightStringOutcomeClassifier(
066          FeaturesEncoder<TreeFeatureVector> featuresEncoder,
067          OutcomeEncoder<String, Integer> outcomeEncoder,
068          Map<Integer, File> models) {
069        super(featuresEncoder, outcomeEncoder);
070        this.models = models;
071      }
072    
073      /**
074       * Classify a features list.
075       * 
076       * @param features
077       *          The feature list to classify.
078       * @return A String of the most likely classification.
079       */
080      public String classify(List<Feature> features) throws CleartkProcessingException {
081        TreeFeatureVector featureVector = this.featuresEncoder.encodeAll(features);
082    
083        int maxScoredIndex = 0;
084        double maxScore = 0;
085        boolean first = true;
086        for (int i : models.keySet()) {
087          double score = score(featureVector, i);
088          if (first || score > maxScore) {
089            first = false;
090            maxScore = score;
091            maxScoredIndex = i;
092          }
093        }
094    
095        return outcomeEncoder.decode(maxScoredIndex);
096      }
097    
098      /**
099       * Score a list of features against the various models used by the One Verse All SVM classifier.
100       * 
101       * @param features
102       *          The features to classify
103       * @param maxResults
104       *          The maximum number of results to return in the list.
105       * @return A list of scored outcomes ordered by likelihood.
106       */
107      @Override
108      public List<ScoredOutcome<String>> score(List<Feature> features, int maxResults)
109          throws CleartkProcessingException {
110        TreeFeatureVector featureVector = this.featuresEncoder.encodeAll(features);
111    
112        List<ScoredOutcome<String>> results = new ArrayList<ScoredOutcome<String>>();
113        for (int i : models.keySet()) {
114          double score = score(featureVector, i);
115          String name = outcomeEncoder.decode(i);
116    
117          results.add(new ScoredOutcome<String>(name, score));
118        }
119        Collections.sort(results);
120    
121        return results.subList(0, Math.min(maxResults, results.size()));
122      }
123    
124      private double score(TreeFeatureVector featureVector, int i) throws CleartkProcessingException {
125        // TD: Use the modelsFile hashmap to get the models, write out the line. Classify the line
126        // versus the model
127        // and get the prediction.
128        File mFile = models.get(i);
129        return TKSVMlightBooleanOutcomeClassifier.tkSvmLightPredict(mFile, featureVector);
130      }
131    }
