039    import org.cleartk.classifier.encoder.features.NumberEncoder;

053     */
054    @Deprecated
055    public class OVATKSVMlightDataWriter extends
056        DataWriter_ImplBase<OVATKSVMlightClassifierBuilder, TreeFeatureVector, String, Integer> {
057    
058      /**
059       * Constructor for the One verse All Tree Kernel SVMlight data writer.
060       */
061      public OVATKSVMlightDataWriter(File outputDirectory) throws IOException {
062        super(outputDirectory);
063        TreeFeatureVectorFeaturesEncoder myFeaturesEncoder = new TreeFeatureVectorFeaturesEncoder();
064        myFeaturesEncoder.addEncoder(new NumberEncoder());
065        myFeaturesEncoder.addEncoder(new BooleanEncoder());
066        myFeaturesEncoder.addEncoder(new StringEncoder());
067        this.setFeaturesEncoder(myFeaturesEncoder);
068        this.setOutcomeEncoder(new StringToIntegerOutcomeEncoder());
069    
070        // aliases to make it easy to remember what the "main" file is being used for
071        allFalseFile = this.trainingDataFile;
072        allFalseWriter = this.trainingDataWriter;
073    
074        // create the output writers
075        trainingDataWriters = new TreeMap<Integer, PrintWriter>();
076      }
077    
078      /**
079       * Write the feature vector into the appropriate files (ultimately, it's one line in each of the
080       * model file.)
081       * 
082       * @param features
083       *          The feature vector to be written.
084       * @param outcome
085       *          The index of the model file which this feature vector corresponds to. If null this
086       *          feature vector has no known "outcome".
087       */
088      @Override
089      public void writeEncoded(TreeFeatureVector features, Integer outcome)
090          throws CleartkProcessingException {
091        if (outcome != null && !trainingDataWriters.containsKey(outcome)) {
092          try {
093            addClass(outcome);
094          } catch (IOException e) {
095            throw new CleartkProcessingException(e);
096          }
097        }
098    
099        StringBuffer featureString = new StringBuffer();
100        featureString.append(TKSVMlightDataWriter.createString(features));
101    
102        StringBuffer output = new StringBuffer();
103        if (outcome == null)
104          output.append("0");
105        else
106          output.append("-1");
107        output.append(featureString);
108        allFalseWriter.println(output);
109    
110        for (int i : trainingDataWriters.keySet()) {
111          output = new StringBuffer();
112          if (outcome == null)
113            output.append("0");
114          else if (outcome == i)
115            output.append("+1");
116          else
117            output.append("-1");
118    
119          output.append(featureString);
120          trainingDataWriters.get(i).println(output);
121        }
122      }
123    
124      /**
125       * Callback called after the last feature vector has been written. Flushes the various streams and
126       * deletes the all false file.
127       */
128      @Override
129      public void finish() throws CleartkProcessingException {
130        // close and remove all-false file
131        allFalseWriter.close();
132        allFalseFile.delete();
133    
134        // flush and close all training data writers
135        for (PrintWriter pw : trainingDataWriters.values()) {
136          pw.flush();
137          pw.close();
138        }
139    
140        // finish in the superclass
141        super.finish();
142      }
143    
144      @Override
145      protected OVATKSVMlightClassifierBuilder newClassifierBuilder() {
146        return new OVATKSVMlightClassifierBuilder();
147      }
148    
149      private void addClass(int label) throws IOException {
150        File newTDFile = this.classifierBuilder.getTrainingDataFile(this.outputDirectory, label);
151        newTDFile.delete();
152    
153        allFalseWriter.flush();
154        copyFile(allFalseFile, newTDFile);
155        trainingDataWriters.put(label, new PrintWriter(new BufferedWriter(new FileWriter(
156            newTDFile,
157            true))));
158      }
159    
160      private void copyFile(File source, File target) throws IOException {
161        // Create channel on the source
162        FileInputStream srcStream = new FileInputStream(source);
163        FileChannel srcChannel = srcStream.getChannel();
164    
165        // Create channel on the destination
166        FileOutputStream dstStream = new FileOutputStream(target);
167        FileChannel dstChannel = dstStream.getChannel();
168    
169        // Copy file contents from source to destination
170        dstChannel.transferFrom(srcChannel, 0, srcChannel.size());
171    
172        // Close the channels
173        srcStream.close();
174        srcChannel.close();
175        dstStream.close();
176        dstChannel.close();
177      }
178    
179      private File allFalseFile;
180    
181      private PrintWriter allFalseWriter;
182    
183      private Map<Integer, PrintWriter> trainingDataWriters;
184    }
