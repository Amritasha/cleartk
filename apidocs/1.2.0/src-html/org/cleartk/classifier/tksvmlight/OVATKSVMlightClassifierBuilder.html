024    package org.cleartk.classifier.tksvmlight;
025    
026    import java.io.BufferedReader;
027    import java.io.BufferedWriter;
028    import java.io.File;
029    import java.io.FileWriter;
030    import java.io.IOException;
031    import java.io.InputStreamReader;
032    import java.util.TreeMap;
033    import java.util.jar.JarEntry;
034    import java.util.jar.JarInputStream;
035    import java.util.jar.JarOutputStream;
036    
037    import org.cleartk.classifier.jar.ClassifierBuilder_ImplBase;
038    import org.cleartk.classifier.jar.JarStreams;
039    
040    /**
041     * A class that provided interfaces to package and unpackage a OVASVMTKClassifier into a jar file.
042     * 
043     * <br>
044     * Copyright (c) 2007-2008, Regents of the University of Colorado <br>
045     * All rights reserved.
046     * 
047     * @author Daryl Lonnon
048     * @version 0.2.1
049     * @deprecated Use {@link TKSVMlightStringOutcomeClassifierBuilder} instead.
050     */
051    @Deprecated
052    public class OVATKSVMlightClassifierBuilder extends
053        ClassifierBuilder_ImplBase<OVATKSVMlightClassifier, TreeFeatureVector, String, Integer> {
054    
055      /**
056       * @param dir
057       *          The directory that contains the all false training data.
058       */
059      public File getTrainingDataFile(File dir) {
060        return new File(dir, "training-data-allfalse.svmlight");
061      }
062    
063      /**
064       * Get the training data file for a specific label.
065       * 
066       * @param dir
067       *          The directory to keep the training data.
068       * @param label
069       *          The integer number that represents an index for a particular label.
070       * @return The file of the training data to be used to train the model.
071       */
072      public File getTrainingDataFile(File dir, int label) {
073        return new File(dir, String.format("training-data-%d.svmlight", label));
074      }
075    
076      /**
077       * Train the OVASVMTK classifier.
078       * 
079       * @param dir
080       *          The directory where the training data has been written.
081       * @param args
082       *          The arguments to be used by the tk_svm_classify command. Note: -t 5 is used to specify
083       *          the use of Tree Kernels.
084       */
085      public void trainClassifier(File dir, String... args) throws Exception {
086        for (File file : dir.listFiles()) {
087          if (file.getName().matches("training-data-\\d+.svmlight")) {
088            TKSVMlightClassifierBuilder.train(file.getPath(), args);
089          }
090        }
091      }
092    
093      /**
094       * Package the classifier into a jar file.
095       */
096      @Override
097      protected void packageClassifier(File dir, JarOutputStream modelStream) throws IOException {
098        super.packageClassifier(dir, modelStream);
099    
100        int label = 1;
101        while (true) {
102          File modelFile = new File(dir, String.format("training-data-%d.svmlight.model", label));
103    
104          if (!modelFile.exists()) {
105            break;
106          }
107    
108          String modelName = String.format("model-%d.svmlight", label);
109          JarStreams.putNextJarEntry(modelStream, modelName, modelFile);
110    
111          label += 1;
112        }
113      }
114    
115      private TreeMap<Integer, File> models;
116    
117      /**
118       * Unpackage the classifier out of a JarInputStream.
119       */
120      @Override
121      protected void unpackageClassifier(JarInputStream modelStream) throws IOException {
122        super.unpackageClassifier(modelStream);
123        this.models = new TreeMap<Integer, File>();
124        File model;
125    
126        int label = 1;
127        while ((model = getNextModel(modelStream, label)) != null) {
128          this.models.put(label, model);
129          label += 1;
130        }
131    
132        if (this.models.isEmpty()) {
133          throw new IOException(String.format("no models found in %s", modelStream));
134        }
135      }
136    
137      /**
138       * Create a OVATKSVMlightClassifier.
139       */
140      @Override
141      protected OVATKSVMlightClassifier newClassifier() {
142        return new OVATKSVMlightClassifier(this.featuresEncoder, this.outcomeEncoder, this.models);
143      }
144    
145      private static File getNextModel(JarInputStream modelStream, int label) throws IOException {
146        File mFile = File.createTempFile(String.format("model-%d", label), ".svmlight");
147        // look for a next entry or return null if there isn't one
148        JarEntry entry = modelStream.getNextJarEntry();
149        if (entry == null) {
150          return null;
151        }
152    
153        // make sure the name was the model we expected
154        String expectedName = String.format("model-%d.svmlight", label);
155        if (!entry.getName().equals(expectedName)) {
156          throw new IOException(String.format(
157              "expected next jar entry to be %s, found %s",
158              expectedName,
159              entry.getName()));
160        }
161    
162        BufferedWriter out = new BufferedWriter(new FileWriter(mFile));
163        BufferedReader in = new BufferedReader(new InputStreamReader(modelStream));
164        String line;
165        while ((line = in.readLine()) != null) {
166          out.append(line);
167          out.append("\n");
168        }
169        out.close();
170        // read the model from the jar stream
171        return mFile;
172      }
173    }
