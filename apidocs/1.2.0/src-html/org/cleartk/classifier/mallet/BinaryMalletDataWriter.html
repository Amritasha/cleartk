024    package org.cleartk.classifier.mallet;
025    
026    import java.io.File;
027    import java.io.IOException;
028    
029    import org.cleartk.classifier.encoder.outcome.BooleanToStringOutcomeEncoder;
030    
031    /**
032     * <br>
033     * Copyright (c) 2007-2008, Regents of the University of Colorado <br>
034     * All rights reserved.
035     * 
036     * 
037     * This training data consumer produces training data suitable for <a
038     * href="http://mallet.cs.umass.edu/index.php/SimpleTagger_example"> Mallet Conditional Random Field
039     * (CRF) tagger</a>.
040     * 
041     * Each line of the training data contains a string representation of each feature followed by the
042     * label/result for that instance.
043     * 
044     * @author Philip Ogren
045     * @deprecated Use {@link MalletBooleanOutcomeDataWriter} instead.
046     */
047    @Deprecated
048    public class BinaryMalletDataWriter extends
049        MalletDataWriter_ImplBase<BinaryMalletClassifierBuilder, Boolean> {
050    
051      public BinaryMalletDataWriter(File outputDirectory) throws IOException {
052        super(outputDirectory);
053        this.setOutcomeEncoder(new BooleanToStringOutcomeEncoder());
054      }
055    
056      @Override
057      protected BinaryMalletClassifierBuilder newClassifierBuilder() {
058        return new BinaryMalletClassifierBuilder();
059      }
060    
061    }
