072    public class CRFSuiteWrapper {
073      static Logger logger = UIMAFramework.getLogger(CRFSuiteWrapper.class);
074    
075      private File executable;
076    
077      public CRFSuiteWrapper() {
078        Executables exec = new Executables();
079        if (exec.isInstalled()) {
080          logger.log(Level.FINE, "The CRFSuite is installed on the system");
081          executable = new File(exec.getExecutableName());
082        } else {
083    
084          this.executable = exec.getExecutable();
085          if (!exec.isInstalled(this.executable.getAbsolutePath())) {
086            logger.log(
087                Level.WARNING,
088                "The CRFSuite binary is not available for the current operation system, please install it!");
089          } else {
090            logger.log(Level.FINE, "The CRFSuite binary is successfully extracted");
091          }
092        }
093      }
094    
095      class Executables {
096        PlatformDetection pd = new PlatformDetection();
097    
098        public Executables() {
099    
100          // windows 64 should also be able to execute win32 files --> change
101          // it to win 32, this can be fixed, when a win 64 Bit executable is
102          // available
103          if (pd.getOs().equals(OS_WINDOWS) && pd.getArch().equals(ARCH_X86_64)) {
104            pd.setArch(ARCH_X86_32);
105          }
106        }
107    
108        public String getExecutablePath() {
109    
110          String[] path = new String[] { "crfsuite", pd.toString(), "bin" };
111          String sep = "/";
112          String p = "";
113          for (String s : path) {
114            p += s + sep;
115          }
116          return p;
117    
118        }
119    
120        public boolean isInstalled() {
121          return isInstalled("crfsuite");
122        }
123    
124        public boolean isInstalled(String path) {
125          ProcessBuilder builder = new ProcessBuilder(path, "-h");
126          builder.redirectErrorStream();
127          Process p;
128    
129          try {
130            p = builder.start();
131            InputStreamHandler<StringBuffer> output = InputStreamHandler.getInputStreamAsBufferedString(p.getInputStream());
132            try {
133              p.waitFor();
134              output.join();
135            } catch (InterruptedException e) {
136              logger.log(Level.WARNING, e.getMessage());
137            }
138            StringBuffer buffer = output.getBuffer();
139            if (buffer.length() < 8) {
140              logger.log(Level.WARNING, "CRFSuite could not be executed!");
141            }
142            return buffer.length() >= 8 && buffer.substring(0, 8).equals("CRFSuite");
143          } catch (IOException e) {
144            // Path is not available
145            logger.log(Level.FINE, e.getMessage());
146          }
147          return false;
148        }
149    
150        public String getExecutableName() {
151          return "crfsuite" + pd.getExecutableSuffix();
152        }
153    
154        public File getExecutable() {
155          String loc = getExecutablePath() + getExecutableName();
156          URL crfExecUrl = getClass().getResource(loc);
157          crfExecUrl = ClassLoader.getSystemResource(loc);
158          logger.log(Level.FINE, "CrfSuite Location " + loc);
159          logger.log(Level.FINE, "CrfSuite Url: " + crfExecUrl);
160          File f;
161          try {
162            if (crfExecUrl != null) {
163              f = new File(ResourceUtils.getUrlAsFile(crfExecUrl, true).toURI().getPath());
164              if (!f.exists()) {
165                f = new File(URLDecoder.decode(f.getAbsolutePath(), ("UTF-8")));
166              }
167              f.setExecutable(true);
168              return f;
169            }
170            logger.log(Level.WARNING, "The executable could not be found at " + loc);
171            return null;
172          } catch (IOException e) {
173            e.printStackTrace();
174    
175            return null;
176          }
177    
178        }
179    
180      }

182      public void trainClassifier(String model, String trainingDataFile, String[] args)
183          throws IOException {
184    
185        StringBuffer cmd = new StringBuffer();
186        cmd.append(executable.getPath());
187        cmd.append(" ");
188        cmd.append("learn");
189        cmd.append(" ");
190        cmd.append("-m");
191        cmd.append(" ");
192        cmd.append(model);
193        cmd.append(" ");
194        for (String a : args) {
195          cmd.append(a);
196          cmd.append(" ");
197        }
198        cmd.append(trainingDataFile);
199        Process p = Runtime.getRuntime().exec(cmd.toString());
200    
201        InputStream stdIn = p.getInputStream();
202        InputStreamHandler<List<String>> ishIn = InputStreamHandler.getInputStreamAsList(stdIn);
203    
204        InputStream stdErr = p.getErrorStream();
205        InputStreamHandler<StringBuffer> ishErr = InputStreamHandler.getInputStreamAsBufferedString(stdErr);
206    
207        try {
208          p.waitFor();
209          ishIn.join();
210          ishErr.join();
211        } catch (InterruptedException e) {
212          logger.log(Level.WARNING, e.getMessage());
213        }
214    
215        logger.log(Level.WARNING, ishErr.getBuffer().toString().replaceAll("(^\\[)|([,])|(]$)", "\n"));
216        logger.log(Level.INFO, ishIn.getBuffer().toString().replaceAll("(^\\[)|([,])|(]$)", "\n"));
217        stdErr.close();
218        stdIn.close();
219    
220      }

222      public List<String> classifyFeatures(String featureFile, String modelFile, int featureSize)
223          throws IOException {
224        return classifyFeatures(new File(featureFile), new File(modelFile), featureSize);
225      }
226    
227      public List<String> classifyFeatures(File featureFile, File modelFile, int featureSize)
228          throws IOException {
229        List<String> result = new ArrayList<String>();
230        result = classifyFeatures(modelFile, featureFile);
231        if (result.size() != featureSize) {
232          throw new IllegalStateException(
233              "The number of extracted classified labels is not equivalent with the number of instanzes ("
234                  + result.size() + "!=" + featureSize + ")");
235        }
236    
237        return result;
238    
239      }
240    
241      public List<String> classifyFeatures(
242          List<List<Feature>> features,
243          OutcomeEncoder<String, String> outcomeEncoder,
244          FeaturesEncoder<List<NameNumber>> featuresEncoder,
245          File modelFile) throws IOException {
246    
247        File featureFile = File.createTempFile("features", ".crfsuite");
248        featureFile.deleteOnExit();
249        logger.log(Level.FINE, "Write features to classify to " + featureFile.getAbsolutePath());
250        try {
251          BufferedWriter out = new BufferedWriter(new FileWriter(featureFile));
252          for (List<Feature> f : features) {
253            List<NameNumber> fe;
254            fe = featuresEncoder.encodeAll(f);
255            for (NameNumber nn : fe) {
256              out.append(nn.name);
257              out.append("\t");
258            }
259            out.append("\n");
260          }
261    
262          out.close();
263          return classifyFeatures(featureFile, modelFile, features.size());
264        } catch (CleartkEncoderException e) {
265          logger.log(Level.WARNING, e.getMessage());
266        }
267        return null;
268    
269      }
270    
271      private List<String> classifyFeatures(File modelFile, File featureFile) throws IOException {
272        List<String> posTags = new ArrayList<String>();
273        StringBuffer cmd = new StringBuffer();
274    
275        cmd.append(executable.getPath());
276        cmd.append(" tag -m ");
277        cmd.append(modelFile.getAbsolutePath());
278        cmd.append(" ");
279        cmd.append(featureFile.getAbsolutePath());
280        cmd.append(" ");
281        Process p = Runtime.getRuntime().exec(cmd.toString());
282    
283        InputStream stdIn = p.getInputStream();
284        InputStreamHandler<List<String>> ishIn = InputStreamHandler.getInputStreamAsList(stdIn);
285    
286        InputStream stdErr = p.getErrorStream();
287        InputStreamHandler<StringBuffer> ishErr = InputStreamHandler.getInputStreamAsBufferedString(stdErr);
288    
289        try {
290          p.waitFor();
291          ishIn.join();
292          ishErr.join();
293        } catch (InterruptedException e) {
294          logger.log(Level.WARNING, e.getMessage());
295        }
296        logger.log(Level.WARNING, ishErr.getBuffer().toString().replaceAll("(^\\[)|([,])|(]$)", "\n"));
297        posTags = ishIn.getBuffer();
298        if (posTags.size() > 0) {
299          if (posTags.get(posTags.size() - 1).trim().length() == 0) {
300            posTags.remove(posTags.size() - 1);
301          }
302        }
303        stdErr.close();
304        stdIn.close();
305        return posTags;
306      }
