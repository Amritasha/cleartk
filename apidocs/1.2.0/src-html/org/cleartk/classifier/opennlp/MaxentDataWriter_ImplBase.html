024    package org.cleartk.classifier.opennlp;
025    
026    import java.io.File;
027    import java.io.IOException;
028    import java.util.List;
029    
030    import opennlp.model.RealValueFileEventStream;
031    
032    import org.cleartk.classifier.CleartkProcessingException;
033    import org.cleartk.classifier.encoder.features.BooleanEncoder;
034    import org.cleartk.classifier.encoder.features.NameNumber;
035    import org.cleartk.classifier.encoder.features.NameNumberFeaturesEncoder;
036    import org.cleartk.classifier.encoder.features.NumberEncoder;
037    import org.cleartk.classifier.encoder.features.StringEncoder;
038    import org.cleartk.classifier.jar.DataWriter_ImplBase;
039    
040    /**
041     * <br>
042     * Copyright (c) 2007-2011, Regents of the University of Colorado <br>
043     * All rights reserved.
044     * 
045     * <p>
046     * 
047     * Each line of the training data contains a label/result for the instance followed by a string
048     * representation of each feature. Models trained with data generated by this class should use
049     * RealValueFileEventStream. For relevant discussion, please see:
050     * 
051     * https://sourceforge.net/forum/forum.php?thread_id=1925312&forum_id=18385
052     * 
053     * 
054     * 
055     * @author Philip Ogren
056     * @author Steven Bethard
057     * @see RealValueFileEventStream
058     */
059    public abstract class MaxentDataWriter_ImplBase<CLASSIFIER_BUILDER_TYPE extends MaxentClassifierBuilder_ImplBase<? extends MaxentClassifier_ImplBase<OUTCOME_TYPE>, OUTCOME_TYPE>, OUTCOME_TYPE>
060        extends DataWriter_ImplBase<CLASSIFIER_BUILDER_TYPE, List<NameNumber>, OUTCOME_TYPE, String> {
061    
062      public MaxentDataWriter_ImplBase(File outputDirectory) throws IOException {
063        super(outputDirectory);
064        NameNumberFeaturesEncoder ftrsNcdr = new NameNumberFeaturesEncoder();
065        ftrsNcdr.addEncoder(new NumberEncoder());
066        ftrsNcdr.addEncoder(new BooleanEncoder());
067        ftrsNcdr.addEncoder(new StringEncoder());
068        this.setFeaturesEncoder(ftrsNcdr);
069      }
070    
071      @Override
072      public void writeEncoded(List<NameNumber> features, String outcome)
073          throws CleartkProcessingException {
074        if (outcome == null) {
075          throw CleartkProcessingException.noInstanceOutcome(features);
076        }
077        this.trainingDataWriter.print(outcome);
078    
079        if (features.size() == 0) {
080          trainingDataWriter.print(" null=0");
081        }
082    
083        // write each of the string features, encoded, into the training data
084        for (NameNumber nameNumber : features) {
085          this.trainingDataWriter.print(' ');
086          if (nameNumber.number.doubleValue() == 1.0)
087            trainingDataWriter.print(nameNumber.name);
088          else
089            trainingDataWriter.print(nameNumber.name + "=" + nameNumber.number);
090    
091        }
092    
093        // complete the feature line
094        this.trainingDataWriter.println();
095      }
096    
097    }
