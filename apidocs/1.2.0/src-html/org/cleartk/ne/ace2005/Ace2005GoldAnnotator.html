052    /**
053     * <br>
054     * Copyright (c) 2007-2008, Regents of the University of Colorado <br>
055     * All rights reserved.
056     * 
057     * 
058     * This parser works with the ACE2005CollectionReader to pre-populate the JCas with ACE named
059     * entities. Currently, it only loads information from the entity and entity_mention tags and adds
060     * NamedEntity and NamedEntityMention objects to the JCas.
061     * 
062     * @author Philip Ogren
063     * 
064     */
065    @SofaCapability(
066        inputSofas = { Ace2005Constants.ACE_2005_APF_URI_VIEW, CAS.NAME_DEFAULT_SOFA },
067        outputSofas = {})
068    public class Ace2005GoldAnnotator extends JCasAnnotator_ImplBase {
069    
070      Pattern ampPattern;
071    
072      @Override
073      public void initialize(UimaContext aContext) throws ResourceInitializationException {
074        super.initialize(aContext);
075        ampPattern = Pattern.compile(Pattern.quote("&amp;"));
076      }
077    
078      @Override
079      public void process(JCas jCas) throws AnalysisEngineProcessException {
080        try {
081          String apfUri = jCas.getView(Ace2005Constants.ACE_2005_APF_URI_VIEW).getSofaDataURI();
082          JCas initialView = jCas.getView(CAS.NAME_DEFAULT_SOFA);
083          String documentText = initialView.getDocumentText();
084          SAXBuilder builder = new SAXBuilder();
085          builder.setDTDHandler(null);
086          URI sofaDataURI = new URI(apfUri);
087          Document doc = builder.build(new File(sofaDataURI));
088    
089          Element apfSource = doc.getRootElement();
090          Element apfDocument = apfSource.getChild("document");
091          for (Element apfEntity : apfDocument.getChildren("entity")) {
092            NamedEntity namedEntity = new NamedEntity(initialView);
093            namedEntity.setEntityType(apfEntity.getAttributeValue("TYPE"));
094            namedEntity.setEntitySubtype(apfEntity.getAttributeValue("SUBTYPE"));
095            namedEntity.setEntityClass(apfEntity.getAttributeValue("CLASS"));
096            namedEntity.setEntityId(apfEntity.getAttributeValue("ID"));
097            namedEntity.addToIndexes();
098    
099            List<NamedEntityMention> mentions = new ArrayList<NamedEntityMention>();
100    
101            for (Element entityMention : apfEntity.getChildren("entity_mention")) {
102              int start = Integer.parseInt(entityMention.getChild("extent").getChild("charseq").getAttributeValue(
103                  "START"));
104              int end = Integer.parseInt(entityMention.getChild("extent").getChild("charseq").getAttributeValue(
105                  "END"));
106              String givenText = entityMention.getChild("extent").getChild("charseq").getText();
107              String parsedText = documentText.substring(start, end + 1);
108              Matcher ampMatcher = ampPattern.matcher(parsedText);
109              parsedText = ampMatcher.replaceAll("&");
110    
111              NamedEntityMention mention = new NamedEntityMention(initialView, start, end + 1);
112              mention.setMentionId(entityMention.getAttributeValue("ID"));
113              mention.setMentionType(entityMention.getAttributeValue("TYPE"));
114              mention.setMentionedEntity(namedEntity);
115    
116              Chunk chunk = new Chunk(initialView, start, end + 1);
117              mention.setAnnotation(chunk);
118    
119              int headStart = Integer.parseInt(entityMention.getChild("head").getChild("charseq").getAttributeValue(
120                  "START"));
121              int headEnd = Integer.parseInt(entityMention.getChild("head").getChild("charseq").getAttributeValue(
122                  "END"));
123              Chunk head = new Chunk(initialView, headStart, headEnd + 1);
124              mention.setHead(head);
125    
126              mention.addToIndexes();
127              mentions.add(mention);
128    
129              givenText = givenText.replaceAll("\\s+", " ");
130              parsedText = givenText.replaceAll("\\s+", " ");
131    
132              // if(!givenText.equals(parsedText))
133              // {
134              //
135              // System.out.println("given text and parsed text differ.");
136              // System.out.println(givenText);
137              // System.out.println(parsedText);
138              // System.out.println(apfDocument.getAttributeValue("DOCID"));
139              // System.out.println(apfEntity.getAttributeValue("ID"));
140              // System.out.println(documentText);
141              // }
142            }
143            namedEntity.setMentions(UIMAUtil.toFSArray(jCas, mentions));
144          }
145    
146        } catch (CASException ce) {
147          throw new AnalysisEngineProcessException(ce);
148        } catch (IOException ioe) {
149          throw new AnalysisEngineProcessException(ioe);
150        } catch (JDOMException je) {
151          throw new AnalysisEngineProcessException(je);
152        } catch (URISyntaxException use) {
153          throw new AnalysisEngineProcessException(use);
154        }
155      }
156    
157    }
