061    public class Ace2005Writer extends JCasAnnotator_ImplBase {
062    
063      public static final String PARAM_OUTPUT_DIRECTORY_NAME = ConfigurationParameterFactory.createConfigurationParameterName(
064          Ace2005Writer.class,
065          "outputDirectoryName");
066    
067      @ConfigurationParameter(
068          mandatory = true,
069          description = "provides the path of the directory where the XML files should be written.")
070      private String outputDirectoryName;
071    
072      private File outputDirectory;
073    
074      private int idIndex = 0;
075    
076      @Override
077      public void initialize(UimaContext context) throws ResourceInitializationException {
078        super.initialize(context);
079    
080        outputDirectory = new File(outputDirectoryName);
081        if (!outputDirectory.exists()) {
082          outputDirectory.mkdirs();
083        }
084      }
085    
086      private Element createExtentElement(String elementName, Annotation annotation) {
087        Element extent = new Element(elementName);
088        Element charseq = new Element("charseq");
089        charseq.setAttribute("START", "" + annotation.getBegin());
090        charseq.setAttribute("END", "" + (annotation.getEnd() - 1));
091        charseq.setText(annotation.getCoveredText());
092        extent.addContent(charseq);
093        return extent;
094      }
095    
096      @Override
097      public void process(JCas jCas) throws AnalysisEngineProcessException {
098        String uri = new File(ViewURIUtil.getURI(jCas)).getName();
099        String docId = uri.substring(0, uri.indexOf(".sgm"));
100        Ace2005Document document;
101        document = JCasUtil.select(jCas, Ace2005Document.class).iterator().next();
102    
103        Document xml = new Document();
104    
105        Element sourceFileElement = new Element("source_file");
106        sourceFileElement.setAttribute("URI", uri);
107        sourceFileElement.setAttribute("SOURCE", document.getAceSource());
108        sourceFileElement.setAttribute("TYPE", document.getAceType());
109        xml.addContent(sourceFileElement);
110    
111        Element documentElement = new Element("document");
112        documentElement.setAttribute("DOCID", docId);
113        sourceFileElement.addContent(documentElement);
114    
115        FSIterator<FeatureStructure> namedEntities = jCas.getFSIndexRepository().getAllIndexedFS(
116            jCas.getCasType(NamedEntity.type));
117        while (namedEntities.hasNext()) {
118          NamedEntity namedEntity = (NamedEntity) namedEntities.next();
119          Element namedEntityElement = new Element("entity");
120          namedEntityElement.setAttribute("ID", "" + idIndex++);
121          namedEntityElement.setAttribute("TYPE", namedEntity.getEntityType());
122          String entitySubtype = namedEntity.getEntitySubtype();
123          if (entitySubtype != null)
124            namedEntityElement.setAttribute("SUBTYPE", entitySubtype);
125          String entityClass = namedEntity.getEntityClass();
126          if (entityClass != null)
127            namedEntityElement.setAttribute("CLASS", entityClass);
128    
129          FSArray namedEntityMentions = namedEntity.getMentions();
130          for (int i = 0; i < namedEntityMentions.size(); i++) {
131            NamedEntityMention namedEntityMention = (NamedEntityMention) namedEntityMentions.get(i);
132            Element namedEntityMentionElement = new Element("entity_mention");
133            namedEntityMentionElement.setAttribute("ID", "" + idIndex++);
134            namedEntityMentionElement.setAttribute("TYPE", namedEntityMention.getMentionType());
135    
136            namedEntityMentionElement.addContent(createExtentElement(
137                "extent",
138                namedEntityMention.getAnnotation()));
139            namedEntityMentionElement.addContent(createExtentElement(
140                "head",
141                namedEntityMention.getHead()));
142            namedEntityElement.addContent(namedEntityMentionElement);
143          }
144          documentElement.addContent(namedEntityElement);
145        }
146    
147        XMLOutputter xmlOut = new XMLOutputter(Format.getPrettyFormat());
148        try {
149          FileOutputStream stream = new FileOutputStream(new File(outputDirectory, docId
150              + ".cleartk.xml"));
151          xmlOut.output(xml, stream);
152          stream.close();
153        } catch (IOException e) {
154          throw new AnalysisEngineProcessException(e);
155        }
156    
157      }
158    
159      public void setOutputDirectoryName(String outputDirectoryName) {
160        this.outputDirectoryName = outputDirectoryName;
161      }
