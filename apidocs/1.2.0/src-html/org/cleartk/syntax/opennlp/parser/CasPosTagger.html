024    package org.cleartk.syntax.opennlp.parser;
025    
026    import java.util.List;
027    
028    import opennlp.tools.postag.POSTagger;
029    import opennlp.tools.util.Sequence;
030    
031    import org.apache.uima.jcas.tcas.Annotation;
032    import org.cleartk.syntax.opennlp.ParserAnnotator;
033    
034    /**
035     * <br>
036     * Copyright (c) 2007-2008, Regents of the University of Colorado <br>
037     * All rights reserved.
038     * 
039     * @author Philipp Wetzler, Phili Ogren
040     * 
041     *         <p>
042     *         This class provides a simple implementation of the OpenNLP POSTagger interface. This
043     *         implementation doesn't perform part-of-speech tagging, per se. Instead, it retrieves
044     *         part-of-speech tags from the CAS using the {@link InputTypesHelper} that it is
045     *         instantiated with. This class is not intended for use outside of the
046     *         {@link ParserAnnotator}.
047     */
048    
049    public class CasPosTagger<TOKEN_TYPE extends Annotation, SENTENCE_TYPE extends Annotation>
050        implements POSTagger {
051    
052      private List<TOKEN_TYPE> tokens;
053    
054      InputTypesHelper<TOKEN_TYPE, SENTENCE_TYPE> inputTypesHelper;
055    
056      public CasPosTagger(InputTypesHelper<TOKEN_TYPE, SENTENCE_TYPE> inputTypesHelper) {
057        this.inputTypesHelper = inputTypesHelper;
058      }
059    
060      public void setTokens(List<TOKEN_TYPE> tokens) {
061        this.tokens = tokens;
062      }
063    
064      public Sequence[] topKSequences(List<String> sentence) {
065        return topKSequences(sentence.toArray(new String[sentence.size()]));
066      }
067    
068      /**
069       * This method will only return a single sequence corresponding to the part-of-speech tags that
070       * have already been assigned to the tokens in the CAS as part of some upstream processing that
071       * precedes running of the {@link ParserAnnotator}.
072       */
073      public Sequence[] topKSequences(String[] sentence) {
074        Sequence[] s = new Sequence[1];
075    
076        s[0] = new Sequence();
077        for (TOKEN_TYPE token : this.tokens) {
078          String pos = inputTypesHelper.getPosTag(token);
079          if (pos == null) {
080            throw new RuntimeException("no part of speech for token: " + token.getCoveredText());
081          }
082          s[0].add(pos, 1.0);
083        }
084    
085        return s;
086      }
087    
088      public List<String> tag(List<String> sentence) {
089        throw new UnsupportedOperationException();
090      }
091    
092      public String[] tag(String[] sentence) {
093        throw new UnsupportedOperationException();
094      }
095    
096      public String tag(String sentence) {
097        throw new UnsupportedOperationException();
098      }
099    
100    }
