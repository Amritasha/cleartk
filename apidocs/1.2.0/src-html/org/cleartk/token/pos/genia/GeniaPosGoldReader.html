067    @SofaCapability(outputSofas = { ViewURIUtil.URI, GeniaPosViewName.GENIA_POS })
068    public class GeniaPosGoldReader extends JCasCollectionReader_ImplBase {
069    
070      public static final String PARAM_GENIA_CORPUS_FILE = ConfigurationParameterFactory.createConfigurationParameterName(
071          GeniaPosGoldReader.class,
072          "geniaCorpusFile");
073    
074      @ConfigurationParameter(
075          description = "names the file that is the Genia corpus to be loaded. A good value is probably '.../GENIAcorpus3.02.pos.xml'.  Please see README in this directory for edits that you may need to make to this file manually.",
076          mandatory = true)
077      private File geniaCorpusFile;
078    
079      public static final String PARAM_LOAD_SENTENCES = ConfigurationParameterFactory.createConfigurationParameterName(
080          GeniaPosGoldReader.class,
081          "loadSentences");
082    
083      @ConfigurationParameter(
084          description = "determines whether sentence annotations will be added from the Genia corpus.",
085          defaultValue = "true")
086      private boolean loadSentences = true;
087    
088      public static final String PARAM_LOAD_TOKENS = ConfigurationParameterFactory.createConfigurationParameterName(
089          GeniaPosGoldReader.class,
090          "loadTokens");
091    
092      @ConfigurationParameter(
093          description = "determines whether tokens annotations will be added from the Genia corpus. ",
094          defaultValue = "true")
095      private boolean loadTokens = true;
096    
097      public static final String PARAM_LOAD_POS_TAGS = ConfigurationParameterFactory.createConfigurationParameterName(
098          GeniaPosGoldReader.class,
099          "loadPosTags");
100    
101      @ConfigurationParameter(
102          description = "determines whether the part of speech tags assigned to each token in the genia corpus will be loaded. The default value of 'true' is used if this "
103              + "parameter is unspecified. If 'loadTokens' is 'false', then 'loadPOSTags' will be treated as 'false' regardless of what is given in the descriptor file.",
104          defaultValue = "true")
105      private boolean loadPosTags = true;
106    
107      public static final String PARAM_ARTICLE_IDS_LIST_FILE = ConfigurationParameterFactory.createConfigurationParameterName(
108          GeniaPosGoldReader.class,
109          "articleIdsListFile");
110    
111      @ConfigurationParameter(
112          description = "names the file used to specify the article ids that should be read in")
113      File articleIdsListFile;
114    
115      private boolean filterArticles;
116    
117      private Set<String> articleIds;
118    
119      private GeniaPOSParser parser;
120    
121      private GeniaParse parse;
122    
123      private int progress = 0;
124    
125      @Override
126      public void initialize(UimaContext context) throws ResourceInitializationException {
127    
128        articleIds = new HashSet<String>();
129    
130        try {
131          if (articleIdsListFile == null) {
132            filterArticles = false;
133          } else {
134            filterArticles = true;
135            String[] ids = FileUtil.loadListOfStrings(articleIdsListFile);
136            for (String id : ids) {
137              articleIds.add(id);
138            }
139          }
140    
141          parser = new GeniaPOSParser(geniaCorpusFile);
142          loadPosTags = loadTokens & loadPosTags;
143        } catch (IOException ioe) {
144          throw new ResourceInitializationException(ioe);
145        } catch (JDOMException je) {
146          throw new ResourceInitializationException(je);
147        }
148      }
149    
150      public void getNext(JCas jCas) throws IOException, CollectionException {
151        if (!hasNext())
152          throw new CollectionException(
153              "Should not be calling getNext() because hasNext returns false",
154              null);
155        try {
156          JCas annotationsView = jCas.getView(CAS.NAME_DEFAULT_SOFA);
157          String text = parse.getText();
158          annotationsView.setDocumentText(text);
159    
160          List<GeniaSentence> sentences = parse.getSentences();
161    
162          for (GeniaSentence geniaSentence : sentences) {
163            if (loadTokens) {
164              List<GeniaTag> posTags = geniaSentence.getPosTags();
165              for (GeniaTag posTag : posTags) {
166                Span tokenSpan = posTag.getSpans().get(0);
167                Token token = new Token(annotationsView, tokenSpan.getBegin(), tokenSpan.getEnd());
168                if (loadPosTags)
169                  token.setPos(posTag.getLabel());
170                token.addToIndexes();
171              }
172            }
173            if (loadSentences) {
174              Sentence sentence = new Sentence(
175                  annotationsView,
176                  geniaSentence.getSpan().getBegin(),
177                  geniaSentence.getSpan().getEnd());
178              sentence.addToIndexes();
179            }
180          }
181    
182          ViewURIUtil.setURI(jCas, new File(parse.getMedline()).toURI());
183    
184          JCas geniaView = jCas.createView(GeniaPosViewName.GENIA_POS);
185          geniaView.setDocumentText(parse.getXml());
186    
187          parse = null;
188        } catch (CASException ce) {
189          throw new CollectionException(ce);
190        }
191    
192      }
193    
194      public void close() throws IOException {
195      }
196    
197      public Progress[] getProgress() {
198        if (filterArticles) {
199          return new Progress[] { new ProgressImpl(progress, articleIds.size(), Progress.ENTITIES) };
200        } else {
201          return new Progress[] { new ProgressImpl(progress, 2000, Progress.ENTITIES) };
202        }
203      }
204    
205      public boolean hasNext() throws IOException, CollectionException {
206        if (parse != null)
207          return true;
208        while (parser.hasNext()) {
209          parse = parser.next();
210          if (!filterArticles) {
211            progress++;
212            return true;
213          }
214          if (articleIds.contains(parse.getMedline())) {
215            progress++;
216            return true;
217          }
218        }
219        return false;
220      }
221    
222      public static CollectionReader getDescription(String geniaCorpusFile)
223          throws ResourceInitializationException {
224        return CollectionReaderFactory.createCollectionReader(
225            GeniaPosGoldReader.class,
226            GeniaPosGoldReader.PARAM_GENIA_CORPUS_FILE,
227            geniaCorpusFile);
228      }
