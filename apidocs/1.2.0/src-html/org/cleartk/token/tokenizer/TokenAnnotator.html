055    public class TokenAnnotator extends JCasAnnotator_ImplBase {
056    
057      public static AnalysisEngineDescription getDescription() throws ResourceInitializationException {
058        return AnalysisEngineFactory.createPrimitiveDescription(TokenAnnotator.class);
059      }
060    
061      public static final String PARAM_TOKENIZER_NAME = ConfigurationParameterFactory.createConfigurationParameterName(
062          TokenAnnotator.class,
063          "tokenizerName");
064    
065      private static final String TOKENIZER_DESCRIPTION = "specifies the class type of the tokenizer that will be used by this annotator. "
066          + "If this parameter is not filled, then the default tokenenizer (org.cleartk.token.util.PennTreebankTokenizer) is used. "
067          + "A tokenenizer is defined as any implementation of the interface defined by org.cleartk.token.util.Tokenizer.";
068    
069      @ConfigurationParameter(
070          description = TOKENIZER_DESCRIPTION,
071          defaultValue = "org.cleartk.token.tokenizer.PennTreebankTokenizer")
072      private String tokenizerName;
073    
074      public static final String PARAM_TOKEN_TYPE_NAME = ConfigurationParameterFactory.createConfigurationParameterName(
075          TokenAnnotator.class,
076          "tokenTypeName");
077    
078      @ConfigurationParameter(
079          description = "class type of the tokens that are created by this annotator. If this parameter is not filled, then tokens of type org.cleartk.token.type.Token will be created.",
080          defaultValue = "org.cleartk.token.type.Token")
081      private String tokenTypeName;
082    
083      public static final String PARAM_WINDOW_TYPE_NAME = ConfigurationParameterFactory.createConfigurationParameterName(
084          TokenAnnotator.class,
085          "windowTypeName");
086    
087      private static final String WINDOW_TYPE_DESCRIPTION = "specifies the class type of annotations that will be tokenized. "
088          + "By default, the tokenizer will tokenize a document sentence by sentence.  If you do not want to precede tokenization with"
089          + "sentence segmentation, then a reasonable value for this parameter is 'org.apache.uima.jcas.tcas.DocumentAnnotation'";
090    
091      @ConfigurationParameter(
092          description = WINDOW_TYPE_DESCRIPTION,
093          defaultValue = "org.cleartk.token.type.Sentence")
094      private String windowTypeName;
095    
096      Tokenizer tokenizer;
097    
098      Class<? extends Annotation> tokenClass;
099    
100      Constructor<? extends Annotation> tokenConstructor;
101    
102      private Class<? extends Annotation> windowClass;
103    
104      private Type windowType = null;
105    
106      private boolean typesInitialized = false;
107    
108      public void initialize(UimaContext uimaContext) throws ResourceInitializationException {
109        try {
110          super.initialize(uimaContext);
111          tokenizer = InitializableFactory.create(uimaContext, tokenizerName, Tokenizer.class);
112          tokenClass = InitializableFactory.getClass(tokenTypeName, Annotation.class);
113          tokenConstructor = tokenClass.getConstructor(new Class[] {
114              JCas.class,
115              Integer.TYPE,
116              Integer.TYPE });
117          if (windowTypeName != null)
118            windowClass = InitializableFactory.getClass(windowTypeName, Annotation.class);
119        } catch (Exception e) {
120          throw new ResourceInitializationException(e);
121        }
122      }
123    
124      private void initializeTypes(JCas jCas) {
125        if (windowClass != null) {
126          windowType = UIMAUtil.getCasType(jCas, windowClass);
127        }
128        typesInitialized = true;
129      }
130    
131      public void process(JCas jCas) throws AnalysisEngineProcessException {
132        try {
133          if (!typesInitialized)
134            initializeTypes(jCas);
135          if (windowType != null) {
136            FSIterator<Annotation> windows = jCas.getAnnotationIndex(windowType).iterator();
137            while (windows.hasNext()) {
138              Annotation window = windows.next();
139              List<Token> pojoTokens = tokenizer.getTokens(window.getCoveredText());
140              createTokens(pojoTokens, window.getBegin(), jCas);
141            }
142          } else {
143            String text = jCas.getDocumentText();
144            List<Token> pojoTokens = tokenizer.getTokens(text);
145            createTokens(pojoTokens, 0, jCas);
146          }
147        } catch (Exception e) {
148          throw new AnalysisEngineProcessException(e);
149        }
150      }
151    
152      private void createTokens(List<Token> pojoTokens, int offset, JCas jCas)
153          throws InstantiationException, InvocationTargetException, IllegalAccessException {
154        for (Token pojoToken : pojoTokens) {
155          int tokenBegin = pojoToken.getBegin() + offset;
156          int tokenEnd = pojoToken.getEnd() + offset;
157          tokenConstructor.newInstance(jCas, tokenBegin, tokenEnd).addToIndexes();
158        }
159      }
160    
161      public void setTokenizerName(String tokenizerName) {
162        this.tokenizerName = tokenizerName;
163      }
164    
165      public void setTokenTypeName(String tokenTypeName) {
166        this.tokenTypeName = tokenTypeName;
167      }
168    
169      public void setWindowTypeName(String windowTypeName) {
170        this.windowTypeName = windowTypeName;
171      }
