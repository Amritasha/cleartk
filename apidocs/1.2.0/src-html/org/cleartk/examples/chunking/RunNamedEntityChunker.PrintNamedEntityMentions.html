048    /**
049     * This class provides a main method that demonstrates how to run a trained
050     * {@link NamedEntityChunker} on new files.
051     * 
052     * <br>
053     * Copyright (c) 2012, Regents of the University of Colorado <br>
054     * All rights reserved.
055     * 
056     * @author Steven Bethard
057     */
058    public class RunNamedEntityChunker {
059    
060      public static class Options extends Options_ImplBase {
061        @Option(name = "--model-dir", usage = "The directory where the model was trained")
062        public File modelDirectory = new File("target/chunking/ne-model");
063    
064        @Option(name = "--text-file", usage = "The file to label with named entities.")
065        public File textFile = new File("src/main/resources/data/sample/2008_Sichuan_earthquake.txt");
066      }
067    
068      public static void main(String[] args) throws Exception {
069        Options options = new Options();
070        options.parseOptions(args);
071    
072        // a reader that loads the URIs of the text file
073        CollectionReader reader = UriCollectionReader.getCollectionReaderFromFiles(Arrays.asList(options.textFile));
074    
075        // assemble the classification pipeline
076        AggregateBuilder aggregate = new AggregateBuilder();
077    
078        // an annotator that loads the text from the training file URIs
079        aggregate.add(UriToDocumentTextAnnotator.getDescription());
080    
081        // annotators that identify sentences, tokens and part-of-speech tags in the text
082        aggregate.add(SentenceAnnotator.getDescription());
083        aggregate.add(TokenAnnotator.getDescription());
084        aggregate.add(PosTaggerAnnotator.getDescription());
085    
086        // our NamedEntityChunker annotator, configured to classify on the new texts
087        aggregate.add(AnalysisEngineFactory.createPrimitiveDescription(
088            NamedEntityChunker.class,
089            CleartkSequenceAnnotator.PARAM_IS_TRAINING,
090            false,
091            GenericJarClassifierFactory.PARAM_CLASSIFIER_JAR_PATH,
092            new File(options.modelDirectory, "model.jar")));
093    
094        // a very simple annotator that just prints out any named entities we found
095        aggregate.add(AnalysisEngineFactory.createPrimitiveDescription(PrintNamedEntityMentions.class));
096    
097        // run the classification pipeline on the new texts
098        SimplePipeline.runPipeline(reader, aggregate.createAggregateDescription());
099      }
100    
101      /**
102       * A simple annotator that just prints out any {@link NamedEntityMention}s in the CAS.
103       * 
104       * A real pipeline would probably decide on an appropriate output format and write files instead
105       * of printing to standard output.
106       */
107      public static class PrintNamedEntityMentions extends JCasAnnotator_ImplBase {
108    
109        @Override
110        public void process(JCas jCas) throws AnalysisEngineProcessException {
111          for (NamedEntityMention mention : JCasUtil.select(jCas, NamedEntityMention.class)) {
112            System.out.printf("%s (%s)\n", mention.getCoveredText(), mention.getMentionType());
113          }
114        }
115    
116      }
117    }
