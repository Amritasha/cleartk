024    package org.cleartk.examples.chunking;
025    
026    import java.util.ArrayList;
027    import java.util.List;
028    
029    import org.apache.uima.UimaContext;
030    import org.apache.uima.analysis_engine.AnalysisEngineProcessException;
031    import org.apache.uima.jcas.JCas;
032    import org.apache.uima.resource.ResourceInitializationException;
033    import org.cleartk.classifier.CleartkSequenceAnnotator;
034    import org.cleartk.classifier.Feature;
035    import org.cleartk.classifier.Instances;
036    import org.cleartk.classifier.chunking.BIOChunking;
037    import org.cleartk.classifier.feature.extractor.CleartkExtractor;
038    import org.cleartk.classifier.feature.extractor.CleartkExtractor.Following;
039    import org.cleartk.classifier.feature.extractor.CleartkExtractor.Preceding;
040    import org.cleartk.classifier.feature.extractor.simple.CharacterCategoryPatternExtractor;
041    import org.cleartk.classifier.feature.extractor.simple.CharacterCategoryPatternExtractor.PatternType;
042    import org.cleartk.classifier.feature.extractor.simple.CombinedExtractor;
043    import org.cleartk.classifier.feature.extractor.simple.CoveredTextExtractor;
044    import org.cleartk.classifier.feature.extractor.simple.SimpleFeatureExtractor;
045    import org.cleartk.classifier.feature.extractor.simple.TypePathExtractor;
046    import org.cleartk.ne.type.NamedEntityMention;
047    import org.cleartk.token.type.Sentence;
048    import org.cleartk.token.type.Token;
049    import org.uimafit.util.JCasUtil;
050    
051    /**
052     * This is the most important class in the named entity chunking example -- it demonstrates how to
053     * write a ClearTK annotator that creates NamedEntityMention annotations by classifying Token
054     * annotations.
055     * 
056     * <br>
057     * Copyright (c) 2012, Regents of the University of Colorado <br>
058     * All rights reserved.
059     * 
060     * @author Steven Bethard
061     */
062    public class NamedEntityChunker extends CleartkSequenceAnnotator<String> {
063    
064      private SimpleFeatureExtractor extractor;
065    
066      private CleartkExtractor contextExtractor;
067    
068      private BIOChunking<Token, NamedEntityMention> chunking;
069    
070      @Override
071      public void initialize(UimaContext context) throws ResourceInitializationException {
072        super.initialize(context);
073    
074        // the token feature extractor: text, char pattern (uppercase, digits, etc.), and part-of-speech
075        this.extractor = new CombinedExtractor(
076            new CoveredTextExtractor(),
077            new CharacterCategoryPatternExtractor(PatternType.REPEATS_MERGED),
078            new TypePathExtractor(Token.class, "pos"));
079    
080        // the context feature extractor: the features above for the 3 preceding and 3 following tokens
081        this.contextExtractor = new CleartkExtractor(
082            Token.class,
083            this.extractor,
084            new Preceding(3),
085            new Following(3));
086    
087        // the chunking definition: Tokens will be combined to form NamedEntityMentions, with labels
088        // from the "mentionType" attribute so that we get B-location, I-person, etc.
089        this.chunking = new BIOChunking<Token, NamedEntityMention>(
090            Token.class,
091            NamedEntityMention.class,
092            "mentionType");
093      }
094    
095      @Override
096      public void process(JCas jCas) throws AnalysisEngineProcessException {
097        for (Sentence sentence : JCasUtil.select(jCas, Sentence.class)) {
098    
099          // extract features for each token in the sentence
100          List<Token> tokens = JCasUtil.selectCovered(jCas, Token.class, sentence);
101          List<List<Feature>> featureLists = new ArrayList<List<Feature>>();
102          for (Token token : tokens) {
103            List<Feature> features = new ArrayList<Feature>();
104            features.addAll(this.extractor.extract(jCas, token));
105            features.addAll(this.contextExtractor.extract(jCas, token));
106            featureLists.add(features);
107          }
108    
109          // during training, convert NamedEntityMentions in the CAS into expected classifier outcomes
110          if (this.isTraining()) {
111    
112            // extract the gold (human annotated) NamedEntityMention annotations
113            List<NamedEntityMention> namedEntityMentions = JCasUtil.selectCovered(
114                jCas,
115                NamedEntityMention.class,
116                sentence);
117    
118            // convert the NamedEntityMention annotations into token-level BIO outcome labels
119            List<String> outcomes = this.chunking.createOutcomes(jCas, tokens, namedEntityMentions);
120    
121            // write the features and outcomes as training instances
122            this.dataWriter.write(Instances.toInstances(outcomes, featureLists));
123          }
124    
125          // during classification, convert classifier outcomes into NamedEntityMentions in the CAS
126          else {
127    
128            // get the predicted BIO outcome labels from the classifier
129            List<String> outcomes = this.classifier.classify(featureLists);
130    
131            // create the NamedEntityMention annotations in the CAS
132            this.chunking.createChunks(jCas, tokens, outcomes);
133          }
134        }
135      }
136    }
