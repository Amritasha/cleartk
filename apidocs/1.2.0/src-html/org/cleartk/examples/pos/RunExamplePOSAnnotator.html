024    package org.cleartk.examples.pos;
025    
026    import java.io.File;
027    import java.util.Arrays;
028    
029    import org.apache.uima.collection.CollectionReader;
030    import org.cleartk.syntax.opennlp.SentenceAnnotator;
031    import org.cleartk.token.stem.snowball.DefaultSnowballStemmer;
032    import org.cleartk.token.tokenizer.TokenAnnotator;
033    import org.cleartk.util.ae.UriToDocumentTextAnnotator;
034    import org.cleartk.util.cr.UriCollectionReader;
035    import org.uimafit.factory.AggregateBuilder;
036    import org.uimafit.factory.AnalysisEngineFactory;
037    import org.uimafit.pipeline.SimplePipeline;
038    
039    /**
040     * <br>
041     * Copyright (c) 2009, Regents of the University of Colorado <br>
042     * All rights reserved.
043     * 
044     * @author Philip Ogren
045     * 
046     *         For examples of using the ExamplePOSAnnotationHandler using different classifiers, please
047     *         see org.cleartk.example.pos.ExamplePosClassifierTest
048     */
049    
050    public class RunExamplePOSAnnotator {
051    
052      public static void main(String[] args) throws Exception {
053    
054        // Create reader
055        CollectionReader reader = UriCollectionReader.getCollectionReaderFromFiles(Arrays.asList(new File(
056            "src/main/resources/data/sample/2008_Sichuan_earthquake.txt")));
057    
058        // Create aggregate engine
059        AggregateBuilder builder = new AggregateBuilder();
060        builder.add(UriToDocumentTextAnnotator.getDescription());
061        builder.add(SentenceAnnotator.getDescription());
062        builder.add(TokenAnnotator.getDescription());
063        builder.add(DefaultSnowballStemmer.getDescription("English"));
064        builder.add(ExamplePOSAnnotator.getClassifierDescription(ExamplePOSAnnotator.DEFAULT_MODEL));
065        builder.add(AnalysisEngineFactory.createPrimitiveDescription(
066            ExamplePOSPlainTextWriter.class,
067            ExamplePOSPlainTextWriter.PARAM_OUTPUT_DIRECTORY_NAME,
068            ExamplePOSPlainTextWriter.DEFAULT_OUTPUT_DIRECTORY));
069    
070        // Run pipeline
071        SimplePipeline.runPipeline(reader, builder.createAggregateDescription());
072    
073        System.out.println("Please look at the file generated by this program: "
074            + ExamplePOSPlainTextWriter.DEFAULT_OUTPUT_DIRECTORY + "/2008_Sichuan_earthquake.txt.pos");
075      }
076    }
