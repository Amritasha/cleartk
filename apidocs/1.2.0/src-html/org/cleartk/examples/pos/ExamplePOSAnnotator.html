068    public class ExamplePOSAnnotator extends CleartkSequenceAnnotator<String> {
069    
070      public static final String DEFAULT_OUTPUT_DIRECTORY = "target/examples/pos";
071    
072      public static final String DEFAULT_MODEL = "target/examples/pos/model.jar";
073    
074      private SimpleFeatureExtractor tokenFeatureExtractor;
075    
076      private List<CleartkExtractor> contextFeatureExtractors;
077    
078      public void initialize(UimaContext context) throws ResourceInitializationException {
079        super.initialize(context);
080        // alias for NGram feature parameters
081        CharacterNGramFeatureFunction.Orientation fromRight = CharacterNGramFeatureFunction.Orientation.RIGHT_TO_LEFT;
082    
083        // a feature extractor that creates features corresponding to the word, the word lower cased
084        // the capitalization of the word, the numeric characterization of the word, and character ngram
085        // suffixes of length 2 and 3.
086        this.tokenFeatureExtractor = new FeatureFunctionExtractor(
087            new CoveredTextExtractor(),
088            new LowerCaseFeatureFunction(),
089            new CapitalTypeFeatureFunction(),
090            new NumericTypeFeatureFunction(),
091            new CharacterNGramFeatureFunction(fromRight, 0, 2),
092            new CharacterNGramFeatureFunction(fromRight, 0, 3));
093    
094        // a list of feature extractors that require the token and the sentence
095        this.contextFeatureExtractors = new ArrayList<CleartkExtractor>();
096        this.contextFeatureExtractors.add(new CleartkExtractor(
097            Token.class,
098            new CoveredTextExtractor(),
099            new Preceding(2),
100            new Following(2)));
101      }
102    
103      public void process(JCas jCas) throws AnalysisEngineProcessException {
104        // generate a list of training instances for each sentence in the
105        // document
106        for (Sentence sentence : JCasUtil.select(jCas, Sentence.class)) {
107          List<Instance<String>> instances = new ArrayList<Instance<String>>();
108          List<Token> tokens = JCasUtil.selectCovered(jCas, Token.class, sentence);
109    
110          // for each token, extract all feature values and the label
111          for (Token token : tokens) {
112            Instance<String> instance = new Instance<String>();
113    
114            // extract all features that require only the token annotation
115            instance.addAll(tokenFeatureExtractor.extract(jCas, token));
116    
117            // extract all features that require the token and sentence annotations
118            for (CleartkExtractor extractor : this.contextFeatureExtractors) {
119              instance.addAll(extractor.extractWithin(jCas, token, sentence));
120            }
121    
122            // set the instance label from the token's part of speech
123            if (this.isTraining()) {
124              instance.setOutcome(token.getPos());
125            }
126    
127            // add the instance to the list
128            instances.add(instance);
129          }
130    
131          // for training, write instances to the data write
132          if (this.isTraining()) {
133            this.dataWriter.write(instances);
134          }
135    
136          // for classification, set the labels as the token POS labels
137          else {
138            Iterator<Token> tokensIter = tokens.iterator();
139            for (String label : this.classify(instances)) {
140              tokensIter.next().setPos(label);
141            }
142          }
143        }
144      }
145    
146      public static AnalysisEngineDescription getClassifierDescription(String modelFileName)
147          throws ResourceInitializationException {
148        return AnalysisEngineFactory.createPrimitiveDescription(
149            ExamplePOSAnnotator.class,
150            GenericJarClassifierFactory.PARAM_CLASSIFIER_JAR_PATH,
151            modelFileName);
152      }
153    
154      public static AnalysisEngineDescription getWriterDescription(String outputDirectory)
155          throws ResourceInitializationException {
156        return AnalysisEngineFactory.createPrimitiveDescription(
157            ExamplePOSAnnotator.class,
158            CleartkSequenceAnnotator.PARAM_DATA_WRITER_FACTORY_CLASS_NAME,
159            ViterbiDataWriterFactory.class.getName(),
160            DirectoryDataWriterFactory.PARAM_OUTPUT_DIRECTORY,
161            outputDirectory,
162            ViterbiDataWriterFactory.PARAM_DELEGATED_DATA_WRITER_FACTORY_CLASS,
163            DefaultDataWriterFactory.class.getName(),
164            DefaultDataWriterFactory.PARAM_DATA_WRITER_CLASS_NAME,
165            MaxentStringOutcomeDataWriter.class.getName(),
166            ViterbiDataWriterFactory.PARAM_OUTCOME_FEATURE_EXTRACTOR_NAMES,
167            new String[] { DefaultOutcomeFeatureExtractor.class.getName() });
168      }
