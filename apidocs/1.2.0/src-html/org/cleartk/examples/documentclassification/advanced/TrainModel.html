024    package org.cleartk.examples.documentclassification.advanced;
025    
026    import java.io.File;
027    import java.util.Arrays;
028    import java.util.List;
029    
030    import org.apache.uima.collection.CollectionReader;
031    import org.cleartk.util.Options_ImplBase;
032    import org.kohsuke.args4j.Option;
033    
034    /**
035     * Copyright (c) 2007-2011, Regents of the University of Colorado <br>
036     * All rights reserved. <br>
037     * 
038     * Main method for running a document classifier model. This is essentially a wrapper for
039     * DocumentClassificationEvaluation, which has the bulk of the pipeline and execution logic.
040     * 
041     * @author Lee Becker
042     * 
043     */
044    public class TrainModel {
045    
046      public static class Options extends Options_ImplBase {
047        @Option(
048            name = "--train-dir",
049            usage = "Specify the directory containing the training documents.  This is used for cross-validation, and for training in a holdout set evaluation. "
050                + "When we run this example we point to a directory containing training data from a subset of the 20 newsgroup corpus - i.e. a directory called '3news-bydate/train'")
051        public File trainDirectory = new File("src/main/resources/data/3news-bydate/train");
052    
053        @Option(
054            name = "--models-dir",
055            usage = "specify the directory in which to write out the trained model files")
056        public File modelsDirectory = new File("target/document_classification/models");
057    
058        @Option(
059            name = "--training-args",
060            usage = "specify training arguments to be passed to the learner.  For multiple values specify -ta for each - e.g. '-ta -t -ta 0'")
061        public List<String> trainingArguments = Arrays.asList("-t", "0");
062      }
063    
064      public static void main(String[] args) throws Exception {
065        Options options = new Options();
066        options.parseOptions(args);
067    
068        DocumentClassificationEvaluation evaluation = new DocumentClassificationEvaluation(
069            options.modelsDirectory,
070            options.trainingArguments);
071    
072        List<File> trainFiles = DocumentClassificationEvaluation.getFilesFromDirectory(options.trainDirectory);
073        CollectionReader collectionReader = evaluation.getCollectionReader(trainFiles);
074        evaluation.train(collectionReader, options.modelsDirectory);
075      }
076    }
