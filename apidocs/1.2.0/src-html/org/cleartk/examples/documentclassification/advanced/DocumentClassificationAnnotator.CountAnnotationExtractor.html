146      public void initialize(UimaContext context) throws ResourceInitializationException {
147        super.initialize(context);
148    
149        try {
150          TfidfExtractor<String> tfIdfExtractor = initTfIdfExtractor();
151          CentroidTfidfSimilarityExtractor<String> simExtractor = initCentroidTfIdfSimilarityExtractor();
152          ZeroMeanUnitStddevExtractor<String> zmusExtractor = initZmusExtractor();
153          MinMaxNormalizationExtractor<String> minmaxExtractor = initMinMaxExtractor();
154          this.extractor = new CombinedExtractor(
155              tfIdfExtractor,
156              simExtractor,
157              zmusExtractor,
158              minmaxExtractor);
159        } catch (IOException e) {
160          throw new ResourceInitializationException(e);
161        }
162      }
163    
164      private TfidfExtractor<String> initTfIdfExtractor() throws IOException {
165        CleartkExtractor countsExtractor = new CleartkExtractor(
166            Token.class,
167            new CoveredTextExtractor(),
168            new CleartkExtractor.Count(new CleartkExtractor.Covered()));
169    
170        TfidfExtractor<String> tfIdfExtractor = new TfidfExtractor<String>(
171            DocumentClassificationAnnotator.TFIDF_EXTRACTOR_KEY,
172            countsExtractor);
173    
174        if (this.tfIdfUri != null) {
175          tfIdfExtractor.load(this.tfIdfUri);
176        }
177        return tfIdfExtractor;
178      }
179    
180      private CentroidTfidfSimilarityExtractor<String> initCentroidTfIdfSimilarityExtractor()
181          throws IOException {
182        CleartkExtractor countsExtractor = new CleartkExtractor(
183            Token.class,
184            new CoveredTextExtractor(),
185            new CleartkExtractor.Count(new CleartkExtractor.Covered()));
186    
187        CentroidTfidfSimilarityExtractor<String> simExtractor = new CentroidTfidfSimilarityExtractor<String>(
188            DocumentClassificationAnnotator.CENTROID_TFIDF_SIM_EXTRACTOR_KEY,
189            countsExtractor);
190    
191        if (this.tfIdfCentroidSimilarityUri != null) {
192          simExtractor.load(this.tfIdfCentroidSimilarityUri);
193        }
194        return simExtractor;
195      }
196    
197      private ZeroMeanUnitStddevExtractor<String> initZmusExtractor() throws IOException {
198        CombinedExtractor featuresToNormalizeExtractor = new CombinedExtractor(
199            new CountAnnotationExtractor(Sentence.class),
200            new CountAnnotationExtractor(Token.class));
201    
202        ZeroMeanUnitStddevExtractor<String> zmusExtractor = new ZeroMeanUnitStddevExtractor<String>(
203            ZMUS_EXTRACTOR_KEY,
204            featuresToNormalizeExtractor);
205    
206        if (this.zmusUri != null) {
207          zmusExtractor.load(this.zmusUri);
208        }
209    
210        return zmusExtractor;
211      }
212    
213      private MinMaxNormalizationExtractor<String> initMinMaxExtractor() throws IOException {
214        CombinedExtractor featuresToNormalizeExtractor = new CombinedExtractor(
215            new CountAnnotationExtractor(Sentence.class),
216            new CountAnnotationExtractor(Token.class));
217    
218        MinMaxNormalizationExtractor<String> minmaxExtractor = new MinMaxNormalizationExtractor<String>(
219            MINMAX_EXTRACTOR_KEY,
220            featuresToNormalizeExtractor);
221    
222        if (this.minmaxUri != null) {
223          minmaxExtractor.load(this.minmaxUri);
224        }
225    
226        return minmaxExtractor;
227      }
228    
229      public void process(JCas jCas) throws AnalysisEngineProcessException {
230        DocumentAnnotation doc = (DocumentAnnotation) jCas.getDocumentAnnotationFs();
231    
232        Instance<String> instance = new Instance<String>();
233        instance.addAll(this.extractor.extract(jCas, doc));
234    
235        if (isTraining()) {
236          UsenetDocument document = JCasUtil.selectSingle(jCas, UsenetDocument.class);
237          instance.setOutcome(document.getCategory());
238          this.dataWriter.write(instance);
239        } else {
240          // This is classification, so classify and create UsenetDocument annotation
241          String result = this.classifier.classify(instance.getFeatures());
242          UsenetDocument document = new UsenetDocument(jCas, 0, jCas.getDocumentText().length());
243          document.setCategory(result);
244          document.addToIndexes();
245          // System.out.println("classified " + ViewURIUtil.getURI(jCas) + " as " + result + ".");
246        }
247      }

249      public static AnalysisEngineDescription getClassifierDescription(File classifierJarFile)
250          throws ResourceInitializationException {
251        return AnalysisEngineFactory.createPrimitiveDescription(
252            DocumentClassificationAnnotator.class,
253            GenericJarClassifierFactory.PARAM_CLASSIFIER_JAR_PATH,
254            classifierJarFile.toString());
255      }

257      public static class CountAnnotationExtractor implements SimpleFeatureExtractor {
258    
259        @SuppressWarnings("rawtypes")
260        private Class annotationType;
261    
262        @SuppressWarnings("rawtypes")
263        public CountAnnotationExtractor(Class annotationType) {
264          this.annotationType = annotationType;
265        }
266    
267        @Override
268        public List<Feature> extract(JCas view, Annotation focusAnnotation)
269            throws CleartkExtractorException {
270    
271          @SuppressWarnings({ "rawtypes", "unchecked" })
272          List annotations = JCasUtil.selectCovered(this.annotationType, focusAnnotation);
273          return Arrays.asList(new Feature("Count_" + annotationType.getName(), annotations.size()));
274        }
275      }
