024    package org.cleartk.examples.documentclassification.advanced;
025    
026    import java.io.File;
027    import java.net.URI;
028    import java.net.URISyntaxException;
029    
030    import org.apache.uima.analysis_engine.AnalysisEngineProcessException;
031    import org.apache.uima.cas.CASException;
032    import org.apache.uima.jcas.JCas;
033    import org.cleartk.examples.type.UsenetDocument;
034    import org.cleartk.util.ViewURIUtil;
035    import org.uimafit.component.JCasAnnotator_ImplBase;
036    
037    /**
038     * <br>
039     * Copyright (c) 2012, Regents of the University of Colorado <br>
040     * All rights reserved.
041     * 
042     * This class will assign the gold-standard document categories for the documents in this example by
043     * looking at their parent directory.
044     * 
045     * @author Lee Becker
046     * 
047     */
048    public class GoldDocumentCategoryAnnotator extends JCasAnnotator_ImplBase {
049    
050      @Override
051      public void process(JCas jCas) throws AnalysisEngineProcessException {
052        try {
053          JCas uriView = jCas.getView(ViewURIUtil.URI);
054          URI uri = new URI(uriView.getSofaDataURI());
055          File file = new File(uri.getPath());
056    
057          // Create UsenetDocument annotation
058          // The document category will come from the document directory structure
059          UsenetDocument document = new UsenetDocument(jCas, 0, jCas.getDocumentText().length());
060          document.setCategory(file.getParentFile().getName());
061          document.addToIndexes();
062    
063        } catch (CASException e) {
064          throw new AnalysisEngineProcessException(e);
065        } catch (URISyntaxException e) {
066          throw new AnalysisEngineProcessException(e);
067        }
068      }
069    
070    }
