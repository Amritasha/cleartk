024    package org.cleartk.examples.documentclassification.basic;
025    
026    import java.io.File;
027    
028    import org.apache.uima.collection.CollectionReader;
029    import org.cleartk.classifier.CleartkAnnotator;
030    import org.cleartk.classifier.jar.GenericJarClassifierFactory;
031    import org.cleartk.syntax.opennlp.SentenceAnnotator;
032    import org.cleartk.token.stem.snowball.DefaultSnowballStemmer;
033    import org.cleartk.token.tokenizer.TokenAnnotator;
034    import org.cleartk.util.Options_ImplBase;
035    import org.cleartk.util.ae.UriToDocumentTextAnnotator;
036    import org.cleartk.util.cr.UriCollectionReader;
037    import org.kohsuke.args4j.Option;
038    import org.uimafit.factory.AggregateBuilder;
039    import org.uimafit.factory.AnalysisEngineFactory;
040    import org.uimafit.pipeline.SimplePipeline;
041    
042    /**
043     * Copyright (c) 2012, Regents of the University of Colorado <br>
044     * All rights reserved. <br>
045     * 
046     * Illustrates how to run a basic document classification annotator, once it is trained using the
047     * TrainModel class. For a more in-depth example that demonstrates ClearTK best practices including
048     * the use of more sophisticated feature extractors and the evaluation framework refer to the
049     * examples in org.cleartk.examples.document.classification.advanced.
050     * 
051     * @author Lee Becker
052     * 
053     */
054    public class RunModel {
055    
056      public static class Options extends Options_ImplBase {
057        @Option(name = "--test-dir", usage = "Specify the directory containing the documents to label.")
058        public File testDirectory = new File("src/main/resources/data/3news-bydate/test");
059    
060        @Option(
061            name = "--models-dir",
062            usage = "specify the directory in which to write out the trained model files")
063        public File modelsDirectory = new File("target/simple_document_classification/models");
064      }
065    
066      public static void main(String[] args) throws Exception {
067        Options options = new Options();
068        options.parseOptions(args);
069    
070        // ////////////////////////////////////////
071        // Create collection reader to load URIs
072        // ////////////////////////////////////////
073        CollectionReader reader = UriCollectionReader.getCollectionReaderFromDirectory(
074            options.testDirectory,
075            UriCollectionReader.RejectSystemFiles.class,
076            UriCollectionReader.RejectSystemDirectories.class);
077    
078        // ////////////////////////////////////////
079        // Create document classification pipeline
080        // ////////////////////////////////////////
081        AggregateBuilder builder = new AggregateBuilder();
082    
083        // Convert URIs in CAS URI View to Plain Text
084        builder.add(UriToDocumentTextAnnotator.getDescription());
085    
086        // NLP pre-processing components
087        builder.add(SentenceAnnotator.getDescription()); // Sentence segmentation
088        builder.add(TokenAnnotator.getDescription()); // Tokenization
089        builder.add(DefaultSnowballStemmer.getDescription("English")); // Stemming
090    
091        // Simple document classification annotator
092        builder.add(AnalysisEngineFactory.createPrimitiveDescription(
093            BasicDocumentClassificationAnnotator.class,
094            CleartkAnnotator.PARAM_IS_TRAINING,
095            false,
096            GenericJarClassifierFactory.PARAM_CLASSIFIER_JAR_PATH,
097            new File(options.modelsDirectory, "model.jar").getPath()));
098    
099        // //////////////////////////////////////////////////////////////////////////////
100        // Run pipeline and classify documents
101        // //////////////////////////////////////////////////////////////////////////////
102        SimplePipeline.runPipeline(reader, builder.createAggregateDescription());
103      }
104    }
