024    package org.cleartk.examples.documentclassification.basic;
025    
026    import java.io.File;
027    import java.util.Arrays;
028    import java.util.List;
029    
030    import org.apache.uima.collection.CollectionReader;
031    import org.cleartk.classifier.jar.DefaultDataWriterFactory;
032    import org.cleartk.classifier.jar.DirectoryDataWriterFactory;
033    import org.cleartk.classifier.jar.JarClassifierBuilder;
034    import org.cleartk.classifier.libsvm.LIBSVMStringOutcomeDataWriter;
035    import org.cleartk.examples.documentclassification.advanced.GoldDocumentCategoryAnnotator;
036    import org.cleartk.syntax.opennlp.SentenceAnnotator;
037    import org.cleartk.token.stem.snowball.DefaultSnowballStemmer;
038    import org.cleartk.token.tokenizer.TokenAnnotator;
039    import org.cleartk.util.Options_ImplBase;
040    import org.cleartk.util.ae.UriToDocumentTextAnnotator;
041    import org.cleartk.util.cr.UriCollectionReader;
042    import org.kohsuke.args4j.Option;
043    import org.uimafit.factory.AggregateBuilder;
044    import org.uimafit.factory.AnalysisEngineFactory;
045    import org.uimafit.pipeline.SimplePipeline;
046    
047    /**
048     * Copyright (c) 2012, Regents of the University of Colorado <br>
049     * All rights reserved. <br>
050     * 
051     * Illustrates how to train a simple document classification annotator. For a more in-depth example
052     * that demonstrates ClearTK best practices including the use of more sophisticated feature
053     * extractors and the evaluation framework refer to the examples in
054     * org.cleartk.examples.document.classification
055     * 
056     * 
057     * @author Lee Becker
058     * 
059     */
060    public class TrainModel {
061    
062      public static class Options extends Options_ImplBase {
063        @Option(
064            name = "--train-dir",
065            usage = "Specify the directory containing the training documents.  This is used for cross-validation, and for training in a holdout set evaluation. "
066                + "When we run this example we point to a directory containing training data from a subset of the 20 newsgroup corpus - i.e. a directory called '3news-bydate/train'")
067        public File trainDirectory = new File("src/main/resources/data/3news-bydate/train");
068    
069        @Option(
070            name = "--models-dir",
071            usage = "specify the directory in which to write out the trained model files")
072        public File modelsDirectory = new File("target/simple_document_classification/models");
073    
074        @Option(
075            name = "--training-args",
076            usage = "specify training arguments to be passed to the learner.  For multiple values specify -ta for each - e.g. '-ta -t -ta 0'")
077        public List<String> trainingArguments = Arrays.asList("-t", "0");
078      }
079    
080      public static void main(String[] args) throws Exception {
081        Options options = new Options();
082        options.parseOptions(args);
083    
084        // ////////////////////////////////////////
085        // Create collection reader to load URIs
086        // ////////////////////////////////////////
087        CollectionReader reader = UriCollectionReader.getCollectionReaderFromDirectory(
088            options.trainDirectory,
089            UriCollectionReader.RejectSystemFiles.class,
090            UriCollectionReader.RejectSystemDirectories.class);
091    
092        // ////////////////////////////////////////
093        // Create document classification pipeline
094        // ////////////////////////////////////////
095        AggregateBuilder builder = new AggregateBuilder();
096    
097        // Convert URIs in CAS URI View to Plain Text
098        builder.add(UriToDocumentTextAnnotator.getDescription());
099    
100        // Label documents with gold labels for training
101        builder.add(AnalysisEngineFactory.createPrimitiveDescription(GoldDocumentCategoryAnnotator.class));
102    
103        // NLP pre-processing components
104        builder.add(SentenceAnnotator.getDescription()); // Sentence segmentation
105        builder.add(TokenAnnotator.getDescription()); // Tokenization
106        builder.add(DefaultSnowballStemmer.getDescription("English")); // Stemming
107    
108        // The simple document classification annotator
109        builder.add(AnalysisEngineFactory.createPrimitiveDescription(
110            BasicDocumentClassificationAnnotator.class,
111            DefaultDataWriterFactory.PARAM_DATA_WRITER_CLASS_NAME,
112            LIBSVMStringOutcomeDataWriter.class.getName(),
113            DirectoryDataWriterFactory.PARAM_OUTPUT_DIRECTORY,
114            options.modelsDirectory));
115    
116        // ///////////////////////////////////////////
117        // Run pipeline to create training data file
118        // ///////////////////////////////////////////
119        SimplePipeline.runPipeline(reader, builder.createAggregateDescription());
120    
121        // //////////////////////////////////////////////////////////////////////////////
122        // Train and write model
123        // //////////////////////////////////////////////////////////////////////////////
124        JarClassifierBuilder.trainAndPackage(
125            options.modelsDirectory,
126            options.trainingArguments.toArray(new String[options.trainingArguments.size()]));
127      }
128    
129    }
