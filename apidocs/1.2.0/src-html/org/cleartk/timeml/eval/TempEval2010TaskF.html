024    package org.cleartk.timeml.eval;
025    
026    import java.io.File;
027    import java.util.ArrayList;
028    import java.util.Arrays;
029    import java.util.List;
030    
031    import org.cleartk.syntax.opennlp.ParserAnnotator;
032    import org.cleartk.syntax.opennlp.PosTaggerAnnotator;
033    import org.cleartk.timeml.corpus.TempEval2010GoldAnnotator;
034    import org.cleartk.timeml.corpus.TempEval2010Writer;
035    import org.cleartk.timeml.tlink.TemporalLinkEventToSubordinatedEventAnnotator;
036    import org.cleartk.timeml.type.TemporalLink;
037    import org.cleartk.token.stem.snowball.DefaultSnowballStemmer;
038    
039    /**
040     * TempEval 2010 task F: event to subordinated event
041     * 
042     * Best reported precision in TempEval 2010: 0.66
043     * 
044     * <br>
045     * Copyright (c) 2011, Regents of the University of Colorado <br>
046     * All rights reserved.
047     * 
048     * @author Steven Bethard
049     */
050    public class TempEval2010TaskF extends TempEval2010Main {
051    
052      public static void main(String[] args) throws Exception {
053        new TempEval2010TaskF().runMain(args);
054      }
055    
056      @Override
057      protected TempEval2010Evaluation getEvaluation(File trainDir, File testDir, File outputDir)
058          throws Exception {
059    
060        List<ModelInfo<TemporalLink>> infos = new ArrayList<ModelInfo<TemporalLink>>();
061        infos.add(new TemporalLinkModelInfo(TemporalLinkEventToSubordinatedEventAnnotator.FACTORY));
062    
063        return new TempEval2010Evaluation(
064            trainDir,
065            testDir,
066            outputDir,
067            Arrays.asList(
068                TempEval2010GoldAnnotator.PARAM_TEXT_VIEWS,
069                TempEval2010GoldAnnotator.PARAM_TIME_EXTENT_VIEWS,
070                TempEval2010GoldAnnotator.PARAM_TIME_ATTRIBUTE_VIEWS,
071                TempEval2010GoldAnnotator.PARAM_EVENT_EXTENT_VIEWS,
072                TempEval2010GoldAnnotator.PARAM_EVENT_ATTRIBUTE_VIEWS),
073            TempEval2010GoldAnnotator.PARAM_TEMPORAL_LINK_EVENT_TO_SUBORDINATED_EVENT_VIEWS,
074            TempEval2010Writer.PARAM_TEMPORAL_LINK_EVENT_TO_SUBORDINATED_EVENT_VIEW,
075            Arrays.asList(
076                DefaultSnowballStemmer.getDescription("English"),
077                PosTaggerAnnotator.getDescription(),
078                ParserAnnotator.getDescription()),
079            infos);
080      }
081    }
