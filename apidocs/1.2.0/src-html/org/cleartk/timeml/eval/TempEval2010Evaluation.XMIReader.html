076    
077      public static final String SYSTEM_VIEW_NAME = CAS.NAME_DEFAULT_SOFA;

079      private File trainDir;

095      public TempEval2010Evaluation(
096          File trainDir,
097          File testDir,
098          File outputDirectory,
099          List<String> goldAnnotatorParamsForViewsRequiredBySystem,
100          String goldAnnotatorParamForViewAnnotatedBySystem,
101          String timemlWriterParamForViewAnnotatedBySystem,
102          List<AnalysisEngineDescription> preprocessingAnnotators,
103          List<? extends ModelInfo<?>> modelInfos) throws Exception {
104        super(outputDirectory);
105    
106        this.trainDir = trainDir;
107        this.testDir = testDir;
108        this.dataDirs = Arrays.asList(trainDir, testDir);
109        this.goldAnnotatorParamsForViewsRequiredBySystem = goldAnnotatorParamsForViewsRequiredBySystem;
110        this.goldAnnotatorParamForViewAnnotatedBySystem = goldAnnotatorParamForViewAnnotatedBySystem;
111        this.timemlWriterParamForViewAnnotatedBySystem = timemlWriterParamForViewAnnotatedBySystem;
112        this.preprocessingAnnotators = preprocessingAnnotators;
113        this.modelInfos = modelInfos;
114      }
115    
116      @Override
117      protected CollectionReader getCollectionReader(List<String> items) throws Exception {
118        return TempEval2010CollectionReader.getCollectionReader(this.dataDirs, new HashSet<String>(
119            items));
120      }
121    
122      @Override
123      protected void train(CollectionReader collectionReader, File directory) throws Exception {
124    
125        // run the XMI reader and the classifier data writers
126        AggregateBuilder builder = new AggregateBuilder();
127        builder.add(AnalysisEngineFactory.createPrimitiveDescription(
128            XMIReader.class,
129            XMIAnnotator.PARAM_XMI_DIRECTORY,
130            this.getXMIDirectory(directory, Stage.TRAIN).getPath()));
131        for (ModelInfo<?> modelInfo : this.modelInfos) {
132          File outputDir = modelInfo.getModelSubdirectory(directory);
133          builder.add(modelInfo.modelFactory.getWriterDescription(outputDir));
134        }
135        SimplePipeline.runPipeline(collectionReader, builder.createAggregateDescription());
136    
137        // train the classifiers
138        for (ModelInfo<?> modelInfo : this.modelInfos) {
139          File modelDir = modelInfo.getModelSubdirectory(directory);
140          JarClassifierBuilder.trainAndPackage(modelDir, modelInfo.trainingArguments);
141        }
142    
143        // if building to the pre-defined training directory, clean up non-model files
144        for (ModelInfo<?> modelInfo : this.modelInfos) {
145          File modelDir = modelInfo.modelFactory.getTrainingDirectory();
146          if (modelDir.exists()) {
147            for (File file : modelDir.listFiles()) {
148              if (!file.isDirectory() && !file.getName().equals("model.jar")) {
149                file.delete();
150              }
151            }
152          }
153        }
154      }
155    
156      @Override
157      protected Map<ModelInfo<?>, AnnotationStatistics<String>> test(
158          CollectionReader collectionReader,
159          File directory) throws Exception {
160        // prepare the XMI reader, the classifiers and the TempEval writer
161        AggregateBuilder builder = new AggregateBuilder();
162        builder.add(AnalysisEngineFactory.createPrimitiveDescription(
163            XMIReader.class,
164            XMIAnnotator.PARAM_XMI_DIRECTORY,
165            this.getXMIDirectory(directory, Stage.TEST).getPath()));
166        for (ModelInfo<?> modelInfo : this.modelInfos) {
167          File modelFile = new File(modelInfo.getModelSubdirectory(directory), "model.jar");
168          builder.add(modelInfo.modelFactory.getAnnotatorDescription(modelFile.getPath()));
169        }
170        builder.add(AnalysisEngineFactory.createPrimitiveDescription(
171            TempEval2010Writer.class,
172            TempEval2010Writer.PARAM_OUTPUT_DIRECTORY,
173            new File(directory, "eval").getPath(),
174            TempEval2010Writer.PARAM_TEXT_VIEW,
175            SYSTEM_VIEW_NAME,
176            this.timemlWriterParamForViewAnnotatedBySystem,
177            SYSTEM_VIEW_NAME));
178    
179        // create statistics for each feature that is classified
180        Map<ModelInfo<?>, AnnotationStatistics<String>> modelInfoToStatistics;
181        modelInfoToStatistics = new HashMap<ModelInfo<?>, AnnotationStatistics<String>>();
182        for (ModelInfo<?> modelInfo : this.modelInfos) {
183          modelInfoToStatistics.put(modelInfo, new AnnotationStatistics<String>());
184        }
185    
186        // gather statistics over all the CASes in the test set
187        AnalysisEngine engine = builder.createAggregate();
188        for (JCas jCas : new JCasIterable(collectionReader, engine)) {
189          JCas goldView = jCas.getView(GOLD_VIEW_NAME);
190          JCas systemView = jCas.getView(SYSTEM_VIEW_NAME);
191          for (ModelInfo<?> modelInfo : this.modelInfos) {
192            AnnotationStatistics<String> statistics = modelInfoToStatistics.get(modelInfo);
193            modelInfo.updateStatistics(statistics, goldView, systemView);
194          }
195        }
196        engine.collectionProcessComplete();
197        return modelInfoToStatistics;
198      }
199    
200      private enum Stage {
201        TRAIN, TEST
202      }
203    
204      private File getXMIDirectory(File directory, Stage stage) throws Exception {
205        int dotIndex = Math.max(0, this.goldAnnotatorParamForViewAnnotatedBySystem.lastIndexOf('.'));
206        String name = this.goldAnnotatorParamForViewAnnotatedBySystem.substring(dotIndex + 1);
207        File xmiDirectory = new File(new File(new File(directory, "xmi"), name), stage.toString());
208    
209        // create XMIs if necessary
210        if (!xmiDirectory.exists()) {
211          Set<String> fileNames = new HashSet<String>();
212          fileNames.addAll(TempEval2010CollectionReader.getAnnotatedFileNames(this.trainDir));
213          fileNames.addAll(TempEval2010CollectionReader.getAnnotatedFileNames(this.testDir));
214          CollectionReader reader = TempEval2010CollectionReader.getCollectionReader(
215              this.dataDirs,
216              fileNames);
217    
218          List<String> viewParams = Arrays.asList(
219              TempEval2010GoldAnnotator.PARAM_TEXT_VIEWS,
220              TempEval2010GoldAnnotator.PARAM_DOCUMENT_CREATION_TIME_VIEWS,
221              TempEval2010GoldAnnotator.PARAM_TIME_EXTENT_VIEWS,
222              TempEval2010GoldAnnotator.PARAM_TIME_ATTRIBUTE_VIEWS,
223              TempEval2010GoldAnnotator.PARAM_EVENT_EXTENT_VIEWS,
224              TempEval2010GoldAnnotator.PARAM_EVENT_ATTRIBUTE_VIEWS,
225              TempEval2010GoldAnnotator.PARAM_TEMPORAL_LINK_EVENT_TO_DOCUMENT_CREATION_TIME_VIEWS,
226              TempEval2010GoldAnnotator.PARAM_TEMPORAL_LINK_EVENT_TO_SAME_SENTENCE_TIME_VIEWS,
227              TempEval2010GoldAnnotator.PARAM_TEMPORAL_LINK_EVENT_TO_SUBORDINATED_EVENT_VIEWS,
228              TempEval2010GoldAnnotator.PARAM_TEMPORAL_LINK_MAIN_EVENT_TO_NEXT_SENTENCE_MAIN_EVENT_VIEWS);
229    
230          // determine the view parameters for the gold annotator
231          List<Object> goldAnnotatorParams = new ArrayList<Object>();
232          for (String viewParam : viewParams) {
233            goldAnnotatorParams.add(viewParam);
234            String[] paramValue;
235            switch (stage) {
236              case TRAIN:
237                // during training, put all required annotations, and the annotations from which the
238                // models will be trained, in the system view
239                paramValue = this.goldAnnotatorParamsForViewsRequiredBySystem.contains(viewParam)
240                    || viewParam.equals(this.goldAnnotatorParamForViewAnnotatedBySystem)
241                    ? new String[] { SYSTEM_VIEW_NAME }
242                    : new String[] {};
243                break;
244    
245              case TEST:
246                // during testing, put required annotation in both views, and the annotations which the
247                // model is supposed to predict only in the gold view
248                if (this.goldAnnotatorParamsForViewsRequiredBySystem.contains(viewParam)) {
249                  paramValue = new String[] { SYSTEM_VIEW_NAME, GOLD_VIEW_NAME };
250                } else if (viewParam.equals(this.goldAnnotatorParamForViewAnnotatedBySystem)) {
251                  paramValue = new String[] { GOLD_VIEW_NAME };
252                } else {
253                  paramValue = new String[] {};
254                }
255                break;
256    
257              default:
258                throw new IllegalArgumentException();
259            }
260            goldAnnotatorParams.add(paramValue);
261          }
262    
263          // run the gold annotator, the preprocessing annotators, and the XMI writer
264          AggregateBuilder builder = new AggregateBuilder();
265          builder.add(AnalysisEngineFactory.createPrimitiveDescription(
266              TempEval2010GoldAnnotator.class,
267              goldAnnotatorParams.toArray()));
268          for (AnalysisEngineDescription desc : this.preprocessingAnnotators) {
269            builder.add(desc);
270          }
271          builder.add(AnalysisEngineFactory.createPrimitiveDescription(
272              XMIWriter.class,
273              XMIAnnotator.PARAM_XMI_DIRECTORY,
274              xmiDirectory.getPath()));
275          SimplePipeline.runPipeline(reader, builder.createAggregateDescription());
276        }
277    
278        return xmiDirectory;
279      }

281      public static abstract class XMIAnnotator extends JCasAnnotator_ImplBase {
282    
283        @ConfigurationParameter(mandatory = true)
284        protected File xmiDirectory;
285    
286        public static final String PARAM_XMI_DIRECTORY = ConfigurationParameterFactory.createConfigurationParameterName(
287            XMIAnnotator.class,
288            "xmiDirectory");
289    
290        protected File getFile(JCas jCas) throws AnalysisEngineProcessException {
291          return new File(this.xmiDirectory, ViewURIUtil.getURI(jCas).getFragment() + ".xmi");
292        }
293    
294      }
295    
296      public static class XMIReader extends XMIAnnotator {
297    
298        @Override
299        public void process(JCas jCas) throws AnalysisEngineProcessException {
300          try {
301            FileInputStream stream = new FileInputStream(this.getFile(jCas));
302            try {
303              XmiCasDeserializer.deserialize(stream, jCas.getCas());
304            } finally {
305              stream.close();
306            }
307          } catch (SAXException e) {
308            throw new AnalysisEngineProcessException(e);
309          } catch (IOException e) {
310            throw new AnalysisEngineProcessException(e);
311          }
312        }
313      }
314    
315      public static class XMIWriter extends XMIAnnotator {
316    
317        @Override
318        public void initialize(UimaContext context) throws ResourceInitializationException {
319          super.initialize(context);
320          if (!this.xmiDirectory.exists()) {
321            this.xmiDirectory.mkdirs();
322          }
323        }
324    
325        @Override
326        public void process(JCas jCas) throws AnalysisEngineProcessException {
327          XmiCasSerializer ser = new XmiCasSerializer(jCas.getTypeSystem());
328          try {
329            FileOutputStream stream = new FileOutputStream(this.getFile(jCas));
330            try {
331              ser.serialize(jCas.getCas(), new XMLSerializer(stream, false).getContentHandler());
332            } finally {
333              stream.close();
334            }
335          } catch (SAXException e) {
336            throw new AnalysisEngineProcessException(e);
337          } catch (IOException e) {
338            throw new AnalysisEngineProcessException(e);
339          }
340        }
341      }
