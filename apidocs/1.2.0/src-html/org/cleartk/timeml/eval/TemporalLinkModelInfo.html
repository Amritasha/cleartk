049    public class TemporalLinkModelInfo extends ModelInfo<TemporalLink> {
050    
051      public TemporalLinkModelInfo(CleartkInternalModelFactory modelFactory, String[] trainingArguments) {
052        super(
053            TemporalLink.class,
054            "relationType",
055            TemporalLinkSpanAndRelation.ANNOTATION_CONVERTER,
056            modelFactory,
057            trainingArguments);
058      }
059    
060      public TemporalLinkModelInfo(CleartkInternalModelFactory modelFactory) {
061        this(modelFactory, new String[0]);
062      }
063    
064      @Override
065      public void updateStatistics(
066          AnnotationStatistics<String> statistics,
067          JCas goldView,
068          JCas systemView) {
069        // restrict evaluation to only the TLINKs that were present in the gold data
070        Collection<TemporalLink> goldTlinks = JCasUtil.select(goldView, this.annotatedClass);
071        Collection<TemporalLink> systemTlinks = JCasUtil.select(systemView, this.annotatedClass);
072        Set<TemporalLinkSpan> goldSpans = new HashSet<TemporalLinkSpan>();
073        for (TemporalLink tlink : goldTlinks) {
074          goldSpans.add(new TemporalLinkSpan(tlink));
075        }
076        List<TemporalLink> systemTlinksWithGoldSpans = new ArrayList<TemporalLink>();
077        for (TemporalLink tlink : systemTlinks) {
078          if (goldSpans.contains(new TemporalLinkSpan(tlink))) {
079            systemTlinksWithGoldSpans.add(tlink);
080          }
081        }
082        statistics.add(
083            goldTlinks,
084            systemTlinksWithGoldSpans,
085            this.annotationConverter,
086            AnnotationStatistics.<TemporalLink> annotationToFeatureValue(this.annotatedFeatureName));
087      }
088    
089      private static class TemporalLinkSpan {
090    
091        private int sourceBegin;
092    
093        private int sourceEnd;
094    
095        private int targetBegin;
096    
097        private int targetEnd;
098    
099        public TemporalLinkSpan(TemporalLink tlink) {
100          Anchor source = tlink.getSource();
101          Anchor target = tlink.getTarget();
102          this.sourceBegin = source.getBegin();
103          this.sourceEnd = source.getEnd();
104          this.targetBegin = target.getBegin();
105          this.targetEnd = target.getEnd();
106        }
107    
108        @Override
109        public int hashCode() {
110          return Objects.hashCode(this.sourceBegin, this.sourceEnd, this.targetBegin, this.targetEnd);
111        }
112    
113        @Override
114        public boolean equals(Object obj) {
115          if (this == obj) {
116            return true;
117          }
118          if (obj == null || this.getClass() != obj.getClass()) {
119            return false;
120          }
121          TemporalLinkSpan that = (TemporalLinkSpan) obj;
122          return this.sourceBegin == that.sourceBegin && this.sourceEnd == that.sourceEnd
123              && this.targetBegin == that.targetBegin && this.targetEnd == that.targetEnd;
124        }
125    
126      }
127    
128      private static class TemporalLinkSpanAndRelation {
129        public static final Function<TemporalLink, TemporalLinkSpanAndRelation> ANNOTATION_CONVERTER = new Function<TemporalLink, TemporalLinkSpanAndRelation>() {
130          @Override
131          public TemporalLinkSpanAndRelation apply(TemporalLink tlink) {
132            return new TemporalLinkSpanAndRelation(tlink);
133          }
134        };
135    
136        private TemporalLinkSpan span;
137    
138        private String relationType;
139    
140        public TemporalLinkSpanAndRelation(TemporalLink tlink) {
141          this.span = new TemporalLinkSpan(tlink);
142          this.relationType = tlink.getRelationType();
143        }
144    
145        @Override
146        public int hashCode() {
147          return Objects.hashCode(this.span, this.relationType);
148        }
149    
150        @Override
151        public boolean equals(Object obj) {
152          if (this == obj) {
153            return true;
154          }
155          if (obj == null || this.getClass() != obj.getClass()) {
156            return false;
157          }
158          TemporalLinkSpanAndRelation that = (TemporalLinkSpanAndRelation) obj;
159          return Objects.equal(this.span, that.span)
160              && Objects.equal(this.relationType, that.relationType);
161        }
162    
163      }
