063    public class TemporalLinkEventToSubordinatedEventAnnotator extends
064        TemporalLinkAnnotator_ImplBase<Event, Event> {
065    
066      public static final CleartkInternalModelFactory FACTORY = new CleartkInternalModelFactory() {
067        @Override
068        public Class<?> getAnnotatorClass() {
069          return TemporalLinkEventToSubordinatedEventAnnotator.class;
070        }
071    
072        @Override
073        public Class<?> getDataWriterClass() {
074          return MaxentStringOutcomeDataWriter.class;
075        }
076    
077        @Override
078        public AnalysisEngineDescription getBaseDescription() throws ResourceInitializationException {
079          return AnalysisEngineFactory.createPrimitiveDescription(TemporalLinkEventToSubordinatedEventAnnotator.class);
080        }
081      };
082    
083      public TemporalLinkEventToSubordinatedEventAnnotator() {
084        super(Event.class, Event.class, "BEFORE", "OVERLAP", "AFTER");
085      }
086    
087      private static final Pattern SUBORDINATE_PATH_PATTERN = Pattern.compile("^(VP>|ADJP>|NP>)?(VP|ADJP|S|SBAR)(<(S|SBAR|PP))*((<VP|<ADJP)*|(<NP)*)$");
088    
089      @Override
090      public void initialize(UimaContext context) throws ResourceInitializationException {
091        super.initialize(context);
092    
093        List<SimpleFeatureExtractor> extractors = Arrays.<SimpleFeatureExtractor> asList(
094            new TypePathExtractor(Event.class, "tense"),
095            new TypePathExtractor(Event.class, "aspect"),
096            new TypePathExtractor(Event.class, "eventClass"),
097            new SyntacticFirstChildOfGrandparentOfLeafExtractor());
098        this.setSourceExtractors(extractors);
099        this.setTargetExtractors(extractors);
100        this.setBetweenExtractors(Arrays.asList(
101            new SyntacticLeafToLeafPathPartsExtractor(),
102            new CleartkExtractor(Token.class, new CoveredTextExtractor(), new Bag(new Covered()))));
103      }
104    
105      @Override
106      public void process(JCas jCas) throws AnalysisEngineProcessException {
107        Map<Event, Map<Event, TemporalLink>> links = this.getLinks(jCas);
108    
109        for (Sentence sentence : JCasUtil.select(jCas, Sentence.class)) {
110          for (Event source : JCasUtil.selectCovered(jCas, Event.class, sentence)) {
111            for (Event target : this.getSubordinateEvents(jCas, source, sentence)) {
112              this.processLink(source, target, links, jCas);
113            }
114          }
115        }
116    
117        this.logSkippedLinks(jCas, links);
118      }
119    
120      private List<Event> getSubordinateEvents(JCas jCas, Event source, Sentence sentence) {
121        List<Event> targets = new ArrayList<Event>();
122        TreebankNode sourceNode = TreebankNodeUtil.selectMatchingLeaf(jCas, source);
123        for (Event target : JCasUtil.selectCovered(jCas, Event.class, sentence)) {
124          TreebankNode targetNode = TreebankNodeUtil.selectMatchingLeaf(jCas, target);
125          if (sourceNode != null && targetNode != null) {
126            String path = noLeavesPath(TreebankNodeUtil.getPath(sourceNode, targetNode));
127            if (SUBORDINATE_PATH_PATTERN.matcher(path).matches()) {
128              targets.add(target);
129            }
130          }
131        }
132        return targets;
133      }
