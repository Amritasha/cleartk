051    /**
052     * Command line utility for annotating plain text files with TimeML annotations. Usage:
053     * 
054     * <pre>
055     * java org.cleartk.timeml.TimeMLAnnotate input-file-or-dir [output-dir]
056     * </pre>
057     * 
058     * <br>
059     * Copyright (c) 2011, Regents of the University of Colorado <br>
060     * All rights reserved.
061     * 
062     * @author Steven Bethard
063     */
064    public class TimeMLAnnotate {
065    
066      private static void error(String message) throws Exception {
067        Logger logger = UimaContextFactory.createUimaContext().getLogger();
068        logger.log(
069            Level.SEVERE,
070            String.format(
071                "%s\nusage: java %s input-file-or-dir [output-dir]",
072                TimeMLAnnotate.class.getName(),
073                message));
074        System.exit(1);
075      }
076    
077      public static void main(String... args) throws Exception {
078        // check arguments
079        if (args.length != 1 && args.length != 2) {
080          error("wrong number of arguments");
081        } else if (!new File(args[0]).exists()) {
082          error("file or directory not found: " + args[0]);
083        }
084    
085        // parse arguments
086        String inputFileOrDir = args[0];
087        File outputDir = new File(args.length == 2 ? args[1] : ".");
088        if (!outputDir.exists()) {
089          outputDir.mkdirs();
090        }
091    
092        // run the components on the selected documents
093        SimplePipeline.runPipeline(
094            FilesCollectionReader.getCollectionReader(inputFileOrDir),
095            SentenceAnnotator.getDescription(),
096            TokenAnnotator.getDescription(),
097            PosTaggerAnnotator.getDescription(),
098            DefaultSnowballStemmer.getDescription("English"),
099            ParserAnnotator.getDescription(),
100            TimeAnnotator.FACTORY.getAnnotatorDescription(),
101            TimeTypeAnnotator.FACTORY.getAnnotatorDescription(),
102            EventAnnotator.FACTORY.getAnnotatorDescription(),
103            EventTenseAnnotator.FACTORY.getAnnotatorDescription(),
104            EventAspectAnnotator.FACTORY.getAnnotatorDescription(),
105            EventClassAnnotator.FACTORY.getAnnotatorDescription(),
106            EventPolarityAnnotator.FACTORY.getAnnotatorDescription(),
107            EventModalityAnnotator.FACTORY.getAnnotatorDescription(),
108            // TemporalLinkEventToDocumentCreationTimeAnnotator.FACTORY.getAnnotatorDescription(),
109            TemporalLinkEventToSameSentenceTimeAnnotator.FACTORY.getAnnotatorDescription(),
110            TemporalLinkMainEventToNextSentenceMainEventAnnotator.FACTORY.getAnnotatorDescription(),
111            // TemporalLinkEventToSubordinatedEventAnnotator.FACTORY.getAnnotatorDescription(),
112            VerbClauseTemporalAnnotator.FACTORY.getAnnotatorDescription(),
113            TimeMLWriter.getDescription(outputDir.getPath()));
114      }
115    }
