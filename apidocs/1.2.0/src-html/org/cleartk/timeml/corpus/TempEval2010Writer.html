193        this.baseWriter = this.createWriter(TimeMLViewName.TEMPEVAL_BASE_SEGMENTATION, this.textView);
194        this.dctWriter = this.createWriter(TimeMLViewName.TEMPEVAL_DCT, this.documentCreationTimeView);
195        this.timexExtentWriter = this.createWriter(
196            TimeMLViewName.TEMPEVAL_TIMEX_EXTENTS,
197            this.timeExtentView);
198        this.timexAttributeWriter = this.createWriter(
199            TimeMLViewName.TEMPEVAL_TIMEX_ATTRIBUTES,
200            this.timeAttributeView);
201        this.eventExtentWriter = this.createWriter(
202            TimeMLViewName.TEMPEVAL_EVENT_EXTENTS,
203            this.eventExtentView);
204        this.eventAttributeWriter = this.createWriter(
205            TimeMLViewName.TEMPEVAL_EVENT_ATTRIBUTES,
206            this.eventAttributeView);
207        this.tlinkDCTEventWriter = this.createWriter(
208            TimeMLViewName.TEMPEVAL_TLINK_DCT_EVENT,
209            this.temporalLinkEventToDocumentCreationTimeView);
210        this.tlinkTimexEventWriter = this.createWriter(
211            TimeMLViewName.TEMPEVAL_TLINK_TIMEX_EVENT,

221      @Override
222      public void process(JCas jCas) throws AnalysisEngineProcessException {
223        // determine the filename
224        URI uri = ViewURIUtil.getURI(jCas);
225        String fileName = uri.getFragment();
226        if (fileName == null) {
227          fileName = new File(uri.getPath()).getName();
228        }
229    
230        // get the view with text, sentences and tokens
231        JCas textJCas = JCasUtil.getView(jCas, this.textView, false);
232    
233        // write the document creation time
234        if (this.documentCreationTimeView != null) {
235          JCas dctJCas = JCasUtil.getView(jCas, this.documentCreationTimeView, false);
236          for (DocumentCreationTime time : JCasUtil.select(dctJCas, DocumentCreationTime.class)) {
237            this.write(this.dctWriter, fileName, time.getValue().replaceAll("-", ""));
238          }
239        }
240    
241        // align tokens to times
242        Map<Token, Time> tokenTimeExtents = new HashMap<Token, Time>();
243        if (this.timeExtentView != null) {
244          JCas timeExtentJCas = JCasUtil.getView(jCas, this.timeExtentView, false);
245          for (Time time : JCasUtil.select(timeExtentJCas, Time.class)) {
246            for (Token token : JCasUtil.selectCovered(textJCas, Token.class, time)) {
247              tokenTimeExtents.put(token, time);
248            }
249          }
250        }
251        Map<Token, Time> tokenTimeAttributes = new HashMap<Token, Time>();
252        if (this.timeAttributeView != null) {
253          JCas timeAttributeJCas = JCasUtil.getView(jCas, this.timeAttributeView, false);
254          for (Time time : JCasUtil.select(timeAttributeJCas, Time.class)) {
255            for (Token token : JCasUtil.selectCovered(textJCas, Token.class, time)) {
256              tokenTimeAttributes.put(token, time);
257            }
258          }
259        }
260    
261        // align tokens to events
262        Map<Token, Event> tokenEventExtents = new HashMap<Token, Event>();
263        if (this.eventExtentView != null) {
264          JCas eventExtentJCas = JCasUtil.getView(jCas, this.eventExtentView, false);
265          for (Event event : JCasUtil.select(eventExtentJCas, Event.class)) {
266            for (Token token : JCasUtil.selectCovered(textJCas, Token.class, event)) {
267              tokenEventExtents.put(token, event);
268            }
269          }
270        }
271    
272        Map<Token, Event> tokenEventAttributes = new HashMap<Token, Event>();
273        if (this.eventAttributeView != null) {
274          JCas eventAttributeJCas = JCasUtil.getView(jCas, this.eventAttributeView, false);
275          for (Event event : JCasUtil.select(eventAttributeJCas, Event.class)) {
276            for (Token token : JCasUtil.selectCovered(textJCas, Token.class, event)) {
277              tokenEventAttributes.put(token, event);
278            }
279          }
280        }
281    
282        // walk through tokens by sentence, writing tokens, times, events, etc.
283        int sentIndex = -1;
284        for (Sentence sentence : JCasUtil.select(textJCas, Sentence.class)) {
285          sentIndex += 1;
286          int tokenIndex = -1;
287          for (Token token : JCasUtil.selectCovered(textJCas, Token.class, sentence)) {
288            tokenIndex += 1;
289    
290            // write the token to the segmentation file
291            this.write(this.baseWriter, fileName, sentIndex, tokenIndex, token.getCoveredText());
292    
293            // write the time extent and attributes
294            this.writeAnchors(
295                this.timexExtentWriter,
296                this.timexAttributeWriter,
297                tokenTimeExtents,
298                tokenTimeAttributes,
299                "timex3",
300                token,
301                fileName,
302                sentIndex,
303                tokenIndex,
304                new AttributeGetter<Time>() {
305                  @Override
306                  public List<Attribute> getAttributes(Time time) {
307                    Attribute value = new Attribute("value", time.getValue());
308                    Attribute type = new Attribute("type", time.getTimeType());
309                    return Arrays.asList(value, type);
310                  }
311                });
312    
313            // write the event extent and attributes
314            this.writeAnchors(
315                this.eventExtentWriter,
316                this.eventAttributeWriter,
317                tokenEventExtents,
318                tokenEventAttributes,
319                "event",
320                token,
321                fileName,
322                sentIndex,
323                tokenIndex,
324                new AttributeGetter<Event>() {
325                  @Override
326                  public List<Attribute> getAttributes(Event event) {
327                    Attribute polarity = new Attribute("polarity", event.getPolarity());
328                    Attribute modality = new Attribute("modality", event.getModality());
329                    Attribute pos = new Attribute("pos", event.getPos());
330                    Attribute tense = new Attribute("tense", event.getTense());
331                    Attribute aspect = new Attribute("aspect", event.getAspect());
332                    Attribute eventClass = new Attribute("class", event.getEventClass());
333                    return Arrays.asList(polarity, modality, pos, tense, aspect, eventClass);
334                  }
335                });
336          }
337        }
338    
339        // write the temporal links
340        this.writeTemporalLinks(
341            this.tlinkDCTEventWriter,
342            jCas,
343            this.temporalLinkEventToDocumentCreationTimeView,
344            fileName);
345        this.writeTemporalLinks(
346            this.tlinkTimexEventWriter,
347            jCas,
348            this.temporalLinkEventToSameSentenceTimeView,
349            fileName);
350        this.writeTemporalLinks(
351            this.tlinkSubordinatedEventsWriter,
352            jCas,
353            this.temporalLinkEventToSubordinatedEventView,
354            fileName);
355        this.writeTemporalLinks(
356            this.tlinkMainEventsWriter,
357            jCas,
358            this.temporalLinkMainEventToNextSentenceMainEventView,
359            fileName);
360      }

378      private PrintWriter createWriter(String tabFileName, String viewParam)
379          throws ResourceInitializationException {
380        PrintWriter writer;
381        if (viewParam != null) {
382          try {
383            writer = new PrintWriter(new FileWriter(new File(this.outputDirectory, tabFileName)));
384          } catch (IOException e) {
385            throw new ResourceInitializationException(e);
386          }
387        } else {
388          writer = new PrintWriter(new OutputStream() {
389            @Override
390            public void write(int b) throws IOException {
391              // do nothing
392            }
393          });
394        }
395        this.writers.add(writer);
396        return writer;
397      }
398    
399      private void write(PrintWriter writer, Object... columns) {
400        writer.println(Joiner.on('\t').join(columns));
401      }

403      private static class Attribute {
404        public String name;
405    
406        public Object value;
407    
408        public Attribute(String name, Object value) {
409          this.name = name;
410          this.value = value;
411        }
412      }
413    
414      private static interface AttributeGetter<T extends Anchor> {
415        public List<Attribute> getAttributes(T anchor);
416      }
417    
418      private <T extends Anchor> void writeAnchors(
419          PrintWriter extentWriter,
420          PrintWriter attrWriter,
421          Map<Token, T> tokenAnchorExtents,
422          Map<Token, T> tokenAnchorAttributes,
423          String anchorType,
424          Token token,
425          String fileName,
426          int sentIndex,
427          int tokenIndex,
428          AttributeGetter<T> attributeGetter) {
429        T anchor = tokenAnchorExtents.get(token);
430        if (anchor != null) {
431          String id = anchor.getId();
432          this.write(extentWriter, fileName, sentIndex, tokenIndex, anchorType, id, "1");
433        }
434        anchor = tokenAnchorAttributes.get(token);
435        if (anchor != null) {
436          String id = anchor.getId();
437          boolean isFirstToken = token.getBegin() == anchor.getBegin();
438          if (isFirstToken) {
439            for (Attribute attr : attributeGetter.getAttributes(anchor)) {
440              if (attr.value != null) {
441                this.write(
442                    attrWriter,
443                    fileName,
444                    sentIndex,
445                    tokenIndex,
446                    anchorType,
447                    id,
448                    "1",
449                    attr.name,
450                    attr.value);
451              }
452            }
453          }
454        }
455      }
456    
457      private void writeTemporalLinks(PrintWriter writer, JCas jCas, String viewName, String fileName) {
458        if (viewName != null) {
459          JCas view = JCasUtil.getView(jCas, viewName, false);
460          for (TemporalLink tlink : JCasUtil.select(view, TemporalLink.class)) {
461            String relation = tlink.getRelationType();
462            if (relation == null) {
463              relation = "NONE";
464            }
465            this.write(writer, fileName, tlink.getSource().getId(), tlink.getTarget().getId(), relation);
466          }
467        }
468      }
