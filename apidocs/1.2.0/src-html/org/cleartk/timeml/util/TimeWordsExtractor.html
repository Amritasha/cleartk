048    public class TimeWordsExtractor implements SimpleFeatureExtractor {
049    
050      private Map<String, Set<String>> groupedWords;
051    
052      public TimeWordsExtractor() {
053        this.groupedWords = new HashMap<String, Set<String>>();
054        this.groupedWords.put("Now", new HashSet<String>(Arrays.asList("now", "current", "currently")));
055        this.groupedWords.put(
056            "TimeOfDay",
057            new HashSet<String>(Arrays.asList(
058                "morning",
059                "noon",
060                "afternoon",
061                "evening",
062                "night",
063                "midnight")));
064        this.groupedWords.put(
065            "Day",
066            new HashSet<String>(Arrays.asList(
067                "Monday",
068                "Tuesday",
069                "Wednesday",
070                "Thursday",
071                "Friday",
072                "Saturday",
073                "Sunday",
074                "today",
075                "tomorrow",
076                "yesterday")));
077        this.groupedWords.put(
078            "Month",
079            new HashSet<String>(Arrays.asList(
080                "January",
081                "February",
082                "March",
083                "April",
084                "May",
085                "June",
086                "July",
087                "August",
088                "September",
089                "October",
090                "November",
091                "December")));
092        this.groupedWords.put(
093            "Season",
094            new HashSet<String>(Arrays.asList("spring", "summer", "fall", "autumn", "winter")));
095        this.groupedWords.put(
096            "TimeDuration",
097            new HashSet<String>(
098                Arrays.asList("minute", "minutes", "second", "seconds", "hour", "hours")));
099        this.groupedWords.put(
100            "DateDuration",
101            new HashSet<String>(Arrays.asList(
102                "day",
103                "days",
104                "week",
105                "weeks",
106                "month",
107                "months",
108                "quarter",
109                "quarters",
110                "season",
111                "seasons",
112                "year",
113                "years",
114                "decade",
115                "decades",
116                "century",
117                "centuries")));
118      }
119    
120      @Override
121      public List<Feature> extract(JCas view, Annotation focusAnnotation) {
122        List<String> types = new ArrayList<String>();
123        String[] words = focusAnnotation.getCoveredText().split("\\W+");
124        for (String word : words) {
125          for (String group : this.groupedWords.keySet()) {
126            if (this.groupedWords.get(group).contains(word)) {
127              types.add(group);
128            }
129          }
130          if (word.matches("^\\d{4}$")) {
131            types.add("Year");
132          }
133        }
134        if (types.isEmpty()) {
135          for (String word : words) {
136            for (String group : this.groupedWords.keySet()) {
137              if (this.groupedWords.get(group).contains(word.toLowerCase())) {
138                types.add(group + "Lower");
139              }
140            }
141          }
142        }
143        if (types.isEmpty()) {
144          types.add("None");
145        }
146        return Arrays.asList(new Feature("TimeType", Joiner.on('_').join(types)));
147      }
148    
149    }
