024    package org.cleartk.timeml.event;
025    
026    import org.apache.uima.UimaContext;
027    import org.apache.uima.analysis_engine.AnalysisEngineDescription;
028    import org.apache.uima.resource.ResourceInitializationException;
029    import org.cleartk.classifier.feature.extractor.CleartkExtractor;
030    import org.cleartk.classifier.feature.extractor.CleartkExtractor.Bag;
031    import org.cleartk.classifier.feature.extractor.CleartkExtractor.Covered;
032    import org.cleartk.classifier.feature.extractor.simple.TypePathExtractor;
033    import org.cleartk.classifier.opennlp.MaxentStringOutcomeDataWriter;
034    import org.cleartk.timeml.type.Event;
035    import org.cleartk.timeml.util.CleartkInternalModelFactory;
036    import org.cleartk.token.type.Token;
037    import org.uimafit.factory.AnalysisEngineFactory;
038    
039    /**
040     * <br>
041     * Copyright (c) 2010, Regents of the University of Colorado <br>
042     * All rights reserved.
043     * 
044     * Annotator for the "class" attribute of TimeML EVENTs.
045     * 
046     * @author Steven Bethard
047     */
048    public class EventClassAnnotator extends EventAttributeAnnotator<String> {
049    
050      public static final CleartkInternalModelFactory FACTORY = new CleartkInternalModelFactory() {
051        @Override
052        public Class<?> getAnnotatorClass() {
053          return EventClassAnnotator.class;
054        }
055    
056        @Override
057        public Class<?> getDataWriterClass() {
058          return MaxentStringOutcomeDataWriter.class;
059        }
060    
061        @Override
062        public AnalysisEngineDescription getBaseDescription() throws ResourceInitializationException {
063          return AnalysisEngineFactory.createPrimitiveDescription(EventClassAnnotator.class);
064        }
065      };
066    
067      @Override
068      public void initialize(UimaContext context) throws ResourceInitializationException {
069        super.initialize(context);
070        this.eventFeatureExtractors.add(new CleartkExtractor(Token.class, new TypePathExtractor(
071            Token.class,
072            "stem"), new Bag(new Covered())));
073        this.eventFeatureExtractors.add(new CleartkExtractor(Token.class, new TypePathExtractor(
074            Token.class,
075            "pos"), new Bag(new Covered())));
076      }
077    
078      @Override
079      protected String getDefaultValue() {
080        return "OCCURRENCE";
081      }
082    
083      @Override
084      protected String getAttribute(Event event) {
085        return event.getEventClass();
086      }
087    
088      @Override
089      protected void setAttribute(Event event, String value) {
090        event.setEventClass(value);
091      }
092    }
