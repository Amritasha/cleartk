046    @Deprecated
047    public class DefaultChunkLabeler extends ChunkLabeler_ImplBase {
048    
049      public static final String PARAM_CHUNK_LABEL_FEATURE_NAME = ConfigurationParameterFactory.createConfigurationParameterName(
050          DefaultChunkLabeler.class,
051          "chunkLabelFeatureName");
052    
053      private static final String CHUNK_LABEL_FEATURE_DESCRIPTION = "names  the feature of the type system chunk type that provides a label for each "
054          + "chunk. The feature is queried and the value of the feature is used as the label for the chunk.  If this parameter has no value, then the name of the "
055          + "chunk type will be used as a label. For example, if the value of the parameter 'org.cleartk.chunk.ChunkLabeler_ImplBase.chunkAnnotationClassName' is 'org.cleartk.type.Chunk', "
056          + "then a good value for this parameter would be 'chunkType'.  This would result in labels corresponding to the values found in the type system feature "
057          + "chunkType.  If the value of the parameter ''org.cleartk.chunk.ChunkLabeler_ImplBase.chunkAnnotationClassName'' is 'org.cleartk.type.Chunk' and no value is given for "
058          + "this parameter, then the label will always be 'Chunk'";
059    
060      @ConfigurationParameter(description = CHUNK_LABEL_FEATURE_DESCRIPTION)
061      private String chunkLabelFeatureName;
062    
063      public void setChunkLabelFeatureName(String chunkLabelFeatureName) {
064        this.chunkLabelFeatureName = chunkLabelFeatureName;
065      }
066    
067      private org.apache.uima.cas.Feature chunkLabelFeature;
068    
069      Constructor<? extends Annotation> chunkAnnotationConstructor;
070    
071      @Override
072      public void initialize(UimaContext context) throws ResourceInitializationException {
073        super.initialize(context);
074    
075        if (chunkLabelFeatureName != null) {
076          chunkLabelFeatureName = chunkAnnotationClass.getCanonicalName() + ":" + chunkLabelFeatureName;
077        }
078    
079        try {
080          chunkAnnotationConstructor = chunkAnnotationClass.getConstructor(new Class[] {
081              JCas.class,
082              java.lang.Integer.TYPE,
083              java.lang.Integer.TYPE });
084        } catch (Exception e) {
085          throw new ResourceInitializationException(e);
086        }
087      }
088    
089      protected void initializeTypes(JCas jCas) throws AnalysisEngineProcessException {
090        super.initializeTypes(jCas);
091        try {
092          if (chunkLabelFeatureName != null) {
093            chunkLabelFeature = jCas.getTypeSystem().getFeatureByFullName(chunkLabelFeatureName);
094            if (chunkLabelFeature == null)
095              throw new AnalysisEngineProcessException("type feature for name '"
096                  + chunkLabelFeatureName + "' not found.  ", null);
097          }
098        } catch (Exception e) {
099          throw new AnalysisEngineProcessException(e);
100        }
101      }
102    
103      public Annotation createChunk(
104          JCas jCas,
105          List<? extends Annotation> labeledAnnotations,
106          String label) throws AnalysisEngineProcessException {
107        try {
108          int begin = labeledAnnotations.get(0).getBegin();
109          int end = labeledAnnotations.get(labeledAnnotations.size() - 1).getEnd();
110          Annotation annotation = chunkAnnotationConstructor.newInstance(new Object[] {
111              jCas,
112              begin,
113              end });
114    
115          if (chunkLabelFeature != null)
116            annotation.setFeatureValueFromString(chunkLabelFeature, label);
117          annotation.addToIndexes();
118          return annotation;
119        } catch (Exception e) {
120          throw new AnalysisEngineProcessException(e);
121        }
122      }
123    
124      @Override
125      public String getChunkLabel(JCas jCas, Annotation chunkAnnotation)
126          throws AnalysisEngineProcessException {
127        if (!typesInitialized)
128          initializeTypes(jCas);
129    
130        if (chunkLabelFeature != null)
131          return chunkAnnotation.getFeatureValueAsString(chunkLabelFeature);
132        else
133          return chunkAnnotationClass.getSimpleName();
134      }
135    }
