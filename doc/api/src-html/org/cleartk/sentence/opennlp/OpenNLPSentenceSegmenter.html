024    package org.cleartk.sentence.opennlp;
025    
026    import java.io.File;
027    import java.io.IOException;
028    import java.util.ArrayList;
029    import java.util.Collections;
030    import java.util.List;
031    import java.util.regex.Matcher;
032    import java.util.regex.Pattern;
033    
034    import opennlp.maxent.MaxentModel;
035    import opennlp.maxent.io.SuffixSensitiveGISModelReader;
036    import opennlp.tools.sentdetect.SentenceDetector;
037    import opennlp.tools.sentdetect.SentenceDetectorME;
038    
039    import org.apache.uima.UimaContext;
040    import org.apache.uima.analysis_component.JCasAnnotator_ImplBase;
041    import org.apache.uima.analysis_engine.AnalysisEngineProcessException;
042    import org.apache.uima.jcas.JCas;
043    import org.apache.uima.resource.ResourceInitializationException;
044    import org.cleartk.type.Sentence;
045    
046    
047    /**
048     * <br>Copyright (c) 2007-2008, Regents of the University of Colorado 
049     * <br>All rights reserved.
050    
051     *
052     * @author Philip Ogren
053     *
054     *This sentence segmenter is a simple wrapper around the OpenNLP SentenceDetector with 
055     *additional sentence detection added that handles multiple newlines (i.e. if multiple
056     *newlines appear (separated only by whitespace) together, then this is treated as 
057     *a sentence delimiter.  
058     *
059     * @see SentenceDetector
060     */
061    
062    public class OpenNLPSentenceSegmenter extends JCasAnnotator_ImplBase
063    { 
064            /**
065             * "org.cleartk.sentence.opennlp.OpenNLPSentenceSegmenter.PARAM_SENTENCE_MODEL_FILE"
066             * is a single, required, string parameter that provides the path of the
067             * OpenNLP sentence segmenter model file, e.g.
068             *   resources/models/OpenNLP.Sentence.English.bin.gz
069             * @see opennlp.maxent.io.SuffixSensitiveGISModelReader
070             */
071            public static final String PARAM_SENTENCE_MODEL_FILE = "org.cleartk.sentence.opennlp.OpenNLPSentenceSegmenter.PARAM_SENTENCE_MODEL_FILE";
072            public static final String multipleNewlinesRegex = "\\s*\\n\\s*\\n\\s*";
073    
074            SentenceDetector sentenceDetector;
075            Pattern multipleNewlinesPattern;
076            Pattern leadingWhitespacePattern;
077            Pattern trailingWhitespacePattern;
078            @Override
079            public void initialize(UimaContext uimaContext) throws ResourceInitializationException
080            {
081                    super.initialize(uimaContext);
082                    try {
083                            String sentenceModelFile = (String) uimaContext.getConfigParameterValue(PARAM_SENTENCE_MODEL_FILE);
084                            MaxentModel model = new SuffixSensitiveGISModelReader(new File(sentenceModelFile)).getModel();
085                            sentenceDetector = new SentenceDetectorME(model);
086                            multipleNewlinesPattern = Pattern.compile(multipleNewlinesRegex, Pattern.MULTILINE | Pattern.DOTALL);
087                            leadingWhitespacePattern = Pattern.compile("^\\s+");
088                            trailingWhitespacePattern = Pattern.compile("\\s+$");
089                    }catch(IOException ioe) {
090                            throw new ResourceInitializationException(ioe);
091                    }
092            }
093            
094            @Override
095            public void process(JCas jCas) throws AnalysisEngineProcessException 
096            {
097                    String text = jCas.getDocumentText();
098                    List<Integer> sentenceOffsets = getSentenceOffsets(text);
099    
100                    int begin = 0;
101                    int end = 0;
102    
103                    //advance the first sentence to first non-whitespace char
104                    Matcher matcher;
105                    matcher = leadingWhitespacePattern.matcher(text);
106                    if(matcher.find()) {
107                            begin+= matcher.group().length();
108                    }
109    
110                    for(Integer offset : sentenceOffsets) {
111                            end = offset;  //offset is really the beginning of the next sentence so we may adjust this below.
112                            String sentenceText = text.substring(begin, end);
113                            //it is possible that there is a duplicate offset or that getSentenceOffsets returned the first sentence offset 
114                            if(sentenceText.trim().length() > 0) {
115                                    matcher = trailingWhitespacePattern.matcher(sentenceText);
116                                    if(matcher.find()) {
117                                            end -= matcher.group().length();
118                                    }
119                                    Sentence sentence = new Sentence(jCas, begin,end);
120                                    sentence.addToIndexes();
121                            }
122                            begin = offset; //we need to advance begin regardless of whether a sentence was created.
123                    }
124                    //take the remaining text if there is any and add it to a sentence.  
125                    //this code will not execute if the text ends with a sentence detected by
126                    //SentenceDetector because it actually returns an offset corresponding to the end
127                    //of the last sentence (see note on getSentenceOffsets)
128                    if(begin < text.length()) {
129                            String sentenceText = text.substring(begin, text.length());
130                            end = text.length();
131                            if(sentenceText.trim().length() > 0) {
132                                    matcher = trailingWhitespacePattern.matcher(sentenceText);
133                                    if(matcher.find()) {
134                                            end -= matcher.group().length();
135                                    }
136                                    Sentence sentence = new Sentence(jCas, begin, end);
137                                    sentence.addToIndexes();
138                            }
139                    }
140            }
141    
142            /**
143             * returns a list of the beginnings of sentences - except (possibly) the 
144             * first sentence - from both the OpenNLP sentence detector and the 
145             * multiple newlines regex. It is possible for this method to return 
146             * duplicate values.  The sentence detector will return an offset
147             * corresponding to the end of the text if the last non-whitespace character
148             * was classified an end of sentence character.  
149             */ 
150            private List<Integer> getSentenceOffsets(String text) {
151                    Matcher matcher = multipleNewlinesPattern.matcher(text);
152                    List<Integer> offsets = new ArrayList<Integer>();
153                    while(matcher.find()) {
154                            offsets.add(matcher.end());
155                    }
156                    
157                    int[] sentenceOffsetsML = sentenceDetector.sentPosDetect(text);
158                    for(int i=0; i<sentenceOffsetsML.length; i++) {
159                            offsets.add(sentenceOffsetsML[i]);
160                    }
161                    Collections.sort(offsets);
162                    return offsets;
163            }
164    }
