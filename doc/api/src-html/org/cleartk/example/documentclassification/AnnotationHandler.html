028    import org.apache.uima.cas.CASException;

031    import org.apache.uima.resource.ResourceInitializationException;
032    import org.cleartk.CleartkException;
033    import org.cleartk.Initializable;
034    import org.cleartk.classifier.Instance;
035    import org.cleartk.classifier.InstanceConsumer;
036    import org.cleartk.classifier.feature.extractor.CountsExtractor;
037    import org.cleartk.classifier.feature.extractor.SimpleFeatureExtractor;
038    import org.cleartk.classifier.feature.extractor.TypePathExtractor;
039    import org.cleartk.type.Token;
040    
041    /**
042     * <br>Copyright (c) 2009, Regents of the University of Colorado

044     *
045     * @author Philipp G. Wetzler
046     *
047     */
048    public class AnnotationHandler implements org.cleartk.classifier.AnnotationHandler<String> , Initializable{
049    
050            public static final String PREDICTION_VIEW_NAME = "ExampleDocumentClassificationPredictionView";
051    
052            public void initialize(UimaContext context) throws ResourceInitializationException {
053                    SimpleFeatureExtractor subExtractor = new TypePathExtractor(Token.class, "stem", false, false, true);
054                    extractor = new CountsExtractor(Token.class, subExtractor);
055            }
056    
057            public void process(JCas jCas, InstanceConsumer<String> consumer) throws AnalysisEngineProcessException, CleartkException {
058                    try {
059                            DocumentAnnotation doc = (DocumentAnnotation) jCas.getDocumentAnnotationFs();
060    
061                            Instance<String> instance = new Instance<String>();
062                            instance.addAll(extractor.extract(jCas, doc));
063    
064                            if( consumer.expectsOutcomes() ) {
065                                    JCas goldView = jCas.getView(GoldAnnotator.GOLD_VIEW_NAME);
066                                    instance.setOutcome(goldView.getSofaDataString());
067                            }
068    
069                            String result = consumer.consume(instance);
070                            
071                            if( result != null ) {
072                                    JCas predictionView = jCas.createView(PREDICTION_VIEW_NAME);
073                                    predictionView.setSofaDataString(result, "text/plain");
074                            }
075                    } catch (CASException e) {
076                            throw new AnalysisEngineProcessException(e);
077                    }
078            }
079    
080            private CountsExtractor extractor;
081    
082    }
