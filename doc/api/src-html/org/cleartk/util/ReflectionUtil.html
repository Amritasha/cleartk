026    import java.lang.reflect.ParameterizedType;
027    import java.lang.reflect.Type;
028    import java.lang.reflect.TypeVariable;

030    import java.util.TreeMap;

036    public class ReflectionUtil {
037            
038            public static Type getTypeArgument(Class<?> genericType, String typeParameterName, Object obj) {
039                    Map<String,Type> typeArguments = getTypeArguments(genericType, obj);
040                    return typeArguments == null ? null : typeArguments.get(typeParameterName);
041            }
042            
043            /**
044             * Try to find the instantiation of all of genericTypes type parameters in objs class.
045             * 
046             * @param genericType   the generic supertype of objs class
047             * @param obj                   an instantiation of a subclass of genericType. All of genericTypes type
048             *                                              parameters must have been instantiated in the inheritance hierarchy.
049             * @return                              a map of genericTypes type parameters (their name in the source code) to
050             *                                              the type they are instantiated as in obj
051             */
052            public static Map<String,Type> getTypeArguments(Class<?> genericType, Object obj) {
053                    Map<String,Type> typeMap = new TreeMap<String,Type>();
054                    return getTypeArguments(genericType, obj.getClass(), typeMap);
055            }
056            
057            public static boolean isAssignableFrom(Type type1, Type type2) {
058                    if (type1 instanceof Class<?> && type2 instanceof Class<?>) {
059                            return ((Class<?>)type1).isAssignableFrom((Class<?>)type2);
060                    } else {
061                            return type1.equals(type2);
062                    }
063            }
064    
065            private static Map<String,Type> getTypeArguments(Class<?> genericType, Type type, Map<String,Type> typeMap) {
066                    if( type instanceof ParameterizedType ) {
067                            return getTypeArguments(genericType, (ParameterizedType)type, typeMap);
068                    } else  if( type instanceof Class ) {
069                            return getTypeArguments(genericType, (Class<?>) type, typeMap);
070                    } else {
071                            throw new IllegalArgumentException();
072                    }
073            }
074    
075            private static Map<String,Type> getTypeArguments(Class<?> genericType, Class<?> classType, Map<String,Type> typeMap) {
076                    if( genericType.isInterface() ) {
077                            for( Type interfaceType : classType.getGenericInterfaces() ) {
078                                    Map<String,Type> result = getTypeArguments(genericType, interfaceType, typeMap);
079                                    if( result != null )
080                                            return result;
081                            }
082                    }
083    
084                    Type superType = classType.getGenericSuperclass();
085                    if( superType != null ) {
086                            return getTypeArguments(genericType, superType, typeMap);
087                    }
088    
089                    return null;
090            }
091            
092            private static Map<String,Type> getTypeArguments(Class<?> genericType, ParameterizedType paramType, Map<String,Type> typeMap) {
093                    Class<?> rawType = (Class<?>) paramType.getRawType();
094                    if( rawType == genericType ) {
095                            // found it!
096                            TypeVariable<?> typeVars[] = rawType.getTypeParameters();
097                            Type actualTypes[] = paramType.getActualTypeArguments();
098                            Map<String,Type> result = new TreeMap<String,Type>();
099                            for( int i=0; i<actualTypes.length; i++ ) {
100                                    while( actualTypes[i] != null && actualTypes[i] instanceof TypeVariable ) {
101                                            String key = typevarString((TypeVariable<?>) actualTypes[i]);
102                                            if( typeMap.containsKey(key) )
103                                                    actualTypes[i] = typeMap.get(key);
104                                            else
105                                                    actualTypes[i] = null;
106                                    }
107                                    result.put(typeVars[i].getName(), actualTypes[i]);
108                            }
109                            return result;
110                    } else {
111                            TypeVariable<?> typeVars[] = rawType.getTypeParameters();
112                            Type actualTypes[] = paramType.getActualTypeArguments();
113                            for( int i=0; i<typeVars.length; i++ )
114                                    typeMap.put(typevarString(typeVars[i]), actualTypes[i]);
115                            return getTypeArguments(genericType, paramType.getRawType(), typeMap);
116                    }
117            }
118            
119            private static String typevarString(TypeVariable<?> tv) {
120                    return tv.getGenericDeclaration().toString() + " " + tv.getName();
121            }
122    }
