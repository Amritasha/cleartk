056            public static <T extends Annotation> T get(JCas jCas, Class<T> cls, int index) {
057                    int type;
058                    try {
059                            type = (Integer) cls.getField("type").get(null);
060                    }
061                    catch (IllegalAccessException e) {
062                            throw new RuntimeException();
063                    }
064                    catch (NoSuchFieldException e) {
065                            throw new RuntimeException();
066                    }
067    
068                    // TODO we should probably iterate from the end rather than
069                    // iterating forward from the begining.
070                    FSIndex fsIndex = jCas.getAnnotationIndex(type);
071                    if (index < 0) index = fsIndex.size() + index;
072    
073                    if (index < 0 || index >= fsIndex.size()) return null;
074                    FSIterator iterator = fsIndex.iterator();
075                    Object returnValue = iterator.next();
076                    for (int i = 0; i < index; i++) {
077                            returnValue = iterator.next();
078                    }
079                    return cls.cast(returnValue);
080            }

086            @SuppressWarnings("unchecked")
087            public static <T extends Annotation> T get(JCas jCas, T annotation, int relativePosition,
088                            Annotation windowAnnotation) {
089                    FSIterator cursor = jCas.getAnnotationIndex(annotation.getType()).iterator();
090                    cursor.moveTo(annotation);
091    
092                    if (relativePosition > 0) {
093                            for (int i = 0; i < relativePosition && cursor.isValid(); i++)
094                                    cursor.moveToNext();
095                    }
096                    else {
097                            for (int i = 0; i < -relativePosition && cursor.isValid(); i++)
098                                    cursor.moveToPrevious();
099                    }
100                    if (cursor.isValid()) {
101                            Annotation relativeAnnotation = (Annotation) cursor.get();
102    
103                            T returnValue = ((Class<T>) annotation.getClass()).cast(relativeAnnotation);
104    
105                            if (windowAnnotation != null) {
106                                    if (AnnotationUtil.contains(windowAnnotation, relativeAnnotation)) return returnValue;
107                                    else return null;
108                            }
109                            else return returnValue;
110                    }
111                    else return null;
112            }

114            public static <RETURN_TYPE extends Annotation> RETURN_TYPE get(JCas jCas, Annotation annotation, Class<RETURN_TYPE> returnCls, int relativePosition) {
115                    if(relativePosition == 0) {
116                            return getFirstAnnotation(jCas, annotation, returnCls);
117                    }
118                    else if(relativePosition < 0) {
119                            RETURN_TYPE returnValue = getAdjacentAnnotation(jCas, annotation, returnCls, true);
120                            if(returnValue == null)
121                                    return null;
122                            if(relativePosition == -1)
123                                    return returnValue;
124                            return get(jCas, returnValue, relativePosition + 1);
125                    } else {
126                            RETURN_TYPE returnValue = getAdjacentAnnotation(jCas, annotation, returnCls, false);
127                            if(returnValue == null)
128                                    return null;
129                            if(relativePosition == 1)
130                                    return returnValue;
131                            return get(jCas, returnValue, relativePosition - 1);
132                    } 
133            }

135            /**
136             * We initially used the AnnotationIndex.subiterator() but ran into issues
137             * that we wanted to work around concerning the use of type priorities when
138             * the window and windowed annotations have the same begin and end offsets.
139             * By using our own iterator we can back up to the first annotation that has
140             * the same beginning as the window annotation and avoid having to set type
141             * priorities.
142             * 
143             * @param jCas
144             * @param windowAnnotation
145             * @return an FSIterator that is at the correct position
146             */
147            private static FSIterator initializeWindowCursor(JCas jCas, Annotation windowAnnotation) {
148    
149                    FSIterator cursor = jCas.getAnnotationIndex().iterator();
150                    
151                    cursor.moveTo(windowAnnotation);
152                    
153                    // if cursor is invalid now we're past the end of the index
154                    if( !cursor.isValid() )
155                            return cursor;
156    
157                    while (cursor.isValid() && ((Annotation) cursor.get()).getBegin() >= windowAnnotation.getBegin()) {
158                            cursor.moveToPrevious();
159                    }
160    
161                    if (cursor.isValid()) {
162                            cursor.moveToNext();
163                    }
164                    else {
165                            cursor.moveToFirst();
166                    }
167    
168                    return cursor;
169            }

171            /**
172             * @param <T>
173             *            determines the return type of the method
174             * @param jCas
175             *            the current jCas or view
176             * @param windowAnnotation
177             *            an annotation that defines a window
178             * @param cls
179             *            determines the return type of the method
180             * @return the last annotation of type cls that is "inside" the window
181             * @see #getAnnotations(JCas, Annotation, Class)
182             */
183            public static <T extends Annotation> T getLastAnnotation(JCas jCas, Annotation windowAnnotation, Class<T> cls) {
184                    FSIterator cursor = initializeWindowCursor(jCas, windowAnnotation);
185    
186                    T currentBestGuess = null;
187                    while (cursor.isValid() && ((Annotation) cursor.get()).getBegin() <= windowAnnotation.getEnd()) {
188                            Annotation annotation = (Annotation) cursor.get();
189    
190                            if (cls.isInstance(annotation) && annotation.getEnd() <= windowAnnotation.getEnd()) currentBestGuess = cls
191                            .cast(annotation);
192                            cursor.moveToNext();
193                    }
194                    return currentBestGuess;
195            }

200             * @param jCas
201             *            the current jCas or view
202             * @param windowAnnotation
203             *            an annotation that defines a window

206             * @return the first annotation of type cls that is "inside" the window
207             * @see #getAnnotations(JCas, Annotation, Class)
208             */
209            public static <T extends Annotation> T getFirstAnnotation(JCas jCas, Annotation windowAnnotation, Class<T> cls) {
210                    FSIterator cursor = initializeWindowCursor(jCas, windowAnnotation);
211    
212                    // I left in the while loop because the first annotation we see might
213                    // not be the right class
214                    while (cursor.isValid() && ((Annotation) cursor.get()).getBegin() <= windowAnnotation.getEnd()) {
215                            Annotation annotation = (Annotation) cursor.get();
216                            if (cls.isInstance(annotation) && annotation.getEnd() <= windowAnnotation.getEnd()) return cls
217                            .cast(annotation);
218                            cursor.moveToNext();
219                    }
220                    return null;
221            }

232            public static <T extends Annotation> T getFirstAnnotation(JCas view, Class<T> cls) {

240                    }

244            /*
245             * Find an annotation of a different type that covers the same span.
246             * 
247             * @param <T> is the type of annotation we're looking for @param jCas is the
248             * current CAS view @param windowAnnotation determines the span of the
249             * annotation @return the first annotation in the index that has the same
250             * span as windowAnnotation
251             */
252            public static <T extends Annotation> T getMatchingAnnotation(JCas jCas, Annotation windowAnnotation, Class<T> cls) {
253                    if (cls.isInstance(windowAnnotation)) return cls.cast(windowAnnotation);
254    
255                    FSIterator cursor = jCas.getAnnotationIndex().iterator();
256    
257                    cursor.moveTo(windowAnnotation);
258                    Annotation cursorAnnotation = (Annotation) cursor.get();
259                    if (cursorAnnotation.getBegin() != windowAnnotation.getBegin()
260                                    || cursorAnnotation.getEnd() != windowAnnotation.getEnd()) return null;
261    
262                    while (cursor.isValid() && cursorAnnotation.getBegin() == windowAnnotation.getBegin()
263                                    && cursorAnnotation.getEnd() == windowAnnotation.getEnd()) {
264                            cursor.moveToPrevious();
265                            if (cursor.isValid()) cursorAnnotation = (Annotation) cursor.get();
266                            else cursorAnnotation = null;
267                    }
268    
269                    if (cursor.isValid()) {
270                            cursor.moveToNext();
271                            cursorAnnotation = (Annotation) cursor.get();
272                    }
273                    else {
274                            cursor.moveToFirst();
275                            cursorAnnotation = (Annotation) cursor.get();
276                    }
277    
278                    while (cursor.isValid() && cursorAnnotation.getBegin() == windowAnnotation.getBegin()
279                                    && cursorAnnotation.getEnd() == windowAnnotation.getEnd()) {
280                            if (cls.isInstance(cursorAnnotation)) return cls.cast(cursorAnnotation);
281                            cursor.moveToNext();
282                            if (cursor.isValid()) cursorAnnotation = (Annotation) cursor.get();
283                            else cursorAnnotation = null;
284                    }
285    
286                    return null;
287            }

289            /**
290             * Returns a List of all annotations of a given type.
291             * 
292             * @param <T>
293             *            The type of Annotation being collected.
294             * @param jCas
295             *            The JCas from which annotations should be collected.
296             * @param annotationClass
297             *            The Annotation class.
298             * @return A List of all Annotations of the given type.
299             */
300            public static <T extends Annotation> List<T> getAnnotations(JCas jCas, Class<T> annotationClass) {
301                    // get the integer type from the annotation class
302                    // not that T.type doesn't work because that gives us Annotation.type
303                    // (why?)
304                    int type;
305                    try {
306                            type = annotationClass.getField("type").getInt(null);
307                    }
308                    catch (IllegalArgumentException e) {
309                            throw new RuntimeException(e);
310                    }
311                    catch (SecurityException e) {
312                            throw new RuntimeException(e);
313                    }
314                    catch (IllegalAccessException e) {
315                            throw new RuntimeException(e);
316                    }
317                    catch (NoSuchFieldException e) {
318                            throw new RuntimeException(e);
319                    }
320    
321                    // collect all annotations of the given type into a list
322                    AnnotationIndex index = jCas.getAnnotationIndex(type);
323                    List<T> annotations = new ArrayList<T>();
324                    for (FSIterator iter = index.iterator(); iter.hasNext();) {
325                            Object next = iter.next();
326                            annotations.add(annotationClass.cast(next));
327                    }
328                    return annotations;
329            }

331            /**
332             * This method returns all annotations (in order) that are inside a "window"
333             * annotation of a particular kind. The functionality provided is similar to
334             * using AnnotationIndex.subiterator(), however we found the use of type
335             * priorities which is documented in detail in their javadocs to be
336             * distasteful. This method does not require that type priorities be set in
337             * order to work as expected for the condition where the window annotation
338             * and the "windowed" annotation have the same size and location.
339             * 
340             * @param <T>
341             *            determines the return type of the method
342             * @param jCas
343             *            the current jCas or view
344             * @param windowAnnotation
345             *            an annotation that defines a window
346             * @param cls
347             *            determines the return type of the method
348             * @return a list of annotations of type cls that are "inside" the window
349             * @see AnnotationIndex#subiterator(org.apache.uima.cas.text.AnnotationFS)
350             */
351            public static <T extends Annotation> List<T> getAnnotations(JCas jCas, Annotation windowAnnotation, Class<T> cls) {
352                    if (windowAnnotation instanceof SplitAnnotation) return getAnnotations(jCas,
353                                    (SplitAnnotation) windowAnnotation, cls);
354    
355                    FSIterator cursor = initializeWindowCursor(jCas, windowAnnotation);
356    
357                    List<T> annotations = new ArrayList<T>();
358                    while (cursor.isValid() && ((Annotation) cursor.get()).getBegin() <= windowAnnotation.getEnd()) {
359                            Annotation annotation = (Annotation) cursor.get();
360    
361                            if (cls.isInstance(annotation) && annotation.getEnd() <= windowAnnotation.getEnd()) annotations.add(cls
362                                            .cast(annotation));
363    
364                            cursor.moveToNext();
365                    }
366                    return annotations;
367            }

369            public static <T extends Annotation> List<T> getAnnotations(JCas jCas, Annotation windowAnnotation, Class<T> cls,
370                            boolean exactSpan) {
371                    if (windowAnnotation instanceof SplitAnnotation) return getAnnotations(jCas,
372                                    (SplitAnnotation) windowAnnotation, cls);
373    
374                    if (!exactSpan) return getAnnotations(jCas, windowAnnotation, cls);
375                    else {
376                            FSIterator cursor = initializeWindowCursor(jCas, windowAnnotation);
377    
378                            List<T> annotations = new ArrayList<T>();
379                            while (cursor.isValid() && ((Annotation) cursor.get()).getBegin() <= windowAnnotation.getEnd()) {
380                                    Annotation annotation = (Annotation) cursor.get();
381    
382                                    if (cls.isInstance(annotation) && annotation.getBegin() == windowAnnotation.getBegin()
383                                                    && annotation.getEnd() == windowAnnotation.getEnd()) annotations.add(cls.cast(annotation));
384    
385                                    cursor.moveToNext();
386                            }
387                            return annotations;
388                    }
389            }

391            public static <T extends Annotation> List<T> getAnnotations(JCas jCas, int begin, int end, Class<T> cls,
392                            boolean exactSpan) {
393                    if(begin > end)
394                            return null;
395                    if (!exactSpan) {
396                            return getAnnotations(jCas, begin, end, cls);
397                    }
398                    else {
399                            return getAnnotations(jCas, new Annotation(jCas, begin, end), cls, exactSpan);
400                    }

404            public static <T extends Annotation> List<T> getAnnotations(JCas jCas, SplitAnnotation splitAnnotation, Class<T> cls) {
405                    List<T> returnValues = new ArrayList<T>();
406    
407                    for (FeatureStructure subAnnotation : splitAnnotation.getAnnotations().toArray()) {
408                            returnValues.addAll(getAnnotations(jCas, (Annotation) subAnnotation, cls));
409                    }
410    
411                    return returnValues;
412            }

417                    Annotation annotation = new Annotation(jCas, begin, end);
418                    return getAnnotations(jCas, annotation, cls);

421            /**
422             * This method provides a way to have multiple annotations define the window
423             * that is used to constrain the annotations in the returned list. The
424             * intersection of the windows is determined and used. This method will
425             * treat any split annotations in the list as a contiguous annotation.
426             * 
427             * @see #getAnnotations(JCas, Annotation, Class)
428             * @see #getAnnotations(JCas, SplitAnnotation, Class)
429             */
430            public static <T extends Annotation> List<T> getAnnotations(JCas jCas, List<Annotation> windowAnnotations,
431                            Class<T> cls) {
432                    if (windowAnnotations == null || windowAnnotations.size() == 0) return null;
433    
434                    int windowBegin = Integer.MIN_VALUE;
435                    int windowEnd = Integer.MAX_VALUE;
436                    for (Annotation windowAnnotation : windowAnnotations) {
437                            if (windowAnnotation.getBegin() > windowBegin) windowBegin = windowAnnotation.getBegin();
438                            if (windowAnnotation.getEnd() < windowEnd) windowEnd = windowAnnotation.getEnd();
439                    }
440                    return getAnnotations(jCas, windowBegin, windowEnd, cls);
441            }

453            /**
454             * This method finds the smallest annotation of type cls that contains the
455             * focus annotation.
456             * 
457             * @param jCas
458             * @param focusAnnotation
459             *            an annotation that will be contained by the returned value
460             * @param cls
461             *            determines the type of the returned annotation
462             * @param exclusiveContain
463             *            if true, then the method will only return an annotation if it
464             *            is larger than the focus annotation. If false, then an
465             *            annotation of the same size can be returned if one exists of
466             *            the type specified by the cls param. This may be undesirable
467             *            if the focus annotation is the same type as the cls param
468             *            because it will return the focus annotation when you might
469             *            actually want the next biggest annotation that contains the
470             *            focus annotation.
471             * @return an annotation of type cls that contains the focus annotation or
472             *         null it one does not exist.
473             */
474    
475            public static <T extends Annotation> T getContainingAnnotation(JCas jCas, Annotation focusAnnotation, Class<T> cls,
476                            boolean exclusiveContain) {
477    
478                    if(focusAnnotation == null)
479                            throw new IllegalArgumentException("focus annotation may not be null");
480    
481    
482                    FSIterator cursor = jCas.getAnnotationIndex().iterator();
483    
484                    //the goal is to move the cursor to the last annotation that begins at the same index as the focus annotation 
485                    cursor.moveTo(focusAnnotation);
486                    while (cursor.isValid()) {
487                            cursor.moveToNext();
488    
489                            //if we have moved past the last element, then move cursor to last element and be done
490                            if(!cursor.isValid()) {
491                                    cursor.moveToLast();
492                                    break;
493                            }
494    
495                            //if we have moved to an annotation that does not be begin at the same index as the focus annotation,
496                            //then move to previous.  
497                            Annotation nextAnnotation = (Annotation) cursor.get();
498                            if(nextAnnotation.getBegin() != focusAnnotation.getBegin()) {
499                                    cursor.moveToPrevious();
500                                    break;
501                            }
502                    }
503    
504    
505                    while (cursor.isValid()) {
506                            Annotation annotation = (Annotation) cursor.get();
507                            if (exclusiveContain) {
508                                    if (cls.isInstance(annotation) && annotation.getBegin() <= focusAnnotation.getBegin()
509                                                    && annotation.getEnd() >= focusAnnotation.getEnd()
510                                                    && AnnotationUtil.size(annotation) > AnnotationUtil.size(focusAnnotation)) {
511                                            return cls.cast(annotation);
512                                    }
513                            }
514                            else {
515                                    if (cls.isInstance(annotation) && annotation.getBegin() <= focusAnnotation.getBegin()
516                                                    && annotation.getEnd() >= focusAnnotation.getEnd()) {
517                                            return cls.cast(annotation);
518                                    }
519                            }
520                            cursor.moveToPrevious();
521                    }
522                    return null;
523    
524            }

526            /**
527             * Finds and returns the annotation of the provided type that is adjacent to
528             * the focus annotation in either to the left or right. Adjacent simply
529             * means the last annotation of the passed in type that ends before the
530             * start of the focus annotation (for the left side) or the first annotation
531             * that starts after the end of the focus annotation (for the right side).
532             * Thus, adacent refers to order of annotations at the annotation level
533             * (e.g. give me the first annotation of type x to the left of the focus
534             * annotation) and does not mean that the annotations are adjacenct at the
535             * character offset level.
536             * 
537             * <br>
538             * <b>note:</b> This method runs <b>much</b> faster if the type of the
539             * passed in annotation and the passed in type are the same.
540             * 
541             * @param jCas
542             * @param focusAnnotation
543             *            an annotation that you want to find an annotation adjacent to
544             *            this.
545             * @param adjacentClass
546             *            the type of annotation that you want to find
547             * @param adjacentBefore
548             *            if true then returns an annotation to the left of the passed
549             *            in annotation, otherwise an annotation to the right will be
550             *            returned.
551             * @return an annotation of type adjacentType or null
552             */
553            public static <T extends Annotation> T getAdjacentAnnotation(JCas jCas, Annotation focusAnnotation,
554                            Class<T> adjacentClass, boolean adjacentBefore) {
555                    try {
556                            Type adjacentType = UIMAUtil.getCasType(jCas, adjacentClass);
557                            if (focusAnnotation.getType().equals(adjacentType)) {
558                                    FSIterator iterator = jCas.getAnnotationIndex(adjacentType).iterator();
559                                    iterator.moveTo(focusAnnotation);
560                                    if (adjacentBefore) iterator.moveToPrevious();
561                                    else iterator.moveToNext();
562                                    return adjacentClass.cast(iterator.get());
563                            }
564                            else {
565                                    FSIterator cursor = jCas.getAnnotationIndex().iterator();
566                                    cursor.moveTo(focusAnnotation);
567                                    if (adjacentBefore) {
568                                            while (cursor.isValid()) {
569                                                    cursor.moveToPrevious();
570                                                    Annotation annotation = (Annotation) cursor.get();
571                                                    if (adjacentClass.isInstance(annotation) && annotation.getEnd() <= focusAnnotation.getBegin()) return adjacentClass
572                                                    .cast(annotation);
573                                            }
574                                    }
575                                    else {
576                                            while (cursor.isValid()) {
577                                                    cursor.moveToNext();
578                                                    Annotation annotation = (Annotation) cursor.get();
579                                                    if (adjacentClass.isInstance(annotation) && annotation.getBegin() >= focusAnnotation.getEnd()) return adjacentClass
580                                                    .cast(annotation);
581                                            }
582                                    }
583                            }
584                    }
585                    catch (NoSuchElementException nsee) {
586                            return null;
587                    }
588                    return null;
589            }
