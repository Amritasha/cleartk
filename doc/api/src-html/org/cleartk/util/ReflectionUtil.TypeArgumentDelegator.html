036    public class ReflectionUtil {
037            
038            /**
039             * Perform an unchecked cast based on a type parameter.
040             * 
041             * @param <T> The type to which the object should be cast.
042             * @param o   The object.
043             * @return    The object, cast to the given type.
044             */
045            @SuppressWarnings("unchecked")
046            public static <T> T uncheckedCast(Object o) {
047                    return (T)o;
048            }
049            
050            public static interface TypeArgumentDelegator {
051                    public Map<String,Type> getTypeArguments(Class<?> genericType);
052            }
053            
054            public static <T> Type getTypeArgument(Class<T> genericType, String typeParameterName, T obj) {
055                    Map<String,Type> typeArguments = getTypeArguments(genericType, obj);
056                    return typeArguments == null ? null : typeArguments.get(typeParameterName);
057            }
058            
059            /**
060             * Try to find the instantiation of all of genericTypes type parameters in objs class.
061             * 
062             * @param genericType   the generic supertype of objs class
063             * @param obj                   an instantiation of a subclass of genericType. All of genericTypes type
064             *                                              parameters must have been instantiated in the inheritance hierarchy.
065             * @return                              a map of genericTypes type parameters (their name in the source code) to
066             *                                              the type they are instantiated as in obj
067             */
068            public static Map<String,Type> getTypeArguments(Class<?> genericType, Object obj) {
069                    if (obj instanceof TypeArgumentDelegator) {
070                            return ((TypeArgumentDelegator)obj).getTypeArguments(genericType);
071                    }
072                    Map<String,Type> typeMap = new TreeMap<String,Type>();
073                    return getTypeArguments(genericType, obj.getClass(), typeMap);
074            }
075            
076            public static boolean isAssignableFrom(Type type1, Type type2) {
077                    if (type1 instanceof Class<?> && type2 instanceof Class<?>) {
078                            return ((Class<?>)type1).isAssignableFrom((Class<?>)type2);
079                    } else {
080                            return type1.equals(type2);
081                    }
082            }
083    
084            private static Map<String,Type> getTypeArguments(Class<?> genericType, Type type, Map<String,Type> typeMap) {
085                    if( type instanceof ParameterizedType ) {
086                            return getTypeArguments(genericType, (ParameterizedType)type, typeMap);
087                    } else  if( type instanceof Class ) {
088                            return getTypeArguments(genericType, (Class<?>) type, typeMap);
089                    } else {
090                            throw new IllegalArgumentException();
091                    }
092            }
093    
094            private static Map<String,Type> getTypeArguments(Class<?> genericType, Class<?> classType, Map<String,Type> typeMap) {
095                    if( genericType.isInterface() ) {
096                            for( Type interfaceType : classType.getGenericInterfaces() ) {
097                                    Map<String,Type> result = getTypeArguments(genericType, interfaceType, typeMap);
098                                    if( result != null )
099                                            return result;
100                            }
101                    }
102    
103                    Type superType = classType.getGenericSuperclass();
104                    if( superType != null ) {
105                            return getTypeArguments(genericType, superType, typeMap);
106                    }
107    
108                    return null;
109            }
110            
111            private static Map<String,Type> getTypeArguments(Class<?> genericType, ParameterizedType paramType, Map<String,Type> typeMap) {
112                    Class<?> rawType = (Class<?>) paramType.getRawType();
113                    if( rawType == genericType ) {
114                            // found it!
115                            TypeVariable<?> typeVars[] = rawType.getTypeParameters();
116                            Type actualTypes[] = paramType.getActualTypeArguments();
117                            Map<String,Type> result = new TreeMap<String,Type>();
118                            for( int i=0; i<actualTypes.length; i++ ) {
119                                    while( actualTypes[i] != null && actualTypes[i] instanceof TypeVariable ) {
120                                            String key = typevarString((TypeVariable<?>) actualTypes[i]);
121                                            if( typeMap.containsKey(key) )
122                                                    actualTypes[i] = typeMap.get(key);
123                                            else
124                                                    actualTypes[i] = null;
125                                    }
126                                    result.put(typeVars[i].getName(), actualTypes[i]);
127                            }
128                            return result;
129                    } else {
130                            TypeVariable<?> typeVars[] = rawType.getTypeParameters();
131                            Type actualTypes[] = paramType.getActualTypeArguments();
132                            for( int i=0; i<typeVars.length; i++ )
133                                    typeMap.put(typevarString(typeVars[i]), actualTypes[i]);
134                            return getTypeArguments(genericType, paramType.getRawType(), typeMap);
135                    }
136            }
137            
138            private static String typevarString(TypeVariable<?> tv) {
139                    return tv.getGenericDeclaration().toString() + " " + tv.getName();
140            }
