053    public class ViewCopier {
054    
055            public ViewCopier(CAS sourceView, CAS targetView) {
056                    this.sourceView = sourceView;
057                    this.targetView = targetView;
058            }
059    
060            public FeatureStructure copyFS(FeatureStructure source) {
061                    assert(source.getCAS() == sourceView);
062                    
063                    if( copiedFSs.containsKey(source) )
064                            return copiedFSs.get(source);
065    
066                    if( source instanceof SofaFS ) {
067                            return targetView.getSofa();
068                    }
069    
070                    Type type = source.getType();
071    
072                    FeatureStructure copy;
073                    if( type.isArray() ) {
074                            copy = copyFSArray(source);
075                            copiedFSs.put(source, copy);
076                    } else {
077                            copy = targetView.createFS(type);
078                            copiedFSs.put(source, copy);
079                            copyFeatures(source, copy);
080                    }
081    
082                    return copy;
083            }
084    
085            private FeatureStructure copyFSArray(FeatureStructure source) {
086                    if (source instanceof StringArrayFS) {
087                            StringArrayFS array = (StringArrayFS) source;
088                            int len = array.size();
089                            StringArrayFS copy = targetView.createStringArrayFS(len);
090                            for (int i = 0; i < len; i++) {
091                                    copy.set(i, array.get(i));
092                            }
093                            return copy;
094                    } else if (source instanceof IntArrayFS) {
095                            IntArrayFS array = (IntArrayFS) source;
096                            int len = array.size();
097                            IntArrayFS copy = targetView.createIntArrayFS(len);
098                            for (int i = 0; i < len; i++) {
099                                    copy.set(i, array.get(i));
100                            }
101                            return copy;
102                    } else if (source instanceof ByteArrayFS) {
103                            ByteArrayFS array = (ByteArrayFS) source;
104                            int len = array.size();
105                            ByteArrayFS copy = targetView.createByteArrayFS(len);
106                            for (int i = 0; i < len; i++) {
107                                    copy.set(i, array.get(i));
108                            }
109                            return copy;
110                    } else if (source instanceof ShortArrayFS) {
111                            ShortArrayFS array = (ShortArrayFS) source;
112                            int len = array.size();
113                            ShortArrayFS copy = targetView.createShortArrayFS(len);
114                            for (int i = 0; i < len; i++) {
115                                    copy.set(i, array.get(i));
116                            }
117                            return copy;
118                    } else if (source instanceof LongArrayFS) {
119                            LongArrayFS array = (LongArrayFS) source;
120                            int len = array.size();
121                            LongArrayFS copy = targetView.createLongArrayFS(len);
122                            for (int i = 0; i < len; i++) {
123                                    copy.set(i, array.get(i));
124                            }
125                            return copy;
126                    } else if (source instanceof FloatArrayFS) {
127                            FloatArrayFS array = (FloatArrayFS) source;
128                            int len = array.size();
129                            FloatArrayFS copy = targetView.createFloatArrayFS(len);
130                            for (int i = 0; i < len; i++) {
131                                    copy.set(i, array.get(i));
132                            }
133                            return copy;
134                    } else if (source instanceof DoubleArrayFS) {
135                            DoubleArrayFS array = (DoubleArrayFS) source;
136                            int len = array.size();
137                            DoubleArrayFS copy = targetView.createDoubleArrayFS(len);
138                            for (int i = 0; i < len; i++) {
139                                    copy.set(i, array.get(i));
140                            }
141                            return copy;
142                    } else if (source instanceof BooleanArrayFS) {
143                            BooleanArrayFS array = (BooleanArrayFS) source;
144                            int len = array.size();
145                            BooleanArrayFS copy = targetView.createBooleanArrayFS(len);
146                            for (int i = 0; i < len; i++) {
147                                    copy.set(i, array.get(i));
148                            }
149                            return copy;
150                    } else {
151                            ArrayFS array = (ArrayFS) source;
152                            int len = array.size();
153                            ArrayFS copy = targetView.createArrayFS(len);
154                            for (int i = 0; i < len; i++) {
155                                    FeatureStructure sourceElement = array.get(i);
156                                    if( sourceElement != null )
157                                            copy.set(i, copyFS(sourceElement));
158                            }
159                            return copy;
160                    }
161            }
162    
163            private void copyFeatures(FeatureStructure source, FeatureStructure target) {
164                    Type type = source.getType();
165                    assert(target.getType() == type);
166    
167                    List<?> features = type.getFeatures();
168                    for( Object o : features ) {
169                            Feature feature = (Feature) o;
170    
171                            if( feature.getRange().isPrimitive() ) {
172                                    String s = source.getFeatureValueAsString(feature);
173                                    target.setFeatureValueFromString(feature, s);
174                            } else {
175                                    FeatureStructure refFS = source.getFeatureValue(feature);
176                                    if( refFS != null ) {
177                                            FeatureStructure cpFS = copyFS(refFS);
178                                            target.setFeatureValue(feature, cpFS);
179                                    }
180                            }
181                    }
182            }
183    
184            private Map<FeatureStructure,FeatureStructure> copiedFSs = new HashMap<FeatureStructure,FeatureStructure>();
185    
186            private CAS sourceView;
187            private CAS targetView;
188    }
