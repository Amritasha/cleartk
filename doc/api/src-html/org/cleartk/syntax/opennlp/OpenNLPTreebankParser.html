123            @Override
124            public void process(JCas jCas) throws AnalysisEngineProcessException {
125                    String text = jCas.getDocumentText();
126                    
127                    FSIterator sentences = jCas.getJFSIndexRepository().getAnnotationIndex(Sentence.type).iterator();
128                    while( sentences.hasNext() ) {
129                            Sentence sentence = (Sentence) sentences.next();
130                            
131                            Parse parse = new Parse(text, new Span(sentence.getBegin(), sentence.getEnd()), "INC", 1, null);
132                            
133                            List<Token> tokenList = AnnotationRetrieval.getAnnotations(jCas, sentence, Token.class); 
134                            Token[] tokens = tokenList.toArray(new Token[tokenList.size()]);
135                            for( Token token : tokens ) {
136                                    //TODO i'm not sure what the value of p should be - so I just put in 1.0f as an initial guess.
137                                    parse.insert(new Parse(text, new Span(token.getBegin(), token.getEnd()), Parser.TOK_NODE, 1.0f, 0));
138                            }       
139    
140                            this.tagger.setTokens(tokens);
141                            parse = this.parser.parse(parse);
142                            
143                            // if the sentence was successfully parsed, add the tree to the sentence
144                            if (parse.getType() == Parser.TOP_NODE) {
145                                    TopTreebankNode topNode = (TopTreebankNode) buildAnnotation(parse, jCas);
146                                    topNode.addToIndexes();
147                                    sentence.setConstituentParse(topNode);
148                            }
149                    }
150            }
151            
152            protected static TreebankNode buildAnnotation(Parse p, JCas jCas) {
153                    TreebankNode myNode;
154                    if( p.getType() == Parser.TOP_NODE ) {
155                            TopTreebankNode topNode = new TopTreebankNode(jCas);
156                            topNode.setParent(null);
157    
158                            StringBuffer sb = new StringBuffer();
159                            p.show(sb);
160                            topNode.setTreebankParse(sb.toString());
161                            
162                            myNode = topNode;
163                    } else {
164                            myNode = new TreebankNode(jCas);
165                    }
166                    
167                    myNode.setNodeType(p.getType());
168                    myNode.setBegin(p.getSpan().getStart());
169                    myNode.setEnd(p.getSpan().getEnd());
170    
171                    if( p.getChildCount() == 1 && p.getChildren()[0].getType() == Parser.TOK_NODE ) {
172                            myNode.setLeaf(true);
173                            myNode.setNodeValue(p.getChildren()[0].toString());
174                            myNode.setChildren(new FSArray(jCas, 0));
175                    } else {
176                            myNode.setNodeValue(null);
177                            myNode.setLeaf(false);
178                            
179                            List<FeatureStructure> cArray = new ArrayList<FeatureStructure>(p.getChildCount());
180                            
181                            for( Parse cp : p.getChildren() ) {
182                                    TreebankNode cNode = buildAnnotation(cp, jCas);
183                                    cNode.setParent(myNode);
184                                    cNode.addToIndexes();
185                                    cArray.add(cNode);
186                            }
187                            
188                            FSArray cFSArray = new FSArray(jCas, cArray.size());
189                            cFSArray.copyFromArray(cArray.toArray(new FeatureStructure[cArray.size()]), 0, 0, cArray.size());
190                            myNode.setChildren(cFSArray);
191                    }
192                    
193                    if( p.getType() == Parser.TOP_NODE ) {
194                            List<TreebankNode> tList = getTerminals(myNode);
195                            FSArray tfsa = new FSArray(jCas, tList.size());
196                            tfsa.copyFromArray(tList.toArray(new FeatureStructure[tList.size()]), 0, 0, tList.size());
197                            ((TopTreebankNode) myNode).setTerminals(tfsa);
198                    }
199                    return myNode;
200            }
201            
202            protected static List<TreebankNode> getTerminals(TreebankNode node) {
203                    List<TreebankNode> tList = new ArrayList<TreebankNode>();
204    
205                    if( node.getChildren().size() == 0 ) {
206                            tList.add(node);
207                            return tList;
208                    }
209                    
210                    TreebankNode[] children = Arrays.asList(node.getChildren().toArray()).toArray(new TreebankNode[node.getChildren().size()]);
211    
212                    for( TreebankNode child : children ) {
213                            tList.addAll(getTerminals(child));
214                    }
215                    return tList;
216            }

218            @Override
219            public void initialize(UimaContext ctx) throws ResourceInitializationException {
220                    super.initialize(ctx);
221                    
222                    String buildModelFile = (String)UIMAUtil.getRequiredConfigParameterValue(
223                                    ctx, OpenNLPTreebankParser.PARAM_BUILD_MODEL_FILE);
224                    String checkModelFile = (String)UIMAUtil.getRequiredConfigParameterValue(
225                                    ctx, OpenNLPTreebankParser.PARAM_CHECK_MODEL_FILE);
226                    String chunkModelFile = (String)UIMAUtil.getRequiredConfigParameterValue(
227                                    ctx, OpenNLPTreebankParser.PARAM_CHUNK_MODEL_FILE);
228                    String headRulesFile = (String)UIMAUtil.getRequiredConfigParameterValue(
229                                    ctx, OpenNLPTreebankParser.PARAM_HEAD_RULES_FILE);
230                    int beamSize = (Integer)UIMAUtil.getDefaultingConfigParameterValue(
231                                    ctx, OpenNLPTreebankParser.PARAM_BEAM_SIZE, Parser.defaultBeamSize);
232                    double advancePercentage = (Double)UIMAUtil.getDefaultingConfigParameterValue(
233                                    ctx, OpenNLPTreebankParser.PARAM_ADVANCE_PERCENTAGE, Parser.defaultAdvancePercentage);
234                    
235                    try {
236                            MaxentModel buildModel = new SuffixSensitiveGISModelReader(new File(buildModelFile)).getModel();
237                            MaxentModel checkModel = new SuffixSensitiveGISModelReader(new File(checkModelFile)).getModel();
238                            
239                            this.tagger = new OpenNLPDummyParserTagger();
240                            
241                            ParserChunker chunker = new ParserChunker(chunkModelFile);
242                            
243                            HeadRules hrules = new HeadRules(headRulesFile);
244                            
245                            this.parser = new Parser(buildModel, checkModel, tagger, chunker, hrules,
246                                                    beamSize, advancePercentage);
247                    } catch (IOException e) {
248                            throw new ResourceInitializationException(e);
249                    }
250            }
