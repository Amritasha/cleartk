051    public class ChunkingParser extends Parser {
052    
053            @Override
054            protected void createAnnotation(CAS cas, int offset, Parse parse) throws AnalysisEngineProcessException {
055                    try {
056                            convert(cas.getJCas(), parse, offset);
057                    } catch(CASException ce) {
058                            throw new AnalysisEngineProcessException(ce);
059                    }
060            }
061    
062    
063            public static org.cleartk.syntax.treebank.type.TopTreebankNode convert(JCas jCas, Parse opennlpParse, int offset) {
064    
065                    org.cleartk.syntax.treebank.type.TopTreebankNode topNode = new org.cleartk.syntax.treebank.type.TopTreebankNode(
066                                    jCas, offset + opennlpParse.getSpan().getStart(), offset + opennlpParse.getSpan().getEnd());
067                    
068                    
069                    List<Token> tokens = AnnotationRetrieval.getAnnotations(jCas, topNode, Token.class);
070                    for(int i=0; i<tokens.size(); i++){
071                            Token token = tokens.get(i);
072                            Span tokenSpan = new Span(token.getBegin() - offset, token.getEnd() - offset);
073                            opennlpParse.insert(new Parse(token.getCoveredText(), tokenSpan, token.getPos(), 1.0, i));
074                    }
075                    
076                    convert(opennlpParse, jCas, topNode, null, offset);
077    
078                    TreebankNodeUtility.initTerminalNodes(topNode, jCas);
079                    topNode.addToIndexes();
080    
081                    return topNode;
082            }
083    
084            public static org.cleartk.syntax.treebank.type.TreebankNode convert(Parse opennlpParse, JCas jCas,
085                            org.cleartk.syntax.treebank.type.TreebankNode uimaNode,
086                            org.cleartk.syntax.treebank.type.TreebankNode parentNode, 
087                            int offset) {
088                    
089                    uimaNode.setNodeType(opennlpParse.getType());
090                    uimaNode.setParent(parentNode);
091    
092                    List<org.cleartk.syntax.treebank.type.TreebankNode> uimaChildren = new ArrayList<org.cleartk.syntax.treebank.type.TreebankNode>();
093    
094                Parse[] parseChildren = opennlpParse.getChildren();
095                if(parseChildren.length > 0)
096                    uimaNode.setLeaf(false);
097                else {
098                    uimaNode.setLeaf(true);
099                }
100                
101                for (Parse child : parseChildren) {
102                            org.cleartk.syntax.treebank.type.TreebankNode childNode = new org.cleartk.syntax.treebank.type.TreebankNode(
103                                            jCas, offset + child.getSpan().getStart(), offset + child.getSpan().getEnd());
104                            uimaChildren.add(convert(child, jCas, childNode, uimaNode, offset));
105                            childNode.addToIndexes();
106                    }
107                    
108                FSArray uimaChildrenFSArray = UIMAUtil.toFSArray(jCas, uimaChildren);
109                    uimaNode.setChildren(uimaChildrenFSArray);
110                    return uimaNode;
111            }
