042    public class TreebankNodeUtility
043    {
044            public static org.cleartk.syntax.treebank.type.TopTreebankNode convert(TopTreebankNode pojoNode, JCas jCas)
045            {
046                    org.cleartk.syntax.treebank.type.TopTreebankNode uimaNode = 
047                            new org.cleartk.syntax.treebank.type.TopTreebankNode(jCas, 
048                                                                                                                                             pojoNode.getTextBegin(), 
049                                                                                                                                             pojoNode.getTextEnd());
050                    convert(pojoNode, jCas, uimaNode, null);
051                    uimaNode.setTreebankParse(pojoNode.getTreebankParse());
052                    initTerminalNodes(uimaNode, jCas);
053                    uimaNode.addToIndexes();
054                    return uimaNode;
055            }
056    
057            public static void initTerminalNodes(org.cleartk.syntax.treebank.type.TopTreebankNode uimaNode, JCas jCas)
058            {
059                    List<org.cleartk.syntax.treebank.type.TreebankNode> terminals = 
060                            new ArrayList<org.cleartk.syntax.treebank.type.TreebankNode>();
061                    _initTerminalNodes(uimaNode, terminals);
062                    
063                    FSArray terminalsFSArray = new FSArray(jCas, terminals.size());
064                    terminalsFSArray.copyFromArray(terminals.toArray(new FeatureStructure[terminals.size()]), 0, 0, terminals.size());
065                    uimaNode.setTerminals(terminalsFSArray);
066            }
067            
068            private static void _initTerminalNodes(org.cleartk.syntax.treebank.type.TreebankNode node,
069                                                                            List<org.cleartk.syntax.treebank.type.TreebankNode> terminals)
070            {
071                    FSArray children = node.getChildren();
072                    for(int i=0; i<children.size(); i++)
073                    {
074                            org.cleartk.syntax.treebank.type.TreebankNode child = (org.cleartk.syntax.treebank.type.TreebankNode) children.get(i);
075                            if(child.getLeaf())
076                                    terminals.add(child);
077                            else
078                                    _initTerminalNodes(child, terminals);
079                    }
080            }
081    
082            
083            
084            public static org.cleartk.syntax.treebank.type.TreebankNode convert(TreebankNode pojoNode, 
085                                                                                                                                         JCas jCas,
086                                                                                                                                         org.cleartk.syntax.treebank.type.TreebankNode uimaNode,
087                                                                                                                                         org.cleartk.syntax.treebank.type.TreebankNode parentNode)
088            {
089                    uimaNode.setNodeType(pojoNode.getType());
090                    uimaNode.setNodeTags(UIMAUtil.toStringArray(jCas, pojoNode.getTags()));
091                    uimaNode.setNodeValue(pojoNode.getValue());
092                    uimaNode.setLeaf(pojoNode.isLeaf());
093                    uimaNode.setParent(parentNode);
094                    
095                    List<org.cleartk.syntax.treebank.type.TreebankNode> uimaChildren = 
096                            new ArrayList<org.cleartk.syntax.treebank.type.TreebankNode>();
097                    for(TreebankNode child : pojoNode.getChildren())
098                    {
099                            org.cleartk.syntax.treebank.type.TreebankNode childNode = 
100                                    new org.cleartk.syntax.treebank.type.TreebankNode(jCas, 
101                                                                                                                                  child.getTextBegin(), 
102                                                                                                                                              child.getTextEnd());
103                            uimaChildren.add(convert(child, jCas, childNode, uimaNode));
104                            childNode.addToIndexes();
105                    }
106                    FSArray uimaChildrenFSArray = new FSArray(jCas, uimaChildren.size());
107                    uimaChildrenFSArray.copyFromArray(uimaChildren.toArray(new FeatureStructure[uimaChildren.size()]), 0, 0, uimaChildren.size());
108                    uimaNode.setChildren(uimaChildrenFSArray);
109                    return uimaNode;
110            }
