066    public class ArgumentAnnotationHandler implements AnnotationHandler<String> {
067            
068            public ArgumentAnnotationHandler() {
069                    SimpleFeatureExtractor[] tokenExtractors = {
070                                    new SpannedTextExtractor(),
071                                    new TypePathExtractor(Token.class, "stem"),
072                                    new TypePathExtractor(Token.class, "pos") };
073    
074                    SimpleFeatureExtractor[] constituentExtractors = {
075                                    new TypePathExtractor(TreebankNode.class, "nodeType"),
076                                    // new TypePathExtractor(TreebankNode.class, "nodeTags"),
077                                    new HeadWordExtractor(new CombinedExtractor(tokenExtractors), true, true) };
078    
079                    predicateTokenExtractor = new CombinedExtractor("PredicateToken",
080                                    tokenExtractors);
081                    predicateNodeExtractor = new SubCategorizationExtractor("PredicateNode");
082    
083                    pathExtractor = new SyntacticPathExtractor("ConstituentPredicate",
084                                    new TypePathExtractor(TreebankNode.class, "nodeType"));
085                    partialPathExtractor = new SyntacticPathExtractor("ConstituentPredicate",
086                                    new TypePathExtractor(TreebankNode.class, "nodeType"),
087                                    true);
088                    relPosExtractor = new RelativePositionExtractor("ConstituentPredicate");
089                    distanceExtractor = new DistanceExtractor("ConstituentPredicate", Token.class);
090    
091                    constituentExtractor = new CombinedExtractor("Constituent",
092                                    constituentExtractors);
093                    leftSiblingExtractor = new CombinedExtractor("LeftSibling",
094                                    constituentExtractors);
095                    rightSiblingExtractor = new CombinedExtractor("RightSibling",
096                                    constituentExtractors);
097                    parentExtractor = new CombinedExtractor("Parent", constituentExtractors);
098                    firstWordExtractor = new WindowExtractor("Constituent", Token.class,
099                                    new CombinedExtractor(tokenExtractors),
100                                    WindowFeature.ORIENTATION_MIDDLE, 0, 1);
101                    lastWordExtractor = new WindowExtractor("Constituent", Token.class,
102                                    new CombinedExtractor(tokenExtractors),
103                                    WindowFeature.ORIENTATION_MIDDLE_REVERSE, 0, 1);
104            }

106            public void initialize(UimaContext context) throws ResourceInitializationException {
107            }
108    
109            public void process(JCas jCas, InstanceConsumer<String> consumer) {
110                    List<Sentence> sentences = AnnotationRetrieval.getAnnotations(jCas, Sentence.class);
111            
112                    for( Sentence sentence : sentences ) {
113                            processSentence(jCas, sentence, consumer);
114                    }               
115            }
116    
117            public void processSentence(JCas jCas, Sentence sentence, InstanceConsumer<String> consumer) {
118    
119                    List<TreebankNode> sentenceConstituents = new ArrayList<TreebankNode>(
120                                    80);
121                    collectConstituents(sentence.getConstituentParse(),
122                                    sentenceConstituents);
123    
124                    List<List<Feature>> sentenceConstituentFeatures = new ArrayList<List<Feature>>(
125                                    sentenceConstituents.size());
126                    for (TreebankNode constituent : sentenceConstituents)
127                            sentenceConstituentFeatures.add(extractConstituentFeatures(jCas,
128                                            constituent, sentence));
129    
130                    List<Predicate> predicates = AnnotationRetrieval.getAnnotations(jCas,
131                                    sentence, Predicate.class);
132                    for (Predicate predicate : predicates) {
133                            processPredicate(jCas, sentence, predicate, sentenceConstituents,
134                                            sentenceConstituentFeatures, consumer);
135                    }
136    
137            }
138    
139            void processPredicate(JCas jCas, Sentence sentence, Predicate predicate,
140                            List<TreebankNode> constituents,
141                            List<List<Feature>> constituentFeatures, InstanceConsumer<String> consumer) {
142                    TreebankNode predicateNode = AnnotationRetrieval.getMatchingAnnotation(
143                                    jCas, predicate.getAnnotation(), TreebankNode.class);
144                    Token predicateToken = AnnotationRetrieval.getMatchingAnnotation(jCas,
145                                    predicate.getAnnotation(), Token.class);
146    
147                    List<Feature> predicateFeatures = new ArrayList<Feature>(20);
148                    predicateFeatures.addAll(predicateTokenExtractor.extract(
149                                    jCas, predicateToken));
150                    predicateFeatures.addAll(predicateNodeExtractor.extract(
151                                    jCas, predicateNode));
152    
153                    for( int i=0; i<constituents.size(); i++ ) {
154                            TreebankNode constituent = constituents.get(i);
155                            
156                            Instance<String> instance = new Instance<String>();
157    
158                            instance.addAll(predicateFeatures);
159                            instance.addAll(constituentFeatures.get(i));
160                            instance.addAll(relPosExtractor.extract(jCas,
161                                            constituent, predicate.getAnnotation()));
162                            instance.addAll(pathExtractor.extract(jCas,
163                                            constituent, predicateNode));
164                            instance.addAll(partialPathExtractor.extract(jCas, constituent, predicateNode));
165                            instance.addAll(distanceExtractor.extract(jCas, constituent, predicateNode));
166                            
167    
168                            instance.setOutcome("NULL");
169                            
170                            for (Argument arg : UIMAUtil.toList(predicate
171                                            .getArguments(), Argument.class)) {
172                                    if( !(arg instanceof SemanticArgument) )
173                                            continue;
174                                    
175                                    SemanticArgument sarg = (SemanticArgument) arg;
176                                    
177                                    if (sarg.getAnnotation() == constituent
178                                                    && !sarg.getLabel().equals("rel")) {
179                                            if (sarg.getFeature() != null) {
180                                                    instance.setOutcome(sarg.getLabel() + "-" + sarg.getFeature());
181                                            } else {
182                                                    instance.setOutcome(sarg.getLabel());
183                                            }
184                                            break;
185                                    }
186                            }
187    
188                            String outcome = consumer.consume(instance);
189                            
190    
191                            if( outcome != null ) {
192                                    if (!outcome.equals("NULL")) {
193                                            SemanticArgument arg = new SemanticArgument(jCas);
194                                            arg.setAnnotation(constituent);
195                                            arg.setBegin(constituent.getBegin());
196                                            arg.setEnd(constituent.getEnd());
197    
198                                            String[] parts = outcome.split("-", 1);
199                                            arg.setLabel(parts[0]);
200                                            if (parts.length > 1)
201                                                    arg.setFeature(parts[1]);
202    
203                                            arg.addToIndexes();
204    
205                                            List<Argument> args;
206                                            if (predicate.getArguments() != null)
207                                                    args = UIMAUtil.toList(predicate.getArguments(),
208                                                                    Argument.class);
209                                            else
210                                                    args = new ArrayList<Argument>(1);
211                                            args.add(arg);
212                                            predicate.setArguments(UIMAUtil.toFSArray(jCas, args));
213                                    }
214                            }
215                    }
216    
217            }
218    
219            List<Feature> extractConstituentFeatures(JCas jCas,
220                            TreebankNode constituent, Sentence sentence) {
221                    List<Feature> features = new ArrayList<Feature>(20);
222                    features.addAll(constituentExtractor.extract(jCas, constituent));
223                    features
224                                    .addAll(firstWordExtractor.extract(jCas, constituent, sentence));
225                    features.addAll(lastWordExtractor.extract(jCas, constituent, sentence));
226    
227                    TreebankNode parent = constituent.getParent();
228                    if (parent != null) {
229                            features.addAll(parentExtractor.extract(jCas, parent));
230    
231                            int constituentPosition = 0;
232                            while (parent.getChildren(constituentPosition) != constituent)
233                                    constituentPosition += 1;
234    
235                            if (constituentPosition > 0)
236                                    features.addAll(leftSiblingExtractor.extract(jCas, parent
237                                                    .getChildren(constituentPosition - 1)));
238    
239                            if (constituentPosition < parent.getChildren().size() - 1)
240                                    features.addAll(rightSiblingExtractor.extract(jCas, parent
241                                                    .getChildren(constituentPosition + 1)));
242                    }
243    
244                    return features;
245            }
246    
247            void collectConstituents(TreebankNode top, List<TreebankNode> constituents) {
248                    if (top == null)
249                            throw new IllegalArgumentException();
250    
251                    if (!(top instanceof TopTreebankNode))
252                            constituents.add(top);
253    
254                    if (top.getChildren() == null)
255                            return;
256    
257                    int numberOfChildren = top.getChildren().size();
258                    for (int i = 0; i < numberOfChildren; i++) {
259                            collectConstituents(top.getChildren(i), constituents);
260                    }
261            }
