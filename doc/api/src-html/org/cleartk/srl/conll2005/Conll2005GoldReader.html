024    package org.cleartk.srl.conll2005;
025    
026    import java.io.BufferedReader;
027    import java.io.File;
028    import java.io.FileInputStream;
029    import java.io.IOException;
030    import java.io.InputStream;
031    import java.io.InputStreamReader;
032    import java.util.zip.GZIPInputStream;
033    
034    import org.apache.uima.cas.CAS;
035    import org.apache.uima.cas.CASException;
036    import org.apache.uima.collection.CollectionException;
037    import org.apache.uima.collection.CollectionReader_ImplBase;
038    import org.apache.uima.jcas.JCas;
039    import org.apache.uima.resource.ResourceInitializationException;
040    import org.apache.uima.util.Progress;
041    import org.apache.uima.util.ProgressImpl;
042    import org.cleartk.ViewNames;
043    import org.cleartk.util.UIMAUtil;
044    import org.cleartk.util.ViewURIUtil;
045

047     * <br>Copyright (c) 2007-2008, Regents of the University of Colorado

049    
050     */
051    public class Conll2005GoldReader extends CollectionReader_ImplBase {
052    
053            /**
054             * "org.cleartk.srl.conll2005.Conll2005GoldReader.PARAM_CONLL_2005_DATA_FILE"
055             * is a single, required, string parameter that provides the path of the
056             * CoNLL 2005 data file.
057             */
058            public static final String PARAM_CONLL_2005_DATA_FILE = "org.cleartk.srl.conll2005.Conll2005GoldReader.PARAM_CONLL_2005_DATA_FILE";
059                                                                                      
060            BufferedReader reader;
061            
062            boolean finished = false;
063            
064            int documentNumber;
065            
066            int totalDocuments;
067            
068            @Override
069            public void initialize() throws ResourceInitializationException
070            {
071                    try
072                    {
073                            File dataFile = new File((String)UIMAUtil.getRequiredConfigParameterValue(
074                                            this.getUimaContext(), PARAM_CONLL_2005_DATA_FILE));
075    
076                            InputStream in;
077                            if( dataFile.getName().endsWith(".gz") )
078                                    in = new GZIPInputStream(new FileInputStream(dataFile));
079                            else
080                                    in = new FileInputStream(dataFile);
081                            reader = new BufferedReader(new InputStreamReader(in));
082                            String line;
083                            totalDocuments = 0;
084                            do {
085                                    line = reader.readLine();
086                                    while( line != null && line.trim().length() == 0 )
087                                            line = reader.readLine();
088                                    
089                                    if( line == null )
090                                            break;
091                                    
092                                    totalDocuments += 1;
093                                    while( line != null && line.trim().length() > 0 )
094                                            line = reader.readLine();
095                            } while( line != null );                        
096                            reader.close();
097                            
098                            if( dataFile.getName().endsWith(".gz") )
099                                    in = new GZIPInputStream(new FileInputStream(dataFile));
100                            else
101                                    in = new FileInputStream(dataFile);
102                            reader = new BufferedReader(new InputStreamReader(in));
103                            documentNumber = 0;
104                            
105                            super.initialize();
106                    } catch (IOException e) {
107                            throw new ResourceInitializationException(e);
108                    }
109            }
110    
111            public void getNext(CAS cas) throws IOException, CollectionException {
112                    try {
113                            JCas conllView = cas.createView(ViewNames.CONLL_2005).getJCas();
114                            
115                            String lineBuffer;
116                            StringBuffer docBuffer = new StringBuffer();
117                            
118                            lineBuffer = reader.readLine();
119                            while( lineBuffer != null && lineBuffer.trim().length() == 0 )
120                                    lineBuffer = reader.readLine();
121                            
122                            if( lineBuffer == null )
123                                    throw new CollectionException("unexpected end of input", null);
124                            
125                            while( lineBuffer != null && lineBuffer.trim().length() != 0 ) {
126                                    docBuffer.append(lineBuffer.trim());
127                                    docBuffer.append("\n");
128                                    lineBuffer = reader.readLine();
129                            }
130                            
131                            documentNumber += 1;
132                            
133                            if( documentNumber == totalDocuments )
134                                    finished = true;
135                            
136                            conllView.setSofaDataString(docBuffer.toString(), "text/plain");
137                            ViewURIUtil.setURI(cas, String.valueOf(documentNumber));
138                    } catch (CASException e) {
139                            throw new CollectionException(e);
140                    }
141            }
142    
143            public void close() throws IOException {
144                    reader.close();
145            }
146    
147            public Progress[] getProgress() {
148                    return new Progress[] { new ProgressImpl(documentNumber, totalDocuments, Progress.ENTITIES) };
149            }
150    
151            public boolean hasNext() throws IOException, CollectionException {
152                    return !finished;
153            }
154    
155    }
