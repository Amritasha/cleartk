047     * <p>
048     * A <em>Proplabel object</em> represents one label of an entry in Propbank.
049     * </p>
050     * 
051     * @author Philipp Wetzler, Steven Bethard
052     */
053    public class Proplabel {
054            /**
055             * Parses one label taken form a Propbank entry and returns its
056             * representation as a <em>Proplabel</em> object.
057             * 
058             * @param lblTxt
059             *            one label part of one line from <tt>prop.txt</tt>
060             * 
061             * @return a <em>Proplabel</em> object representing <b>lblTxt</b>
062             */
063            static Proplabel fromString(String lblTxt) {
064                    // split the string by hyphens and catch some simple errors
065                    String[] columns = lblTxt.split("-");
066                    if (columns.length < 1) {
067                            throw new PropbankFormatException(String.format(
068                                            "Missing label: %s", lblTxt));
069                    }
070                    if (!Proplabel.labels.contains(columns[1])) {
071                            throw new PropbankFormatException(String.format(
072                                            "Invalid label: %s", columns[1]));
073                    }
074                    
075            
076                    // set the relation and label
077                    Proplabel proplabel = new Proplabel();
078                    proplabel.setRelation(PropbankRelation.fromString(columns[0]));
079                    proplabel.setLabel(columns[1]);
080                    
081                    // second column may be feature, hyphen tag or preposition
082                    // third column may only be hyphen tag following feature
083                    int expectedLength = 2;
084                    if (columns.length > 2) {
085                            if (Proplabel.features.contains(columns[2])) {
086                                    proplabel.setFeature(columns[2]);
087                                    if (columns.length > 3) {
088                                            if (Proplabel.hyphenTags.contains(columns[3])) {
089                                                    proplabel.setHyphenTag(columns[3]);
090                                            }
091                                            expectedLength = 4;
092                                    } else {
093                                            expectedLength = 3;
094                                    }
095                            } else if (Proplabel.hyphenTags.contains(columns[2])) {
096                                    proplabel.setHyphenTag(columns[2]);
097                                    expectedLength = 3;
098                            } else {
099                                    proplabel.setPreposition(columns[2]);
100                                    expectedLength = 3;
101                            }
102                    }
103                    
104                    // throw some exceptions for bad input
105                    if (columns.length != expectedLength) {
106                            throw new PropbankFormatException(String.format(
107                                            "Expected %d items, found %d", expectedLength, columns.length));
108                    }
109                    if (Proplabel.labelsRequiringFeatures.contains(columns[1])) {
110                            if (proplabel.getFeature() == null) {
111                                    throw new PropbankFormatException(String.format(
112                                                    "Label %s requires a feature", proplabel.getLabel()));
113                            }
114                    }
115                    return proplabel;
116            }

175            /**
176             * Convert to ClearTK <em>SemanticArgument</em> annotation and add it to
177             * <b>view</b>.
178             * 
179             * @param view
180             * @param topNode
181             *            the top node annotation of the corresponding Treebank parse
182             * @return the generated <em>SemanticArgument</em> annotation
183             */
184            public SemanticArgument convert(JCas view, TopTreebankNode topNode) {
185                    SemanticArgument argument = new SemanticArgument(view);
186                    argument.setLabel(this.label);
187                    argument.setFeature(this.feature);
188                    argument.setPreposition(this.preposition);
189                    argument.setHyphenTag(this.hyphenTag);
190                    if (this.relation instanceof PropbankCorefRelation) {
191                            List<SimpleAnnotation> annotations = new ArrayList<SimpleAnnotation>();
192                            List<SimpleAnnotation> substantiveAnnotations = new ArrayList<SimpleAnnotation>();
193    
194                            for (PropbankRelation rel : ((PropbankCorefRelation) this.relation)
195                                            .getCorefRelations()) {
196                                    SimpleAnnotation a = rel.convert(view, topNode);
197                                    annotations.add(a);
198                                    if (a.getBegin() != a.getEnd())
199                                            substantiveAnnotations.add(a);
200                            }
201                            argument.setCoreferenceAnnotations(UIMAUtil.toFSArray(view,
202                                            annotations));
203    
204                            Span extent = AnnotationUtil
205                                            .getAnnotationsExtent(substantiveAnnotations);
206                            argument.setBegin(extent.getStart());
207                            argument.setEnd(extent.getEnd());
208    
209                            if (substantiveAnnotations.size() == 1) {
210                                    argument.setAnnotation(substantiveAnnotations.get(0));
211                            }
212                    } else {
213                            argument.setAnnotation(this.relation.convert(view, topNode));
214                            argument.setBegin(argument.getAnnotation().getBegin());
215                            argument.setEnd(argument.getAnnotation().getEnd());
216                    }
217                    argument.addToIndexes();
218    
219                    return argument;
220            }
221    
222            /**
223             * Re-generate the Propbank text that this object was parsed from.
224             */
225            @Override
226            public String toString() {
227                    StringBuffer buffer = new StringBuffer();
228    
229                    buffer.append(getRelation().toString());
230                    buffer.append("-" + getLabel());
231                    if (getFeature() != null)
232                            buffer.append("-" + getFeature());
233                    if (getHyphenTag() != null)
234                            buffer.append("-" + getHyphenTag());
235                    if (getPreposition() != null)
236                            buffer.append("-" + getPreposition());
237    
238                    return buffer.toString();
239            }

241            private static final Set<String> labels = new HashSet<String>(Arrays.asList(
242                            "rel|Support|ARG0|ARG1|ARG2|ARG3|ARG4|ARG5|ARGA|ARGM".split("\\|")));
243            private static final Set<String> labelsRequiringFeatures = new HashSet<String>(Arrays.asList(
244                            new String[]{"ARGM"}));
245            private static final Set<String> features = new HashSet<String>(Arrays.asList(
246                            "ADV|CAU|DIR|DIS|EXT|LOC|MNR|MOD|NEG|PNC|PRD|REC|TMP".split("\\|")));
247            private static final Set<String> hyphenTags = new HashSet<String>(Arrays.asList(
248                            "H0|H1|H2|H3|H4|H5|H6|H7|H8|H9|XX".split("\\|")));
