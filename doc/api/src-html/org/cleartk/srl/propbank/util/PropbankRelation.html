024    package org.cleartk.srl.propbank.util;
025    
026    import org.apache.uima.jcas.JCas;
027    import org.cleartk.syntax.treebank.type.TopTreebankNode;
028    import org.cleartk.type.SimpleAnnotation;
029    
030    
031    /**
032     * <br>Copyright (c) 2007-2008, Regents of the University of Colorado 
033     * <br>All rights reserved.
034    
035     *
036     * 
037     * <p>
038     * A <em>PropbankRelation object</em> represents the relation (or a
039     * sub-relation) of one label of an entry in Propbank.
040     * </p>
041     * 
042     * @author Philipp Wetzler
043     */
044    public abstract class PropbankRelation {
045            /**
046             * Parses a relation taken form a Propbank entry and returns its
047             * representation as the appropriate <em>PropbankRelation</em> object.
048             * 
049             * @param s
050             *            the textual representation of a relation taken from
051             *            <tt>prop.txt</tt>
052             * 
053             * @return a <em>PropbankRelation</em> object representing <b>s</b>
054             */
055            public static PropbankRelation fromString(String s) {
056                    if (s.contains("*")) {
057                            return PropbankCorefRelation.fromString(s);
058                    } else if (s.contains(",")) {
059                            return PropbankSplitRelation.fromString(s);
060                    } else if (s.contains(":")) {
061                            return PropbankNodeRelation.fromString(s);
062                    } else {
063                            return PropbankTerminalRelation.fromString(s);
064                    }
065            }
066    
067            /**
068             * Convert to an appropriate ClearTK annotation and add it to <b>view</b>
069             * if necessary.
070             * 
071             * @param jCas
072             * @param topNode
073             *            the top node annotation of the corresponding Treebank parse
074             * @return the corresponding annotation
075             */
076            public abstract SimpleAnnotation convert(JCas jCas, TopTreebankNode topNode);
077    
078            /**
079             * Re-generate the text that this object was parsed from.
080             */
081            @Override
082            public abstract String toString();
083    }
