064     * 
065     * <p>
066     * ArgumentIdentifier can work in 3 modes: <il>
067     * <li> <b>training mode</b>: take in fully annotated Propbank style data and
068     * generate training data for detection of arguments </li>
069     * <li> <b>filter mode</b>: take in fully annotated Propbank style data and a
070     * model, then add annotations for falsely detected arguments, and remove
071     * annotations for missed arguments; this is to facilitate training of
072     * AnnotationClassifier </li>
073     * <li> <b>annotation mode</b>: take in unlabeled Treebank style data and
074     * annotate all detected arguments (no labeling is done by this annotator) </li>
075     * </il>
076     * </p>
077     * 
078     * @author Philipp Wetzler, Philip Ogren
079     */
080    public class ArgumentIdentificationHandler implements AnnotationHandler<Boolean> {
081    
082            public ArgumentIdentificationHandler() {
083                    SimpleFeatureExtractor[] tokenExtractors = {
084                                    new SpannedTextExtractor(),
085                                    new TypePathExtractor(Token.class, "stem"),
086                                    new TypePathExtractor(Token.class, "pos") };
087    
088                    SimpleFeatureExtractor[] constituentExtractors = {
089                                    new TypePathExtractor(TreebankNode.class, "nodeType"),
090                                    // new TypePathExtractor(TreebankNode.class, "nodeTags"),
091                                    new HeadWordExtractor(new CombinedExtractor(tokenExtractors)) };
092    
093                    predicateTokenExtractor = new CombinedExtractor("PredicateToken",
094                                    tokenExtractors);
095                    predicateNodeExtractor = new SubCategorizationExtractor("PredicateNode");
096    
097                    constituentExtractor = new CombinedExtractor("Constituent",
098                                    constituentExtractors);
099                    // leftSiblingExtractor = new CombinedExtractor("LeftSibling",
100                    // constituentExtractors);
101                    // rightSiblingExtractor = new CombinedExtractor("RightSibling",
102                    // constituentExtractors);
103                    // parentExtractor = new CombinedExtractor("Parent",
104                    // constituentExtractors);
105    
106                    firstWordExtractor = new WindowExtractor("Constituent", Token.class,
107                                    new CombinedExtractor(tokenExtractors),
108                                    WindowFeature.ORIENTATION_MIDDLE, 0, 1);
109    
110                    lastWordExtractor = new WindowExtractor("Constituent", Token.class,
111                                    new CombinedExtractor(tokenExtractors),
112                                    WindowFeature.ORIENTATION_MIDDLE_REVERSE, 0, 1);
113    
114                    pathExtractor = new SyntacticPathExtractor("ConstituentPredicate",
115                                    new TypePathExtractor(TreebankNode.class, "nodeType"));
116                    relPosExtractor = new RelativePositionExtractor("ConstituentPredicate");
117            }

119            public void initialize(UimaContext context)
120            throws ResourceInitializationException {
121                    Boolean value = (Boolean) context.getConfigParameterValue("FilterMode");
122                    if (value != null && value)
123                            this.filterMode = true;
124            }

126            public void process(JCas jCas, InstanceConsumer<Boolean> consumer) {
127                    /*
128                     * Iterate over sentences in document
129                     */
130                    List<Sentence> sentences = AnnotationRetrieval.getAnnotations(jCas, Sentence.class);
131    
132                    for( Sentence sentence : sentences ) {
133                            processSentence(jCas, sentence, consumer);
134                    }
135            }
136    
137            void processSentence(JCas jCas, Sentence sentence, InstanceConsumer<Boolean> consumer) {
138    
139                    /*
140                     * Pre-compute sentence level data: sentenceConstituents: list of all
141                     * constituents in sentence
142                     */
143                    List<TreebankNode> sentenceConstituents = new ArrayList<TreebankNode>(
144                                    200);
145                    collectConstituents(sentence.getConstituentParse(),
146                                    sentenceConstituents);
147    
148                    /*
149                     * Compute constituent features for all constituents in sentence
150                     */
151                    List<List<Feature>> sentenceConstituentFeatures = new ArrayList<List<Feature>>(
152                                    sentenceConstituents.size());
153                    for (TreebankNode constituent : sentenceConstituents) {
154                            sentenceConstituentFeatures.add(extractConstituentFeatures(jCas, constituent, sentence));
155                    }
156    
157                    /*
158                     * Iterate over predicates in sentence
159                     */
160                    List<Predicate> predicates = AnnotationRetrieval.getAnnotations(jCas,
161                                    sentence, Predicate.class);
162                    for (Predicate predicate : predicates) {
163                            processPredicate(jCas, predicate, sentenceConstituents,
164                                            sentenceConstituentFeatures, consumer);
165                    }
166            }
167    
168            public void processPredicate(JCas jCas, Predicate predicate,
169                            List<TreebankNode> sentenceConstituents,
170                            List<List<Feature>> sentenceConstituentFeatures,
171                            InstanceConsumer<Boolean> consumer) {
172                    /*
173                     * Pre-compute predicate level data: predicateArguments: all semantic
174                     * arguments of the predicate predicateNode: the constituent of the
175                     * predicate predicateToken: the token of the predicate (first if there
176                     * are multiple)
177                     */
178                    Map<Integer, SemanticArgument> predicateArguments = new TreeMap<Integer, SemanticArgument>();
179                    int numberOfArgs = predicate.getArguments().size();
180                    for (int i = 0; i < numberOfArgs; i++) {
181                            Argument arg = predicate.getArguments(i);
182                            if (arg instanceof SemanticArgument
183                                            && arg.getAnnotation() instanceof TreebankNode) {
184                                    predicateArguments.put(buildKey(arg.getAnnotation()),
185                                                    (SemanticArgument) arg);
186                            }
187                    }
188                    boolean modifiedPredicateArguments = false;
189    
190    
191                    TreebankNode predicateNode = AnnotationRetrieval.getMatchingAnnotation(
192                                    jCas, predicate.getAnnotation(), TreebankNode.class);
193                    Token predicateToken = AnnotationRetrieval.getMatchingAnnotation(jCas,
194                                    predicate.getAnnotation(), Token.class);
195    
196    
197                    /*
198                     * Compute predicate features: predicateTokenFeatures
199                     * predicateNodeFeatures
200                     */
201                    List<Feature> predicateFeatures = new ArrayList<Feature>(12);
202                    predicateFeatures.addAll(predicateTokenExtractor.extract(jCas,
203                                    predicateToken));
204                    predicateFeatures.addAll(predicateNodeExtractor.extract(jCas,
205                                    predicateNode));
206    
207    
208                    /*
209                     * Iterate over constituents in sentence
210                     */
211                    for (int i = 0; i < sentenceConstituents.size(); i++) {
212                            TreebankNode constituent = sentenceConstituents.get(i);
213    
214                            Instance<Boolean> instance = new Instance<Boolean>();
215    
216                            /*
217                             * Compute predicate-constituent features: relPosFeatures
218                             * pathFeatures
219                             */
220                            instance.addAll(relPosExtractor.extract(jCas, constituent,
221                                            predicate.getAnnotation()));
222                            instance.addAll(pathExtractor.extract(jCas, constituent,
223                                            predicateNode));
224    
225                            /*
226                             * Add constituent features
227                             */
228                            instance.addAll(sentenceConstituentFeatures.get(i));
229    
230                            /*
231                             * Add predicate features
232                             */
233                            instance.addAll(predicateFeatures);
234    
235                            instance.setOutcome(predicateArguments.containsKey(buildKey(constituent)));
236    
237    
238                            Boolean outcome = consumer.consume(instance);
239    
240                            if( outcome != null ) {
241                                    if (this.filterMode) {
242                                            boolean isIdentified = outcome;
243                                            boolean isArgument = predicateArguments
244                                            .containsKey(buildKey(constituent));
245    
246                                            if (!isIdentified && isArgument) {
247                                                    predicateArguments.get(buildKey(constituent))
248                                                    .removeFromIndexes();
249    
250                                                    predicateArguments.remove(buildKey(constituent));
251                                                    modifiedPredicateArguments = true;
252                                            } else if (isIdentified && !isArgument) {
253                                                    SemanticArgument arg = new SemanticArgument(jCas);
254                                                    arg.setAnnotation(constituent);
255                                                    arg.setBegin(constituent.getBegin());
256                                                    arg.setEnd(constituent.getEnd());
257                                                    arg.setLabel("NULL");
258                                                    arg.addToIndexes();
259    
260                                                    predicateArguments.put(buildKey(constituent), arg);
261                                                    modifiedPredicateArguments = true;
262                                            }
263                                    } else {
264                                            boolean isArgument = outcome;
265    
266                                            if (isArgument) {
267                                                    SemanticArgument arg = new SemanticArgument(jCas);
268                                                    arg.setAnnotation(constituent);
269                                                    arg.setBegin(constituent.getBegin());
270                                                    arg.setEnd(constituent.getEnd());
271                                                    arg.setLabel("?");
272                                                    arg.addToIndexes();
273    
274                                                    predicateArguments.put(buildKey(constituent), arg);
275                                            }
276                                    }
277                            }
278    
279                            /*
280                             * Update predicates list of arguments in the CAS, if it was modified
281                             */
282                            if (modifiedPredicateArguments) {
283                                    FSArray args = new FSArray(jCas, predicateArguments.size());
284                                    int index = 0;
285                                    for (SemanticArgument arg : predicateArguments.values()) {
286                                            args.set(index, arg);
287                                            index += 1;
288                                    }
289                                    predicate.setArguments(args);
290                            }
291                    }
292            }
293    
294            Integer buildKey(Annotation annotation) {
295                    return annotation.getAddress();
296            }
297    
298            /**
299             * Recursively build a list of constituents under a TreebankNode.
300             * 
301             * @param top
302             *            the root of the parse tree to operate on; <b>top</b> itself
303             *            will also be added, unless it is of type TopTrebankNode
304             * @param constituents
305             *            list of nodes to add to
306             */
307            protected void collectConstituents(TreebankNode top,
308                            List<TreebankNode> constituents) {
309                    if (top == null)
310                            throw new IllegalArgumentException();
311    
312                    if (!(top instanceof TopTreebankNode))
313                            constituents.add(top);
314    
315                    if (top.getChildren() == null)
316                            return;
317    
318                    int numberOfChildren = top.getChildren().size();
319                    for (int i = 0; i < numberOfChildren; i++) {
320                            collectConstituents(top.getChildren(i), constituents);
321                    }
322            }
323

326                    features.addAll(this.constituentExtractor
327                                    .extract(jCas, constituent));
328                    features.addAll(this.firstWordExtractor.extract(jCas, constituent, sentence));
329                    features.addAll(this.lastWordExtractor.extract(jCas, constituent, sentence));
330    
331                    // TreebankNode parent = constituent.getParent();

334                    //

337                    // constituent)

339                    //
340                    // if (constituentPosition > 0)
341                    // features.addAll(this.leftSiblingExtractor.extract(jCas,
342                    // parent.getChildren(constituentPosition - 1)));
343                    // if (constituentPosition < parent.getChildren().size() - 1)

345                    // jCas, parent
346                    // .getChildren(constituentPosition + 1)));
347                    // }
348    
349                    return features;
350            }
