172            public static <OUTCOME_TYPE> AnalysisEngineDescription createViterbiDataWriterAnnotator(
173                            Class<? extends SequentialAnnotationHandler<OUTCOME_TYPE>> annotationHandlerClass,
174                            Class<? extends DataWriterFactory<OUTCOME_TYPE>> delegatedDataWriterFactoryClass,

176            throws ResourceInitializationException {
177                    return AnalysisEngineFactory.createPrimitiveAnalysisEngineDescription(
178                                    SequentialDataWriterAnnotator.class, TYPE_SYSTEM_DESCRIPTION, TYPE_PRIORITIES,

184                                                    SequentialDataWriterAnnotator.PARAM_OUTPUT_DIRECTORY, outputDir,
185                                                    ViterbiDataWriter.PARAM_DELEGATED_DATAWRITER_FACTORY_CLASS,
186                                                    delegatedDataWriterFactoryClass.getName(),

188                                                    new String[]{DefaultOutcomeFeatureExtractor.class.getName()}));

264            private static String getParameterValue(String paramName, String defaultValue) {
265                    String value = System.getProperty(paramName);
266                    if (value == null) {
267                            value = defaultValue;
268                    }
269                    return value;
270            }
271            
272            private static Object[] combineParams(Object[] oldParams, Object ... newParams) {
273                    Object[] combined = new Object[oldParams.length + newParams.length];
274                    System.arraycopy(oldParams, 0, combined, 0, oldParams.length);
275                    System.arraycopy(newParams, 0, combined, oldParams.length, newParams.length);
276                    return combined;
277            }
