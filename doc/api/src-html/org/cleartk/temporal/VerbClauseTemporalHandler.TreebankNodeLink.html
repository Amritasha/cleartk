086            public void initialize(UimaContext context) throws ResourceInitializationException {
087                    this.eventID = 1;
088                    this.tokenFeatureExtractors = new ArrayList<SimpleFeatureExtractor>();
089                    this.tokenFeatureExtractors.add(new SpannedTextExtractor());
090                    this.tokenFeatureExtractors.add(new TypePathExtractor(Token.class, "pos"));
091                    this.tokenFeatureExtractors.add(new TypePathExtractor(Token.class, "stem"));
092                    this.pathExtractor = new SyntacticPathExtractor(
093                                    new TypePathExtractor(TreebankNode.class, "nodeType"));
094            }

096            public void process(JCas jCas, InstanceConsumer<String> consumer)
097                            throws AnalysisEngineProcessException {
098                    int docEnd = jCas.getDocumentText().length();
099    
100                    // collect TLINKs if necessary
101                    Map<String, TemporalLink> tlinks = null;
102                    if (consumer.expectsOutcomes()) {
103                            tlinks = this.getTemporalLinks(jCas);
104                    }
105    
106                    // look for verb-clause pairs in each sentence in the document
107                    for (Sentence sentence: AnnotationRetrieval.getAnnotations(jCas, Sentence.class)) {
108                            TopTreebankNode tree = sentence.getConstituentParse();
109                            if (tree == null) {
110                                    throw new AnalysisEngineProcessException(new Exception(String.format(
111                                                    "%s: missing syntactic parses", DocumentUtil.getIdentifier(jCas))));
112                            }
113    
114                            // iterate over all verb-clause pairs
115                            List<TreebankNodeLink> links = new ArrayList<TreebankNodeLink>();
116                            this.collectVerbClausePairs(tree, links);
117                            for (TreebankNodeLink link: links) {
118                                    
119                                    Token sourceToken = AnnotationRetrieval.getAnnotations(
120                                                    jCas, link.source, Token.class).get(0);
121                                    Token targetToken = AnnotationRetrieval.getAnnotations(
122                                                    jCas, link.target, Token.class).get(0);
123                                    
124                                    // create an instance and populate it with features
125                                    Instance<String> instance = new Instance<String>();
126                                    for (SimpleFeatureExtractor extractor: this.tokenFeatureExtractors) {
127                                            instance.addAll(extractor.extract(jCas, sourceToken));
128                                            instance.addAll(extractor.extract(jCas, targetToken));
129                                    }
130                                    instance.addAll(this.pathExtractor.extract(
131                                                    jCas, link.source, link.target));
132                                    
133                                    // find source and target anchors if they're available
134                                    Anchor source = AnnotationRetrieval.getContainingAnnotation(
135                                                    jCas, link.source, Anchor.class);
136                                    Anchor target = AnnotationRetrieval.getContainingAnnotation(
137                                                    jCas, link.target, Anchor.class);
138    
139                                    // if the consumer expects outcomes, get the relation
140                                    // type from a TLINK annotation
141                                    if (consumer.expectsOutcomes()) {
142                                            if (source != null && target != null) {
143                                                    String key = String.format("%s:%s", source.getId(), target.getId());
144                                                    TemporalLink tlink = tlinks.remove(key);
145                                                    if (tlink != null) {
146                                                            instance.setOutcome(tlink.getRelationType());
147                                                            consumer.consume(instance);
148                                                    }
149                                            }
150                                    }
151                                    
152                                    // if the consumer doesn't expect outcomes, send all instances
153                                    // to the consumer, and create a new TLINK with the result
154                                    else {
155                                            String relationType = consumer.consume(instance);
156                                            source = this.getOrCreateEvent(jCas, source, link.source);
157                                            target = this.getOrCreateEvent(jCas, target, link.target);
158                                            TemporalLink tlink = new TemporalLink(jCas, docEnd, docEnd);
159                                            tlink.setSource(source);
160                                            tlink.setTarget(target);
161                                            tlink.setRelationType(relationType);
162                                            tlink.setEventID(source.getId());
163                                            tlink.setRelatedToEvent(target.getId());
164                                            tlink.addToIndexes();
165                                    }
166                            }                       
167                    }
168    //              if (!tlinks.isEmpty()) {
169    //                      System.err.println();
170    //                      System.err.println(DocumentUtil.getIdentifier(jCas));
171    //                      System.err.println();
172    //                      System.err.println("TLINKs not used:");
173    //                      System.err.println(tlinks.keySet());
174    //                      for (String key: tlinks.keySet()) {
175    //                              TemporalLink tlink = tlinks.get(key); 
176    //                              Event source = (Event)tlink.getSource();
177    //                              Event target = (Event)tlink.getTarget();
178    //                              System.err.format("%s  %s (%s) -> %s (%s) \n", key,
179    //                                              source.getCoveredText(), source.getEventInstanceID(),
180    //                                              target.getCoveredText(), target.getEventInstanceID());
181    //                              Sentence sentence = AnnotationRetrieval.getContainingAnnotation(jCas, source, Sentence.class);
182    //                              System.err.println(sentence.getCoveredText());
183    //                              System.err.println(sentence.getConstituentParse().getTreebankParse());
184    //                      }

189            private Event getOrCreateEvent(JCas jCas, Anchor anchor, TreebankNode node) {
190                    if (anchor != null && anchor instanceof Event) {
191                            return (Event)anchor;
192                    }
193                    Event event = new Event(jCas, node.getBegin(), node.getEnd());
194                    event.setId("e" + this.eventID);
195                    this.eventID++;
196                    event.addToIndexes();
197                    return event;
198            }
199            
200            private Map<String, TemporalLink> getTemporalLinks(JCas jCas) {
201                    Map<String, TemporalLink> tlinks = new HashMap<String, TemporalLink>();
202                    for (TemporalLink tlink: AnnotationRetrieval.getAnnotations(jCas, TemporalLink.class)) {
203                            String sourceID = tlink.getSource().getId();
204                            String targetID = tlink.getTarget().getId();
205                            String key = String.format("%s:%s", sourceID, targetID);
206                            tlinks.put(key, tlink);
207                    }
208                    return tlinks;
209            }
210            
211            private void collectVerbClausePairs(TreebankNode node, List<TreebankNodeLink> links) {
212                    if (this.isVerbPhrase(node)) {
213                            List<TreebankNode> sources = new ArrayList<TreebankNode>();
214                            List<TreebankNode> targets = new ArrayList<TreebankNode>();
215                            this.collectHeads(node, sources);
216                            
217                            // look for clauses in descendants
218                            for (int i = 0; i < node.getChildren().size(); i++) {
219                                    TreebankNode child = node.getChildren(i);
220                                    if (this.isClause(child)) {
221                                            
222                                            // pair the verb phrase heads with the clause heads
223                                            targets.clear();
224                                            this.collectHeads(child, targets);
225                                            for (TreebankNode source: sources) {
226                                                    for (TreebankNode target: targets) {
227                                                            
228                                                            // skip pairs where the head of the VP is inside the clause
229                                                            if (!this.contains(child, source)) {
230                                                                    links.add(new TreebankNodeLink(source, target));
231                                                            }
232                                                    }
233                                            }
234                                    }
235                            }
236                    }
237                    // look for verb phrases in descendants
238                    for (int i = 0; i < node.getChildren().size(); i++) {
239                            TreebankNode child = node.getChildren(i);
240                            this.collectVerbClausePairs(child, links);
241                    }
242            }
243            
244            private void collectHeads(TreebankNode node, List<TreebankNode> heads) {
245                    if (node.getLeaf()) {
246                            heads.add(node);
247                    }
248                    String[] headTypes = VerbClauseTemporalHandler.headMap.get(node.getNodeType());
249                    if (headTypes != null) {
250                            for (String headType: headTypes) {
251                                    boolean foundChildWithHeadType = false;
252                                    for (int i = 0; i < node.getChildren().size(); i++) {
253                                            TreebankNode child = node.getChildren(i);
254                                            if (child.getNodeType().equals(headType)) {
255                                                    String text = child.getCoveredText();
256                                                    if (!VerbClauseTemporalHandler.stopWords.contains(text)) {
257                                                            this.collectHeads(child, heads);
258                                                            foundChildWithHeadType = true;
259                                                    }
260                                            }
261                                    }
262                                    if (foundChildWithHeadType) {
263                                            break;
264                                    }
265                            }
266                    }
267            }
268            
269            private boolean contains(TreebankNode node, TreebankNode descendant) {
270                    if (node == descendant) {
271                            return true;
272                    }
273                    for (int i = 0; i < node.getChildren().size(); i ++) {
274                            boolean result = this.contains(node.getChildren(i), descendant);
275                            if (result) {
276                                    return true;
277                            }
278                    }
279                    return false;
280            }
281            
282            private boolean isVerbPhrase(TreebankNode node) {
283                    return node.getNodeType().startsWith("VP");
284            }
285            
286            private boolean isClause(TreebankNode node) {
287                    return node.getNodeType().startsWith("S");
288            }
289            
290            private class TreebankNodeLink {
291                    public TreebankNode source;
292                    public TreebankNode target;
293                    public TreebankNodeLink(TreebankNode source, TreebankNode target) {
294                            this.source = source;
295                            this.target = target;
296                    }
297            }
