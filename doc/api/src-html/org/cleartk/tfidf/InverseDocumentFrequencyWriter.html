055    public class InverseDocumentFrequencyWriter extends JCasAnnotator_ImplBase {
056            
057            /**
058             * "org.cleartk.tfidf.InverseDocumentFrequencyWriter.PARAM_OUTPUT_FILE"
059             * is a single, required, string parameter that provides the path where the
060             * Map<String, Double> of inverse document frequencies should be written.
061             */
062            public static final String PARAM_OUTPUT_FILE = "org.cleartk.tfidf.InverseDocumentFrequencyWriter.PARAM_OUTPUT_FILE";
063    
064            @Override
065            public void initialize(UimaContext context) throws ResourceInitializationException {
066                    super.initialize(context);
067                    
068                    this.documentFrequencies = new HashMap<String, Integer>();
069                    this.documentCount = 0.0;
070                    this.outputFilePath = (String)UIMAUtil.getRequiredConfigParameterValue(
071                                    context, InverseDocumentFrequencyWriter.PARAM_OUTPUT_FILE);
072            }
073            
074            @Override
075            public void process(JCas jCas) throws AnalysisEngineProcessException {
076                    // iterate over each token in the document
077                    Set<String> stemSet = new HashSet<String>();
078                    DocumentAnnotation doc = AnnotationRetrieval.getDocument(jCas);
079                    for (Token token: AnnotationRetrieval.getAnnotations(jCas, doc, Token.class)) {
080                            String stem = token.getStem();
081                            if (stem == null) {
082                                    stem = token.getCoveredText();
083                            }
084                            stemSet.add(stem);
085                    }
086                    
087                    // increment the document count for each stem found
088                    for (String stem: stemSet) {
089                            int count;
090                            if (documentFrequencies.containsKey(stem)) {
091                                    count = documentFrequencies.get(stem) + 1;
092                            } else {
093                                    count = 1;
094                            }
095                            documentFrequencies.put(stem, count);
096                    }
097                    
098                    // increment the total document count
099                    this.documentCount++;
100            }
101    
102            @Override
103            public void collectionProcessComplete() throws AnalysisEngineProcessException {
104                    super.collectionProcessComplete();
105                    
106                    // convert term-document counts to inverse document frequencies
107                    Map<String, Double> inverseDocumentFrequencies = new HashMap<String, Double>();
108                    for (String term: this.documentFrequencies.keySet()) {
109                            int documentFrequency = this.documentFrequencies.get(term);
110                            double inverseDocumentFrequency = this.documentCount / documentFrequency;
111                            inverseDocumentFrequencies.put(term, Math.log(inverseDocumentFrequency));
112                    }
113                    inverseDocumentFrequencies.put(null, Math.log(this.documentCount));
114    
115                    // save the inverse document frequency map to the file
116                    try {
117                            ObjectOutput output = new ObjectOutputStream(new BufferedOutputStream(
118                                            new FileOutputStream(this.outputFilePath)));
119                            try {
120                                    output.writeObject(inverseDocumentFrequencies);
121                            } finally {
122                                    output.close();
123                            }
124                    } catch (IOException e) {
125                            throw new AnalysisEngineProcessException(e);
126                    }
127            }
128    
129            private Map<String, Integer> documentFrequencies;
130            private double documentCount;
131            private String outputFilePath;
132    
133    }
