024    package org.cleartk.tfidf;
025

027    import java.io.File;
028    import java.io.FileOutputStream;
029    import java.io.IOException;
030    import java.io.ObjectOutput;
031    import java.io.ObjectOutputStream;
032    import java.util.HashMap;
033    import java.util.List;
034    import java.util.Map;
035    
036    import org.apache.uima.UimaContext;
037    import org.apache.uima.analysis_engine.AnalysisEngineProcessException;
038    import org.apache.uima.resource.ResourceInitializationException;
039    import org.cleartk.classifier.Feature;
040    import org.cleartk.classifier.Instance;
041    import org.cleartk.classifier.InstanceConsumer_ImplBase;
042    import org.cleartk.classifier.feature.Counts;
043    import org.cleartk.util.UIMAUtil;
044    
045    
046    /**

050     * @author Philipp G. Wetzler
051     *
052     */
053    public class IDFMapWriter<OUTCOME_TYPE> extends InstanceConsumer_ImplBase<OUTCOME_TYPE> {
054    
055            /**
056             * "org.cleartk.tfidf.IDFMapWriter.PARAM_IDFMAP_FILE"
057             * is a single, required, string parameter that provides the path where the
058             * Map<String, Double> of inverse document frequencies should be written.
059             */
060            public static final String PARAM_IDFMAP_FILE = "org.cleartk.tfidf.IDFMapWriter.PARAM_IDFMAP_FILE";
061    
062            @Override
063            public void initialize(UimaContext context) throws ResourceInitializationException {
064                    super.initialize(context);
065    
066                    idfMapFile = new File((String)UIMAUtil.getRequiredConfigParameterValue(
067                                    context, PARAM_IDFMAP_FILE)).getAbsoluteFile();
068    
069                    if( ! idfMapFile.getParentFile().exists() )
070                            throw new ResourceInitializationException();
071            }
072    
073            public OUTCOME_TYPE consume(Instance<OUTCOME_TYPE> instance) {
074                    for( Feature feature : instance.getFeatures() ) {
075                            if( feature.getValue() instanceof Counts ) {
076                                    String name = feature.getName();
077                                    Counts counts = (Counts) feature.getValue();
078                                    consumeCountsFeature(name, counts);
079                            }
080                    }
081    
082                    return null;
083            }
084    
085            @Override
086            public void collectionProcessComplete() throws AnalysisEngineProcessException {
087                    super.collectionProcessComplete();
088    
089                    try {
090                            if( counters.size() == 1 ) {
091                                    IDFCounter counter = counters.values().iterator().next();
092                                    writeMap(counter, idfMapFile);
093                            } else {
094                                    for( String name : counters.keySet() ) {
095                                            File outputFile = new File(idfMapFile.getParentFile(), idfMapFile.getName() + "-" + name);
096                                            IDFCounter counter = counters.get(name);
097                                            writeMap(counter, outputFile);
098                                    }
099                            }
100                    }
101                    catch (IOException e) {
102                            throw new AnalysisEngineProcessException(e);
103                    }
104            }
105    
106            private void writeMap(IDFCounter counter, File outputFile) throws IOException {
107                    ObjectOutput output = new ObjectOutputStream(new BufferedOutputStream(
108                                    new FileOutputStream(idfMapFile)));
109                    try {
110                            output.writeObject(counter.getIDFMap());
111                    } finally {
112                            output.close();
113                    }
114            }
115    
116            private void consumeCountsFeature(String name, Counts counts) {
117                    String indexedName = name + "__" + counts.getFeatureName();
118    
119                    IDFCounter counter = getCounter(indexedName);
120                    for( Object o : counts.getValues() ) {
121                            String key = o.toString();
122                            int count = counts.getCount(o);
123                            counter.increment(key, count);
124                    }
125            }
126    
127            private IDFCounter getCounter(String name) {
128                    if( counters.containsKey(name) )
129                            return counters.get(name);
130                    else {
131                            IDFCounter counter = new IDFCounter();
132                            counters.put(name, counter);
133                            return counter;
134                    }
135            }
136    
137            public List<OUTCOME_TYPE> consumeSequence(List<Instance<OUTCOME_TYPE>> instances) {
138                    for( Instance<OUTCOME_TYPE> instance : instances )
139                            consume(instance);
140                    return null;
141            }
142    
143            public boolean expectsOutcomes() {
144                    return false;
145            }
146    
147            private Map<String, IDFCounter> counters = new HashMap<String, IDFCounter>();
148            private File idfMapFile;
149    
150            private static class IDFCounter {
151    
152                    public void increment(String key, int c) {
153                            int count = 0;
154                            if( documentFrequencies.containsKey(key) )
155                                    count = documentFrequencies.get(key);
156    
157                            count += c;
158                            documentCount += c;
159    
160                            documentFrequencies.put(key, count);
161                    }
162    
163                    public void increment(String key) {
164                            increment(key, 1);
165                    }
166    
167                    public Map<String, Double> getIDFMap() {
168                            Map<String, Double> idfMap = new HashMap<String, Double>();
169                            for( String key : documentFrequencies.keySet() ) {
170                                    int documentFrequency = documentFrequencies.get(key);
171                                    double inverseDocumentFrequency = documentCount / documentFrequency;
172                                    idfMap.put(key, inverseDocumentFrequency);
173                            }
174                            idfMap.put(null, Math.log(documentCount));
175    
176                            return idfMap;
177                    }
178    
179                    private Map<String, Integer> documentFrequencies = new HashMap<String, Integer>();
180                    private int documentCount = 0;
181            }
182
