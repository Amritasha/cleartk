060    
061    public class TermFinderAnnotator extends JCasAnnotator_ImplBase {
062    
063            /**
064             * "TermListListing" is a single, required string parameter that points to a
065             * file that contains a listing of term list files that are to be loaded.
066             * Each line of the file should contain the name of a term list followed by
067             * the name of the file that contains the terms, a boolean ('true' or
068             * 'false') that indicates whether the file should be treated as case
069             * sensitive followed optionally by separator string to be used to separate
070             * an id from a term if the file contains ids. The values on each line
071             * should be tab delimited. Also, see data/lexicons/README
072             */
073    
074            public static final String PARAM_TERM_LIST_LISTING = "TermListListing";
075    
076            /**
077             * "SentenceClass" is a single, optional, string parameter that names the
078             * class of the type system type from which to extract tokens. Any
079             * annotation that contains tokens can be used (e.g. sentence, paragraph,
080             * document). If no value is given for this parameter, then all tokens will
081             * be searched. An example value might be: <br>
082             * <code>org.cleartk.type.Sentence</code>
083             */
084            public static final String PARAM_SENTENCE_CLASS = "SentenceClass";
085    
086            /**
087             * "TokenClass" is a single, required, string parameter that names the class
088             * of the type system type corresponding to tokens. A good value for this
089             * would be: <br>
090             * <code>org.cleartk.type.Token</code>
091             */
092            public static final String PARAM_TOKEN_CLASS = "TokenClass";
093    
094            /**
095             * "TermMatchAnnotationCreator" is a single, optional, string parameter that
096             * provides the class name of a class that extends
097             * org.cleartk.bio.TermMatchAnnotationCreator. If this parameter is
098             * not given a value, then the parameter PARAM_TERM_MATCH_ANNOTATION_CLASS
099             * must be given a value.
100             */
101            public static final String PARAM_TERM_MATCH_ANNOTATION_CREATOR = "TermMatchAnnotationCreator";
102    
103            /**
104             * "TermMatchAnnotationClass" is a single, optional, string parameter that
105             * names the class of the type system type that specifies the annotations
106             * created of found term matches. One annotation is created for each term
107             * match found of the given type specified by this parameter. This parameter
108             * is ignored if PARAM_TERM_MATCH_ANNOTATION_CREATOR is given a value. An
109             * example value might be something like: <br>
110             * <code>org.cleartk.type.ne.NamedEntityMention</code>
111             */
112            public static final String PARAM_TERM_MATCH_ANNOTATION_CLASS = "TermMatchAnnotationClass";
113    
114            TermFinder caseSensitiveTermFinder;
115    
116            TermFinder caseInsensitiveTermFinder;
117    
118            boolean allTokens = true;
119    
120            boolean typesInitialized = false;
121    
122            protected Class<? extends Annotation> sentenceClass;
123    
124            protected Type sentenceType;
125    
126            protected Class<? extends Annotation> tokenClass;
127    
128            protected Type tokenType;
129    
130            TermMatchAnnotationCreator annotationCreator;
131    
132            Constructor<? extends Annotation> annotationConstructor;
133    
134            @Override
135            public void initialize(UimaContext context) throws ResourceInitializationException {
136                    try {
137                            // load the term lists
138                            String termLists = (String) context.getConfigParameterValue(PARAM_TERM_LIST_LISTING);
139                            BufferedReader input = new BufferedReader(new FileReader(termLists));
140                            String line;
141                            while ((line = input.readLine()) != null) {
142                                    line = line.trim();
143                                    String[] columns = line.split("\t");
144                                    String termListName = columns[0];
145                                    String fileName = columns[1];
146                                    boolean caseSensitive = Boolean.parseBoolean(columns[2]);
147                                    String separator = columns.length == 4 ? columns[3] : null;
148    
149                                    if (caseSensitive && caseSensitiveTermFinder == null) caseSensitiveTermFinder = new SimpleTermFinder(
150                                                    true, new PennTreebankTokenizer());
151                                    if (!caseSensitive && caseInsensitiveTermFinder == null) caseInsensitiveTermFinder = new SimpleTermFinder(
152                                                    false, new PennTreebankTokenizer());
153    
154                                    TermList termList = TermList.loadSimpleFile(termListName, new File(fileName), separator);
155                                    if (caseSensitive) caseSensitiveTermFinder.addTermList(termList);
156                                    else caseInsensitiveTermFinder.addTermList(termList);
157                            }
158    
159                            // sentence class
160                            String sentenceClassName = (String) context.getConfigParameterValue(PARAM_SENTENCE_CLASS);
161                            if (sentenceClassName != null) {
162                                    allTokens = false;
163                                    Class<?> cls = Class.forName(sentenceClassName);
164                                    sentenceClass = cls.asSubclass(Annotation.class);
165                            }
166    
167                            // token class
168                            String tokenClassName = (String) context.getConfigParameterValue(PARAM_TOKEN_CLASS);
169                            Class<?> cls = Class.forName(tokenClassName);
170                            tokenClass = cls.asSubclass(Annotation.class);
171    
172                            // annotation creator or constructor
173                            String termMatchAnnotationCreatorClassName = (String) context
174                                            .getConfigParameterValue(PARAM_TERM_MATCH_ANNOTATION_CREATOR);
175                            if (termMatchAnnotationCreatorClassName != null && !termMatchAnnotationCreatorClassName.equals("")) {
176                                    cls = Class.forName(termMatchAnnotationCreatorClassName);
177                                    Class<? extends TermMatchAnnotationCreator> annotationCreatorClass = cls
178                                                    .asSubclass(TermMatchAnnotationCreator.class);
179                                    annotationCreator = annotationCreatorClass.newInstance();
180                                    annotationCreator.initialize(context);
181                            }
182                            else {
183                                    // annotationConstructor
184                                    String annotationClassName = (String) context
185                                                    .getConfigParameterValue(PARAM_TERM_MATCH_ANNOTATION_CLASS);
186                                    cls = Class.forName(annotationClassName);
187                                    Class<? extends Annotation> annotationClass = cls.asSubclass(Annotation.class);
188                                    annotationConstructor = annotationClass.getConstructor(new Class[] { JCas.class,
189                                                    java.lang.Integer.TYPE, java.lang.Integer.TYPE });
190                            }
191                    }
192                    catch (Exception e) {
193                            throw new ResourceInitializationException(e);
194                    }
195            }
196    
197            private void initializeTypes(JCas jCas) {
198                    if (!allTokens) sentenceType = UIMAUtil.getCasType(jCas, sentenceClass);
199                    tokenType = UIMAUtil.getCasType(jCas, tokenClass);
200                    typesInitialized = true;
201            }
202    
203            private Token createToken(Annotation annotation) {
204                    return new Token(annotation.getBegin(), annotation.getEnd(), annotation.getCoveredText());
205            }
206    
207            @Override
208            public void process(JCas jCas) throws AnalysisEngineProcessException {
209                    if (!typesInitialized) initializeTypes(jCas);
210    
211                    List<Token> tokens = new ArrayList<Token>();
212                    if (allTokens) {
213                            FSIterator tokenAnnotations = jCas.getAnnotationIndex(tokenType).iterator();
214                            while (tokenAnnotations.hasNext()) {
215                                    Annotation tokenAnnotation = (Annotation) tokenAnnotations.next();
216                                    tokens.add(createToken(tokenAnnotation));
217                            }
218                            findTerms(jCas, tokens);
219                    }
220                    else {
221                            FSIterator sentenceAnnotations = jCas.getAnnotationIndex(sentenceType).iterator();
222                            while (sentenceAnnotations.hasNext()) {
223                                    tokens.clear();
224                                    Annotation sentenceAnnotation = (Annotation) sentenceAnnotations.next();
225                                    FSIterator tokenAnnotations = jCas.getAnnotationIndex(tokenType).subiterator(sentenceAnnotation);
226                                    while (tokenAnnotations.hasNext()) {
227                                            Annotation tokenAnnotation = (Annotation) tokenAnnotations.next();
228                                            tokens.add(createToken(tokenAnnotation));
229                                    }
230                                    findTerms(jCas, tokens);
231                            }
232                    }
233            }
234    
235            public void findTerms(JCas jCas, List<Token> tokens) throws AnalysisEngineProcessException {
236    
237                    if (caseSensitiveTermFinder != null) {
238                            findTerms(jCas, tokens, caseSensitiveTermFinder);
239                    }
240                    if (caseInsensitiveTermFinder != null) {
241                            findTerms(jCas, tokens, caseInsensitiveTermFinder);
242                    }
243            }
244    
245            public void findTerms(JCas jCas, List<Token> tokens, TermFinder termFinder) throws AnalysisEngineProcessException {
246    
247                    List<TermMatch> termMatches = termFinder.getMatches(tokens);
248                    for (TermMatch termMatch : termMatches) {
249                            if (annotationCreator != null) {
250                                    annotationCreator.createTermMatchAnnotation(jCas, termMatch);
251                            }
252                            else {
253                                    try {
254                                            int begin = termMatch.getBegin();
255                                            int end = termMatch.getEnd();
256                                            Annotation annotation = annotationConstructor.newInstance(new Object[] { jCas, begin, end });
257                                            annotation.addToIndexes();
258                                    }
259                                    catch (Exception e) {
260                                            throw new AnalysisEngineProcessException(e);
261                                    }
262                            }
263    
264                    }
265            }
