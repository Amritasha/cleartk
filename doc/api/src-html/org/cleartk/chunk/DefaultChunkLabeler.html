040    
041    public class DefaultChunkLabeler extends ChunkLabeler_ImplBase {
042    
043            /**
044             * "ChunkLabelFeature" is a single, optional, string parameter that names
045             * the feature of the type system chunk type that provides a label for each
046             * chunk. The feature is queried and the value of the feature is used as the
047             * label for the chunk. If this parameter has no value, then the name of the
048             * chunk type will be used as a label. For example, if the value of the
049             * parameter "ChunkAnnotationClass" is
050             * <code>org.cleartk.type.Chunk</code>, then a good value for
051             * this parameter would be <code>chunkType</code>. This would result in
052             * labels corresponding to the values found in the type system feature
053             * chunkType. If the value of the parameter "ChunkAnnotationClass" is
054             * <code>org.cleartk.type.Chunk</code> and no value is given for
055             * this parameter, then the label will always be <code>Chunk</code>.
056             */
057            public static final String PARAM_CHUNK_LABEL_FEATURE = "ChunkLabelFeature";
058    
059            private String chunkLabelFeatureName;
060    
061            private org.apache.uima.cas.Feature chunkLabelFeature;
062    
063            Constructor<? extends Annotation> chunkAnnotationConstructor;
064    
065            @Override
066            public void initialize(UimaContext context) throws ResourceInitializationException {
067                    super.initialize(context);
068                    try {
069                            String paramValue = (String) context.getConfigParameterValue(PARAM_CHUNK_LABEL_FEATURE);
070                            if (paramValue != null) chunkLabelFeatureName = chunkAnnotationClass.getCanonicalName() + ":" + paramValue;
071    
072                            chunkAnnotationConstructor = chunkAnnotationClass.getConstructor(new Class[] { JCas.class,
073                                            java.lang.Integer.TYPE, java.lang.Integer.TYPE });
074                    }
075                    catch (Exception e) {
076                            throw new ResourceInitializationException(e);
077                    }
078            }
079    
080            protected void initializeTypes(JCas jCas) throws AnalysisEngineProcessException {
081                    super.initializeTypes(jCas);
082                    try {
083                            if (chunkLabelFeatureName != null) {
084                                    chunkLabelFeature = jCas.getTypeSystem().getFeatureByFullName(chunkLabelFeatureName);
085                                    if (chunkLabelFeature == null) throw new AnalysisEngineProcessException("type feature for name '"
086                                                    + chunkLabelFeatureName + "' not found.  ", null);
087                            }
088                    }
089                    catch (Exception e) {
090                            throw new AnalysisEngineProcessException(e);
091                    }
092            }
093    
094            public Annotation createChunk(JCas jCas, List<? extends Annotation> labeledAnnotations, String label)
095                            throws AnalysisEngineProcessException {
096                    try {
097                            int begin = labeledAnnotations.get(0).getBegin();
098                            int end = labeledAnnotations.get(labeledAnnotations.size() - 1).getEnd();
099                            Annotation annotation = chunkAnnotationConstructor.newInstance(new Object[] { jCas, begin, end });
100    
101                            if (chunkLabelFeature != null) annotation.setFeatureValueFromString(chunkLabelFeature, label);
102                            annotation.addToIndexes();
103                            return annotation;
104                    }
105                    catch (Exception e) {
106                            throw new AnalysisEngineProcessException(e);
107                    }
108            }
109    
110            @Override
111            public String getChunkLabel(JCas jCas, Annotation chunkAnnotation) throws AnalysisEngineProcessException {
112                    if (!typesInitialized) initializeTypes(jCas);
113    
114                    if (chunkLabelFeature != null) return chunkAnnotation.getFeatureValueAsString(chunkLabelFeature);
115                    else return chunkAnnotationClass.getSimpleName();
116            }
117    }
