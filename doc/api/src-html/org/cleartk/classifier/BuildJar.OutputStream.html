042    public class BuildJar {
043            
044            public static final String MODEL_FILE_NAME = "model.jar";
045            
046            public static void main(String ... args) throws Exception {
047                    String programName = BuildJar.class.getName();
048                    String usage = String.format(
049                                    "usage: java %s DIR\n\n" + 
050                                    "The directory DIR should contain the MANIFEST.MF file as created by\n" + 
051                                    "a classifier DataWriter, and all files created by training a model\n" +
052                                    "from the corresponding training-data.xxx file\n", programName);
053                    
054                    // usage message for wrong number of arguments
055                    if (args.length != 1) {
056                            System.err.format("error: wrong number of arguments\n%s", usage);
057                            System.exit(1);
058                    }
059                    File dir = new File(args[0]);
060                    
061                    // get the classifier class from the manifest
062                    ClassifierManifest manifest = new ClassifierManifest(dir);
063                    ClassifierBuilder<?> classifierBuilder = manifest.getClassifierBuilder();
064                    
065                    // clip the first item off the command line arguments, and call buildJar
066                    String[] remainingArgs = new String[args.length - 1];
067                    System.arraycopy(args, 1, remainingArgs, 0, remainingArgs.length);
068                    classifierBuilder.buildJar(dir, remainingArgs);
069            }
070            
071            public static class OutputStream extends JarOutputStream {
072                    public OutputStream(File dir) throws IOException {
073                            super(getOutputStream(dir), new ClassifierManifest(dir));
074                            String encodersFileName = FeaturesEncoder_ImplBase.ENCODERS_FILE_NAME;
075                            File encodersFile = new File(dir, encodersFileName);
076                            if(encodersFile.exists())
077                                    this.write(encodersFileName, encodersFile);
078                    }
079                    
080                    public void write(String entryName, File file) throws IOException {
081                            this.putNextEntry(new JarEntry(entryName));
082                            FileInputStream stream = new FileInputStream(file);
083                            byte[] byteArray = new byte[stream.available()];
084                            stream.read(byteArray);
085                            this.write(byteArray);
086                    }
087                    
088                    private static FileOutputStream getOutputStream(File dir) throws IOException {
089                            return new FileOutputStream(new File(dir, MODEL_FILE_NAME));
090                    }
091            }
092    
093    }
