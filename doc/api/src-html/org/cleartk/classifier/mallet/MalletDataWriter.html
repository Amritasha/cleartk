024    package org.cleartk.classifier.mallet;
025    
026    import java.io.File;
027    import java.io.IOException;
028    import java.io.PrintWriter;
029    import java.util.List;
030    
031    import org.cleartk.CleartkException;
032    import org.cleartk.classifier.ClassifierBuilder;
033    import org.cleartk.classifier.DataWriter_ImplBase;
034    import org.cleartk.classifier.encoder.features.NameNumber;
035    import org.cleartk.classifier.encoder.features.NameNumberFeaturesEncoder;
036    
037    /**
038     * <br>
039     * Copyright (c) 2007-2008, Regents of the University of Colorado <br>
040     * All rights reserved.
041     * 
042     * 
043     * This training data consumer produces training data suitable for <a
044     * href="http://mallet.cs.umass.edu/index.php/SimpleTagger_example"> Mallet
045     * Conditional Random Field (CRF) tagger</a>.
046     * 
047     * Each line of the training data contains a string representation of each
048     * feature followed by the label/result for that instance.
049     * 
050     * @author Philip Ogren
051     */
052    public class MalletDataWriter extends DataWriter_ImplBase<String, String, List<NameNumber>> {
053    
054            public static final String TRAINING_DATA_FILE_NAME ="training-data.mallet"; 
055            public MalletDataWriter(File outputDirectory) throws IOException {
056                    super(outputDirectory);
057    
058                    // initialize output writer and Classifier class
059                    this.trainingDataWriter = this.getPrintWriter(TRAINING_DATA_FILE_NAME);
060    
061            }
062    
063            @Override
064            public void finish() throws CleartkException {
065                    super.finish();
066    
067                    try {
068                            if (featuresEncoder instanceof NameNumberFeaturesEncoder) {
069                                    NameNumberFeaturesEncoder dfe = (NameNumberFeaturesEncoder) featuresEncoder;
070                                    if (dfe.isCompressFeatures()) dfe.writeNameLookup(this
071                                                    .getPrintWriter(NameNumberFeaturesEncoder.LOOKUP_FILE_NAME));
072                            }
073                    }
074                    catch (IOException ioe) {
075                            throw new CleartkException(ioe);
076                    }
077    
078            }
079    
080            @Override
081            public void writeEncoded(List<NameNumber> features, String outcome)  throws CleartkException{
082                    if(outcome == null) {
083                            throw new CleartkException("all consumed instances must have an outcome.  outcome="+outcome);
084                    }
085                    if (features.size() == 0) {
086                            trainingDataWriter.print("null:0 ");
087                    }
088                    else {
089                            for (NameNumber nameNumber : features) {
090                                    trainingDataWriter.print(nameNumber.name + ":" + nameNumber.number + " ");
091                            }
092                    }
093                    this.trainingDataWriter.print(outcome);
094                    this.trainingDataWriter.println();
095            }
096    
097            public Class<? extends ClassifierBuilder<String>> getDefaultClassifierBuilderClass() {
098                    return MalletClassifierBuilder.class;
099            }
100    
101            protected PrintWriter trainingDataWriter;
102    
103    }
