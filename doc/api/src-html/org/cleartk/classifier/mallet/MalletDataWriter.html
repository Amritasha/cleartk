024    package org.cleartk.classifier.mallet;
025    
026    import java.io.PrintWriter;
027    import java.util.List;
028    
029    import org.apache.uima.UimaContext;
030    import org.apache.uima.resource.ResourceInitializationException;
031    import org.cleartk.classifier.ClassifierBuilder;
032    import org.cleartk.classifier.DataWriter_ImplBase;
033    import org.cleartk.classifier.Instance;
034    import org.cleartk.classifier.encoder.EncoderFactory;
035    import org.cleartk.classifier.encoder.factory.ContextValueEncoderFactory;
036    import org.cleartk.classifier.encoder.features.contextvalue.ContextValue;
037    
038    /**
039     * <br>Copyright (c) 2007-2008, Regents of the University of Colorado 
040     * <br>All rights reserved.
041    
042     *
043     * This training data consumer produces training data suitable for 
044     * <a href="http://mallet.cs.umass.edu/index.php/SimpleTagger_example">
045     * Mallet Conditional Random Field (CRF) tagger</a>.  
046     * 
047     * Each line of the training data contains a string representation of each feature followed
048     * by the label/result for that instance.  
049     * 
050     * @author Philip Ogren
051     */
052    public class MalletDataWriter extends DataWriter_ImplBase<String,String,List<ContextValue>> {
053    
054            protected PrintWriter trainingDataWriter;
055            
056            @Override
057            public void initialize(UimaContext context) throws ResourceInitializationException {
058                    super.initialize(context);
059                    
060                    // initialize output writer and Classifier class
061                    this.trainingDataWriter = this.getPrintWriter("training-data.mallet");
062            }
063    
064            public String consume(Instance<String> instance) {
065                    String outcomeString = this.outcomeEncoder.encode(instance.getOutcome());
066                    
067                    List<ContextValue> contextValues = this.featuresEncoder.encodeAll(instance.getFeatures());
068                    for (ContextValue contextValue : contextValues) {
069                            trainingDataWriter.print(contextValue.getContext() + ":" + contextValue.getValue()+" ");
070                    }
071                    this.trainingDataWriter.print(outcomeString);
072                    this.trainingDataWriter.println();
073                    
074                    // no labels created
075                    return null;
076            }
077    
078            
079            @Override
080            protected Class<? extends ClassifierBuilder<? extends String>> getDefaultClassifierBuilderClass() {
081                    return MalletClassifierBuilder.class;
082            }
083    
084            @Override
085            protected Class<? extends EncoderFactory> getDefaultEncoderFactoryClass() {
086                    return ContextValueEncoderFactory.class;
087            }
088    }
