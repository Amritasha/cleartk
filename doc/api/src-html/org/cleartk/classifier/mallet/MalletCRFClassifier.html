024    package org.cleartk.classifier.mallet;
025    
026    import java.io.ObjectInputStream;
027    import java.util.ArrayList;
028    import java.util.List;
029    import java.util.jar.JarFile;
030    import java.util.zip.ZipEntry;
031    
032    import org.cleartk.classifier.Classifier_ImplBase;
033    import org.cleartk.classifier.Feature;
034    
035    import cc.mallet.fst.Transducer;
036    import cc.mallet.pipe.Pipe;
037    import cc.mallet.types.Instance;
038    import cc.mallet.types.Sequence;
039    
040    /**
041     * <br>Copyright (c) 2007-2008, Regents of the University of Colorado 
042     * <br>All rights reserved.
043    
044     *
045     * @author Philip Ogren
046     *
047     * This classifier provides an interface to the 
048     * <a href="http://mallet.cs.umass.edu/index.php/SimpleTagger_example">
049     * Mallet Conditional Random Field (CRF) tagger</a>.  
050     * Annotators that use a sequential learner such as this one will need to 
051     * support classification of a sequence of instances.  There are two main 
052     * considerations that need to be addressed.  The first is that the annotator
053     * should call this classifier with a sequence of "instances" (i.e. lists of features,
054     * one list of features per instance).  For example, all the tokens in a sentence
055     * could be classified for pos tags or named entity chunk tags - rather than passing
056     * in one token at a time.  The other consideration is has to do with creating the training
057     * data which is discussed in the
058     * {@link MalletCRFDataWriter#consume(org.cleartk.classifier.Instance)}.
059     * 
060     */
061    public class MalletCRFClassifier extends Classifier_ImplBase<String,String,List<String>>
062    {
063            protected Transducer transducer;
064            
065            public MalletCRFClassifier(JarFile modelFile) throws Exception {
066                    super(modelFile);
067                    
068                    ZipEntry modelEntry = modelFile.getEntry("model.malletcrf");
069                    ObjectInputStream objectStream = new ObjectInputStream(modelFile.getInputStream(modelEntry));
070                    this.transducer = (Transducer) objectStream.readObject();
071        }
072            
073            /**
074             * This method simply throws an UnsupportedOperationException because
075             * CRF is a sequential classifier.
076             */
077            @Override
078            public String classify(List<Feature> features) throws UnsupportedOperationException
079            {
080                    throw new UnsupportedOperationException("Mallet CRF is a sequential learner which means that classifing one instance at a time does not work well.");
081            }
082    
083            /**
084             * This method classifies several instances at once
085             * @param features a list of lists of features - each list 
086             * in the list represents one instance to be classified.  The
087             * list should correspond to some logical sequence of instances
088             * to be classified (e.g. tokens in a sentence or lines in a document)
089             * that corresponds to the model that has been built for this classifier.
090             */
091            @SuppressWarnings("unchecked")
092            @Override
093            public List<String> classifySequence(List<List<Feature>> features)
094            {
095                    String[][] featureStringArray = toStrings(features);
096                    Pipe pipe = transducer.getInputPipe();
097                    
098                    Instance instance = new Instance(featureStringArray, null, null, null); 
099                    instance = pipe.instanceFrom(instance);
100                    
101                    Sequence<String> sequence = transducer.transduce((Sequence<String>)instance.getData());
102                    
103                    List<String> returnValues = new ArrayList<String>();
104                    
105                    for(int i=0; i<sequence.size(); i++) {
106                            String encodedOutcome = (String) sequence.get(i);
107                            returnValues.add(outcomeEncoder.decode(encodedOutcome));
108                    }
109                    return returnValues;
110            }
111    
112            /**
113             * Converts the features into a 2D string array that Mallet can
114             * use.  The only thing mildly tricky about this method is that
115             * the length of each string array is one more than the size of each
116             * feature list - i.e. <code>returnValues[0].length == features.get(0).size() + 1</code>
117             * where the last element in each string array is an empty string.
118             * @param features the features to be converted.  
119             * @return a 2D string array that Mallet can use. 
120             */
121            private String[][] toStrings(List<List<Feature>> features) {
122                    List<List<String>> encodedFeatures = new ArrayList<List<String>>(features.size());
123                    for( List<Feature> features1 : features ) {
124                            List<String> encodedFeatures1 = this.featuresEncoder.encodeAll(features1);
125                            encodedFeatures.add(encodedFeatures1);
126                    }
127                    
128                    String[][] encodedFeaturesArray = new String[encodedFeatures.size()][];
129                    for( int i=0; i<encodedFeatures.size(); i++ ) {
130                            String[] encodedFeaturesArray1 = encodedFeatures.get(i).toArray(new String[0]);
131                            encodedFeaturesArray[i] = encodedFeaturesArray1;
132                    }
133                    
134                    return encodedFeaturesArray;
135            }
136            
137            /**
138             * returns true
139             */
140            @Override
141            public boolean isSequential() {
142                    return true;
143            }
144            
145    
146    }
