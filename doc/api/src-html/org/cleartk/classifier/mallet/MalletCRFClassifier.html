024    package org.cleartk.classifier.mallet;
025    
026    import java.io.ObjectInputStream;
027    import java.util.ArrayList;
028    import java.util.List;
029    import java.util.jar.JarFile;
030    import java.util.zip.ZipEntry;
031    
032    import org.cleartk.CleartkException;
033    import org.cleartk.classifier.Feature;
034    import org.cleartk.classifier.SequentialClassifier_ImplBase;
035    import org.cleartk.classifier.encoder.features.NameNumber;
036    import org.cleartk.util.ReflectionUtil;
037    
038    import cc.mallet.fst.Transducer;
039    import cc.mallet.pipe.Pipe;
040    import cc.mallet.types.Instance;
041    import cc.mallet.types.Sequence;
042    
043    /**
044     * <br>Copyright (c) 2007-2008, Regents of the University of Colorado 
045     * <br>All rights reserved.
046    
047     *
048     * @author Philip Ogren
049     *
050     * This classifier provides an interface to the 
051     * <a href="http://mallet.cs.umass.edu/index.php/SimpleTagger_example">
052     * Mallet Conditional Random Field (CRF) tagger</a>.  
053     * Annotators that use a sequential learner such as this one will need to 
054     * support classification of a sequence of instances.  
055     */
056    public class MalletCRFClassifier extends SequentialClassifier_ImplBase<String,String,List<NameNumber>>
057    {
058            protected Transducer transducer;
059            
060            public MalletCRFClassifier(JarFile modelFile) throws Exception {
061                    super(modelFile);
062                    
063                    ZipEntry modelEntry = modelFile.getEntry("model.malletcrf");
064                    ObjectInputStream objectStream = new ObjectInputStream(modelFile.getInputStream(modelEntry));
065                    this.transducer = (Transducer) objectStream.readObject();
066        }
067            
068            /**
069             * This method classifies several instances at once
070             * @param features a list of lists of features - each list 
071             * in the list represents one instance to be classified.  The
072             * list should correspond to some logical sequence of instances
073             * to be classified (e.g. tokens in a sentence or lines in a document)
074             * that corresponds to the model that has been built for this classifier.
075             * @throws CleartkException 
076             */
077            public List<String> classifySequence(List<List<Feature>> features) throws CleartkException
078            {
079                    String[][] featureStringArray = toStrings(features);
080                    Pipe pipe = transducer.getInputPipe();
081                    
082                    Instance instance = new Instance(featureStringArray, null, null, null); 
083                    instance = pipe.instanceFrom(instance);
084                    
085                    Sequence<?> data = (Sequence<?>)instance.getData();
086                    Sequence<?> untypedSequence = transducer.transduce(data);
087                    Sequence<String> sequence = ReflectionUtil.uncheckedCast(untypedSequence);
088                    
089                    List<String> returnValues = new ArrayList<String>();
090                    
091                    for(int i=0; i<sequence.size(); i++) {
092                            String encodedOutcome = sequence.get(i);
093                            returnValues.add(outcomeEncoder.decode(encodedOutcome));
094                    }
095                    return returnValues;
096            }
097    
098            /**
099             * Converts the features into a 2D string array that Mallet can
100             * use.  The only thing mildly tricky about this method is that
101             * the length of each string array is one more than the size of each
102             * feature list - i.e. <code>returnValues[0].length == features.get(0).size() + 1</code>
103             * where the last element in each string array is an empty string.
104             * @param features the features to be converted.  
105             * @return a 2D string array that Mallet can use. 
106             * @throws CleartkException 
107             */
108            private String[][] toStrings(List<List<Feature>> features) throws CleartkException {
109                    List<List<String>> encodedFeatures = new ArrayList<List<String>>(features.size());
110                    for( List<Feature> features1 : features ) {
111                            List<NameNumber> nameNumbers = this.featuresEncoder.encodeAll(features1);
112                            List<String> encodedFeatures1 = new ArrayList<String>();
113                            for(NameNumber nameNumber : nameNumbers) {
114                                    encodedFeatures1.add(nameNumber.name);
115                            }
116                            encodedFeatures.add(encodedFeatures1);
117                    }
118                    
119                    String[][] encodedFeaturesArray = new String[encodedFeatures.size()][];
120                    for( int i=0; i<encodedFeatures.size(); i++ ) {
121                            String[] encodedFeaturesArray1 = encodedFeatures.get(i).toArray(new String[0]);
122                            encodedFeaturesArray[i] = encodedFeaturesArray1;
123                    }
124                    
125                    return encodedFeaturesArray;
126            }
127    }
