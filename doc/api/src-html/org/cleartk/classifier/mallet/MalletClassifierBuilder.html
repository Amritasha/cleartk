047    public class MalletClassifierBuilder implements ClassifierBuilder<String> {
048    
049            @SuppressWarnings("unchecked")
050            public void train(File dir, String[] args) throws Exception {
051    
052                    InstanceListCreator instanceListCreator = new InstanceListCreator();
053                    InstanceList instanceList = instanceListCreator.createInstanceList(new File(dir, "training-data.mallet"));
054                    instanceList.save(new File(dir, "training-data.ser"));
055                    
056                    String factoryName = args[0];
057                    Class<ClassifierTrainerFactory> factoryClass = createTrainerFactory(factoryName);
058                    if(factoryClass == null) {
059                            String factoryName2 = "org.cleartk.classifier.mallet.factory."+factoryName+"TrainerFactory";
060                            factoryClass = createTrainerFactory(factoryName2);
061                    }
062                    if(factoryClass == null) {
063                            throw new IllegalArgumentException(String.format("name for classifier trainer factory is not valid: name given ='%1$s'", factoryName));
064                    }
065                    
066                    String[] factoryArgs = new String[args.length - 1];
067                    System.arraycopy(args, 1, factoryArgs, 0, factoryArgs.length);
068    
069                    ClassifierTrainerFactory factory = factoryClass.newInstance();
070                    ClassifierTrainer trainer = null;
071                    try {
072                            trainer = factory.createTrainer(factoryArgs);
073                    } catch(Throwable t) {
074                            throw new IllegalArgumentException("Unable to create trainer.  Usage for "+factoryClass.getCanonicalName()+": "+factory.getUsageMessage(),t);
075                    }
076                    
077                    cc.mallet.classify.Classifier classifier = trainer.train(instanceList);
078    
079                    ObjectOutputStream oos = new ObjectOutputStream (new FileOutputStream (new File(dir, "model.mallet")));
080                    oos.writeObject (classifier);
081                    oos.close();
082    
083            }
084    
085            @SuppressWarnings("unchecked")
086            private Class<ClassifierTrainerFactory> createTrainerFactory(String className){
087                    try {
088                            return (Class<ClassifierTrainerFactory>)Class.forName(className);
089                    } catch(ClassNotFoundException cnfe) {
090                            return null;
091                    }
092            }
093            
094            public void buildJar(File dir, String[] args) throws Exception {
095                    BuildJar.OutputStream stream = new BuildJar.OutputStream(dir);
096                    stream.write("model.mallet", new File(dir, "model.mallet"));
097                    stream.close();
098            }
099    
100            public Class<? extends Classifier<String>> getClassifierClass() {
101                    return MalletClassifier.class;
102            }
103    
104    }
