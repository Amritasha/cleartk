051    public class DefaultMalletCRFDataWriterFactory extends SequentialDataWriterFactory_ImplBase<List<NameNumber>, String, String> {
052    
053            /**
054             * "org.cleartk.classifier.mallet.DefaultMalletCRFDataWriterFactory.PARAM_COMPRESS"
055             * is a single, optional, boolean parameter, defaulting to false, that when true
056             * indicates that the FeaturesEncoder should compress the feature names.
057             * @see NameNumberFeaturesEncoder
058             */
059            public static final String PARAM_COMPRESS = "org.cleartk.classifier.mallet.DefaultMalletCRFDataWriterFactory.PARAM_COMPRESS";
060    
061            /**
062             * "org.cleartk.classifier.mallet.DefaultMalletCRFDataWriterFactory.PARAM_SORT_NAME_LOOKUP"
063             * is a single, optional, boolean parameter, defaulting to false, that when true
064             * indicates that the FeaturesEncoder should write the feature names in sorted order.
065             * @see NameNumberFeaturesEncoder
066             */
067            public static final String PARAM_SORT_NAME_LOOKUP = "org.cleartk.classifier.mallet.DefaultMalletCRFDataWriterFactory.PARAM_SORT_NAME_LOOKUP";
068    
069            @Override
070            public void initialize(UimaContext context) throws ResourceInitializationException {
071                    super.initialize(context);
072                    compress = (Boolean)UIMAUtil.getDefaultingConfigParameterValue(context, PARAM_COMPRESS, false);
073                    sort = (Boolean)UIMAUtil.getDefaultingConfigParameterValue(context, PARAM_SORT_NAME_LOOKUP, false);
074            }
075            
076            public SequentialDataWriter<String> createSequentialDataWriter(File outputDirectory) throws IOException {
077                    MalletCRFDataWriter mdw = new MalletCRFDataWriter(outputDirectory);
078                    
079                    if(!this.setEncodersFromFileSystem(mdw)) {
080                            NameNumberFeaturesEncoder featuresEncoder = new NameNumberFeaturesEncoder(compress, sort);
081                            featuresEncoder.addEncoder(new NumberEncoder());
082                            featuresEncoder.addEncoder(new BooleanEncoder());
083                            featuresEncoder.addEncoder(new StringEncoder());
084                            mdw.setFeaturesEncoder(featuresEncoder);
085                            
086                            mdw.setOutcomeEncoder(new StringToStringOutcomeEncoder());
087                    }
088    
089                    return mdw;
090            }
091    
092            private boolean compress;
093            private boolean sort;
094    
095    }
