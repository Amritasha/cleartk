046    public abstract class SequentialClassifier_ImplBase<INPUTOUTCOME_TYPE,OUTPUTOUTCOME_TYPE,FEATURES_TYPE> implements SequentialClassifier<INPUTOUTCOME_TYPE> {
047            
048            protected FeaturesEncoder<FEATURES_TYPE> featuresEncoder;
049            protected OutcomeEncoder<INPUTOUTCOME_TYPE,OUTPUTOUTCOME_TYPE> outcomeEncoder;
050            
051            public SequentialClassifier_ImplBase(JarFile modelFile) throws IOException {
052                    
053                    // de-serialize the encoders
054                    ZipEntry zipEntry = modelFile.getEntry(FeaturesEncoder_ImplBase.ENCODERS_FILE_NAME);
055                    ObjectInputStream is = new ObjectInputStream(modelFile.getInputStream(zipEntry));
056                    FeaturesEncoder<?> genericFeaturesEncoder;
057                    OutcomeEncoder<?,?> genericOutcomeEncoder;
058                    try {
059                            genericFeaturesEncoder = (FeaturesEncoder<?>) is.readObject();
060                            genericOutcomeEncoder = (OutcomeEncoder<?,?>) is.readObject();
061                    } catch (ClassNotFoundException e) {
062                            throw new RuntimeException(e);
063                    } finally {
064                            is.close();
065                    }
066                    this.featuresEncoder = featuresEncoderCast(genericFeaturesEncoder);
067                    this.featuresEncoder.allowNewFeatures(false);
068                    this.outcomeEncoder = outcomeEncoderCast(genericOutcomeEncoder);
069                    
070            }
071    
072            protected Class<?> getMyTypeArgument(String parameterName) {
073                    return getTypeArgument(SequentialClassifier_ImplBase.class, parameterName, this);
074            }
075            
076            protected  Class<?> getTypeArgument(Class<?> cls, String parameterName, Object instance) {
077                    Map<String,Type> typeArguments = ReflectionUtil.getTypeArguments(cls, instance);
078                    Type t = typeArguments.get(parameterName);
079                    if( t instanceof Class )
080                            return (Class<?>)t;
081                    else
082                            return null;
083            }
084            
085            protected FeaturesEncoder<FEATURES_TYPE> featuresEncoderCast(FeaturesEncoder<?> encoder) {
086                    Class<?> myFEATURESTYPE = getMyTypeArgument("FEATURES_TYPE");
087                    Class<?> feFEATURESTYPE = getTypeArgument(FeaturesEncoder.class, "FEATURES_TYPE", encoder);
088                    
089                    if( myFEATURESTYPE != feFEATURESTYPE )
090                            throw new ClassCastException();
091    
092                    return ReflectionUtil.uncheckedCast(encoder);
093            }
094            
095            protected OutcomeEncoder<INPUTOUTCOME_TYPE,OUTPUTOUTCOME_TYPE> outcomeEncoderCast(OutcomeEncoder<?,?> encoder) {
096                    Class<?> myINPUTOUTCOME_TYPE = getMyTypeArgument("INPUTOUTCOME_TYPE");
097                    Class<?> oeINPUTOUTCOME_TYPE = getTypeArgument(OutcomeEncoder.class, "INPUTOUTCOME_TYPE", encoder);
098                    
099                    if( myINPUTOUTCOME_TYPE != oeINPUTOUTCOME_TYPE )
100                            throw new ClassCastException();
101                    
102                    Class<?> myOUTPUTOUTCOME_TYPE = getMyTypeArgument("OUTPUTOUTCOME_TYPE");
103                    Class<?> oeOUTPUTOUTCOME_TYPE = getTypeArgument(OutcomeEncoder.class, "OUTPUTOUTCOME_TYPE", encoder);
104                    
105                    if( myOUTPUTOUTCOME_TYPE != oeOUTPUTOUTCOME_TYPE )
106                            throw new ClassCastException();
107                    
108                    return ReflectionUtil.uncheckedCast(encoder);
109            }
110            
111            
112    
113            public List<ScoredOutcome<List<INPUTOUTCOME_TYPE>>> scoreSequence(List<List<Feature>> features, int maxResults) {
114                    throw new UnsupportedOperationException("there is no default implementation of the score method.");
115            }
116    }
