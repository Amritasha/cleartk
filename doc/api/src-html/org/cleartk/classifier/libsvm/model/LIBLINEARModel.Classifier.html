047    public class LIBLINEARModel {
048            
049            public static LIBLINEARModel fromFile(File modelFile) throws IOException, CleartkException {
050                    InputStream modelStream = new FileInputStream(modelFile);
051                    LIBLINEARModel model = fromInputStream(modelStream);
052                    modelStream.close();
053                    return model;
054            }
055            
056            public static LIBLINEARModel fromInputStream(InputStream modelStream) throws IOException, CleartkException {
057                    LIBLINEARModel model = new LIBLINEARModel();
058                    
059                    BufferedReader in = new BufferedReader(new InputStreamReader(modelStream));
060                    String buffer;
061                    String[] fields;
062                    
063                    buffer = in.readLine(); // solver_type
064    
065                    buffer = in.readLine(); // nr_class
066                    model.numberOfClasses = Integer.valueOf(buffer.split(" ")[1]);
067                    if( model.numberOfClasses == 2 )
068                            model.numberOfClassifiers = 1;
069                    else
070                            model.numberOfClassifiers = model.numberOfClasses;
071                    
072                    model.classifiers = new Classifier[model.numberOfClassifiers];
073                    for( int i=0; i < model.numberOfClassifiers; i++ )
074                            model.classifiers[i] = new Classifier();
075                    
076                    buffer = in.readLine(); // label
077                    fields = buffer.split(" ");
078                    for( int i = 0; i < model.numberOfClassifiers; i++ ) 
079                            model.classifiers[i].label = Integer.valueOf(fields[i+1]);
080                    if( model.numberOfClasses == 2 )
081                            model.fallbackLabel = Integer.valueOf(fields[2]);
082                    
083                    buffer = in.readLine(); // nr_feature
084                    int numberOfFeatures = Integer.valueOf(buffer.split(" ")[1]);
085                    
086                    buffer = in.readLine(); // bias
087                    double bias = Double.valueOf(buffer.split(" ")[1]);
088                    
089                    buffer = in.readLine(); // w
090                    
091                    for( int i=0; i<numberOfFeatures; i++ ) {
092                            buffer = in.readLine();
093                            fields = buffer.trim().split(" ");
094                            for( int j=0; j<model.numberOfClassifiers; j++ ) {
095                                    model.classifiers[j].weightVector.set(i+1, Double.valueOf(fields[j]));
096                            }
097                    }
098                    
099                    if( bias >= 0 ) {
100                            buffer = in.readLine();
101                            fields = buffer.trim().split(" ");
102                            for( int j=0; j<model.numberOfClassifiers; j++ )
103                                    model.classifiers[j].bias = Double.valueOf(fields[j]) * bias;
104                    }
105                    
106                    in.close();
107                    
108                    return model;
109            }
110            
111            int numberOfClasses;
112            
113            int numberOfClassifiers;
114            
115            double[] biases;
116            
117            Classifier[] classifiers;
118    
119            int fallbackLabel;
120            
121            LIBLINEARModel() {
122            }
123    
124            public int predict(FeatureVector featureVector) {
125                    double[] values = new double[numberOfClassifiers];
126                    
127                    for( int i=0; i<numberOfClassifiers; i++ )
128                            values[i] = classifiers[i].evaluate(featureVector);
129                    
130                    if( numberOfClasses == 2 ) {
131                            if( values[0] > 0 ) {
132                                    return classifiers[0].label;
133                            } else {
134                                    return fallbackLabel;
135                            }
136                    } else {
137                            int max_index = 0;
138                            
139                            for( int i=1; i<numberOfClassifiers; i++ ) {
140                                    if( values[i] > values[max_index] )
141                                            max_index = i;
142                            }
143                            return classifiers[max_index].label;
144                    }
145                    
146            }
147            
148            static class Classifier {
149                    FeatureVector weightVector;
150                    double bias;
151                    int label;
152                    
153                    Classifier() {
154                            this.weightVector = new ArrayFeatureVector();
155                            this.bias = 0.0;
156                            this.label = 0;
157                    }
158                    
159                    double evaluate(FeatureVector featureVector) {
160                            return weightVector.innerProduct(featureVector) + bias;
161                    }
162            }
163    }
