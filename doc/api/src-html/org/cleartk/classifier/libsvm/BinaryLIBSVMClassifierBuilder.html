024    package org.cleartk.classifier.libsvm;
025    
026    import java.io.File;
027    import java.io.IOException;
028    import java.io.InputStream;
029    import java.io.PrintStream;
030    
031    import org.cleartk.classifier.BuildJar;
032    import org.cleartk.classifier.Classifier;
033    import org.cleartk.classifier.ClassifierBuilder;
034    
035    
036    /**
037     * <br>Copyright (c) 2007-2008, Regents of the University of Colorado 
038     * <br>All rights reserved.
039    
040    */
041    
042    public class BinaryLIBSVMClassifierBuilder implements ClassifierBuilder<Boolean> {
043            
044            public String getCommand() {
045                    return "svm-train";
046            }
047            
048            public String getModelName() {
049                    return LIBSVMClassifier.MODEL_NAME;
050            }
051    
052            public void train(File dir, String[] args) throws Exception {
053                    String[] command = new String[args.length + 3];
054                    command[0] = this.getCommand();
055                    System.arraycopy(args, 0, command, 1, args.length);
056                    command[command.length - 2] = new File(dir, "training-data.libsvm").getPath();
057                    command[command.length - 1] = new File(dir, this.getModelName()).getPath();
058                    Process process = Runtime.getRuntime().exec(command);
059                    output(process.getInputStream(), System.out);
060                    output(process.getErrorStream(), System.err);
061                    process.waitFor();
062            }
063    
064            public void buildJar(File dir, String[] args) throws Exception {
065                    BuildJar.OutputStream stream = new BuildJar.OutputStream(dir);
066                    stream.write(this.getModelName(), new File(dir, this.getModelName()));
067                    stream.close();
068            }
069    
070            public Class<? extends Classifier<Boolean>> getClassifierClass() {
071                    return BinaryLIBSVMClassifier.class;
072            }
073    
074            private static void output(InputStream input, PrintStream output) throws IOException {
075                    byte[] buffer = new byte[128];
076                    int count = input.read(buffer);
077                    while (count != -1) {
078                            output.write(buffer, 0, count);
079                            count = input.read(buffer);
080                    }
081            }
082    
083    }
