024    package org.cleartk.classifier;
025    
026    import java.io.File;
027    import java.io.FileInputStream;
028    import java.io.FileOutputStream;
029    import java.io.IOException;
030    import java.util.jar.Attributes;
031    import java.util.jar.JarFile;
032    import java.util.jar.Manifest;
033    
034    /**
035     * <br>Copyright (c) 2007-2008, Regents of the University of Colorado 
036     * <br>All rights reserved.
037    
038     * 
039     * @author Steven Bethard, Philipp Wetzler
040     */
041    public class ClassifierManifest extends Manifest {
042            
043            public static final Attributes.Name CLASSIFIER_BUILDER_ATTRIBUTE = new Attributes.Name("classifierBuilderClass");
044            
045            private ClassifierBuilder<?> classifierBuilder;
046            
047            public ClassifierManifest() {
048                    Attributes attributes = this.getMainAttributes();
049                    attributes.put(Attributes.Name.MANIFEST_VERSION, "1.0");
050            }
051            
052            public ClassifierManifest(ClassifierBuilder<?> classifierBuilder) {
053                    Attributes attributes = this.getMainAttributes();
054                    attributes.put(Attributes.Name.MANIFEST_VERSION, "1.0");
055                    attributes.put(CLASSIFIER_BUILDER_ATTRIBUTE, classifierBuilder.getClass().getName());
056                    this.classifierBuilder = classifierBuilder;
057            }
058            
059            public ClassifierManifest(File dir) throws IOException {
060                    super(new FileInputStream(getFile(dir)));
061                    this.loadClassifierBuilder(getFile(dir).getPath());
062            }
063            
064            public ClassifierManifest(JarFile jarFile) throws IOException {
065                    super(jarFile.getManifest());
066                    this.loadClassifierBuilder(jarFile.getName());
067            }
068            
069            public void write(File dir) throws IOException {
070                    FileOutputStream manifestStream = new FileOutputStream(getFile(dir));
071                    this.write(manifestStream);
072                    manifestStream.close();
073            }
074            
075            public void setClassifierBuilder(ClassifierBuilder<?> classifierBuilder) {
076                    Attributes attributes = this.getMainAttributes();
077                    attributes.put(CLASSIFIER_BUILDER_ATTRIBUTE, classifierBuilder.getClass().getName());
078                    this.classifierBuilder = classifierBuilder;
079            }
080            
081            public ClassifierBuilder<?> getClassifierBuilder() {
082                    return this.classifierBuilder;
083            }
084            
085            private void loadClassifierBuilder(String path) throws IOException {
086                    Attributes attributes = this.getMainAttributes();
087                    String classifierBuilderClassName = attributes.getValue(CLASSIFIER_BUILDER_ATTRIBUTE);
088                    if (classifierBuilderClassName == null) {
089                            throw new IOException(String.format("Missing %s attribute in manifest %s",
090                                            CLASSIFIER_BUILDER_ATTRIBUTE, path));
091                    }
092                    Exception exception = null;
093                    try {
094                            this.classifierBuilder = Class.forName(classifierBuilderClassName).asSubclass(ClassifierBuilder.class).newInstance();
095                    } catch (ClassNotFoundException e) {
096                            exception = e;
097                    } catch (InstantiationException e) {
098                            exception = e;
099                    } catch (IllegalAccessException e) {
100                            exception = e;
101                    }
102                    if (exception != null) {
103                            throw new IOException(String.format("Invalid %s attribute in manifest %s",
104                                            CLASSIFIER_BUILDER_ATTRIBUTE, path));
105                    }
106            }
107            
108            private static File getFile(File dir) {
109                    return new File(dir, "MANIFEST.MF");
110            }
111            
112    }
