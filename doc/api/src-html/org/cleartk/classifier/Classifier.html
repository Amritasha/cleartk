024    package org.cleartk.classifier;
025    
026    import java.util.List;
027    
028    import org.cleartk.classifier.mallet.MalletCRFClassifier;
029    import org.cleartk.classifier.opennlp.MaxentClassifier;
030    
031    /**
032     * <br>Copyright (c) 2007-2008, Regents of the University of Colorado 
033     * <br>All rights reserved.
034    
035     * <p>
036     * 
037     * This class provides an abstraction for interfacing with various ML learning
038     * libraries such as Mallet, Libsvm, OpenNLP Maxent, etc. Each subclass must
039     * override the abstract methods (naturally!) each of which is documented below.
040     * Each subclass must also be able to be instantiated from a single jar file
041     * that contains all the necessary meta-data and data (e.g. the model). The
042     * contents of the jar file will vary for each implementation. However, each jar
043     * file should have a manifest file with two attributes, "returnType" and
044     * "classifier". The jar command can be used to set the manifest of a jar file.
045     * A manifest file should look something like this:
046     * 
047     * <pre>
048     *  Manifest-Version: 1.0
049     *  classifier: org.cleartk.classifier.OpenNLPMaxentClassifier
050     *  returnType: string
051     * </pre>
052     * 
053     * @author Philipp Wetzler
054     * @author Philip Ogren
055     */
056    
057    public interface Classifier<OUTCOME_TYPE> {
058            /**
059             * Classifies a list of features. This method may not be supported for
060             * sequential classifiers.
061             * 
062             * @param features
063             * @return A Value holding the classification result
064             */
065            public abstract OUTCOME_TYPE classify(List<Feature> features);
066    
067            /**
068             * This method provides a default implementation for classifying a bunch of
069             * instances at once. The values returned are the same as they would if
070             * <code>classify(List&lt;Feature&gt;)</code> was called individually on
071             * each list of features. Sequential classifiers should override this
072             * method.
073             * 
074             * @param features
075             *            a list of lists of features.
076             * @return a list of values corresponding to the classifications made.
077             * @see MalletCRFClassifier#classify(List)
078             */
079            public List<OUTCOME_TYPE> classifySequence(List<List<Feature>> features);
080    
081            /**
082             * Provides a way to get a score associated with a classification. The
083             * 'score' in the scored value could be anything but is often a probability.
084             * The only expectation is that a higher score should correspond to higher
085             * confidence in the classification. No other assumptions should be made.
086             * 
087             * @param features
088             *            a list of features to be classified
089             * @return the classification score
090             */
091            public ScoredValue<OUTCOME_TYPE> score(List<Feature> features);
092    
093            /**
094             * Provides a way to get a scores associated with the top N scores for the
095             * best N classifications. The 'score' in the scored value could be anything
096             * but is often a probability. The only expectation is that a higher score
097             * should correspond to higher confidence in the classification. No other
098             * assumptions should be made.
099             * 
100             * @param features
101             *            a list of features to be classified
102             * @param maxResults
103             *            the maximum number of classifications to return.
104             * @return a sorted list of the best N classifications with their scores.
105             */
106            public List<ScoredValue<OUTCOME_TYPE>> score(List<Feature> features, int maxResults);
107    
108            /**
109             * A classifier that is based on a sequence of labels (i.e.
110             * computes/optimizes label transition probabilities) such as the
111             * MalletCRFClassifier should return true.
112             * 
113             * Here is the expectation for implementors. If true is returned, then the
114             * classifier may or may not implement
115             * <code>classify(List&lt;Feature&gt;)</code> (usually not we expect) and
116             * must override <code>classify(List&lt;List&lt;Feature&gt;&gt;)</code>.
117             * If false is returned, then the classifier is required to implement
118             * <code>classify(List&lt;Feature&gt;)</code> and may or many not override
119             * <code>classify(List&lt;List&lt;Feature&gt;&gt;)</code>. If you do not
120             * want to implement <code>classify(List&lt;Feature&gt;)</code> for your
121             * sequential classifier, then simply throw an
122             * UnsupportedOperationException.
123             * 
124             * @return true if this is a sequential classifier, false otherwise.
125             * @see MalletCRFClassifier#isSequential()
126             * @see MaxentClassifier#isSequential()
127             */
128            public abstract boolean isSequential();
129    
130    }
