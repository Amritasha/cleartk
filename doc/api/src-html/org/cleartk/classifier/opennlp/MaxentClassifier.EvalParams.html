024    package org.cleartk.classifier.opennlp;

026    import java.io.DataInputStream;
027    import java.io.IOException;

030    import java.util.List;

035    import org.cleartk.classifier.Classifier_ImplBase;
036    import org.cleartk.classifier.Feature;
037    import org.cleartk.classifier.ScoredValue;
038    import org.cleartk.classifier.encoder.features.contextvalue.ContextValue;
039    
040    import opennlp.maxent.MaxentModel;
041    import opennlp.maxent.io.BinaryGISModelReader;
042    
043    /**
044     * <br>Copyright (c) 2007-2008, Regents of the University of Colorado

049     *
050     */
051    public class MaxentClassifier extends Classifier_ImplBase<String,String,List<ContextValue>> {
052    
053            protected MaxentModel model;
054            
055            public MaxentClassifier(JarFile modelFile) throws IOException {
056                    super(modelFile);
057                    ZipEntry modelEntry = modelFile.getEntry("model.maxent");
058                    this.model = new BinaryGISModelReader(new DataInputStream(new GZIPInputStream(modelFile.getInputStream(modelEntry)))).getModel();
059            }
060    
061            @Override
062            public String classify(List<Feature> features) {
063                    EvalParams evalParams = convertToEvalParams(features);
064                    String encodedOutcome = this.model.getBestOutcome(this.model.eval(evalParams.getContext(), evalParams.getValues()));
065                    return outcomeEncoder.decode(encodedOutcome);
066            }
067    
068            @Override
069            public ScoredValue<String> score(List<Feature> features) {
070                    EvalParams evalParams = convertToEvalParams(features);
071                    double[] evalResults = this.model.eval(evalParams.getContext(), evalParams.getValues());
072    
073                    String encodedBestOutcome = outcomeEncoder.decode(this.model.getBestOutcome(evalResults));
074                    double bestResult = evalResults[this.model.getIndex(encodedBestOutcome)];
075                    return new ScoredValue<String>(encodedBestOutcome, bestResult);
076            }
077    
078            @Override
079            public List<ScoredValue<String>> score(List<Feature> features, int maxResults) {
080                    EvalParams evalParams = convertToEvalParams(features);
081                    double[] evalResults = this.model.eval(evalParams.getContext(), evalParams.getValues());
082                    String[] encodedOutcomes = (String[]) this.model.getDataStructures()[2];
083                    
084                    List<ScoredValue<String>> returnValues = new ArrayList<ScoredValue<String>>();
085                    for(int i=0; i<evalResults.length; i++) {
086                            returnValues.add(new ScoredValue<String>(outcomeEncoder.decode(encodedOutcomes[i]), evalResults[i]));
087                    }
088                    
089                    Collections.sort(returnValues);
090                    if(returnValues.size() > maxResults) {
091                            return returnValues.subList(0, maxResults);
092                    } else {
093                            return returnValues;
094                    }
095            }
096            
097            /**
098             * @return false
099             */
100            @Override
101            public boolean isSequential() {
102                    return false;
103            }
104            
105            private EvalParams convertToEvalParams(List<Feature> features) {
106                    String[] context = new String[features.size()];
107                    float[] values = new float[features.size()];
108                    
109                    List<ContextValue> contexts = featuresEncoder.encodeAll(features);
110                    
111                    for(int i=0; i<contexts.size(); i++) {
112                            ContextValue contextValue = contexts.get(i);
113                            context[i] = contextValue.getContext();
114                            values[i] = contextValue.getValue();
115                    }
116    
117                    return new EvalParams(context, values);
118            }
119    
120            public class EvalParams{
121                    private String[] context;
122                    private float[] values;
123                    public String[] getContext() {
124                            return context;
125                    }
126                    public float[] getValues() {
127                            return values;
128                    }
129                    public EvalParams(String[] context, float[] values) {
130                            this.context = context;
131                            this.values = values;
132                    }
133                    
134            }
135    
136    }
