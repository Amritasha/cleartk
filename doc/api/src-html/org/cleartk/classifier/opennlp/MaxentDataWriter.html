024    package org.cleartk.classifier.opennlp;
025    
026    import java.io.IOException;
027    import java.io.PrintWriter;
028    import java.util.List;
029    
030    import opennlp.maxent.RealValueFileEventStream;
031    
032    import org.apache.uima.UimaContext;
033    import org.apache.uima.analysis_engine.AnalysisEngineProcessException;
034    import org.apache.uima.resource.ResourceInitializationException;
035    import org.cleartk.classifier.ClassifierBuilder;
036    import org.cleartk.classifier.DataWriter_ImplBase;
037    import org.cleartk.classifier.Instance;
038    import org.cleartk.classifier.encoder.EncoderFactory;
039    import org.cleartk.classifier.encoder.factory.ContextValueEncoderFactory;
040    import org.cleartk.classifier.encoder.features.contextvalue.ContextValue;
041    import org.cleartk.classifier.encoder.features.contextvalue.ContextValueFeaturesEncoder;
042    
043    
044    /**
045     * <br>Copyright (c) 2007-2008, Regents of the University of Colorado 
046     * <br>All rights reserved.
047    
048     * <p>
049     * 
050     * This training data consumer produces training data suitable for <a
051     * href="http://maxent.sourceforge.net/"> OpenNLP Maxent package</a>.
052     * 
053     * Each line of the training data contains a label/result for the instance
054     * followed by a string representation of each feature. Models trained with data
055     * generated by this class should use RealValueFileEventStream. For relevant
056     * discussion, please see:
057     * 
058     * https://sourceforge.net/forum/forum.php?thread_id=1925312&forum_id=18385
059     * 
060     * 
061     * 
062     * @author Philip Ogren
063     * @author Steven Bethard
064     * @see RealValueFileEventStream
065     */
066    public class MaxentDataWriter extends DataWriter_ImplBase<String, String, List<ContextValue>> {
067    
068            public static final String FEATURE_LOOKUP_FILE_NAME = "feature-lookup.txt";
069    
070            protected PrintWriter trainingDataWriter;
071    
072            @Override
073            public void initialize(UimaContext context) throws ResourceInitializationException {
074                    super.initialize(context);
075    
076                    // initialize output writer and Classifier class
077                    this.trainingDataWriter = this.getPrintWriter("training-data.maxent");
078            }
079    
080            public String consume(Instance<String> instance) {
081                    // write the label
082                    String outcomeString = this.outcomeEncoder.encode(instance.getOutcome());
083                    this.trainingDataWriter.print(outcomeString);
084    
085                    // aggregate the features
086                    List<ContextValue> features = this.featuresEncoder.encodeAll(instance.getFeatures());
087    
088                    // write each of the string features, encoded, into the training data
089                    for (ContextValue contextValue : features) {
090                            this.trainingDataWriter.print(' ');
091                            if(contextValue.getValue() == 1)
092                                    trainingDataWriter.print(contextValue.getContext());
093                            else
094                                    trainingDataWriter.print(contextValue.getContext() + "=" + contextValue.getValue());
095    
096                    }
097    
098                    // complete the feature line
099                    this.trainingDataWriter.println();
100    
101                    // no labels created, so return null
102                    return null;
103            }
104    
105            @Override
106            public void collectionProcessComplete() throws AnalysisEngineProcessException {
107                    if (featuresEncoder instanceof ContextValueFeaturesEncoder) {
108                            try {
109                                    ContextValueFeaturesEncoder cvfe = (ContextValueFeaturesEncoder) featuresEncoder;
110                                    if(cvfe.isCompressFeatures())
111                                            cvfe.writeKeys(this.getPrintWriter(FEATURE_LOOKUP_FILE_NAME));
112                            }
113                            catch (ResourceInitializationException e) {
114                                    throw new AnalysisEngineProcessException(e);
115                            }
116                            catch (IOException e) {
117                                    throw new AnalysisEngineProcessException(e);
118                            }
119                    }
120                    
121                    super.collectionProcessComplete();
122            }
123    
124            @Override
125            protected Class<? extends ClassifierBuilder<? extends String>> getDefaultClassifierBuilderClass() {
126                    return MaxentClassifierBuilder.class;
127            }
128    
129            @Override
130            protected Class<? extends EncoderFactory> getDefaultEncoderFactoryClass() {
131                    return ContextValueEncoderFactory.class;
132            }
133    }
