024    package org.cleartk.classifier.svmlight;
025    
026    import java.io.File;
027    import java.io.IOException;
028    import java.io.InputStream;
029    import java.io.PrintStream;
030    
031    import org.cleartk.classifier.BuildJar;
032    import org.cleartk.classifier.Classifier;
033    import org.cleartk.classifier.ClassifierBuilder;
034    
035    /**
036     * <br>Copyright (c) 2007-2008, Regents of the University of Colorado 
037     * <br>All rights reserved.
038    
039    */
040    
041    public class SVMlightClassifierBuilder implements ClassifierBuilder<Boolean> {
042    
043            public static void train(String filePath, String[] args) throws Exception {
044                    String[] command = new String[args.length + 3];
045                    command[0] = "svm_learn";
046                    System.arraycopy(args, 0, command, 1, args.length);
047                    command[command.length - 2] = new File(filePath).getPath();
048                    command[command.length - 1] = new File(filePath + ".model").getPath();
049                    Process process = Runtime.getRuntime().exec(command);
050                    output(process.getInputStream(), System.out);
051                    output(process.getErrorStream(), System.err);
052                    process.waitFor();
053            }
054    
055            public void train(File dir, String[] args) throws Exception {
056                    train(new File(dir, "training-data.svmlight").getPath(), args);
057            }
058            
059            public void buildJar(File dir, String[] args) throws Exception {
060                    BuildJar.OutputStream stream = new BuildJar.OutputStream(dir);
061                    stream.write("model.svmlight", new File(dir, "training-data.svmlight.model"));
062                    stream.close();
063            }
064    
065            public Class<? extends Classifier<Boolean>> getClassifierClass() {
066                    return SVMlightClassifier.class;
067            }
068            
069            private static void output(InputStream input, PrintStream output) throws IOException {
070                    byte[] buffer = new byte[128];
071                    int count = input.read(buffer);
072                    while (count != -1) {
073                            output.write(buffer, 0, count);
074                            count = input.read(buffer);
075                    }
076            }
077    
078    }
