024    package org.cleartk.classifier.svmlight;
025    
026    import java.io.IOException;
027    import java.util.List;
028    import java.util.Map;
029    import java.util.TreeMap;
030    import java.util.jar.JarFile;
031    import java.util.zip.ZipEntry;
032    
033    import org.cleartk.classifier.Classifier_ImplBase;
034    import org.cleartk.classifier.Feature;
035    import org.cleartk.classifier.svmlight.model.SVMlightModel;
036    import org.cleartk.classifier.util.featurevector.FeatureVector;
037    
038    /**
039     * <br>Copyright (c) 2007-2008, Regents of the University of Colorado 
040     * <br>All rights reserved.
041    
042     */
043    public class OVASVMlightClassifier extends Classifier_ImplBase<String,Integer,FeatureVector> {
044            
045            Map<Integer, SVMlightModel> models;
046            
047    
048            public OVASVMlightClassifier(JarFile modelFile) throws IOException {
049                    super(modelFile);
050                    this.models = new TreeMap<Integer,SVMlightModel>();
051                    
052                    int i = 1;
053                    ZipEntry modelEntry = modelFile.getEntry(String.format("model-%d.svmlight", i));
054                    while( modelEntry != null ) {
055                            SVMlightModel m = SVMlightModel.fromInputStream(modelFile.getInputStream(modelEntry));
056                            this.models.put(i, m);
057                            i += 1;
058                            modelEntry = modelFile.getEntry(String.format("model-%d.svmlight", i));
059                    }
060                    
061                    if (this.models.isEmpty()) {
062                            throw new IOException(String.format("no models found in %s", modelFile.getName()));
063                    }
064            }
065    
066    
067            @Override
068            public String classify(List<Feature> features) {
069                    FeatureVector featureVector = this.featuresEncoder.encodeAll(features);
070                    
071                    int maxScoredIndex = 0;
072                    double maxScore = 0;
073                    boolean first = true;
074                    for( int i : models.keySet() ) {
075                            double score = models.get(i).evaluate(featureVector);
076                            if( first || score > maxScore ) {
077                                    first = false;
078                                    maxScore = score;
079                                    maxScoredIndex = i;
080                            }
081                    }
082                    
083                    return outcomeEncoder.decode(maxScoredIndex);
084            }
085            
086    
087            @Override
088            public boolean isSequential() {
089                    return false;
090            }
091    
092    }
