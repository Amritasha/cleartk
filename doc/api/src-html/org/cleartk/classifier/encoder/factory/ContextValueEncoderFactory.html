024    package org.cleartk.classifier.encoder.factory;
025    
026    import java.util.List;
027    
028    import org.apache.uima.UimaContext;
029    import org.cleartk.classifier.encoder.EncoderFactory;
030    import org.cleartk.classifier.encoder.features.FeaturesEncoder;
031    import org.cleartk.classifier.encoder.features.contextvalue.ContextValue;
032    import org.cleartk.classifier.encoder.features.contextvalue.ContextValueFeatureEncoder;
033    import org.cleartk.classifier.encoder.features.contextvalue.ContextValueFeaturesEncoder;
034    import org.cleartk.classifier.encoder.outcome.OutcomeEncoder;
035    import org.cleartk.classifier.encoder.outcome.StringToStringOutcomeEncoder;
036    
037    
038    /**
039     * <br>Copyright (c) 2007-2008, Regents of the University of Colorado 
040     * <br>All rights reserved.
041    
042    */
043    
044    public class ContextValueEncoderFactory implements EncoderFactory {
045    
046            public static final String PARAM_COMPRESS = "Compress";
047    
048            public FeaturesEncoder<List<ContextValue>> createFeaturesEncoder(UimaContext context) {
049                    Boolean compress = (Boolean) context.getConfigParameterValue(PARAM_COMPRESS);
050                    if(compress == null)
051                            compress = false;
052                    
053                    return new ContextValueFeaturesEncoder(new ContextValueFeatureEncoder(compress));
054            }
055    
056            public OutcomeEncoder<?, ?> createOutcomeEncoder(UimaContext context) {
057                    return new StringToStringOutcomeEncoder();
058            }
059    
060    }
