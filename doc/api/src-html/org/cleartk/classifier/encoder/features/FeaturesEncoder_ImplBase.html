037    
038    public abstract class FeaturesEncoder_ImplBase<T, U> implements FeaturesEncoder<T> {
039            
040            private static final long serialVersionUID = 7789468136107138541L;
041            
042            public static final String ENCODER_FILE_NAME = "encoding.ser";
043    
044            public abstract T encodeAll(Iterable<Feature> features);
045            
046            public void addEncoder(FeatureEncoder<U> encoder) {
047                    featureEncoders.add(encoder);
048            }
049            
050            public void allowNewFeatures(boolean flag) {}
051    
052            protected List<U> encode(Feature feature) {
053                    for( FeatureEncoder<U> featureEncoder : featureEncoders ) {
054                            if( featureEncoder.encodes(feature) )
055                                    return featureEncoder.encode(feature);
056                    }
057                    
058                    throw new IllegalArgumentException();
059            }
060            
061            protected List<FeatureEncoder<U>> featureEncoders = new ArrayList<FeatureEncoder<U>>();
062    }
