024    package org.cleartk.classifier;
025    
026    import java.io.File;
027    import java.io.IOException;
028    import java.util.List;
029    
030    import org.apache.uima.UimaContext;
031    import org.apache.uima.analysis_engine.AnalysisEngineProcessException;
032    import org.apache.uima.resource.ResourceInitializationException;
033    import org.cleartk.CleartkException;
034    import org.cleartk.util.ReflectionUtil;
035    import org.cleartk.util.UIMAUtil;
036    
037    public class SequentialDataWriterAnnotator<OUTCOME_TYPE> extends SequentialInstanceConsumer_ImplBase<OUTCOME_TYPE> {
038    
039            /**
040             * "org.cleartk.classifier.SequentialDataWriterAnnotator.PARAM_OUTPUT_DIRECTORY"
041             * is a single, required, string parameter that provides the name of the
042             * directory where the training data will be written.
043             */
044            public static final String PARAM_OUTPUT_DIRECTORY = "org.cleartk.classifier.SequentialDataWriterAnnotator.PARAM_OUTPUT_DIRECTORY";
045            
046            /**
047             * "org.cleartk.classifier.SequentialDataWriterAnnotator.PARAM_DATAWRITER_FACTORY_CLASS"
048             * is a single, required, string parameter that provides the full name of
049             * the SequentialDataWriterFactory class to be used.  
050             */
051            public static final String PARAM_DATAWRITER_FACTORY_CLASS = "org.cleartk.classifier.SequentialDataWriterAnnotator.PARAM_DATAWRITER_FACTORY_CLASS";
052    
053            private SequentialDataWriter<OUTCOME_TYPE> sequentialDataWriter;
054            
055            @Override
056            public void initialize(UimaContext context) throws ResourceInitializationException {
057                    super.initialize(context);
058                    
059                    String outputDirectoryPath = (String) UIMAUtil.getRequiredConfigParameterValue(
060                                    context, PARAM_OUTPUT_DIRECTORY);
061                    File outputDirectory = new File(outputDirectoryPath);
062    
063                    // create the factory and instantiate the data writer
064                    SequentialDataWriterFactory<?> factory = UIMAUtil.create(
065                                    context, PARAM_DATAWRITER_FACTORY_CLASS, SequentialDataWriterFactory.class);
066                    SequentialDataWriter<?> untypedDataWriter;
067                    try {
068                            untypedDataWriter = factory.createSequentialDataWriter(outputDirectory);
069                    } catch (IOException e) {
070                            throw new ResourceInitializationException(e);
071                    }
072                    UIMAUtil.initialize(untypedDataWriter, context);
073                    
074                    // check the type of the DataWriter and assign the instance variable
075                    this.checkOutcomeType(SequentialDataWriter.class, "OUTCOME_TYPE", untypedDataWriter);
076                    this.sequentialDataWriter = ReflectionUtil.uncheckedCast(untypedDataWriter);
077            }
078    
079            
080            @Override
081            public void collectionProcessComplete() throws AnalysisEngineProcessException {
082                    super.collectionProcessComplete();
083    
084                    try {
085                            sequentialDataWriter.finish();
086                    }
087                    catch (CleartkException ctke) {
088                            throw new AnalysisEngineProcessException(ctke);
089                    }
090            }
091    
092            public boolean expectsOutcomes() {
093                    return true;
094            }
095            
096            public List<OUTCOME_TYPE> consumeSequence(List<Instance<OUTCOME_TYPE>> instances)  throws CleartkException{
097                            sequentialDataWriter.writeSequence(instances);
098                            return null;
099            }
100    
101    }
