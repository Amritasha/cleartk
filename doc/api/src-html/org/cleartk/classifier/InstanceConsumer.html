027    
028    /**

031

034     * @author Steven Bethard
035     */
036    public interface InstanceConsumer<OUTCOME_TYPE> {
037    
038            /**
039             * Consume the instance and return the label assigned to the instance. If
040             * the consumer does not assign labels to instances (e.g. a training data
041             * consumer), this method should return null.
042             * 
043             * @param instance
044             *            The instance (features and label) to be consumed.
045             * @return The assigned label, or null if a label was not assigned.
046             */
047            public OUTCOME_TYPE consume(Instance<OUTCOME_TYPE> instance);
048    
049            /**
050             * Consume the instances and return the labels assigned to the instances. If
051             * the consumer does not assign labels to instances (e.g. a training data
052             * consumer), this method should return null.
053             * 
054             * @param instances
055             *            The instances to be consumed.
056             * @return The assigned labels, or null if labels were not assigned. Labels
057             *         should be in the same order as the original instances.
058             */
059            public List<OUTCOME_TYPE> consumeAll(List<Instance<OUTCOME_TYPE>> instances);
060    
061            /**
062             * This method provides an annotation handler (or anything else using an
063             * InstanceConsumer) a way to determine whether or not the instance consumer
064             * expects the Instances it consumes to have outcomes. For example, if your
065             * instance consumer corresponds to a training data writer, then it will
066             * expect outcomes. However, if your instance consumer is
067             * ClassifierAnnotator, then it will not expect outcomes and therefore your
068             * annotation handler should not bother worrying about whether the instances
069             * is passes to the consume method have outcomes or not.
070             * 
071             * This method says nothing about whether values are returned from consume
072             * and consumeAll methods.
073             * 
074             * @return True if the consumer expects the classification Instances to have
075             *         outcomes, and false otherwise.
076             */
077            public boolean expectsOutcomes();
078    }
