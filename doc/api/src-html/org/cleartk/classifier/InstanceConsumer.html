024    package org.cleartk.classifier;
025    
026    import org.cleartk.CleartkException;
027    
028    /**
029     * <br>
030     * Copyright (c) 2009, Regents of the University of Colorado <br>
031     * All rights reserved.
032     * 
033     * <p>
034     * 
035     * @author Steven Bethard
036     * @author Philip Ogren
037     */
038    public interface InstanceConsumer<OUTCOME_TYPE> {
039    
040            /**
041             * "org.cleartk.classifier.InstanceConsumer.PARAM_ANNOTATION_HANDLER"
042             * is a single, required, string parameter that provides the full name of
043             * the AnnotationHandler class that will be used with this InstanceConsumer.
044             */
045            public static final String PARAM_ANNOTATION_HANDLER = "org.cleartk.classifier.InstanceConsumer.PARAM_ANNOTATION_HANDLER";
046    
047            /**
048             * Consume the instance and return the outcome (classification) assigned to
049             * the instance. If the consumer does not assign outcomes to instances (e.g.
050             * a training data consumer), this method should return null.
051             * 
052             * @param instance
053             *            The instance (features and label) to be consumed.
054             * @return The assigned outcome (classification), or null if an outcome was
055             *         not assigned.
056             */
057            public OUTCOME_TYPE consume(Instance<OUTCOME_TYPE> instance) throws CleartkException;
058    
059            /**
060             * This method provides an annotation handler (or anything else using an
061             * InstanceConsumer) a way to determine whether or not the instance consumer
062             * expects the Instances it consumes to have outcomes. For example, if your
063             * instance consumer corresponds to a training data writer, then it will
064             * expect outcomes. However, if your instance consumer is
065             * ClassifierAnnotator, then it will not expect outcomes and therefore your
066             * annotation handler should not bother worrying about whether the instances
067             * is passes to the consume method have outcomes or not.
068             * 
069             * This method says nothing about whether values are returned from consume.
070             * 
071             * @return true if the consumer expects the classification Instances to have
072             *         outcomes, and false otherwise.
073             */
074            public abstract boolean expectsOutcomes();
075    }
