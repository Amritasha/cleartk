024    package org.cleartk.classifier;
025    
026    import java.io.BufferedWriter;
027    import java.io.File;
028    import java.io.FileOutputStream;
029    import java.io.FileWriter;
030    import java.io.IOException;
031    import java.io.ObjectOutputStream;
032    import java.io.PrintWriter;
033    import java.util.ArrayList;
034    import java.util.List;
035    
036    import org.cleartk.CleartkException;
037    import org.cleartk.classifier.encoder.features.FeaturesEncoder;
038    import org.cleartk.classifier.encoder.features.FeaturesEncoder_ImplBase;
039    import org.cleartk.classifier.encoder.outcome.OutcomeEncoder;
040    
041    /**
042     * <br>Copyright (c) 2007-2008, Regents of the University of Colorado 
043     * <br>All rights reserved.
044     */
045    public abstract class DataWriter_ImplBase<INPUTOUTCOME_TYPE, OUTPUTOUTCOME_TYPE, FEATURES_TYPE> implements DataWriter<INPUTOUTCOME_TYPE> {
046    
047            public DataWriter_ImplBase(File outputDirectory) {
048                    // Initialize the output directory and list of output writers
049                    this.outputDirectory = outputDirectory;
050                    this.writers = new ArrayList<PrintWriter>();
051    
052                    // Initialize the Manifest
053                    this.classifierManifest = new ClassifierManifest();
054            }
055    
056            public void write(Instance<INPUTOUTCOME_TYPE> instance) throws CleartkException {
057                    FEATURES_TYPE features = featuresEncoder.encodeAll(instance.getFeatures());
058                    OUTPUTOUTCOME_TYPE outcome = outcomeEncoder.encode(instance.getOutcome());
059                    writeEncoded(features, outcome);
060            }
061    
062            public abstract void writeEncoded(FEATURES_TYPE features, OUTPUTOUTCOME_TYPE outcome) throws CleartkException;
063    
064            public void finish() throws CleartkException {
065                    try {
066                            // close out the file writers
067                            for (PrintWriter writer : this.writers) {
068                                    writer.flush();
069                                    writer.close();
070                            }
071            
072                            // serialize the features encoder
073                            ObjectOutputStream os = new ObjectOutputStream(new FileOutputStream(
074                                            getFile(FeaturesEncoder_ImplBase.ENCODERS_FILE_NAME)));
075                            os.writeObject(this.featuresEncoder);
076                            os.writeObject(this.outcomeEncoder);
077                            os.close();
078            
079                            // set manifest values
080                            try {
081                                    Class<? extends ClassifierBuilder<? extends INPUTOUTCOME_TYPE>> classifierBuilderClass = this
082                                                    .getDefaultClassifierBuilderClass();
083                                    this.classifierManifest.setClassifierBuilder(classifierBuilderClass.newInstance());
084                            }
085                            catch (InstantiationException e) {
086                                    throw new RuntimeException(e);
087                            }
088                            catch (IllegalAccessException e) {
089                                    throw new RuntimeException(e);
090                            }
091            
092                            // write the manifest file
093                            classifierManifest.write(this.outputDirectory);
094                    } catch(IOException ioe) {
095                            throw new CleartkException(ioe);
096                    }
097            }
098    
099            public void setFeaturesEncoder(FeaturesEncoder<FEATURES_TYPE> featuresEncoder) {
100                    this.featuresEncoder = featuresEncoder;
101            }
102    
103            public void setOutcomeEncoder(OutcomeEncoder<INPUTOUTCOME_TYPE, OUTPUTOUTCOME_TYPE> outcomeEncoder) {
104                    this.outcomeEncoder = outcomeEncoder;
105            }
106    
107            public ClassifierManifest getClassifierManifest() {
108                    return classifierManifest;
109            }
110    
111            protected File getFile(String fileName) {
112                    return new File(this.outputDirectory, fileName);
113            }
114    
115            protected PrintWriter getPrintWriter(String fileName) throws IOException {
116                    File file = this.getFile(fileName);
117                    if (!file.getParentFile().exists()) {
118                            file.getParentFile().mkdirs();
119                    }
120    
121                    PrintWriter writer = new PrintWriter(new BufferedWriter(new FileWriter(file)));
122                    this.writers.add(writer);
123                    return writer;
124            }
125    
126            private File outputDirectory;
127            private List<PrintWriter> writers;
128            protected ClassifierManifest classifierManifest;
129            protected FeaturesEncoder<FEATURES_TYPE> featuresEncoder;
130            protected OutcomeEncoder<INPUTOUTCOME_TYPE, OUTPUTOUTCOME_TYPE> outcomeEncoder;
131    }
