041    public class  ViterbiClassifierBuilder<OUTCOME_TYPE> implements ClassifierBuilder<OUTCOME_TYPE>{
042    
043            public static final String DELEGATED_MODEL_FILE_NAME = "delegated-model.jar";
044    
045            public void buildJar(File dir, String[] args) throws Exception {
046                    File delegatedOutputDirectory = new File(dir, ViterbiDataWriter.DELEGATED_MODEL_DIRECTORY_NAME);
047                    FileUtil.copyFile(new File(delegatedOutputDirectory, BuildJar.MODEL_FILE_NAME), new File(dir, DELEGATED_MODEL_FILE_NAME));
048                    
049                    BuildJar.OutputStream stream = new BuildJar.OutputStream(dir);
050                    stream.write(DELEGATED_MODEL_FILE_NAME, new File(dir, DELEGATED_MODEL_FILE_NAME));
051                    stream.write(ViterbiDataWriter.OUTCOME_FEATURE_EXTRACTOR_FILE_NAME, new File(dir, ViterbiDataWriter.OUTCOME_FEATURE_EXTRACTOR_FILE_NAME));
052                    stream.close();
053    
054            }
055    
056            public Class<?> getClassifierClass() {
057                    return ViterbiClassifier.class;
058            }
059    
060            public void train(File dir, String[] args) throws Exception {
061                    File delegatedOutputDirectory = new File(dir, ViterbiDataWriter.DELEGATED_MODEL_DIRECTORY_NAME);
062                    String[] delegatedArgs = new String[args.length + 1];
063                    System.arraycopy(args, 0, delegatedArgs, 1, args.length);
064                    delegatedArgs[0] = delegatedOutputDirectory.getPath();
065                    Train.main(delegatedArgs);
066            }
067    
068    }
