024    package org.cleartk.classifier.viterbi;
025    
026    import java.io.File;
027    import java.io.FileOutputStream;
028    import java.io.ObjectOutputStream;
029    import java.lang.reflect.Type;
030    import java.util.ArrayList;
031    import java.util.List;
032    import java.util.Map;
033    import java.util.jar.Attributes;
034    
035    import org.apache.uima.UimaContext;
036    import org.apache.uima.resource.ResourceInitializationException;
037    import org.cleartk.CleartkException;
038    import org.cleartk.Initializable;
039    import org.cleartk.classifier.ClassifierBuilder;
040    import org.cleartk.classifier.ClassifierManifest;
041    import org.cleartk.classifier.DataWriter;
042    import org.cleartk.classifier.DataWriterFactory;
043    import org.cleartk.classifier.Feature;
044    import org.cleartk.classifier.Instance;
045    import org.cleartk.classifier.SequentialDataWriter;
046    import org.cleartk.classifier.feature.extractor.outcome.OutcomeFeatureExtractor;
047    import org.cleartk.util.ReflectionUtil;
048    import org.cleartk.util.UIMAUtil;
049    import org.cleartk.util.ReflectionUtil.TypeArgumentDelegator;
050    
051    public class ViterbiDataWriter<OUTCOME_TYPE> implements
052                    SequentialDataWriter<OUTCOME_TYPE>, Initializable, TypeArgumentDelegator {
053    
054            public static final String OUTCOME_FEATURE_EXTRACTOR_FILE_NAME = "outcome-features-extractors.ser";
055    
056            public static final String DELEGATED_MODEL_DIRECTORY_NAME = "delegated-model";
057    
058            /**
059             * "org.cleartk.classifier.viterbi.ViterbiDataWriter.PARAM_OUTCOME_FEATURE_EXTRACTORS"
060             * is an optional, multi-valued, string parameter that specifies which
061             * OutcomeFeatureExtractors should be used. Each value of this parameter
062             * should be the name of a class that implements
063             * {@link OutcomeFeatureExtractor}. One valid value that you might use is"org.cleartk.classifier.feature.extractor.outcome.DefaultOutcomeFeatureExtractor"
064             * .
065             */
066            public static final String PARAM_OUTCOME_FEATURE_EXTRACTORS = "org.cleartk.classifier.viterbi.ViterbiDataWriter.PARAM_OUTCOME_FEATURE_EXTRACTORS";
067    
068            /**
069             * "org.cleartk.classifier.viterbi.ViterbiDataWriter.PARAM_DELEGATED_DATAWRITER_FACTORY_CLASS"
070             * is a single, required, string parameter that provides the full name of
071             * the DataWriterFactory class that will be wrapped.
072             */
073            public static final String PARAM_DELEGATED_DATAWRITER_FACTORY_CLASS = "org.cleartk.classifier.viterbi.ViterbiDataWriter.PARAM_DELEGATED_DATAWRITER_FACTORY_CLASS";
074    
075            public static final Attributes.Name DELEGATED_CLASSIFIER_BUILDER_ATTRIBUTE = new Attributes.Name(
076                            "delegatedClassifierBuilderClass");
077    
078            protected DataWriter<OUTCOME_TYPE> delegatedDataWriter;
079    
080            private File outputDirectory;
081            private File delegatedOutputDirectory;
082    
083            private OutcomeFeatureExtractor[] outcomeFeatureExtractors;
084    
085            public ViterbiDataWriter(File outputDirectory) {
086                    this.outputDirectory = outputDirectory;
087                    this.delegatedOutputDirectory = new File(this.outputDirectory, DELEGATED_MODEL_DIRECTORY_NAME);
088            }
089    
090            public void initialize(UimaContext context) throws ResourceInitializationException {
091                    try {
092                            DataWriterFactory<?> dataWriterFactory = UIMAUtil.create(context, PARAM_DELEGATED_DATAWRITER_FACTORY_CLASS,
093                                            DataWriterFactory.class);
094                            this.delegatedDataWriter = ReflectionUtil.uncheckedCast(dataWriterFactory.createDataWriter(delegatedOutputDirectory));
095                            UIMAUtil.initialize(this.delegatedDataWriter, context);
096            
097                            String[] outcomeFeatureExtractorNames = (String[]) UIMAUtil.getDefaultingConfigParameterValue(context,
098                                            PARAM_OUTCOME_FEATURE_EXTRACTORS, null);
099                            if (outcomeFeatureExtractorNames == null) {
100                                    outcomeFeatureExtractors = new OutcomeFeatureExtractor[0];
101                            }
102                            else {
103                                    outcomeFeatureExtractors = new OutcomeFeatureExtractor[outcomeFeatureExtractorNames.length];
104                                    for (int i = 0; i < outcomeFeatureExtractorNames.length; i++) {
105                                            Class<?> cls = Class.forName(outcomeFeatureExtractorNames[i]);
106                                            Class<? extends OutcomeFeatureExtractor> outcomeFeatureExtractorClass = cls
107                                                            .asSubclass(OutcomeFeatureExtractor.class);
108                                            outcomeFeatureExtractors[i] = outcomeFeatureExtractorClass.newInstance();
109                                            outcomeFeatureExtractors[i].initialize(context);
110                                    }
111                            }
112                    }
113                    catch(Exception e){
114                            throw new ResourceInitializationException(e);
115                    }
116            }
117    
118            public void writeSequence(List<Instance<OUTCOME_TYPE>> instances) throws CleartkException {
119                    List<Object> outcomes = new ArrayList<Object>();
120                    for (Instance<OUTCOME_TYPE> instance : instances) {
121                            List<Feature> instanceFeatures = instance.getFeatures();
122                            for (OutcomeFeatureExtractor outcomeFeatureExtractor : outcomeFeatureExtractors) {
123                                    instanceFeatures.addAll(outcomeFeatureExtractor.extractFeatures(outcomes));
124                            }
125                            outcomes.add(instance.getOutcome());
126                            delegatedDataWriter.write(instance);
127                    }
128    
129            }
130    
131            
132            public void finish() throws CleartkException {
133                    try {
134                            this.delegatedDataWriter.finish();
135    
136                            ObjectOutputStream os = new ObjectOutputStream(new FileOutputStream(new File(outputDirectory, OUTCOME_FEATURE_EXTRACTOR_FILE_NAME)));
137                            os.writeObject(this.outcomeFeatureExtractors);
138                            os.close();
139    
140                            ClassifierManifest classifierManifest = new ClassifierManifest();
141                            Class<? extends ClassifierBuilder<? extends OUTCOME_TYPE>> classifierBuilderClass = this.getDefaultClassifierBuilderClass();
142                            classifierManifest.setClassifierBuilder(classifierBuilderClass.newInstance());
143                            classifierManifest.write(this.outputDirectory);
144    
145                    }
146                    catch (Exception e) {
147                            throw new CleartkException(e);
148                    }
149            }
150    
151            public Class<? extends ClassifierBuilder<OUTCOME_TYPE>> getDefaultClassifierBuilderClass() {
152                    return ReflectionUtil.uncheckedCast(ViterbiClassifierBuilder.class);
153            }
154    
155            public Map<String, Type> getTypeArguments(Class<?> genericType) {
156                    if (genericType.equals(SequentialDataWriter.class)) {
157                            genericType = DataWriter.class;
158                    }
159                    return ReflectionUtil.getTypeArguments(genericType, this.delegatedDataWriter);
160            }
161    }
