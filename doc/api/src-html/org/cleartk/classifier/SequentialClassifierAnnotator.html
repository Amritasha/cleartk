024    package org.cleartk.classifier;
025    
026    import java.io.IOException;
027    import java.util.ArrayList;
028    import java.util.List;
029    
030    import org.apache.uima.UimaContext;
031    import org.apache.uima.resource.ResourceInitializationException;
032    import org.cleartk.CleartkException;
033    import org.cleartk.util.ReflectionUtil;
034    import org.cleartk.util.UIMAUtil;
035    
036    /**
037     * <br>
038     * Copyright (c) 2009, Regents of the University of Colorado <br>
039     * All rights reserved.
040     * 
041     * 
042     * ClassifierAnnotator objects are used to take the classification labels
043     * produced by a Classifier model, and add these as annotations to new
044     * documents. In order to create a new ClassifierAnnotator, you need: (1) A
045     * Classifier object stored in a jar file (2) A AnnotationHandler which defines
046     * a feature extraction routine and how the classification labels should be
047     * turned into annotations
048     * 
049     * For each document, AnnotationHandler.produce(JCas, InstanceConsumer) is
050     * called, passing the document as a JCas and this object as the consumer. The
051     * AnnotationHandler should then extract lists of features and pass them back to
052     * the ClassifierAnnotator as ClassifierInstance objects, using one of:
053     * InstanceConsumer.consume(ClassifierInstance)
054     * InstanceConsumer.consumeAll(List) The consumer (this class) will then pass
055     * the features on to the classifier, collect the classification labels, and
056     * then return those labels to the producer. The AnnotationHandler should then
057     * add those labels as annotations to the document in whatever way it sees fit.
058     * 
059     * @see org.cleartk.classifier.Classifier
060     * @see org.cleartk.classifier.AnnotationHandler
061     * @see org.cleartk.example.pos.ExamplePOSAnnotationHandler
062     * 
063     * @author Steven Bethard
064     * @author Philip Ogren
065     */
066    public class SequentialClassifierAnnotator<OUTCOME_TYPE> extends SequentialInstanceConsumer_ImplBase<OUTCOME_TYPE> {
067    
068            /**
069             * "org.cleartk.classifier.SequentialClassifierAnnotator.PARAM_CLASSIFIER_JAR"
070             * is a single, required, string parameter that provides the path to the
071             * jar file that should be used to instantiate the classifier.
072             */
073            public static final String PARAM_CLASSIFIER_JAR = "org.cleartk.classifier.SequentialClassifierAnnotator.PARAM_CLASSIFIER_JAR";
074    
075            @Override
076            public void initialize(UimaContext context) throws ResourceInitializationException {
077                    super.initialize(context);
078                    // get the SequentialClassifier jar file path and load the SequentialClassifier
079                    String jarPath = (String)UIMAUtil.getRequiredConfigParameterValue(
080                                    context, PARAM_CLASSIFIER_JAR);
081                    SequentialClassifier<?> untypedClassifier;
082                    try {
083                            untypedClassifier = ClassifierFactory.createSequentialClassifierFromJar(jarPath);
084                    } catch (IOException e) {
085                            throw new ResourceInitializationException(e);
086                    }
087                    
088                    // check that the SequentialClassifier matches the AnnotationHandler type
089                    this.checkOutcomeType(SequentialClassifier.class, "OUTCOME_TYPE", untypedClassifier);
090                    this.classifier = ReflectionUtil.uncheckedCast(untypedClassifier);
091                    UIMAUtil.initialize(this.classifier, context);
092            }
093    
094            public List<OUTCOME_TYPE> consumeSequence(List<Instance<OUTCOME_TYPE>> instances) throws CleartkException {
095                    List<List<Feature>> instanceFeatures = new ArrayList<List<Feature>>();
096                    for (Instance<OUTCOME_TYPE> instance : instances) {
097                            instanceFeatures.add(instance.getFeatures());
098                    }
099                    return this.classifier.classifySequence(instanceFeatures);
100            }
101    
102            private SequentialClassifier<OUTCOME_TYPE> classifier;
103    
104            public boolean expectsOutcomes() {
105                    return false;
106            }
107    
108    }
