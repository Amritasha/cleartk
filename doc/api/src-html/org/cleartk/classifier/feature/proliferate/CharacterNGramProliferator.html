025    
026    import java.util.Collections;
027    import java.util.List;
028    
029    import org.cleartk.classifier.Feature;
030    
031    
032    /**

040    public class CharacterNGramProliferator extends FeatureProliferator{
041    
042            public static final int RIGHT_TO_LEFT = 0;
043            public static final int LEFT_TO_RIGHT = 1;
044    
045            int orientation;
046            int start;
047            int end;
048            int minimumValueLength;
049            boolean lowerCase;
050            
051    /**
052     * This proliferator serves up character n-grams based on StringValued features. For example, if you wanted trigram 
053     * suffixes (e.g. 'ion' of 'emotion') for words that are of length 7 or more you could call the constructor with the 
054     * following: CharacterNGramProliferator(RIGHT_TO_LEFT, 0, 3, 7, false)
055     * @param featureName a user-specified name for the proliferator, to be included in all feature names.
056     * @param orientation must be one of LEFT_TO_RIGHT or RIGHT_TO_LEFT.  The orientation determines whether index 0 corresponds to 
057     * the first character of the string value or the last.  The orientation does not affect the ordering of the characters in the
058     * n-gram which are always returned in left-to-right order.  
059     * @param start the start of the n-gram (typically 0 for both orientations)
060     * @param end the end of the n-gram (typically n for both orientations)
061     * @param minimumValueLength This parameter allows you to skip string values that are too short.  It must be greater than or equal to end.  
062     * @param lowerCase if true than the n-gram used as the feature value will be lowercased.  
063     */     
064            public CharacterNGramProliferator(String featureName, int orientation, int start, int end, int minimumValueLength, boolean lowerCase) {
065                    super(Feature.createName(
066                                    "NGram",
067                                    orientation == RIGHT_TO_LEFT ? "Right" : "Left",
068                                    String.valueOf(start),
069                                    String.valueOf(end),
070                                    String.valueOf(minimumValueLength),
071                                    lowerCase ? "lower" : null,
072                                    featureName));
073                    if (orientation != RIGHT_TO_LEFT && orientation != LEFT_TO_RIGHT) {
074                            throw new IllegalArgumentException("orientation must be one of CharacterNGramProliferator.RIGHT_TO_LEFT or CharacterNGramProliferator.LEFT_TO_RIGHT ");
075                    }
076                    if (minimumValueLength < end) {
077                            throw new IllegalArgumentException("minimumValueLength must be greater than or equal to the parameter end.");
078                    }
079                    this.orientation = orientation;
080                    this.start = start;
081                    this.end = end;
082                    this.minimumValueLength = minimumValueLength;
083                    this.lowerCase = lowerCase;
084            }
085            
086            public CharacterNGramProliferator(int orientation, int start, int end, int minimumValueLength, boolean lowerCase) {
087                    this(null, orientation, start, end, minimumValueLength, lowerCase);
088            }
089    
090            public CharacterNGramProliferator(String featureName, int orientation, int start, int end) {
091                    this(featureName, orientation, start, end, end - start, false);
092            }
093    
094            public CharacterNGramProliferator(int orientation, int start, int end) {
095                    this(null, orientation, start, end);
096            }
097    
098            /**
099             * @return will return an empty list if the value of the feature is not a StringValue or is
100             * not as long as the minimumValueLength.  
101             * @see CharacterNGramProliferator#CharacterNGramProliferator(int, int, int, int, boolean) 
102             */
103            @Override
104            public List<Feature> proliferate(Feature feature) {
105                    String featureName = Feature.createName(this.getFeatureName(), feature.getName());
106                    Object featureValue = feature.getValue();
107                    if(featureValue == null || !(featureValue instanceof String)) return Collections.emptyList();
108                    
109                    String value = featureValue.toString();
110                    if(value == null || value.length() < minimumValueLength) return Collections.emptyList();
111    
112                    String ngram;
113                    if(orientation == LEFT_TO_RIGHT) {
114                             ngram = value.substring(start, end);
115                    }else {
116                            ngram = value.substring(value.length()-end, value.length()-start);
117                    }
118                    if(lowerCase)
119                            ngram = ngram.toLowerCase();
120                    
121                    return Collections.singletonList(new Feature(featureName, ngram));
122            }
123    
124    
125            public int getEnd() {
126                    return end;
127            }
128    
129    
130            public boolean isLowerCase() {
131                    return lowerCase;
132            }
133    
134    
135            public int getMinimumValueLength() {
136                    return minimumValueLength;
137            }
138    
139    
140            public int getOrientation() {
141                    return orientation;
142            }
143    
144    
145            public int getStart() {
146                    return start;
147            }
148    }
