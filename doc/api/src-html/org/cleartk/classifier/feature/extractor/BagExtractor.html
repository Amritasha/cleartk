024    package org.cleartk.classifier.feature.extractor;
025    
026    import java.util.ArrayList;
027    import java.util.HashSet;
028    import java.util.List;
029    import java.util.Set;
030    
031    import org.apache.uima.jcas.JCas;
032    import org.apache.uima.jcas.tcas.Annotation;
033    import org.cleartk.classifier.Feature;
034    import org.cleartk.util.AnnotationRetrieval;
035    
036    /**
037     * <br>Copyright (c) 2007-2008, Regents of the University of Colorado 
038     * <br>All rights reserved.
039    
040     *
041     * A class for extracting bag-of-words style features. It allows any type of
042     * Annotation to serve as the "words", and allows any type of feature extractor
043     * to be applied to each of these "words".
044     * 
045     * @author Steven Bethard
046     */
047    public class BagExtractor implements SimpleFeatureExtractor {
048    
049            /**
050             * Extracts bag-of-words style features, running a feature extractor over all
051             * "word" Annotations within the focus Annotation.
052             * 
053             * @param extractorName    A name identifying this extractor, which will be prefixed
054             *                         to the context string of all features extracted.
055             * @param annotationClass  The class of Annotation representing "words".
056             * @param featureExtractor The feature extractor to be applied to each "word".
057             */
058            public BagExtractor(
059                            String extractorName,
060                            Class<? extends Annotation> annotationClass,
061                            SimpleFeatureExtractor featureExtractor) {
062                    this.annotationClass = annotationClass;
063                    this.simpleAnnotationName = annotationClass.getSimpleName();
064                    this.featureExtractor = featureExtractor;
065                    this.extractorName = extractorName;
066            }
067    
068            /**
069             * Extracts bag-of-words style features, running a feature extractor over all
070             * "word" Annotations within the focus Annotation.
071             * 
072             * @param annotationClass  The class of Annotation representing "words".
073             * @param featureExtractor The feature extractor to be applied to each "word".
074             */
075            public BagExtractor(
076                            Class<? extends Annotation> annotationClass,
077                            SimpleFeatureExtractor featureExtractor) {
078                    this(null, annotationClass, featureExtractor);
079            }
080    
081            public List<Feature> extract(JCas jCas, Annotation focusAnnotation) {
082                    
083                    // list for collecting features and set of values seen
084                    List<Feature> features = new ArrayList<Feature>();
085                    Set<String> seenValues = new HashSet<String>();
086    
087                    // get the Type of the chosen Annotation
088                    for (Annotation ann: AnnotationRetrieval.getAnnotations(
089                                    jCas, focusAnnotation, this.annotationClass)) {
090                            for (Feature feature: this.featureExtractor.extract(jCas, ann)) {
091                                    String featureValue = feature.getValue().toString();
092                                    if (!seenValues.contains(featureValue)) {
093                                            String featureName = Feature.createName(
094                                                            "BagOf",
095                                                            this.simpleAnnotationName,
096                                                            feature.getName(),
097                                                            this.extractorName);
098                                            features.add(new Feature(featureName, featureValue));
099                                            seenValues.add(featureValue);
100                                    }
101                            }
102                    }
103    
104                    // return the collected features
105                    return features;
106            }
107            private Class<? extends Annotation> annotationClass;
108            private String simpleAnnotationName;
109            private SimpleFeatureExtractor featureExtractor;
110            private String extractorName;
111    
112    }
