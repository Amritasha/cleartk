024    package org.cleartk.classifier.feature.extractor;
025    
026    import java.util.ArrayList;
027    import java.util.List;
028    import java.util.ListIterator;
029    
030    import org.apache.uima.jcas.JCas;
031    import org.cleartk.classifier.Feature;
032    import org.cleartk.syntax.treebank.type.TreebankNode;
033    
034    
035    /**
036     * <br>Copyright (c) 2007-2008, Regents of the University of Colorado 
037     * <br>All rights reserved.
038    
039     *
040     * @author Philipp Wetzler
041     *
042     */
043    public class SyntacticPathExtractor {
044            protected final String UP_SEPARATOR = "::";
045    
046            protected final String DOWN_SEPARATOR = ";;";
047    
048            protected SimpleFeatureExtractor pathMemberExtractor;
049    
050            protected String name;
051    
052            protected boolean isPartial;
053    
054            /**
055             * 
056             * @param name
057             *            this name will be given to the context of the generated
058             *            feature.
059             * @param pathMemberExtractor
060             *            this extractor will be used to get a feature for every node on
061             *            the path, which will then be combined to form a single string.
062             *            The extractor should preferably generate exactly one
063             *            <tt>StringFeature</tt>, but must generate at least one
064             *            <tt>StringFeature</tt>, <tt>LongFeature</tt>,
065             *            <tt>DoubleFeature</tt>, or <tt>BooleanFeature</tt>. Only
066             *            the first feature will then be used and naively converted to a
067             *            string.
068             * @param partial
069             *            if true, generate a partial path only, i.e. from the first
070             *            node up to the lowest common ancestor of the two
071             */
072            public SyntacticPathExtractor(String name,
073                            SimpleFeatureExtractor pathMemberExtractor, boolean partial) {
074                    this.pathMemberExtractor = pathMemberExtractor;
075                    this.name = name;
076                    this.isPartial = partial;
077            }
078    
079            /**
080             * This constructor defaults to a full rather than a partial path.
081             */
082            public SyntacticPathExtractor(String name,
083                            SimpleFeatureExtractor pathMemberExtractor) {
084                    this(name, pathMemberExtractor, false);
085            }
086    
087            /**
088             * This constructor doesn't assign a name to the generated feature.
089             */
090            public SyntacticPathExtractor(SimpleFeatureExtractor pathMemberExtractor,
091                            boolean partial) {
092                    this(null, pathMemberExtractor, partial);
093            }
094    
095            /**
096             * This constructor doesn't assign a name to the generated feature and
097             * default to a full rather than a partial path.
098             */
099            public SyntacticPathExtractor(SimpleFeatureExtractor pathMemberExtractor) {
100                    this(null, pathMemberExtractor, false);
101            }
102    
103            /**
104             * Extract a string representation of a path feature.
105             * 
106             * @param firstAnnotation
107             *            the first node of the path
108             * 
109             * @param lastAnnotation
110             *            the last node of the path
111             * 
112             * @return List of one <em>StringFeature</em>, which contains a string
113             *         representation of the path between the two nodes.
114             * 
115             */
116            public List<Feature> extract(JCas jCas, TreebankNode firstAnnotation,
117                            TreebankNode lastAnnotation) {
118                    List<TreebankNode> fromStart = getPathToRoot(firstAnnotation);
119                    List<TreebankNode> fromEnd = getPathToRoot(lastAnnotation);
120                    String pathFeatureName = null;
121                    String lengthFeatureName = null;
122    
123                    fromEnd.remove(fromEnd.size() - 1);
124                    while (fromStart.size() > 1
125                                    && fromEnd.size() > 0
126                                    && fromStart.get(fromStart.size() - 2) == fromEnd.get(fromEnd
127                                                    .size() - 1)) {
128                            fromStart.remove(fromStart.size() - 1);
129                            fromEnd.remove(fromEnd.size() - 1);
130                    }
131                    
132                    int length = fromStart.size();
133                    if( !isPartial )
134                            length += fromEnd.size();
135    
136                    try {
137                            ListIterator<TreebankNode> it = fromStart.listIterator();
138                            StringBuffer pathBuffer = new StringBuffer();
139                            boolean first = true;
140                            while (it.hasNext()) {
141                                    Feature feature = this.pathMemberExtractor.extract(jCas,
142                                                    it.next()).get(0);
143                                    if (first) {
144                                            String s = feature.getName();
145                                            if( isPartial ) {
146                                                    pathFeatureName = Feature.createName(name, "PartialSyntacticPath", s);
147                                                    lengthFeatureName = Feature.createName(name, "PartialSyntacticPath", "Length");
148                                            } else {
149                                                    pathFeatureName = Feature.createName(name, "SyntacticPath", s);
150                                                    lengthFeatureName = Feature.createName(name, "SyntacticPath", "Length");
151                                            }
152                                            first = false;
153                                    } else {
154                                            pathBuffer.append(this.UP_SEPARATOR);
155                                    }
156                                    pathBuffer.append(feature.getValue().toString());
157                            }
158    
159                            if (!isPartial) {
160                                    it = fromEnd.listIterator(fromEnd.size());
161                                    while (it.hasPrevious()) {
162                                            Feature feature = this.pathMemberExtractor.extract(jCas,
163                                                            it.previous()).get(0);
164                                            pathBuffer.append(this.DOWN_SEPARATOR);
165                                            pathBuffer.append(feature.getValue().toString());
166                                    }
167                            }
168                            
169                            List<Feature> features = new ArrayList<Feature>(2);
170                            features.add(new Feature(pathFeatureName, pathBuffer.toString()));
171                            features.add(new Feature(lengthFeatureName,     (long) length));
172                            
173                            return features;
174    
175                    } catch (IndexOutOfBoundsException e) {
176                            return new ArrayList<Feature>(0);
177                    }
178            }
179    
180            /**
181             * Find the path from a TreebankNode to the root of the tree it belongs to.
182             * 
183             * @param startNode
184             *            The start node of the path
185             * 
186             * @return A list of TreebankNodes that make up the path from <b>startNode</b>
187             *         to the root of the tree
188             */
189            protected static List<TreebankNode> getPathToRoot(TreebankNode startNode) {
190                    List<TreebankNode> nlist = new ArrayList<TreebankNode>(20);
191                    TreebankNode cursorNode = startNode;
192    
193                    while (cursorNode != null) {
194                            nlist.add(cursorNode);
195                            cursorNode = cursorNode.getParent();
196                    }
197    
198                    return nlist;
199            }
200    }
