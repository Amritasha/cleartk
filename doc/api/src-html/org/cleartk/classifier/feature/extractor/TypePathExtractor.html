085            /**
086             * This extractor creates features from attributes of an annotation. For
087             * example, if you had a type called Token with an attribute called 'pos'
088             * you could use this extractor to extract a pos attribute for tokens. This
089             * would be done by calling the constructor with the values "pos", "pos",
090             * false, and false.
091             * 
092             * The value you want may be nested within the type structure of the
093             * annotation that is being examined. For example, if you might have a type
094             * called 'NamedEntity' that has an attribute of type 'ResourceEntry' that
095             * has an attribute of type 'DbRecord' that has a String attribute called
096             * 'identifier'. You may want to extract the value of the identifier as a
097             * feature of the NamedEntity annotation. This could be done by calling the
098             * constructor with the values "identifier",
099             * "resourceEntry/dbRecord/identifier", false, and false (or something
100             * similar).
101             * 
102             * The value you want for your featured may be multi-valued
103             * 
104             * @param focusClass
105             *            the type of annotation that you are doing feature extraction
106             *            on.
107             * @param typePath
108             *            a string representation of the path that should be traversed
109             *            to extract a feature value (e.g.
110             *            "resourceEntry/dbRecord/identifier" or "pos" from the examples
111             *            above.)
112             * @param traverseAllPaths
113             *            The path you traverse to the value you are trying to retrieve
114             *            may include attributes that have multiple-values. If true,
115             *            then all paths are examined and features for each traversal
116             *            are added if possible. If false, then the first path that
117             *            results in a non-null value will be examined.
118             * @param returnAllValues
119             *            The last node of the path may be multi-valued. If true, then
120             *            all values found in the last node will be returned as
121             *            features. If false, then only the first value of the last node
122             *            is returned. If traverseAllPaths and returnAllValues are both
123             *            false, then a list of size 0 or 1 will be returned. The other
124             *            three combinations are each valid and may return a list of
125             *            size 0 or greater.
126             * @param uniqueValues
127             *            if true, then the returned values of the extract method will
128             *            be unique.
129             */
130    
131            public TypePathExtractor(Class<? extends Annotation> focusClass, String typePath, boolean traverseAllPaths,
132                            boolean returnAllValues, boolean uniqueValues) {
133                    this.focusClass = focusClass;
134                    this.path = typePath;
135                    this.allPaths = traverseAllPaths;
136                    this.allValues = returnAllValues;
137                    this.uniqueValues = uniqueValues;
138            }

142                    this(focusClass, typePath, traverseAllPaths, returnAllValues, uniqueValues);
143                    this.name = name;
144            }

159            public List<org.cleartk.classifier.Feature> extract(JCas jCas, Annotation focusAnnotation) {
160                    if (this.type == null) this.type = UIMAUtil.getCasType(jCas, this.focusClass);
161    
162                    this.typeSystem = jCas.getTypeSystem();
163    
164                    if (!isValidPath(jCas)) throw new IllegalArgumentException("The path " + path + " is not valid for the type "
165                                    + type.getName());
166    
167                    String[] pathMembers = path.split("/");
168                    List<Object> pathValues = new ArrayList<Object>();
169                    _extract(jCas, focusAnnotation, pathMembers, pathValues);
170    
171                    List<org.cleartk.classifier.Feature> returnValues = new ArrayList<org.cleartk.classifier.Feature>();
172                    Set<Object> values = new HashSet<Object>();
173                    for (Object pathValue : pathValues) {
174                            if (uniqueValues) {
175                                    if (!values.contains(pathValue)) {
176                                            org.cleartk.classifier.Feature feature = new org.cleartk.classifier.feature.TypePathFeature(
177                                                            this.name, pathValue, this.path);
178                                            returnValues.add(feature);
179                                            values.add(pathValue);
180                                    }
181                            }
182                            else {
183                                    org.cleartk.classifier.Feature feature = new org.cleartk.classifier.feature.TypePathFeature(
184                                                    this.name, pathValue, this.path);
185                                    returnValues.add(feature);
186                            }
187                    }
188    
189                    return returnValues;
190            }

192            private void _extract(JCas jCas, FeatureStructure featureStructure, String[] pathMembers, List<Object> pathValues) {
193                    if (pathMembers.length == 1) {
194                            Feature feature = featureStructure.getType().getFeatureByBaseName(pathMembers[0]);
195                            if (feature == null) {
196                                    return;
197                            }
198                            Type type = feature.getRange();
199                            if (type.isPrimitive()) {
200                                    Object pathValue = getPrimitiveFeatureValue(jCas, featureStructure, feature);
201                                    if (pathValue != null) pathValues.add(pathValue);
202                            }
203                            else if (typeSystem.subsumes(typeSystem.getType("uima.tcas.Annotation"), type)) {
204                                    String coveredText = ((Annotation) featureStructure.getFeatureValue(feature)).getCoveredText();
205                                    if (coveredText != null) pathValues.add(coveredText);
206                            }
207                            else if (type.isArray()) {
208                                    Type componentType = type.getComponentType();
209                                    if (componentType.isPrimitive()) {
210                                            Object[] values = getPrimitiveArrayFeatureValue(jCas, featureStructure, feature);
211                                            if (allValues) pathValues.addAll(Arrays.asList(values));
212                                            else if (values.length > 0) pathValues.add(values[0]);
213                                    }
214                                    else if (typeSystem.subsumes(typeSystem.getType("uima.tcas.Annotation"), componentType)) {
215                                            ArrayFS fsArray = (ArrayFS) featureStructure.getFeatureValue(feature);
216                                            FeatureStructure[] array = fsArray.toArray();
217                                            if (allValues) {
218                                                    for (FeatureStructure ftr : array)
219                                                            pathValues.add(((Annotation) ftr).getCoveredText());
220                                            }
221                                            else {
222                                                    if (array.length > 0) pathValues.add(((Annotation) array[0]).getCoveredText());
223                                            }
224                                    }
225                            }
226                            // TODO else if type is a List type
227                    }
228                    else {
229                            String[] remainingPathMembers = new String[pathMembers.length - 1];
230                            System.arraycopy(pathMembers, 1, remainingPathMembers, 0, pathMembers.length - 1);
231    
232                            Feature feature = featureStructure.getType().getFeatureByBaseName(pathMembers[0]);
233                            FeatureStructure featureValue = featureStructure.getFeatureValue(feature);
234                            if (featureValue == null) return;
235                            if (featureValue instanceof FSArray) {
236                                    FSArray fsArray = (FSArray) featureValue;
237                                    if (allPaths) {
238                                            for (int i = 0; i < fsArray.size(); i++) {
239                                                    FeatureStructure fs = fsArray.get(i);
240                                                    _extract(jCas, fs, remainingPathMembers, pathValues);
241                                            }
242                                    }
243                                    else {
244                                            if (fsArray.size() > 0) _extract(jCas, fsArray.get(0), remainingPathMembers, pathValues);
245                                    }
246                            }
247                            // TODO else if(featureValue instanceof FSList)
248                            else {
249                                    _extract(jCas, featureValue, remainingPathMembers, pathValues);
250                            }
251                    }
252            }

254            private boolean isValidPath(JCas jCas) {
255                    if (!pathChecked) {
256                            boolean validPath = isValidPath(type, path, jCas);
257                            if (validPath) pathChecked = true;
258                            return validPath;
259                    }
260                    else return true;
261            }

263            // TODO should be possible to just get the Feature from the type system and
264            // return
265            // true if it is not null.
266            public static boolean isValidPath(Type type, String path, JCas jCas) {
267                    String[] pathMembers = path.split("/");
268                    Type pathMemberType = type; // will be set to type of last path member
269                    // feature type
270                    for (String pathMember : pathMembers) {
271                            Feature feature = pathMemberType.getFeatureByBaseName(pathMember);
272                            if (feature == null) {
273                                    return false;
274                            }
275                            pathMemberType = feature.getRange();
276                            if (pathMemberType.isArray()) pathMemberType = pathMemberType.getComponentType();
277                    }
278                    return isValidType(pathMemberType, jCas.getTypeSystem());
279            }

281            private static final String[] HANDLED_TYPES = new String[] { "uima.cas.Boolean", "uima.cas.BooleanArray",
282                            "uima.cas.Byte", "uima.cas.ByteArray", "uima.cas.Double", "uima.cas.DoubleArray", "uima.cas.Float",
283                            "uima.cas.FloatArray", "uima.cas.FloatList", "uima.cas.Integer", "uima.cas.IntegerArray",
284                            "uima.cas.IntegerList", "uima.cas.Long", "uima.cas.LongArray", "uima.cas.Short", "uima.cas.ShortArray",
285                            "uima.cas.String", "uima.cas.StringArray", "uima.cas.StringList", "uima.tcas.Annotation" };
286    
287            public static boolean isValidType(Type type, TypeSystem typeSystem) {
288                    String typeName = type.getName();
289                    for (String handledType : HANDLED_TYPES) {
290                            if (typeName.equals(handledType)) return true;
291                    }
292    
293                    // see section 2.3.4 of UIMA References
294                    if (typeSystem.subsumes(typeSystem.getType("uima.cas.String"), type)) return true;
295                    if (typeSystem.subsumes(typeSystem.getType("uima.tcas.Annotation"), type)) return true;
296    
297                    return false;
298            }

300            /**
301             * see section 4.2.1 of the UIMA References documentation.
302             * 
303             * @param jCas
304             * @param featureStructure
305             * @param feature
306             * @return The feature value.
307             */
308            private static Object getPrimitiveFeatureValue(JCas jCas, FeatureStructure featureStructure, Feature feature) {
309                    TypeSystem typeSystem = jCas.getTypeSystem();
310                    Type type = feature.getRange();
311                    if (type.equals(typeSystem.getType("uima.cas.Boolean"))) return featureStructure.getBooleanValue(feature);
312                    else if (type.equals(typeSystem.getType("uima.cas.Double"))) return featureStructure.getDoubleValue(feature);
313                    else if (type.equals(typeSystem.getType("uima.cas.Float"))) return featureStructure.getFloatValue(feature);
314                    else if (type.equals(typeSystem.getType("uima.cas.Byte"))) return featureStructure.getByteValue(feature);
315                    else if (type.equals(typeSystem.getType("uima.cas.Short"))) return featureStructure.getShortValue(feature);
316                    else if (type.equals(typeSystem.getType("uima.cas.Integer"))) return featureStructure.getIntValue(feature);
317                    else if (type.equals(typeSystem.getType("uima.cas.Long"))) return featureStructure.getLongValue(feature);
318                    else if (type.equals(typeSystem.getType("uima.cas.String"))) return featureStructure.getStringValue(feature);
319                    else throw new IllegalArgumentException("The feature of the featureStructure is not primitive.  "
320                                    + "Feature domain type = " + feature.getDomain() + ".  Feature range type = " + feature.getRange());
321            }
322    
323            private static Object[] getPrimitiveArrayFeatureValue(JCas jCas, FeatureStructure featureStructure, Feature feature) {
324                    TypeSystem typeSystem = jCas.getTypeSystem();
325                    Type type = feature.getRange();
326                    if (type.isArray()) {
327                            Type componentType = type.getComponentType();
328                            FeatureStructure featureValue = featureStructure.getFeatureValue(feature);
329                            if (componentType.equals(typeSystem.getType("uima.cas.String"))) {
330                                    return ((StringArray)featureValue).toArray();
331                            } else if (componentType.equals(typeSystem.getType("uima.cas.Boolean"))) {
332                                    return Arrays.asList(((BooleanArray)featureValue).toArray()).toArray();
333                            } else if (componentType.equals(typeSystem.getType("uima.cas.Double"))) {
334                                    return Arrays.asList(((DoubleArray)featureValue).toArray()).toArray();
335                            } else if (componentType.equals(typeSystem.getType("uima.cas.Float"))) {
336                                    return Arrays.asList(((FloatArray)featureValue).toArray()).toArray();
337                            } else if (componentType.equals(typeSystem.getType("uima.cas.Byte"))) {
338                                    return Arrays.asList(((ByteArray)featureValue).toArray()).toArray();
339                            } else if (componentType.equals(typeSystem.getType("uima.cas.Short"))) {
340                                    return Arrays.asList(((ShortArray)featureValue).toArray()).toArray();
341                            } else if (componentType.equals(typeSystem.getType("uima.cas.Integer"))) {
342                                    return Arrays.asList(((IntegerArray)featureValue).toArray()).toArray();
343                            } else if (componentType.equals(typeSystem.getType("uima.cas.Long"))) {
344                                    return Arrays.asList(((LongArray)featureValue).toArray()).toArray();
345                            }
346                    }
347                    else throw new IllegalArgumentException("The feature of the featureStructure is not a primitive array.  "
348                                    + "Feature domain type = " + feature.getDomain() + ".  Feature range type = " + feature.getRange());
349                    return null;
350            }
