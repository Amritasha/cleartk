044    public class NGramExtractor implements SimpleFeatureExtractor {
045            
046            public NGramExtractor(
047                            String name,
048                            int n,
049                            Class<? extends Annotation> annotationClass, 
050                            SimpleFeatureExtractor subExtractor) {
051                    this.name = name;
052                    this.n = n;
053                    this.subExtractor = subExtractor;
054                    this.annotationClass = annotationClass;
055            }
056    
057            public NGramExtractor(
058                            int n,
059                            Class<? extends Annotation> annotationClass, 
060                            SimpleFeatureExtractor subExtractor) {
061                    this(null, n, annotationClass, subExtractor);
062            }
063    
064            public List<Feature> extract(JCas jCas, Annotation focusAnnotation)
065                            throws UnsupportedOperationException {
066                    List<? extends Annotation> annotations;
067                    
068                    List<Feature> ngramFeatures = new ArrayList<Feature>();
069    
070                    annotations = AnnotationRetrieval.getAnnotations(jCas, focusAnnotation, annotationClass);
071    
072                    Map<Annotation, List<Feature>> annotationFeatures = new HashMap<Annotation, List<Feature>>();
073                    for( Annotation annotation : annotations ) {
074                            List<Feature> features = this.subExtractor.extract(jCas, annotation);
075                            annotationFeatures.put(annotation, features);
076                    }
077                    
078                    for( int i = 0; i < annotations.size() - n + 1; i++ ) {
079                            List<Annotation> slice = new ArrayList<Annotation>(n);
080                            for( int j = 0; j < n; j++ ) {
081                                    slice.add(annotations.get(i+j));
082                            }
083                            
084                            
085                            List<String> values = new ArrayList<String>(n);
086                            List<String> names = new ArrayList<String>(n + 1);
087    
088                            names.add(this.name);
089                            names.add("Ngram");
090                            for( Annotation annotation : slice ) {
091                                    for( Feature feature : annotationFeatures.get(annotation) ) {
092                                            values.add(feature.getValue().toString());
093                                            names.add(feature.getName());
094                                    }
095                            }
096                            
097                            // create name
098                            String[] nameArray = new String[names.size()];
099                            String name = Feature.createName(names.toArray(nameArray));
100    
101                            // create value
102                            StringBuffer valueBuffer = new StringBuffer();
103                            for( String subValue : values ) {
104                                    valueBuffer.append(subValue);
105                                    valueBuffer.append(valueSeparator);
106                            }
107                            if( valueBuffer.length() > 0 )
108                                    valueBuffer.deleteCharAt(valueBuffer.length()-1);
109                            String value = valueBuffer.toString();
110                            
111                            // add feature
112                            ngramFeatures.add(new Feature(name, value));
113                    }
114                    
115                    return ngramFeatures;
116            }
117            
118            private String name;
119            private SimpleFeatureExtractor subExtractor;
120            private int n;
121            Class<? extends Annotation> annotationClass;
122    
123            private String valueSeparator = "|";
124    
125            public String getValueSeparator() {
126                    return valueSeparator;
127            }
128    
129            public void setValueSeparator(String valueSeparator) {
130                    this.valueSeparator = valueSeparator;
131            }
132            
133    }
