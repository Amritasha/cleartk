024    package org.cleartk.classifier.feature.extractor;
025    
026    import java.util.ArrayList;
027    import java.util.List;
028    
029    import org.apache.uima.jcas.JCas;
030    import org.apache.uima.jcas.tcas.Annotation;
031    import org.cleartk.classifier.Feature;
032    
033    
034    /**
035     * <br>Copyright (c) 2007-2008, Regents of the University of Colorado 
036     * <br>All rights reserved.
037    
038     * <p>
039     * 
040     * @author Philipp Wetzler
041     * 
042     */
043    public class CombinedExtractor implements SimpleFeatureExtractor {
044    
045            private SimpleFeatureExtractor[] extractors;
046    
047            private String name;
048    
049            public SimpleFeatureExtractor[] getExtractors() {
050                    return extractors;
051            }
052    
053            public String getName() {
054                    return name;
055            }
056    
057            /**
058             * @param name
059             *            the name to give to the top level context of all generated
060             *            features.
061             * @param extractors
062             *            the array of sub-extractors to be called by this one.
063             */
064            public CombinedExtractor(String name, SimpleFeatureExtractor ... extractors) {
065                    this(extractors);
066                    this.name = name;
067            }
068    
069            /**
070             * This constructor doesn't set the name. Consequently, the extract method
071             * will not change the context name of features generated by the
072             * sub-extractors.
073             * 
074             * @param extractors
075             *            the array of sub-extractors to be called by this one.
076             */
077            public CombinedExtractor(SimpleFeatureExtractor ... extractors) {
078                    this.extractors = extractors;
079            }
080    
081            /**
082             * Extract features from the <tt>Annotation</tt> using the sub-extractors.
083             * The parameters are passed on as they are.
084             * 
085             * @param jCas
086             * @param focusAnnotation
087             * 
088             * @return the combined list of features generated by the sub-extractors. If
089             *         <tt>name</tt> was set in the constructor, the top-level context
090             *         of all features will have that as their name.
091             */
092            public List<Feature> extract(JCas jCas, Annotation focusAnnotation) throws UnsupportedOperationException {
093                    List<Feature> result = new ArrayList<Feature>();
094                    for (SimpleFeatureExtractor extractor : this.extractors) {
095                            result.addAll(extractor.extract(jCas, focusAnnotation));
096                    }
097    
098                    if (name != null) {
099                            for (Feature feature : result) {
100                                    feature.setName(this.name);
101                            }
102                    }
103                    return result;
104            }
105    
106    }
