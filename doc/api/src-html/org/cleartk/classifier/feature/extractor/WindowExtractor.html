051    public class WindowExtractor {
052    
053            Class<? extends Annotation> featureClass;
054    
055            Type featureType;
056    
057            SimpleFeatureExtractor featureExtractor;
058    
059            String windowOrientation;
060    
061            int windowStart;
062    
063            int windowEnd;
064    
065            protected String name;
066    
067            public WindowExtractor(Class<? extends Annotation> featureClass, SimpleFeatureExtractor featureExtractor,
068                            String windowOrientation, int windowStart, int windowEnd) {
069                    this.featureClass = featureClass;
070                    this.featureExtractor = featureExtractor;
071                    this.windowOrientation = windowOrientation;
072                    this.windowStart = windowStart;
073                    this.windowEnd = windowEnd;
074            }
075    
076            /**
077             * This feature extractor provides a mechanism for getting features such as:
078             * <ul>
079             * <li>the three words to the left or right of a token</li>
080             * <li>get the three words to the left and right of a constituent bounded
081             * by a sentence.</li>
082             * <li>get the words "inside" the constituent</li>
083             * <li>the three pos tags of the three words to the left of a token</li>
084             * <li>the concept ids of the three normalized named entities to the left
085             * of a consituent inside a paragraph/section annotation.</li>
086             * </ul>
087             * 
088             * An example call to this constructor might look something like this: <br>
089             * <code>
090             * <pre>
091             *       new WindowExtractor(Token.class, 
092             *                                                                               new SpannedTextExtractor(),
093             *                                                                               WindowExtractor.ORIENTATION_LEFT,
094             *                                                                               0,3)
095             * 
096             * &#064;param name the name of the extractor
097             * &#064;param featureClass the type of the annotations that we are going to extract a feature from
098             * &#064;param featureExtractor a feature extractor that will extract features from the annotations of type featureType
099             * &#064;param windowOrientation one of ORIENTATION_LEFT, ORIENTATION_RIGHT, ORIENTATION_MIDDLE, ORIENTATION_MIDDLE_REVERSE
100             * &#064;param windowStart the index (inclusive) of the first annotation of type featureType to examine
101             * &#064;param windowEnd the index (exclusive) of the last annotation to examine. 
102             * 
103             * &#064;see WindowFeature
104             * 
105             */
106            public WindowExtractor(String name, Class<? extends Annotation> featureClass,
107                            SimpleFeatureExtractor featureExtractor, String windowOrientation, int windowStart, int windowEnd) {
108                    this(featureClass, featureExtractor, windowOrientation, windowStart, windowEnd);
109                    this.name = name;
110            }
111    
112            public List<Feature> extract(JCas jCas, Annotation focusAnnotation, Class<? extends Annotation> cls) {
113                    Annotation windowAnnotation = AnnotationRetrieval.getContainingAnnotation(jCas, focusAnnotation, cls);
114                    return extract(jCas, focusAnnotation, windowAnnotation);
115            }
116    
117            public List<Feature> extract(JCas jCas, Annotation focusAnnotation, Annotation windowAnnotation) {
118                    if (this.featureType == null) this.featureType = UIMAUtil.getCasType(jCas, this.featureClass);
119    
120                    List<Feature> returnValues = new ArrayList<Feature>();
121    
122                    Annotation startAnnotation = getStartAnnotation(jCas, focusAnnotation);
123    
124                    FSIterator featureAnnotationIterator = jCas.getAnnotationIndex(featureType).iterator();
125    
126                    if (startAnnotation != null) {
127                            featureAnnotationIterator.moveTo(startAnnotation);
128                    }
129    
130                    int outOfBoundsDistance = 0;
131    
132                    for (int i = 0; i < windowEnd; i++) {
133                            // if start annotation is null, then we will mistakenly create a
134                            // featureAnnotation for whatever
135                            // annotation the iterator starts at. Therefore, we check for null
136                            // here.
137                            if (outOfBoundsDistance == 0 && startAnnotation != null && featureAnnotationIterator.isValid()) {
138                                    Annotation featureAnnotation = (Annotation) featureAnnotationIterator.get();
139                                    if (isWithinBoundaries(featureAnnotation, focusAnnotation, windowAnnotation)) {
140                                            if (i >= windowStart) {
141                                                    returnValues.addAll(extractWindowedFeatures(jCas, i, featureAnnotation));
142                                            }
143                                            moveIterator(featureAnnotationIterator);
144                                            continue;
145                                    }
146                            }
147                            outOfBoundsDistance++;
148                            if (i >= windowStart) {
149                                    Feature feature = new WindowFeature(this.name, null, windowOrientation, i, outOfBoundsDistance);
150                                    returnValues.add(feature);
151                            }
152    
153                    }
154                    return returnValues;
155            }
156    
157            private boolean isWithinBoundaries(Annotation featureAnnotation, Annotation focusAnnotation,
158                            Annotation windowAnnotation) {
159                    if (windowOrientation.equals(WindowFeature.ORIENTATION_LEFT)
160                                    || windowOrientation.equals(WindowFeature.ORIENTATION_RIGHT)) {
161                            return AnnotationUtil.contains(windowAnnotation, featureAnnotation);
162                    }
163                    else return AnnotationUtil.contains(windowAnnotation, featureAnnotation)
164                                    && AnnotationUtil.contains(focusAnnotation, featureAnnotation);
165            }
166    
167            private void moveIterator(FSIterator windowIterator) {
168                    if (windowOrientation.equals(WindowFeature.ORIENTATION_LEFT)
169                                    || windowOrientation.equals(WindowFeature.ORIENTATION_MIDDLE_REVERSE)) windowIterator.moveToPrevious();
170                    else if (windowOrientation.equals(WindowFeature.ORIENTATION_RIGHT)
171                                    || windowOrientation.equals(WindowFeature.ORIENTATION_MIDDLE)) windowIterator.moveToNext();
172            }
173    
174            private List<Feature> extractWindowedFeatures(JCas jCas, int i, Annotation annotation) {
175                    List<Feature> windowedFeatures = featureExtractor.extract(jCas, annotation);
176                    List<Feature> returnValues = new ArrayList<Feature>();
177                    if (windowedFeatures != null && windowedFeatures.size() > 0) {
178                            for (Feature windowedFeature : windowedFeatures) {
179                                    Feature feature = new WindowFeature(this.name, windowedFeature.getValue(), windowOrientation, i,
180                                                    windowedFeature, 0);
181                                    returnValues.add(feature);
182                            }
183                            return returnValues;
184                    }
185    
186                    Feature feature = new WindowFeature(this.name, null, windowOrientation, i, 0);
187                    return Collections.singletonList(feature);
188            }
189    
190            /**
191             * This method returns the annotation at index 0 w.r.t. the orientation of
192             * the extractor, the annotation we are extracting features for and the
193             * window the provides the boundaries for feature extraction.
194             * 
195             * @param annotation
196             *            the annotation of focus for which we are extracting features
197             *            for
198             * @return the annotation at index 0:
199             *         <ul>
200             *         <li>for ORIENTATION_LEFT return the annotation adjacent and to
201             *         the left of annotation
202             *         <li>for ORIENTATION_RIGHT return the annotation adjacent and to
203             *         the right of annotation
204             *         <li>for ORIENTATION_MIDDLE return the left most annotation
205             *         completely contained in annotation, null if does not exist
206             *         <li>for ORIENTATION_MIDDLE_REVERSE return the right most
207             *         annotation completely contained in annotation, null if does not
208             *         exist
209             *         </ul>
210             */
211            public Annotation getStartAnnotation(JCas jCas, Annotation annotation) {
212                    if (windowOrientation.equals(WindowFeature.ORIENTATION_LEFT)) {
213                            return AnnotationRetrieval.getAdjacentAnnotation(jCas, annotation, featureClass, true);
214                    }
215                    else if (windowOrientation.equals(WindowFeature.ORIENTATION_RIGHT)) {
216                            return AnnotationRetrieval.getAdjacentAnnotation(jCas, annotation, featureClass, false);
217                    }
218                    else if (windowOrientation.equals(WindowFeature.ORIENTATION_MIDDLE)) {
219                            return AnnotationRetrieval.getFirstAnnotation(jCas, annotation, featureClass);
220                    }
221                    else if (windowOrientation.equals(WindowFeature.ORIENTATION_MIDDLE_REVERSE)) {
222                            return AnnotationRetrieval.getLastAnnotation(jCas, annotation, featureClass);
223                    }
224                    return null;
225            }

228                    return featureClass;
229            }

244                    return windowOrientation;
245            }

248                    return windowStart;
249            }
