061     * This collection reader reads in the CoNLL 2003 named entity data.  The 
062     * data can be retrieved from http://www.cnts.ua.ac.be/conll2003/ner/
063     * 
064     */
065    public class Conll2003GoldReader extends CollectionReader_ImplBase
066    {
067            /**
068             * "org.cleartk.corpus.conll2003.Conll2003GoldReader.PARAM_CONLL_2003_DATA_FILE"
069             * is a single, required string parameter that points to CoNLL data
070             * (e.g. ner/eng.train).
071             */
072            public static final String PARAM_CONLL_2003_DATA_FILE = "org.cleartk.corpus.conll2003.Conll2003GoldReader.PARAM_CONLL_2003_DATA_FILE";
073    
074            /**
075             * "org.cleartk.corpus.conll2003.Conll2003GoldReader.PARAM_LOAD_NAMED_ENTITIES"
076             * is a single, required boolean parameter that determines if the named entities
077             * are loaded (i.e. named entity mention annotations are created) or if just the
078             * plain text from the files is loaded.
079             */
080            public static final String PARAM_LOAD_NAMED_ENTITIES = "org.cleartk.corpus.conll2003.Conll2003GoldReader.PARAM_LOAD_NAMED_ENTITIES";
081            public static final String DOCSTART = "-DOCSTART-";
082                                                                                      
083            BufferedReader reader;
084            
085            boolean hasNext = true;
086            
087            int documentIndex = 0;
088            String documentPath;
089            
090            boolean loadNamedEntities = true;
091            int entityIdIndex = 0;
092            
093            public void initialize() throws ResourceInitializationException
094            {
095                    try
096                    {
097                            documentPath = (String) UIMAUtil.getRequiredConfigParameterValue(getUimaContext(), PARAM_CONLL_2003_DATA_FILE);
098                            loadNamedEntities = (Boolean) UIMAUtil.getDefaultingConfigParameterValue(getUimaContext(), PARAM_LOAD_NAMED_ENTITIES, true);
099                            
100                            File conllFile = new File(documentPath);
101                        reader = new BufferedReader(new FileReader(conllFile));
102                            //advance the reader past the first occurrence of a document start and
103                        //blank line.
104                        String line;
105                        while((line = reader.readLine()) != null)
106                            {
107                                    if(line.trim().startsWith(DOCSTART))
108                                    {
109                                            reader.readLine(); //read the blank line
110                                            break;
111                                    }
112                            }
113    
114                        sentenceTokens = new ArrayList<Token>();
115                            sentenceChunks = new ArrayList<Chunk>();
116                            chunkTokens = new ArrayList<Token>();
117                            namedEntityTokens = new ArrayList<Token>();
118                            
119                            super.initialize();
120                    }
121                    catch(FileNotFoundException fnfe)
122                    {
123                            throw new ResourceInitializationException(fnfe);
124                    }
125                    catch(IOException ioe)
126                    {
127                            throw new ResourceInitializationException(ioe);
128                    }
129            }
130    
131            List<String> documentData;  //contains every line from for the current document
132            StringBuffer documentText;  //collects the text of the tokens and then we set the document text with contents
133            int sentenceStart;          //the start (character offset) of the current sentence
134            List<Token> sentenceTokens; //the tokens for the current sentence
135            List<Chunk> sentenceChunks; //the chunks for the current sentence
136            
137            int tokenPosition;             //the index of the current token
138    
139            int chunkStart;             //the start (char offset) of the current chunk
140            String currentChunkType;    //the type of the current chunk (including the I- or B- prefix)
141            List<Token> chunkTokens;    //the tokens for the current chunk
142            
143            int namedEntityStart;       //the start (char offset) of the current named entity
144            String currentNamedEntityType; //the type of the current named entity (including the I- or B- prefix)
145            List<Token> namedEntityTokens; //the tokens for the current named entity
146            
147            public void getNext(CAS cas) throws IOException, CollectionException
148            {
149                    try
150                    {
151                            JCas jCas = cas.getJCas();
152                            
153                            //read in the data for the next document from the reader
154                            documentData = new ArrayList<String>();
155                            String line;
156                            while((line = reader.readLine()) != null &&
157                                       !line.startsWith(DOCSTART))
158                            {
159                                    documentData.add(line.trim());
160                            }
161    
162                            if(line == null)
163                                    hasNext = false;
164                            else
165                                    line = reader.readLine().trim(); //advance past the blank line that follows "-DOCSTART- -X- O O" 
166                                                                                                     //(we don't want an empty sentence on the front end!)
167                            documentText = new StringBuffer();
168                            
169                            initSentence();
170                            tokenPosition = 0;
171                            chunkStart = 0;
172                            currentChunkType = "";
173                            chunkTokens.clear();
174                            namedEntityStart = 0;
175                            currentNamedEntityType = "";
176                            namedEntityTokens.clear();
177                            
178                            for(String dataLine : documentData)
179                            {
180                                    if(dataLine.trim().equals(""))
181                                    {
182                                            createChunk(jCas);
183                                            currentChunkType = "";
184                                            createNamedEntity(jCas);
185                                            currentNamedEntityType = "";
186                                            
187                                            Sentence sentence = new Sentence(jCas, sentenceStart, documentText.length());
188                                            sentence.addToIndexes();
189                                            
190                                            initSentence();
191                                    }
192                                    else
193                                    {
194                                            String[] dataPieces = dataLine.split(" ");
195                                            String tok = dataPieces[0];
196                                            String pos = dataPieces[1];
197                                            String chunkType = dataPieces[2];
198                                            if(currentChunkType.equals(""))
199                                                    initChunk(chunkType);
200                                            String namedEntityType = dataPieces[3];
201                                            if(currentNamedEntityType.equals(""))
202                                                    initNamedEntity(namedEntityType);
203                                            
204                                            
205                                            Token token = new Token(jCas, documentText.length(), documentText.length()+tok.length());
206                                            token.setPos(pos);
207                                            token.addToIndexes();
208                                            
209                                            boolean chunkStartsWithB = startsWithB(currentChunkType, chunkType);
210                                            if(!chunkType.equals(currentChunkType) && !chunkStartsWithB){ 
211                                                    createChunk(jCas);
212                                                    initChunk(chunkType);
213                                            }
214                                            
215                                            boolean namedEntityStartsWithB = startsWithB(currentNamedEntityType, namedEntityType);
216                                            
217                                            if(!namedEntityType.equals(currentNamedEntityType) && !namedEntityStartsWithB)
218                                            {
219                                                    createNamedEntity(jCas);
220                                                    initNamedEntity(namedEntityType);
221                                            }
222    
223                                            sentenceTokens.add(token);
224                                            chunkTokens.add(token);
225                                            namedEntityTokens.add(token);
226                                            documentText.append(tok+" ");
227                                    }
228                            }
229                            
230                            jCas.setDocumentText(documentText.toString());
231                            
232                            String identifier = String.format("%s#%s", documentPath, documentIndex);
233                            ViewURIUtil.setURI(cas, identifier);
234                            ++documentIndex;
235    
236                    }
237                    catch(CASException ce)
238                    {
239                            throw new CollectionException(ce);
240                    }
241    
242            }
243    
244            private void initSentence()
245            {
246                    sentenceStart = documentText.length();
247                    sentenceTokens.clear();
248                    sentenceChunks.clear();
249            }
250    
251            private void createChunk(JCas jCas)
252            {
253                    if(!currentChunkType.equals("O"))
254                    {
255                            Chunk chunk = new Chunk(jCas, chunkStart, documentText.length()-1);
256                            chunk.setChunkType(currentChunkType.substring(2));
257                            chunk.setTokens(UIMAUtil.toFSArray(jCas, chunkTokens));
258                            chunk.addToIndexes();
259                            sentenceChunks.add(chunk);
260                    }
261            }
262            
263            private void initChunk(String chunkType)
264            {
265                    chunkStart = documentText.length();
266                    chunkTokens.clear();
267                    currentChunkType = chunkType;
268            }
269            
270            private void createNamedEntity(JCas jCas)
271            {
272                    if(!currentNamedEntityType.equals("O") && loadNamedEntities)
273                    {
274                            NamedEntity ne = new NamedEntity(jCas);
275                            ne.setEntityClass("SPC");
276                            ne.setEntityId(""+entityIdIndex++);
277                            ne.setEntityType(currentNamedEntityType.substring(2));
278                            ne.setEntitySubtype(currentNamedEntityType.substring(2));
279                            ne.addToIndexes();
280                            
281                            NamedEntityMention nem = new NamedEntityMention(jCas, namedEntityStart, documentText.length()-1);
282                            nem.setMentionType("NAM");
283                            SimpleAnnotation annotation = new SimpleAnnotation(jCas, namedEntityStart, documentText.length()-1);
284                            annotation.addToIndexes();
285    //                      Chunk chunk = new Chunk(jCas, namedEntityStart, documentText.length()-1);
286    //                      chunk.setTokens(UIMAUtil.toFSArray(jCas, namedEntityTokens));
287    //                      chunk.setChunkType("CoNLL NEM annotation");
288    //                      chunk.addToIndexes();
289                            nem.setAnnotation(annotation);
290                            nem.setHead(annotation);
291                            nem.setMentionedEntity(ne);
292                            nem.addToIndexes();
293                            
294                            ne.setMentions(UIMAUtil.toFSArray(jCas, Collections.singletonList(nem)));
295                    }
296            }
297            
298            private void initNamedEntity(String namedEntityType)
299            {
300                    namedEntityStart = documentText.length();
301                    namedEntityTokens.clear();
302                    currentNamedEntityType = namedEntityType;
303            }
304    
305            /**
306             * Determines if the read type is the same as the current type with the only
307             * difference being that the current type (bType) starts with "B-" and the
308             * read type starts with "I-".  
309             * @param bType - the current type for the chunk or named entity
310             * @param iType - the read chunk or named entity type for the token being examined.
311             * @return true if we should consder iType to be the same as bType so we know
312             * not to make a new chunk or named entity.
313             */
314            private boolean startsWithB(String bType, String iType)
315            {
316                    if(bType.startsWith("B") && 
317                       iType.startsWith("I") &&
318                       iType.substring(1).equals(bType.substring(1)))
319                    {
320                            return true;
321                    }
322                    return false;
323            }
324            
325            public void close() throws IOException
326            {
327                    reader.close();
328            }
329    
330            public Progress[] getProgress()
331            {
332                return new Progress[] { new ProgressImpl(documentIndex, 5000, Progress.ENTITIES) };
333            }
334    
335            public boolean hasNext() throws IOException, CollectionException
336            {
337                    return hasNext;
338            }
