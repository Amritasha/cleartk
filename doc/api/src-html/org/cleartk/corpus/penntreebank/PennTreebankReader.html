024    package org.cleartk.corpus.penntreebank;
025    
026    import java.io.File;
027    import java.io.IOException;
028    import java.util.Collections;
029    import java.util.LinkedList;
030    import java.util.List;
031    
032    import org.apache.uima.cas.CAS;
033    import org.apache.uima.cas.CASException;
034    import org.apache.uima.collection.CollectionException;
035    import org.apache.uima.collection.CollectionReader_ImplBase;
036    import org.apache.uima.jcas.JCas;
037    import org.apache.uima.resource.ResourceInitializationException;
038    import org.apache.uima.util.FileUtils;
039    import org.apache.uima.util.Level;
040    import org.apache.uima.util.Progress;
041    import org.apache.uima.util.ProgressImpl;
042    import org.cleartk.syntax.treebank.TreebankGoldAnnotator;
043    import org.cleartk.util.DocumentUtil;
044    import org.cleartk.util.ListSpecification;
045    import org.cleartk.util.UIMAUtil;
046    
047    
048    /**
049    /**
050     * <br>
051     * Copyright (c) 2007-2008, Regents of the University of Colorado <br>
052     * All rights reserved.
053     * 
054     * 
055     * 
056     * <p>
057     * PennTreebankReader reads in the PennTreebank (PTB) data distributed by the
058     * LDC. It simply reads the raw treebank data into a view called "TreebankView".
059     * To actually parse the treebank data and post it to the CAS, you will need to
060     * use the TreebankGoldAnnotator which does the real work of parsing the
061     * treebank format. In general, treebank data can be read in by a
062     * PlainTextCollectionReader or some other simple collection reader. This class
063     * exists because the PennTreebank has a specific directory structure that
064     * corresponds to sections which are often used in specific ways to conduct
065     * experiments - e.g. section 02-20 for training and sections 21-24 for testing.
066     * This collection reader makes it easy to read in specific sections for later
067     * processing.  Only files ending with ".mrg" will be read in.  
068     * </p>
069     * <p>
070     * The acronym WSJ stands for Wall Street Journal which is the source of the
071     * articles treebanked by PTB.
072     * <p>
073     * If you are concerned that your copy of PTB is not working correctly with this collection reader (or TreebankGoldAnnotator)
074     * then please see the corresponding unit tests.
075     * 
076     * @see TreebankGoldAnnotator
077     * 
078     * @author Philip Ogren, Philipp Wetzler
079     */
080    public class PennTreebankReader extends CollectionReader_ImplBase {
081            /**
082             * "CorpusDirectory" is a single, required, string parameter that species
083             * the location of the treebank files. The directory should contain
084             * subdirectories corresponding to the sections (e.g. "00", "01", etc.) That
085             * is, if a local copy of PennTreebank sits at C:\Data\PTB\wsj\mrg, then the
086             * the subdirectory C:\Data\PTB\wsj\mrg\00 should exist. There are 24
087             * sections in PTB corresponding to the directories 00, 01, 02, ... 24.
088             */
089            public static final String PARAM_CORPUS_DIRECTORY = "CorpusDirectory";
090    
091            /**
092             * "Sections" is a single, optional, string parameter that specifies which
093             * sections of PTB to read in. The format allows for comma-separated section
094             * numbers and section ranges, for example "02,07-12,16". If not value is
095             * given for this parameter, then a default value of "00-24" will be used.
096             */
097            public static final String PARAM_SECTIONS = "Sections";
098    
099            public static final String TREEBANK_VIEW_NAME = "TreebankView";
100            
101            protected File directory;
102    
103            protected LinkedList<File> files;
104    
105            protected int numberOfFiles;
106    
107            protected ListSpecification sections;
108    
109            @Override
110            public void initialize() throws ResourceInitializationException {
111                    String sectionsParam = (String) UIMAUtil.getDefaultingConfigParameterValue(getUimaContext(), PARAM_SECTIONS, "00-24");
112                    this.sections = new ListSpecification(sectionsParam);
113    
114                    String directoryName = (String) UIMAUtil.getRequiredConfigParameterValue(getUimaContext(), PARAM_CORPUS_DIRECTORY);
115                    
116                    this.directory = new File(directoryName);
117                    this.files = new LinkedList<File>();
118                    collectSections(new File(directory.getPath()), this.files, this.sections);
119                    Collections.sort(files);
120                    this.numberOfFiles = files.size();
121    
122                    super.initialize();
123            }
124    
125            /**
126             * This will add all the <tt>.mrg</tt> files in the given WSJ sections to
127             * <em>treebankFiles</em>.
128             * 
129             * @param wsjDirectory
130             *            is the top level of the WSJ part of Treebank. Underneath here
131             *            are the section subdirectories.
132             * @param treebankFiles
133             * @param wsjSections
134             *            is the set of sections to include.
135             */
136            public static void collectSections(File wsjDirectory, List<File> treebankFiles, ListSpecification wsjSections) {
137                    if (!wsjDirectory.isDirectory()) return;
138    
139                    for (File subFile : wsjDirectory.listFiles()) {
140                            if (!subFile.isDirectory()) continue;
141    
142                            try {
143                                    int section = Integer.valueOf(subFile.getName());
144    
145                                    if (!wsjSections.contains(section)) continue;
146                            }
147                            catch (NumberFormatException e) {
148                                    continue;
149                            }
150    
151                            collectFiles(subFile, treebankFiles);
152                    }
153            }
154    
155            static void collectFiles(File file, List<File> treebankFiles) {
156                    if (file.isFile() && file.getName().endsWith(".mrg")) {
157                            treebankFiles.add(file);
158                    }
159                    else if (file.isDirectory()) {
160                            for (File subFile : file.listFiles()) {
161                                    collectFiles(subFile, treebankFiles);
162                            }
163                    }
164            }
165    
166            /**
167             * Reads the next file and stores its text in <b>cas</b> as the
168             * "TreebankView" SOFA.
169             * 
170             * @param cas
171             * 
172             * @throws IOException
173             * @throws CollectionException
174             */
175            public void getNext(CAS cas) throws IOException, CollectionException {
176                    try {
177                            JCas tbView = cas.createView(TREEBANK_VIEW_NAME).getJCas();
178                            File treebankFile = files.removeFirst();
179    
180                            getUimaContext().getLogger().log(Level.FINEST, "reading treebank file: " + treebankFile.getPath());
181    
182                            tbView.setSofaDataString(FileUtils.file2String(treebankFile, "ascii"), "text/plain");
183    
184                            DocumentUtil.createDocument(tbView, treebankFile.getName(), treebankFile.getName());
185                    }
186                    catch (CASException casException) {
187                            throw new CollectionException(casException);
188                    }
189            }
190    
191            public void close() throws IOException {
192            }
193    
194            public Progress[] getProgress() {
195                    return new Progress[] { new ProgressImpl(numberOfFiles - files.size(), numberOfFiles, Progress.ENTITIES) };
196            }
197    
198            public boolean hasNext() throws IOException, CollectionException {
199                    if (files.size() > 0) return true;
200                    else return false;
201            }
202    
203    }
