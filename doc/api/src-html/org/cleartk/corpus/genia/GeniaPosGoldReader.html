061    public class GeniaPosGoldReader extends CollectionReader_ImplBase {
062    
063            public static final String GENIA_POS_VIEW_NAME = "GeniaPOSView";
064    
065            /**
066             * "ArticleIdsList" is a single, optional, string parameter that names the
067             * file used to specify the article ids that should be read in.
068             */
069            public static final String PARAM_ARTICLE_IDS_LIST = "ArticleIdsList";
070    
071            /**
072             * "GeniaCorpus" is a single, required, string parameter that names the file
073             * that is the Genia corpus to be loaded. A good value is probably
074             * "data/genia/GENIAcorpus3.02.pos.xml"
075             */
076            public static final String PARAM_GENIA_CORPUS = "GeniaCorpus";
077    
078            /**
079             * "LoadSentences" is a single, optional, boolean parameter that determines
080             * whether sentence annotations will be added from the Genia corpus. The
081             * default value of "true" is used if this parameter is unspecified.
082             */
083            public static final String PARAM_LOAD_SENTENCES = "LoadSentences";
084    
085            /**
086             * "LoadTokens" is a single, optional, boolean parameter that determines
087             * whether tokens annotations will be added from the Genia corpus. The
088             * default value of "true" is used if this parameter is unspecified.
089             */
090            public static final String PARAM_LOAD_TOKENS = "LoadTokens";
091    
092            /**
093             * "LoadPOSTags" is a single, optional, boolean parameter that determines
094             * whether the part of speech tags assigned to each token in the genia
095             * corpus will be loaded. The default value of "true" is used if this
096             * parameter is unspecified. If "LoadTokens" is 'false', then "LoadPOSTags"
097             * will also be 'false' regardless of what is given in the descriptor file.
098             */
099            public static final String PARAM_LOAD_POS_TAGS = "LoadPOSTags";
100    
101            boolean filterArticles;
102    
103            Set<String> articleIds;
104    
105            GeniaPOSParser parser;
106    
107            GeniaParse parse;
108    
109            int progress = 0;
110    
111            int tokenIndex = 0;
112    
113            boolean loadSentences = true;
114    
115            boolean loadTokens = true;
116    
117            boolean loadPOSTags = true;
118    
119            @Override
120            public void initialize() throws ResourceInitializationException {
121                    articleIds = new HashSet<String>();
122    
123                    try {
124                            String articleIdsList = (String) getConfigParameterValue(PARAM_ARTICLE_IDS_LIST);
125                            if (articleIdsList == null) {
126                                    filterArticles = false;
127                            }
128                            else {
129                                    filterArticles = true;
130                                    String[] ids = FileUtil.loadListOfStrings(new File(articleIdsList));
131                                    for (String id : ids) {
132                                            articleIds.add(id);
133                                    }
134                            }
135    
136                            String geniaCorpus = (String) getConfigParameterValue(PARAM_GENIA_CORPUS);
137                            parser = new GeniaPOSParser(new File(geniaCorpus));
138    
139                            Boolean bool = (Boolean) getConfigParameterValue(PARAM_LOAD_SENTENCES);
140                            if (bool != null) loadSentences = bool;
141                            bool = (Boolean) getConfigParameterValue(PARAM_LOAD_TOKENS);
142                            if (bool != null) loadTokens = bool;
143                            bool = (Boolean) getConfigParameterValue(PARAM_LOAD_POS_TAGS);
144                            if (bool != null) loadPOSTags = bool;
145                            loadPOSTags = loadTokens & loadPOSTags;
146    
147                    }
148                    catch (IOException ioe) {
149                            throw new ResourceInitializationException(ioe);
150                    }
151                    catch (JDOMException je) {
152                            throw new ResourceInitializationException(je);
153                    }
154            }
155    
156            public void getNext(CAS cas) throws IOException, CollectionException {
157                    if(!hasNext()) 
158                            throw new CollectionException("Should not be calling getNext() because hasNext returns false", null);
159                    try {
160                            JCas jCas = cas.getJCas();
161                            String text = parse.getText();
162                            jCas.setDocumentText(text);
163    
164                            List<GeniaSentence> sentences = parse.getSentences();
165    
166                            for (GeniaSentence geniaSentence : sentences) {
167                                    if (loadTokens) {
168                                            List<GeniaTag> posTags = geniaSentence.getPosTags();
169                                            for (GeniaTag posTag : posTags) {
170                                                    Span tokenSpan = posTag.getSpans().get(0);
171                                                    Token token = new Token(jCas, tokenSpan.getBegin(), tokenSpan.getEnd());
172                                                    if (loadPOSTags) token.setPos(posTag.getLabel());
173                                                    token.addToIndexes();
174                                            }
175                                    }
176                                    if (loadSentences) {
177                                            Sentence sentence = new Sentence(jCas, geniaSentence.getSpan().getBegin(), geniaSentence.getSpan()
178                                                            .getEnd());
179                                            sentence.addToIndexes();
180                                    }
181                            }
182    
183                            Document document = new Document(jCas);
184                            document.setIdentifier(parse.getMedline());
185                            document.addToIndexes();
186    
187                            JCas geniaView = jCas.createView(GENIA_POS_VIEW_NAME);
188                            geniaView.setDocumentText(parse.getXml());
189    
190                            parse = null;
191                    }
192                    catch (CASException ce) {
193                            throw new CollectionException(ce);
194                    }
195    
196            }
197    
198            public void close() throws IOException {
199            }
200    
201            public Progress[] getProgress() {
202                    if (filterArticles) {
203                            return new Progress[] { new ProgressImpl(progress, articleIds.size(), Progress.ENTITIES) };
204                    }
205                    else {
206                            return new Progress[] { new ProgressImpl(progress, 2000, Progress.ENTITIES) };
207                    }
208            }
209    
210            public boolean hasNext() throws IOException, CollectionException {
211                    if (parse != null) return true;
212                    while (parser.hasNext()) {
213                            parse = parser.next();
214                            if (!filterArticles) {
215                                    progress++;
216                                    return true;
217                            }
218                            if (articleIds.contains(parse.getMedline())) {
219                                    progress++;
220                                    return true;
221                            }
222                    }
223                    return false;
224            }
225    }
