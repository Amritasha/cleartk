060    public class TimeMLGoldAnnotator extends JCasAnnotator_ImplBase {
061            
062            /**
063             * "org.cleartk.corpus.timeml.TimeMLGoldAnnotator.PARAM_LOAD_TLINKS"
064             * is a single, optional, boolean parameter, defaulting to true, that when false
065             * indicates that annotation should not be created for TLINKs (though annotations
066             * will still be created for TIMEX3s, EVENTs, etc.). 
067             */
068            public static final String PARAM_LOAD_TLINKS = "org.cleartk.corpus.timeml.TimeMLGoldAnnotator.PARAM_LOAD_TLINKS";
069            
070            private boolean loadTLINKs;
071    
072            @Override
073            public void initialize(UimaContext context) throws ResourceInitializationException {
074                    super.initialize(context);
075                    this.loadTLINKs = (Boolean)UIMAUtil.getDefaultingConfigParameterValue(
076                                    context, TimeMLGoldAnnotator.PARAM_LOAD_TLINKS, true);
077            }
078    
079            @Override
080            public void process(JCas jCas) throws AnalysisEngineProcessException {
081                    JCas timemlView;
082                    JCas initialView;
083                    try {
084                            timemlView = jCas.getView(ViewNames.TIMEML);
085                            initialView = jCas.getView(ViewNames.DEFAULT);
086                    } catch (CASException e) {
087                            throw new AnalysisEngineProcessException(e);
088                    }
089                    
090                    String timeML = timemlView.getDocumentText();
091                    SAXBuilder builder = new SAXBuilder();
092                    builder.setDTDHandler(null);
093                    Element root;
094                    try {
095                            Document doc = builder.build(new StringReader(timeML));
096                            root = doc.getRootElement();
097                    } catch (JDOMException e) {
098                            throw new AnalysisEngineProcessException(e);
099                    } catch (IOException e) {
100                            throw new AnalysisEngineProcessException(e);
101                    }
102                    
103                    
104                    StringBuffer textBuffer = new StringBuffer();
105                    Map<String, Anchor> anchors = new HashMap<String, Anchor>();
106                    this.addAnnotations(initialView, root, textBuffer, anchors);
107                    initialView.setDocumentText(textBuffer.toString());
108            }
109            
110            private void addAnnotations(
111                            JCas jCas, Element element, StringBuffer textBuffer,
112                            Map<String, Anchor> anchors) {
113                    int startOffset = textBuffer.length();
114                    for (Object content: element.getContent()) {
115                            if (content instanceof org.jdom.Text) {
116                                    textBuffer.append(((org.jdom.Text)content).getText());
117                            } else if (content instanceof Element) {
118                                    this.addAnnotations(jCas, (Element)content, textBuffer, anchors);
119                            }
120                    }
121                    int endOffset = textBuffer.length();
122                    
123                    
124                    if (element.getName().equals("TIMEX3")) {
125                            Time time = new Time(jCas, startOffset, endOffset);
126                            TimeMLUtil.copyAttributes(element, time, jCas);
127                            anchors.put(time.getId(), time);
128                            time.addToIndexes();
129                    } else if (element.getName().equals("EVENT")) {
130                            Event event = new Event(jCas, startOffset, endOffset);
131                            TimeMLUtil.copyAttributes(element, event, jCas);
132                            anchors.put(event.getId(), event);
133                            event.addToIndexes();
134                    } else if (element.getName().equals("MAKEINSTANCE")) {
135                            String eventID = element.getAttributeValue("eventID");
136                            String eventInstanceID = element.getAttributeValue("eiid");
137                            Event event = (Event)anchors.get(eventID);
138                            anchors.put(eventInstanceID, event);
139                            eventInstanceID = event.getEventInstanceID();
140                            if (eventInstanceID == null) {
141                                    TimeMLUtil.copyAttributes(element, event, jCas);
142                            } else {
143                                    TimeMLUtil.removeInconsistentAttributes(element, event, jCas);
144                                    event.setId(eventID);
145                                    event.setEventInstanceID(eventInstanceID);
146                            }
147                    } else if (element.getName().equals("TLINK") && this.loadTLINKs) {
148                            TemporalLink temporalLink = new TemporalLink(jCas, startOffset, endOffset);
149                            TimeMLUtil.copyAttributes(element, temporalLink, jCas);
150                            String sourceID = this.getOneOf(element,
151                                            "eventInstanceID", "eventID", "timeID");
152                            String targetID = this.getOneOf(element,
153                                            "relatedToEventInstance", "relatedToEvent", "relatedToTime");
154                            Anchor source = this.getAnchor(anchors, sourceID);
155                            Anchor target = this.getAnchor(anchors, targetID);
156                            if (source instanceof Event) {
157                                    temporalLink.setEventID(source.getId());
158                            }
159                            if (target instanceof Event) {
160                                    temporalLink.setRelatedToEvent(target.getId());
161                            }
162                            temporalLink.setSource(source);
163                            temporalLink.setTarget(target);
164                            temporalLink.addToIndexes();
165                    } else if (element.getName().equals("TEXT")) {
166                            Text text = new Text(jCas, startOffset, endOffset);
167                            text.addToIndexes();
168                    }
169            }
170            
171            private String getOneOf(Element element, String ... attributeNames) {
172                    for (String name: attributeNames) {
173                            String result = element.getAttributeValue(name);
174                            if (result != null) {
175                                    return result;
176                            }
177                    }
178                    throw new RuntimeException(String.format(
179                                    "unable to find in %s any of the following attributes: %s",
180                                    element, Arrays.asList(attributeNames)));
181            }
182            
183            private Anchor getAnchor(Map<String, Anchor> anchors, String id) {
184                    Anchor anchor = anchors.get(id);
185                    if (anchor == null) {
186                            throw new RuntimeException(String.format("no anchor for id %s", id));
187                    }
188                    return anchor;
189            }
