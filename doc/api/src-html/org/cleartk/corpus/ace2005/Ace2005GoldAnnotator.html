054    /**
055     * <br>Copyright (c) 2007-2008, Regents of the University of Colorado 
056     * <br>All rights reserved.
057    
058     *
059     * This parser works with the ACE2005CollectionReader to pre-populate the JCas with 
060     * ACE named entities.  Currently, it only loads information from the entity and 
061     * entity_mention tags and adds NamedEntity and NamedEntityMention objects to the 
062     * JCas.  
063     * 
064     * @author Philip Ogren
065     *
066     */
067    public class Ace2005GoldAnnotator extends JCasAnnotator_ImplBase
068    {
069    
070            Pattern ampPattern;
071            
072            @Override
073            public void initialize(UimaContext aContext) throws ResourceInitializationException
074            {
075                    ampPattern = Pattern.compile(Pattern.quote("&amp;"));
076            }
077    
078            @Override
079            public void process(JCas jCas) throws AnalysisEngineProcessException
080            {
081                    try
082                    {
083                            String apfUri = jCas.getView(ViewNames.ACE_APF_URI).getSofaDataURI();
084                            JCas initialView = jCas.getView(ViewNames.DEFAULT);
085                            String documentText = initialView.getDocumentText();
086                            SAXBuilder builder = new SAXBuilder();
087                            builder.setDTDHandler(null);
088                            URI sofaDataURI = new URI(apfUri);
089                            Document doc = builder.build(new File(sofaDataURI));
090                            
091                            Element apfSource = doc.getRootElement();
092                            Element apfDocument = apfSource.getChild("document");
093                            List<?> apfEntities = apfDocument.getChildren("entity");
094                            for(int i=0; i<apfEntities.size(); i++)
095                            {
096                                    Element apfEntity = (Element) apfEntities.get(i);
097                                    NamedEntity namedEntity = new NamedEntity(initialView);
098                                    namedEntity.setEntityType(apfEntity.getAttributeValue("TYPE"));
099                                    namedEntity.setEntitySubtype(apfEntity.getAttributeValue("SUBTYPE"));
100                                    namedEntity.setEntityClass(apfEntity.getAttributeValue("CLASS"));
101                                    namedEntity.setEntityId(apfEntity.getAttributeValue("ID"));
102                                    namedEntity.addToIndexes();
103                                    
104                                    List<NamedEntityMention> mentions = new ArrayList<NamedEntityMention>();
105                                    
106                                    List<?> entityMentions = apfEntity.getChildren("entity_mention");
107                                    for(int j=0; j<entityMentions.size(); j++)
108                                    {
109                                            Element entityMention = (Element) entityMentions.get(j);
110                                            int start = Integer.parseInt(entityMention.getChild("extent").getChild("charseq").getAttributeValue("START"));
111                                            int end = Integer.parseInt(entityMention.getChild("extent").getChild("charseq").getAttributeValue("END"));
112                                            String givenText = entityMention.getChild("extent").getChild("charseq").getText();
113                                            String parsedText = documentText.substring(start, end+1);
114                                            Matcher ampMatcher = ampPattern.matcher(parsedText);
115                                            parsedText = ampMatcher.replaceAll("&");
116    
117                                            NamedEntityMention mention = new NamedEntityMention(initialView, start, end+1);
118                                            mention.setMentionType(entityMention.getAttributeValue("TYPE"));
119                                            mention.setMentionedEntity(namedEntity);
120                                            
121                                            Chunk chunk = new Chunk(initialView, start, end+1);
122                                            List<Token> chunkTokens = AnnotationRetrieval.getAnnotations(initialView, chunk, Token.class);
123                                            chunk.setTokens(UIMAUtil.toFSArray(initialView, chunkTokens));
124                                            mention.setAnnotation(chunk);
125                                            
126                                            int headStart = Integer.parseInt(entityMention.getChild("head").getChild("charseq").getAttributeValue("START"));
127                                            int headEnd = Integer.parseInt(entityMention.getChild("head").getChild("charseq").getAttributeValue("END"));
128                                            Chunk head = new Chunk(initialView, headStart, headEnd+1);
129                                            List<Token> headTokens = AnnotationRetrieval.getAnnotations(initialView, head, Token.class);
130                                            head.setTokens(UIMAUtil.toFSArray(initialView, headTokens));
131                                            mention.setHead(head);
132                                            
133                                            mention.addToIndexes();
134                                            mentions.add(mention);
135    
136                                            givenText = givenText.replaceAll("\\s+", " ");
137                                            parsedText = givenText.replaceAll("\\s+", " ");
138    
139    //                                      if(!givenText.equals(parsedText))
140    //                                      {
141    //                                              
142    //                                              System.out.println("given text and parsed text differ.");
143    //                                              System.out.println(givenText);
144    //                                              System.out.println(parsedText);
145    //                                              System.out.println(apfDocument.getAttributeValue("DOCID"));
146    //                                              System.out.println(apfEntity.getAttributeValue("ID"));
147    //                                              System.out.println(documentText);
148    //                                      }
149                                    }
150                                    namedEntity.setMentions(UIMAUtil.toFSArray(jCas, mentions));
151                            }
152    
153                    }
154                    catch(CASException ce)
155                    {
156                            throw new AnalysisEngineProcessException(ce);
157                    }
158                    catch(IOException ioe)
159                    {
160                            throw new AnalysisEngineProcessException(ioe);
161                    }
162                    catch(JDOMException je)
163                    {
164                            throw new AnalysisEngineProcessException(je);
165                    }
166                    catch(URISyntaxException use)
167                    {
168                            throw new AnalysisEngineProcessException(use);
169                    }
170            }
