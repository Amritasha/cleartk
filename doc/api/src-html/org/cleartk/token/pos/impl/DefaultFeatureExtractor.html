058    public class DefaultFeatureExtractor implements POSFeatureExtractor<Token, Sentence>, Initializable{
059    
060            private List<SimpleFeatureExtractor> simpleExtractors;
061            private List<WindowExtractor> windowExtractors;
062            private List<WindowNGramExtractor> windowNGramExtractors;
063            
064            public void initialize(UimaContext context) throws ResourceInitializationException {
065                    simpleExtractors = new ArrayList<SimpleFeatureExtractor>();
066    
067                    SimpleFeatureExtractor wordExtractor = new SpannedTextExtractor();
068                    SimpleFeatureExtractor stemExtractor = new TypePathExtractor(Token.class, "stem");
069    
070                    int fromLeft = CharacterNGramProliferator.LEFT_TO_RIGHT;
071                    int fromRight = CharacterNGramProliferator.RIGHT_TO_LEFT;
072                    simpleExtractors.add(new ProliferatingExtractor(
073                                    wordExtractor,
074                                    new LowerCaseProliferator(),
075                                    new CapitalTypeProliferator(),
076                                    new NumericTypeProliferator(),
077                                    new CharacterNGramProliferator(fromLeft, 0, 2),
078                                    new CharacterNGramProliferator(fromLeft, 0, 3),
079                                    new CharacterNGramProliferator(fromRight, 0, 2),
080                                    new CharacterNGramProliferator(fromRight, 0, 3)));
081                    simpleExtractors.add(stemExtractor);
082    
083                    
084                    windowExtractors = new ArrayList<WindowExtractor>();
085                    
086                    windowExtractors.add(new WindowExtractor(
087                                    Token.class, stemExtractor, WindowFeature.ORIENTATION_LEFT, 0, 3));
088                    windowExtractors.add(new WindowExtractor(
089                                    Token.class, stemExtractor, WindowFeature.ORIENTATION_RIGHT, 0, 3));
090                    
091                    windowNGramExtractors = new ArrayList<WindowNGramExtractor>();
092                    windowNGramExtractors.add(new WindowNGramExtractor(
093                                    Token.class, stemExtractor, WindowNGramFeature.ORIENTATION_LEFT,WindowNGramFeature.DIRECTION_LEFT_TO_RIGHT,"_", 0, 2));
094                    windowNGramExtractors.add(new WindowNGramExtractor(
095                                    Token.class, stemExtractor, WindowNGramFeature.ORIENTATION_LEFT,WindowNGramFeature.DIRECTION_LEFT_TO_RIGHT,"_", 1, 3));
096                    windowNGramExtractors.add(new WindowNGramExtractor(
097                                    Token.class, stemExtractor, WindowNGramFeature.ORIENTATION_RIGHT,WindowNGramFeature.DIRECTION_LEFT_TO_RIGHT,"_", 0, 2));
098                    windowNGramExtractors.add(new WindowNGramExtractor(
099                                    Token.class, stemExtractor, WindowNGramFeature.ORIENTATION_RIGHT,WindowNGramFeature.DIRECTION_LEFT_TO_RIGHT,"_", 1, 3));
100    
101                    
102            }
103    
104            public List<Feature> extractFeatures(JCas jCas, Token token, Sentence sentence) {
105                    List<Feature> features = new ArrayList<Feature>();
106    
107                    for (SimpleFeatureExtractor extractor: simpleExtractors) {
108                            features.addAll(extractor.extract(jCas, token));
109                    }
110                    
111                    for (WindowExtractor extractor: windowExtractors) {
112                            features.addAll(extractor.extract(jCas, token, sentence));
113                    }
114    
115                    for (WindowNGramExtractor extractor: windowNGramExtractors) {
116                            features.add(extractor.extract(jCas, token, sentence));
117                    }
118                    
119                    return features;
120            }
121    
122    
123    }
