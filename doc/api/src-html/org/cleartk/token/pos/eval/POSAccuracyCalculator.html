044            public Results calculate(List<Tag> sysoutTags, Map<Integer, Tag> goldstandardTags, boolean tokensOnly) {
045                    List<Tag> tps = new ArrayList<Tag>();
046                    List<Tag> fps = new ArrayList<Tag>();
047                    List<Tag> fns = new ArrayList<Tag>();
048                    
049                    for (Tag sysoutTag : sysoutTags) {
050                            if (goldstandardTags.containsKey(sysoutTag.getBegin())) {
051                                    Tag goldstandardTag = goldstandardTags.get(sysoutTag.getBegin());
052                                    if (sysoutTag.getText().equals(goldstandardTag.getText())
053                                                    && (tokensOnly || sysoutTag.getPos().equals(goldstandardTag.getPos()))) {
054                                            tps.add(sysoutTag);
055                                            goldstandardTags.remove(goldstandardTag.getBegin());
056                                    } else {
057                                            fps.add(sysoutTag);
058                                    }
059                            } else {
060                                    fps.add(sysoutTag);
061                            }
062                    }
063                    fns.addAll(goldstandardTags.values());
064    
065                    return new Results(tps, fps, fns);
066            }
067    
068            private Tag parseTag(String line) {
069                    String[] columns = line.split("\t");
070                    String text = columns[0];
071                    int begin = Integer.parseInt(columns[1]);
072                    int end = Integer.parseInt(columns[2]);
073                    String pos = columns[3];
074                    return new Tag(text, pos, begin, end);
075            }
076            // assert there is no two tags with the same begin
077             private List<Tag> getTags(List<String> lines){
078                     List<Tag> tags = new ArrayList<Tag>();
079                     for(String line : lines) {
080                             tags.add(parseTag(line));
081                     }
082                     return tags;
083             }
084             
085             private Map<Integer, Tag> getTagMap(List<Tag> tags){
086                     Map<Integer, Tag> tagMap = new HashMap<Integer, Tag>();
087                     for(Tag tag : tags) {
088                             tagMap.put(tag.getBegin(), tag);
089                     }
090                     return tagMap;
091             }
092    
093             private List<String> getLines(String[] allLines, int offset){
094                     List<String> lines = new ArrayList<String>();
095                     for(int i=offset; i<allLines.length; i++) {
096                             String line = allLines[i];
097                             if(line.indexOf("\t") == -1)
098                                     return lines;
099                             lines.add(line);
100                             
101                     }
102                     return lines;
103             }
104             public static void main(String[] args) {
105                    try {
106                            String goldStandardFileName = args[0];
107                            String systemOutputFileName = args[1];
108                            File resultsDirectory = new File(args[2]);
109                            if(!resultsDirectory.exists()) {
110                                    resultsDirectory.mkdirs();
111                            }
112                            File resultsPrefix = new File(args[3]);
113                            boolean evaluateTokensOnly = Boolean.parseBoolean(args[4]);
114                            
115                            PrintStream tpsOut = new PrintStream(new File(resultsDirectory, resultsPrefix+"TPs.txt"));
116                            PrintStream fpsOut = new PrintStream(new File(resultsDirectory, resultsPrefix+"FPs.txt"));
117                            PrintStream fnsOut = new PrintStream(new File(resultsDirectory, resultsPrefix+"FNs.txt"));
118                            PrintStream summaryOut = new PrintStream(new File(resultsDirectory, resultsPrefix+"summary.txt"));
119                            PrintStream resultsOut = new PrintStream(new File(resultsDirectory, resultsPrefix+"results.txt"));
120                            resultsOut.println("document\tTPs\tFPs\tFNs");
121                            
122                            int totalTPs = 0;
123                            int totalFPs = 0;
124                            int totalFNs = 0;
125                            
126                            POSAccuracyCalculator calculator = new POSAccuracyCalculator();
127                            
128                            String[] goldStandardLines = FileUtil.loadListOfStrings(new File(goldStandardFileName));
129                            String[] systemOutLines = FileUtil.loadListOfStrings(new File(systemOutputFileName));
130                            
131                            int gsOffset = 0;
132                            int soOffset = 0;
133                            
134                            
135                            while(gsOffset < goldStandardLines.length) {
136                                    String goldStandardId = goldStandardLines[gsOffset++];
137                                    String systemOutId = systemOutLines[soOffset++];
138                                    if(!goldStandardId.equals(systemOutId)) {
139                                            System.out.println("identifiers differ:");
140                                            System.out.println("\tgold-standard id = "+goldStandardId);
141                                            System.out.println("\tsystem output id = "+systemOutId);
142                                            break;
143                                    }
144                                    
145                                    List<String> gsLines = calculator.getLines(goldStandardLines, gsOffset);
146                                    gsOffset += gsLines.size();
147                                    
148                                    Map<Integer, Tag> goldstandardTags = calculator.getTagMap(calculator.getTags(gsLines));
149                                    
150                                    List<String> sysoutLines = calculator.getLines(systemOutLines, soOffset);
151                                    soOffset += sysoutLines.size();
152                                    
153                                    List<Tag> sysoutTags = calculator.getTags(sysoutLines);
154                                    Results results = calculator.calculate(sysoutTags, goldstandardTags, evaluateTokensOnly);
155                                    totalTPs += results.getTps().size();
156                                    totalFPs += results.getFps().size();
157                                    totalFNs += results.getFns().size();
158                                    
159                                    tpsOut.println(goldStandardId);
160                                    for(Tag tag : results.getTps()) {
161                                            tpsOut.println(tag.getText()+"\t"+tag.getBegin()+"\t"+tag.getEnd()+"\t"+tag.getPos());
162                                    }
163    
164                                    fpsOut.println(goldStandardId);
165                                    for(Tag tag : results.getFps()) {
166                                            fpsOut.println(tag.getText()+"\t"+tag.getBegin()+"\t"+tag.getEnd()+"\t"+tag.getPos());
167                                    }
168    
169                                    fnsOut.println(goldStandardId);
170                                    for(Tag tag : results.getFns()) {
171                                            fnsOut.println(tag.getText()+"\t"+tag.getBegin()+"\t"+tag.getEnd()+"\t"+tag.getPos());
172                                    }
173                                    
174                                    resultsOut.println(goldStandardId+"\t"+results.getTps().size()+"\t"+results.getFps().size()+"\t"+results.getFns().size());
175                                    
176                            }
177    
178                            summaryOut.println("true positives = "+totalTPs);
179                            summaryOut.println("false positives = "+totalFPs);
180                            summaryOut.println("false negatives = "+totalFNs);
181                            
182                            double precision = (double) totalTPs / (double) (totalTPs + totalFPs);
183                            double recall = (double) totalTPs / (double) (totalTPs + totalFNs);
184                            double fmeasure = 2 * (double) totalTPs / (double) (2*totalTPs + totalFPs + totalFNs);
185                            
186                            summaryOut.println("precision = "+precision);
187                            summaryOut.println("recall = "+recall);
188                            summaryOut.println("fmeasure = "+fmeasure);
189                            summaryOut.println(""+precision+"\t"+recall+"\t"+fmeasure);
190    
191                    } catch(Exception e) {
192                            e.printStackTrace();
193                    }
194            }
