024    package org.cleartk.token.util;

026    import java.util.ArrayList;
027    import java.util.List;
028    
029    /**

032

034     */
035    public abstract class Tokenizer_ImplBase implements Tokenizer {
036    
037            public abstract String[] getTokenTexts(String text);
038    
039            /**
040             * performs tokenization on the input text and returns the character offsets
041             * corresponding to the begining and end of each token w.r.t. the input
042             * text.
043             * 
044             * @param text
045             *            the text that you want tokenized.
046             * @return offsets corresponding to the begining and end of each token. The
047             *         dimensions of the returned array will be number of tokens
048             *         returned by getTokens by 2 (one for token begin and one for token
049             *         end).
050             * 
051             *         <pre>
052             * 
053             *         <code> int[][] offsets = getTokenOffsets("example text"); int
054             *         beginFirstToken = offsets[0][0]; int endFirstToken =
055             *         offsets[0][1]; int begin4thToken = offsets[3][0]; int end4thToken
056             *         = offsets[3][1];
057             * 
058             *         </pre>
059             * 
060             *         </code>
061             * 
062             */
063            public List<Token> getTokens(String text) {
064                    List<Token> returnValues = new ArrayList<Token>();
065                    String[] tokenTexts = getTokenTexts(text);
066                    if (text.length() > 0) {
067                            int offset = 0;
068                            for (String tokenText : tokenTexts) {
069                                    while (text.charAt(offset) != tokenText.charAt(0))
070                                            offset++;
071    
072                                    Token token = new Token(offset, offset + tokenText.length(), tokenText);
073                                    returnValues.add(token);
074                                    offset += tokenText.length();
075                            }
076                    }
077    
078                    return returnValues;
079            }
080    
081    }
