059    public class TokenAnnotator extends JCasAnnotator_ImplBase {
060            /**
061             * "org.cleartk.token.TokenAnnotator.PARAM_TOKENIZER" is a single, optional,
062             * string parameter that specifies the class type of the tokenizer that will
063             * be used by this annotator. If this parameter is not filled, then the
064             * default tokenenizer (org.cleartk.token.util.PennTreebankTokenizer) is
065             * used. A tokenenizer is defined as any implementation of the interface
066             * defined by org.cleartk.token.util.Tokenizer.
067             * 
068             * @see Tokenizer
069             * @see PennTreebankTokenizer
070             */
071            public static final String PARAM_TOKENIZER = "org.cleartk.token.TokenAnnotator.PARAM_TOKENIZER";
072    
073            /**
074             * "org.cleartk.token.TokenAnnotator.PARAM_TOKEN_TYPE" is a single,
075             * optional, string parameter the class type of the tokens that are created
076             * by this annotator. If this parameter is not filled, then tokens of type
077             * org.cleartk.type.Token will be created.
078             */
079            public static final String PARAM_TOKEN_TYPE = "org.cleartk.token.TokenAnnotator.PARAM_TOKEN_TYPE";
080    
081            /**
082             * "org.cleartk.token.TokenAnnotator.PARAM_WINDOW_TYPE" is a single,
083             * optional, string parameter that specifies the class type of annotations
084             * that will be tokenized. If no value is given, then the entire document
085             * will be tokenized at once. A good value for this parameter would be
086             * <code>org.cleartk.type.Sentence</code> (especially when using the
087             * PennTreebankTokenizer).
088             */
089            public static final String PARAM_WINDOW_TYPE = "org.cleartk.token.TokenAnnotator.PARAM_WINDOW_TYPE";
090    
091            Tokenizer tokenizer;
092    
093            Class<? extends Annotation> tokenClass;
094    
095            Constructor<? extends Annotation> tokenConstructor;
096    
097            private Class<? extends Annotation> windowAnnotationClass;
098    
099            private Type windowAnnotationType = null;
100    
101            private boolean typesInitialized = false;
102    
103            public void initialize(UimaContext uimaContext) throws ResourceInitializationException {
104                    try {
105                            super.initialize(uimaContext);
106    
107                            String tokenCreatorClassName = (String) uimaContext.getConfigParameterValue(PARAM_TOKENIZER);
108                            if (tokenCreatorClassName != null && !tokenCreatorClassName.trim().equals("")) {
109                                    Class<?> cls = Class.forName(tokenCreatorClassName);
110                                    Class<? extends Tokenizer> tokensClass = cls.asSubclass(Tokenizer.class);
111                                    tokenizer = tokensClass.newInstance();
112                            }
113                            else {
114                                    tokenizer = new PennTreebankTokenizer();
115                                    windowAnnotationClass = Sentence.class;
116                            }
117                            String tokenClassName = (String) uimaContext.getConfigParameterValue(PARAM_TOKEN_TYPE);
118                            if (tokenClassName != null && !tokenClassName.trim().equals("")) {
119                                    Class<?> cls = Class.forName(tokenClassName);
120                                    tokenClass = cls.asSubclass(Annotation.class);
121                            }
122                            else {
123                                    tokenClass = org.cleartk.type.Token.class;
124                            }
125                            tokenConstructor = tokenClass.getConstructor(new Class[] { JCas.class, Integer.TYPE, Integer.TYPE });
126    
127                            String windowAnnotationClassName = (String) uimaContext.getConfigParameterValue(PARAM_WINDOW_TYPE);
128                            if (windowAnnotationClassName != null && !windowAnnotationClassName.trim().equals("")) {
129                                    Class<?> cls = Class.forName(windowAnnotationClassName);
130                                    windowAnnotationClass = cls.asSubclass(Annotation.class);
131                            }
132    
133                    }
134                    catch (Exception e) {
135                            throw new ResourceInitializationException(e);
136                    }
137    
138            }
139    
140            private void initializeTypes(JCas jCas) throws AnalysisEngineProcessException {
141                    if (windowAnnotationClass != null) {
142                            windowAnnotationType = UIMAUtil.getCasType(jCas, windowAnnotationClass);
143                    }
144                    typesInitialized = true;
145            }
146    
147            public void process(JCas jCas) throws AnalysisEngineProcessException {
148                    try {
149                            if (!typesInitialized) initializeTypes(jCas);
150                            if (windowAnnotationType != null) {
151                                    FSIterator windows = jCas.getAnnotationIndex(windowAnnotationType).iterator();
152                                    while (windows.hasNext()) {
153                                            Annotation window = (Annotation) windows.next();
154                                            List<Token> pojoTokens = tokenizer.getTokens(window.getCoveredText());
155                                            createTokens(pojoTokens, window.getBegin(), jCas);
156                                    }
157                            }
158                            else {
159                                    String text = jCas.getDocumentText();
160                                    List<Token> pojoTokens = tokenizer.getTokens(text);
161                                    createTokens(pojoTokens, 0, jCas);
162                            }
163                    }
164                    catch (Exception e) {
165                            throw new AnalysisEngineProcessException(e);
166                    }
167            }
168    
169            private void createTokens(List<Token> pojoTokens, int offset, JCas jCas) throws InstantiationException,
170                            InvocationTargetException, IllegalAccessException {
171                    for (Token pojoToken : pojoTokens) {
172                            int tokenBegin = pojoToken.getBegin() + offset;
173                            int tokenEnd = pojoToken.getEnd() + offset;
174                            tokenConstructor.newInstance(jCas, tokenBegin, tokenEnd).addToIndexes();
175                    }
176            }
177    
178            public static AnalysisEngineDescription getDescription() throws ResourceInitializationException {
179                    return AnalysisEngineFactory.createPrimitiveAnalysisEngineDescription(TokenAnnotator.class,
180                                    ClearTKComponents.TYPE_SYSTEM_DESCRIPTION, ClearTKComponents.TYPE_PRIORITIES);
181    
182            }
183    
184            public static AnalysisEngineDescription getDescription(TypeSystemDescription typeSystemDescription,
185                            TypePriorities typePriorities, Class<? extends Tokenizer> tokenizerClass,
186                            Class<? extends Annotation> tokenClass, Class<? extends Annotation> windowClass)
187                            throws ResourceInitializationException {
188                    return AnalysisEngineFactory.createPrimitiveAnalysisEngineDescription(TokenAnnotator.class,
189                                    typeSystemDescription, typePriorities, TokenAnnotator.PARAM_TOKENIZER, tokenizerClass.getName(),
190                                    TokenAnnotator.PARAM_TOKEN_TYPE, tokenClass.getName(), TokenAnnotator.PARAM_WINDOW_TYPE, windowClass
191                                                    .getName());
192            }
